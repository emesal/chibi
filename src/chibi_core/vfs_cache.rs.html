1//! VFS-backed tool output cache.
2//!
3//! Stores large tool outputs under `vfs:///sys/tool_cache/<context>/<id>`,
4//! written as SYSTEM_CALLER and world-readable. Replaces the old `cache.rs`
5//! flat-file system.
6
7use std::collections::hash_map::DefaultHasher;
8use std::hash::{Hash, Hasher};
9use std::time::{SystemTime, UNIX_EPOCH};
10
11/// Generate a unique cache ID.
12/// Format: `{tool}_{timestamp_hex}_{args_hash}` — globally unique, no collision risk.
13pub fn generate_cache_id(tool_name: &str, args: &serde_json::Value) -> String {
14    let timestamp = SystemTime::now()
15        .duration_since(UNIX_EPOCH)
16        .unwrap_or_default()
17        .as_secs();
18
19    let mut hasher = DefaultHasher::new();
20    args.to_string().hash(&mut hasher);
21    let hash = hasher.finish();
22
23    format!("{}_{:x}_{:08x}", tool_name, timestamp, hash as u32)
24}
25
26/// Return the VFS path (without URI scheme) for a cache entry.
27pub fn vfs_path_for(context_name: &str, cache_id: &str) -> String {
28    format!("/sys/tool_cache/{}/{}", context_name, cache_id)
29}
30
31/// Return the `vfs:///` URI for a cache entry.
32///
33/// `vfs_path_for` always returns a path beginning with `/`, so `"vfs://"` +
34/// `"/sys/..."` concatenates to `"vfs:///sys/..."` — three slashes as required.
35pub fn vfs_uri_for(context_name: &str, cache_id: &str) -> String {
36    format!("vfs://{}", vfs_path_for(context_name, cache_id))
37}
38
39/// Check if content should be cached based on size threshold.
40/// Does not cache empty or whitespace-only content.
41pub fn should_cache(content: &str, threshold: usize) -> bool {
42    if content.trim().is_empty() {
43        return false;
44    }
45    content.len() > threshold
46}
47
48/// Generate the truncated stub message shown to the LLM instead of the full output.
49pub fn truncated_message(
50    vfs_uri: &str,
51    tool_name: &str,
52    content: &str,
53    preview_chars: usize,
54) -> String {
55    let char_count = content.len();
56    let token_estimate = char_count / 4;
57    let line_count = content.lines().count();
58
59    let preview: String = content.chars().take(preview_chars).collect();
60    let preview = if let Some(pos) = preview.rfind('\n') {
61        &preview[..pos]
62    } else {
63        &preview
64    };
65
66    format!(
67        "[Output cached: {vfs_uri}]\n\
68         Tool: {tool_name} | Size: {char_count} chars, ~{token_estimate} tokens | Lines: {line_count}\n\
69         Preview:\n\
70         ---\n\
71         {preview}\n\
72         ---\n\
73         Use file_head, file_tail, file_lines, file_grep with path=\"{vfs_uri}\" to examine."
74    )
75}

77#[cfg(test)]
78mod tests {
79    use super::*;
80
81    #[test]
82    fn test_should_cache_threshold() {
83        assert!(!should_cache("short", 100));
84        assert!(should_cache(&"x".repeat(101), 100));
85    }
86
87    #[test]
88    fn test_should_cache_empty() {
89        assert!(!should_cache("", 10));
90        assert!(!should_cache("   ", 10));
91        assert!(!should_cache("\n\n", 10));
92    }
93
94    #[test]
95    fn test_generate_cache_id_format() {
96        let id = generate_cache_id("web_fetch", &serde_json::json!({"url": "x"}));
97        assert!(id.starts_with("web_fetch_"));
98    }
99
100    #[test]
101    fn test_vfs_path_for() {
102        let p = vfs_path_for("myctx", "tool_abc_123");
103        assert_eq!(p, "/sys/tool_cache/myctx/tool_abc_123");
104    }
105
106    #[test]
107    fn test_vfs_uri_for() {
108        let uri = vfs_uri_for("myctx", "tool_abc_123");
109        assert_eq!(uri, "vfs:///sys/tool_cache/myctx/tool_abc_123");
110        assert!(crate::vfs::VfsPath::is_vfs_uri(&uri));
111    }
112
113    #[test]
114    fn test_truncated_message_contains_uri() {
115        let uri = "vfs:///sys/tool_cache/ctx/web_fetch_1_2";
116        let msg = truncated_message(uri, "web_fetch", "line1\nline2\nline3", 200);
117        assert!(msg.contains(uri));
118        assert!(msg.contains("web_fetch"));
119        assert!(msg.contains("file_head"));
120    }
121
122    #[test]
123    fn test_truncated_message_preview_truncates_at_line() {
124        let uri = "vfs:///sys/tool_cache/ctx/x";
125        // Preview of 6 chars from "abc\ndef" should stop at newline → "abc"
126        let msg = truncated_message(uri, "t", "abc\ndef\nghi", 6);
127        assert!(msg.contains("abc"));
128        assert!(!msg.contains("def"));
129    }
