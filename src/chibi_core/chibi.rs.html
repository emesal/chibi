1//! High-level facade for embedding chibi.
2//!
3//! The `Chibi` struct provides a clean, high-level API for embedding chibi
4//! in other applications. It wraps `AppState` and tool loading, providing
5//! a simpler interface for common operations.
6//!
7//! # Example
8//!
9//! ```no_run
10//! // Requires ~/.chibi directory with config.toml and models.toml.
11//! use chibi_core::{Chibi, CollectingSink, ResolvedConfig};
12//! use chibi_core::api::PromptOptions;
13//!
14//! #[tokio::main]
15//! async fn main() -> std::io::Result<()> {
16//!     let mut chibi = Chibi::load()?;
17//!
18//!     // Get the config for the default context
19//!     let config = chibi.resolve_config("default", None)?;
20//!
21//!     // Create options and a sink to collect the response
22//!     let options = PromptOptions::new(false, false, &[], false);
23//!     let mut sink = CollectingSink::new();
24//!
25//!     // Send a prompt to the default context
26//!     chibi.send_prompt_streaming("default", "Hello!", &config, &options, &mut sink).await?;
27//!
28//!     println!("Response: {}", sink.text);
29//!     Ok(())
30//! }
31//! ```

44/// Permission handler for gated operations (file writes, shell execution).
45///
46/// Receives hook data as JSON (containing tool_name, path/command, etc.).
47/// Returns `Ok(true)` to allow the operation, `Ok(false)` to deny.
48///
49/// The frontend (e.g. CLI) registers a handler that prompts the user
50/// interactively. When no handler is set, operations fail-safe to deny.
51pub type PermissionHandler = Box<dyn Fn(&serde_json::Value) -> io::Result<bool>>;

53/// Options for loading a Chibi instance.
54///
55/// Use `Default::default()` for standard behavior, or customize as needed.
56///
57/// # Example
58///
59/// ```
60/// use chibi_core::LoadOptions;
61/// use std::path::PathBuf;
62///
63/// // Default options use ~/.chibi or CHIBI_HOME
64/// let default_opts = LoadOptions::default();
65/// assert!(!default_opts.verbose);
66/// assert!(default_opts.home.is_none());
67/// assert!(default_opts.project_root.is_none());
68///
69/// // Custom options with verbose output and custom home
70/// let custom_opts = LoadOptions {
71///     verbose: true,
72///     home: Some(PathBuf::from("/custom/chibi")),
73///     project_root: Some(PathBuf::from("/my/project")),
74/// };
75/// ```
76#[derive(Debug, Clone, Default)]
77pub struct LoadOptions {
78    /// Print diagnostic info about tool loading to stderr.
79    pub verbose: bool,
80    /// Override the chibi home directory.
81    /// If `None`, uses `CHIBI_HOME` env var or `~/.chibi`.
82    pub home: Option<PathBuf>,
83    /// Override the project root directory.
84    /// If `None`, uses `CHIBI_PROJECT_ROOT` env var or current working directory.
85    pub project_root: Option<PathBuf>,
86}

88/// High-level facade for chibi embedding.
89///
90/// Provides a clean API for common chibi operations without exposing
91/// internal implementation details. Wraps `AppState` and manages tool loading.
92pub struct Chibi {
93    /// The underlying application state.
94    pub app: AppState,
95    /// Loaded tools (plugins from ~/.chibi/plugins/).
96    pub tools: Vec<Tool>,
97    /// Project root directory (always resolved, never None).
98    pub project_root: PathBuf,
99    /// Optional permission handler for gated operations.
100    /// If `None`, gated operations fail-safe to deny (unless a plugin approves).
101    permission_handler: Option<PermissionHandler>,
102}

104impl Chibi {
105    /// Load chibi from default home directory.
106    ///
107    /// Convenience method equivalent to `load_with_options(LoadOptions::default())`.
108    /// Kept as a shortcut for library users who don't need custom options.
109    ///
110    /// Uses the following precedence for the chibi directory:
111    /// 1. `CHIBI_HOME` environment variable
112    /// 2. `~/.chibi` default
113    pub fn load() -> io::Result<Self> {
114        Self::load_with_options(LoadOptions::default())
115    }
116
117    /// Load chibi from a specific home directory.
118    ///
119    /// Convenience method for `load_with_options(LoadOptions { home: Some(...), ..Default::default() })`.
120    /// Kept as a shortcut for library users who only need to override the home directory.
121    ///
122    /// This overrides both `CHIBI_HOME` and the default `~/.chibi`.
123    pub fn from_home(home: &Path) -> io::Result<Self> {
124        Self::load_with_options(LoadOptions {
125            home: Some(home.to_path_buf()),
126            ..Default::default()
127        })
128    }
129
130    /// Load chibi with custom options.
131    ///
132    /// This is the most flexible way to load chibi, allowing control over
133    /// both the home directory and verbose output.
134    ///
135    /// Project root resolution order: `options.project_root` > `CHIBI_PROJECT_ROOT` env > cwd.
136    ///
137    /// Sets `CHIBI_PROJECT_ROOT` and `CHIBI_INDEX_DB` environment variables so
138    /// plugins and hooks can discover the project root and index database path.
139    ///
140    /// # Example
141    ///
142    /// ```no_run
143    /// // Requires a chibi home directory with config.toml and models.toml.
144    /// use chibi_core::{Chibi, LoadOptions};
145    ///
146    /// let chibi = Chibi::load_with_options(LoadOptions {
147    ///     verbose: true,
148    ///     home: Some("/custom/path".into()),
149    ///     ..Default::default()
150    /// })?;
151    /// # Ok::<(), std::io::Error>(())
152    /// ```
153    pub fn load_with_options(options: LoadOptions) -> io::Result<Self> {
154        let app = AppState::load(options.home)?;
155        // CLI flag overrides config setting
156        let verbose = options.verbose || app.config.verbose;
157        let tools = tools::load_tools(&app.plugins_dir, verbose)?;
158        let project_root = resolve_project_root(options.project_root)?;
159
160        // Expose project root to plugins/hooks via environment variables.
161        // SAFETY: Called once during single-threaded initialization, before any
162        // plugin/hook child processes are spawned.
163        unsafe {
164            std::env::set_var("CHIBI_PROJECT_ROOT", &project_root);
165            std::env::set_var("CHIBI_INDEX_DB", project_index_db_path(&project_root));
166        }
167
168        Ok(Self {
169            app,
170            tools,
171            project_root,
172            permission_handler: None,
173        })
174    }

176    /// Set the permission handler for gated operations.
177    ///
178    /// The handler is called when a gated tool (write_file, file_edit, shell_exec)
179    /// is invoked and no plugin has denied the operation. If no handler is set,
180    /// operations fail-safe to deny.
181    pub fn set_permission_handler(&mut self, handler: PermissionHandler) {
182        self.permission_handler = Some(handler);
183    }

185    /// Initialize the session.
186    ///
187    /// Executes `OnStart` hooks. Call this once at the start of a session,
188    /// before any prompts are sent.
189    ///
190    /// # Example
191    ///
192    /// ```no_run
193    /// use chibi_core::Chibi;
194    ///
195    /// # fn example() -> std::io::Result<()> {
196    /// let chibi = Chibi::load()?;
197    /// chibi.init()?;
198    /// // ... use chibi ...
199    /// chibi.shutdown()?;
200    /// # Ok(())
201    /// # }
202    /// ```
203    pub fn init(&self) -> io::Result<Vec<(String, serde_json::Value)>> {
204        let hook_data = serde_json::json!({
205            "chibi_home": self.app.chibi_dir.to_string_lossy(),
206            "project_root": self.project_root.to_string_lossy(),
207            "tool_count": self.tools.len(),
208        });
209        tools::execute_hook(&self.tools, tools::HookPoint::OnStart, &hook_data)
210    }
211
212    /// Shutdown the session.
213    ///
214    /// Executes `OnEnd` hooks. Call this once at the end of a session,
215    /// after all prompts are complete.
216    pub fn shutdown(&self) -> io::Result<Vec<(String, serde_json::Value)>> {
217        let hook_data = serde_json::json!({
218            "chibi_home": self.app.chibi_dir.to_string_lossy(),
219            "project_root": self.project_root.to_string_lossy(),
220            "tool_count": self.tools.len(),
221        });
222        tools::execute_hook(&self.tools, tools::HookPoint::OnEnd, &hook_data)
223    }

225    /// Clear a context, executing PreClear/PostClear hooks.
226    ///
227    /// This wraps `AppState::clear_context` with hook execution.
228    pub fn clear_context(&self, context_name: &str) -> io::Result<()> {
229        // Get context info for hook data before clearing
230        let context = self.app.get_or_create_context(context_name)?;
231
232        let pre_hook_data = serde_json::json!({
233            "context_name": context_name,
234            "message_count": context.messages.len(),
235            "summary": context.summary,
236        });
237        let _ = tools::execute_hook(&self.tools, tools::HookPoint::PreClear, &pre_hook_data);
238
239        self.app.clear_context(context_name)?;
240
241        let post_hook_data = serde_json::json!({
242            "context_name": context_name,
243        });
244        let _ = tools::execute_hook(&self.tools, tools::HookPoint::PostClear, &post_hook_data);
245
246        Ok(())
247    }

249    /// Send a prompt with streaming output via a ResponseSink.
250    ///
251    /// This is the primary method for sending prompts to the LLM. The sink
252    /// receives streaming events as they occur, allowing for real-time output
253    /// or collection for later processing.
254    ///
255    /// # Arguments
256    ///
257    /// * `context_name` - The context to use for this prompt
258    /// * `prompt` - The user's prompt text
259    /// * `config` - Resolved configuration for this request
260    /// * `options` - Options controlling prompt execution behavior
261    /// * `sink` - A sink to receive response events
262    ///
263    /// # Example
264    ///
265    /// ```no_run
266    /// // Requires ~/.chibi directory and valid API key in config.
267    /// # use chibi_core::{Chibi, CollectingSink};
268    /// # use chibi_core::api::PromptOptions;
269    /// # async fn example() -> std::io::Result<()> {
270    /// # let chibi = Chibi::load()?;
271    /// # let config = chibi.resolve_config("default", None)?;
272    /// # let options = PromptOptions::new(false, false, &[], false);
273    /// let mut sink = CollectingSink::new();
274    /// chibi.send_prompt_streaming("default", "Hello", &config, &options, &mut sink).await?;
275    /// println!("Got response: {}", sink.text);
276    /// # Ok(())
277    /// # }
278    /// ```
279    pub async fn send_prompt_streaming<S: ResponseSink>(
280        &self,
281        context_name: &str,
282        prompt: &str,
283        config: &ResolvedConfig,
284        options: &PromptOptions<'_>,
285        sink: &mut S,
286    ) -> io::Result<()> {
287        send_prompt(
288            &self.app,
289            context_name,
290            prompt.to_string(),
291            &self.tools,
292            config,
293            options,
294            sink,
295            self.permission_handler.as_ref(),
296            self.home_dir(),
297            &self.project_root,
298        )
299        .await
300    }

302    /// Execute a tool by name with the given arguments.
303    ///
304    /// Tries built-in tools first, then falls back to loaded plugins.
305    ///
306    /// # Arguments
307    ///
308    /// * `context_name` - The context to use for file tools
309    /// * `name` - The tool name
310    /// * `args` - JSON arguments for the tool
311    ///
312    /// # Returns
313    ///
314    /// The tool's output as a string, or an error if the tool wasn't found
315    /// or execution failed.
316    pub async fn execute_tool(
317        &self,
318        context_name: &str,
319        name: &str,
320        args: serde_json::Value,
321    ) -> io::Result<String> {
322        // Try built-in tools first
323        if let Some(result) =
324            tools::execute_builtin_tool(&self.app, context_name, name, &args, None)
325        {
326            return result;
327        }
328
329        // Try file tools
330        if tools::is_file_tool(name) {
331            let config = self.app.resolve_config(context_name, None)?;
332            if let Some(result) =
333                tools::execute_file_tool(&self.app, context_name, name, &args, &config)
334            {
335                return result;
336            }
337        }
338
339        // Try coding tools
340        if tools::is_coding_tool(name) {
341            let project_root = std::env::var("CHIBI_PROJECT_ROOT")
342                .map(std::path::PathBuf::from)
343                .unwrap_or_else(|_| std::env::current_dir().unwrap_or_default());
344            if let Some(result) =
345                tools::execute_coding_tool(name, &args, &project_root, &self.tools).await
346            {
347                return result;
348            }
349        }
350
351        // Try agent tools
352        if tools::is_agent_tool(name) {
353            let config = self.app.resolve_config(context_name, None)?;
354            return tools::execute_agent_tool(&config, name, &args, &self.tools).await;
355        }
356
357        // Try plugins
358        if let Some(tool) = tools::find_tool(&self.tools, name) {
359            return tools::execute_tool(tool, &args, false);
360        }
361
362        Err(io::Error::new(
363            io::ErrorKind::NotFound,
364            format!("Tool '{}' not found", name),
365        ))
366    }

413/// Resolve project root: explicit path > `CHIBI_PROJECT_ROOT` env > VCS root > cwd.
414fn resolve_project_root(explicit: Option<PathBuf>) -> io::Result<PathBuf> {
415    if let Some(root) = explicit {
416        return Ok(root);
417    }
418    if let Ok(env_root) = std::env::var("CHIBI_PROJECT_ROOT")
419        && !env_root.is_empty()
420    {
421        return Ok(PathBuf::from(env_root));
422    }
423    let cwd = std::env::current_dir()?;
424    Ok(crate::vcs::detect_project_root(&cwd).unwrap_or(cwd))
425}

427/// Return the project-local chibi directory (`<project_root>/.chibi/`), creating it if absent.
428pub fn project_chibi_dir(root: &Path) -> io::Result<PathBuf> {
429    let dir = root.join(".chibi");
430    if !dir.exists() {
431        std::fs::create_dir_all(&dir)?;
432    }
433    Ok(dir)
434}

436/// Return the path to the project's codebase index database.
437pub fn project_index_db_path(root: &Path) -> PathBuf {
438    root.join(".chibi").join("codebase.db")
439}
