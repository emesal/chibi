1//! API request building utilities.
2//!
3//! This module handles constructing the request body for LLM API calls,
4//! applying all configuration parameters.
5
6use crate::config::{ResolvedConfig, ToolChoice, ToolChoiceMode};
7use crate::input::DebugKey;
8use serde_json::json;
9
10/// Options for controlling prompt execution behavior
11#[derive(Debug, Clone)]
12pub struct PromptOptions<'a> {
13    pub use_reflection: bool,
14    pub debug: &'a [DebugKey],
15    pub force_render: bool,
16    /// Optional override for the fallback handoff target
17    pub fallback_override: Option<crate::tools::HandoffTarget>,
18}
19
20impl<'a> PromptOptions<'a> {
21    pub fn new(use_reflection: bool, debug: &'a [DebugKey], force_render: bool) -> Self {
22        Self {
23            use_reflection,
24            debug,
25            force_render,
26            fallback_override: None,
27        }
28    }
29
30    /// Set the fallback handoff target override
31    pub fn with_fallback(mut self, fallback: crate::tools::HandoffTarget) -> Self {
32        self.fallback_override = Some(fallback);
33        self
34    }
35}
36
37/// Build the request body for the LLM API, applying all API parameters from ResolvedConfig
38pub fn build_request_body(
39    config: &ResolvedConfig,
40    messages: &[serde_json::Value],
41    tools: Option<&[serde_json::Value]>,
42    stream: bool,
43) -> serde_json::Value {
44    let mut body = json!({
45        "model": config.model,
46        "messages": messages,
47        "stream": stream,
48    });
49
50    // Add tools if provided
51    if let Some(tools) = tools
52        && !tools.is_empty()
53    {
54        body["tools"] = json!(tools);
55    }
56
57    // Apply API parameters from resolved config
58    let api = &config.api;
59
60    // Temperature
61    if let Some(temp) = api.temperature {
62        body["temperature"] = json!(temp);
63    }
64
65    // Max tokens
66    if let Some(max_tokens) = api.max_tokens {
67        body["max_tokens"] = json!(max_tokens);
68    }
69
70    // Top P
71    if let Some(top_p) = api.top_p {
72        body["top_p"] = json!(top_p);
73    }
74
75    // Stop sequences
76    if let Some(ref stop) = api.stop {
77        body["stop"] = json!(stop);
78    }
79
80    // Tool choice
81    if let Some(ref tool_choice) = api.tool_choice {
82        match tool_choice {
83            ToolChoice::Mode(mode) => {
84                let mode_str = match mode {
85                    ToolChoiceMode::Auto => "auto",
86                    ToolChoiceMode::None => "none",
87                    ToolChoiceMode::Required => "required",
88                };
89                body["tool_choice"] = json!(mode_str);
90            }
91            ToolChoice::Function { type_, function } => {
92                body["tool_choice"] = json!({
93                    "type": type_,
94                    "function": { "name": function.name }
95                });
96            }
97        }
98    }
99
100    // Parallel tool calls
101    if let Some(parallel) = api.parallel_tool_calls {
102        body["parallel_tool_calls"] = json!(parallel);
103    }
104
105    // Frequency penalty
106    if let Some(freq) = api.frequency_penalty {
107        body["frequency_penalty"] = json!(freq);
108    }
109
110    // Presence penalty
111    if let Some(pres) = api.presence_penalty {
112        body["presence_penalty"] = json!(pres);
113    }
114
115    // Seed
116    if let Some(seed) = api.seed {
117        body["seed"] = json!(seed);
118    }
119
120    // Response format
121    if let Some(ref format) = api.response_format {
122        body["response_format"] = serde_json::to_value(format).unwrap_or(json!(null));
123    }
124
125    // Reasoning configuration (OpenRouter-specific)
126    // Either effort OR max_tokens can be set, plus optional exclude/enabled
127    if !api.reasoning.is_empty() {
128        let mut reasoning_obj = json!({});
129        if let Some(ref effort) = api.reasoning.effort {
130            reasoning_obj["effort"] = json!(effort.as_str());
131        }
132        if let Some(max_tokens) = api.reasoning.max_tokens {
133            reasoning_obj["max_tokens"] = json!(max_tokens);
134        }
135        if let Some(exclude) = api.reasoning.exclude {
136            reasoning_obj["exclude"] = json!(exclude);
137        }
138        if let Some(enabled) = api.reasoning.enabled {
139            reasoning_obj["enabled"] = json!(enabled);
140        }
141        body["reasoning"] = reasoning_obj;
142    }
143
144    body
145}
