2//!
3//! This module provides different compaction strategies:
4//! - Rolling compaction: strips messages and integrates them into the summary
5//! - Full compaction: summarizes all messages and starts fresh

22/// Calculate how many messages to drop during rolling compaction.
23///
24/// Formula: `(count * percentage / 100).round().max(1)`
25pub(crate) fn drop_count(message_count: usize, drop_percentage: f32) -> usize {
26    ((message_count as f32 * drop_percentage / 100.0).round() as usize).max(1)
27}

29/// Collect `tool_call_id`s from messages that have `tool_calls` arrays.
30///
31/// Used to atomically drop tool results when their assistant message is dropped.
32pub(crate) fn collect_tool_call_ids(
33    messages: &[serde_json::Value],
34) -> std::collections::HashSet<String> {
35    let mut ids = std::collections::HashSet::new();
36    for m in messages {
37        if let Some(tool_calls) = m["tool_calls"].as_array() {
38            for tc in tool_calls {
39                if let Some(id) = tc["id"].as_str() {
40                    ids.insert(id.to_string());
41                }
42            }
43        }
44    }
45    ids
46}

48/// Filter messages for compaction: keep system messages unconditionally,
49/// drop messages whose `_id` is in `drop_ids`, and drop tool results whose
50/// `tool_call_id` matches a dropped assistant message's tool call.
51pub(crate) fn filter_messages(
52    messages: &[serde_json::Value],
53    drop_ids: &std::collections::HashSet<String>,
54    drop_tool_call_ids: &std::collections::HashSet<String>,
55) -> Vec<serde_json::Value> {
56    messages
57        .iter()
58        .filter(|m| {
59            let role = m["role"].as_str().unwrap_or("");
60            if role == "system" {
61                return true;
62            }
63            let id = m["_id"].as_str().unwrap_or("");
64            if drop_ids.contains(id) {
65                return false;
66            }
67            // Atomic tool exchange: drop tool results whose call was dropped
68            if role == "tool"
69                && let Some(tc_id) = m["tool_call_id"].as_str()
70                && drop_tool_call_ids.contains(tc_id)
71            {
72                return false;
73            }
74            true
75        })
76        .cloned()
77        .collect()
78}

80/// Rolling compaction: strips messages and integrates them into the summary
81/// This is triggered automatically when context exceeds threshold
82/// The LLM decides which messages to drop based on goals/tasks, with fallback to percentage
83pub async fn rolling_compact(
84    app: &AppState,
85    context_name: &str,
86    resolved_config: &ResolvedConfig,
87    sink: &dyn OutputSink,
88) -> io::Result<()> {
89    let mut context = app.get_or_create_context(context_name)?;
90
91    // Skip system messages when counting
92    let non_system_messages: Vec<&serde_json::Value> = context
93        .messages
94        .iter()
95        .filter(|m| m["role"].as_str() != Some("system"))
96        .collect();
97
98    if non_system_messages.len() <= 4 {
99        // Not enough messages to strip - too few to meaningfully compact
100        return Ok(());
101    }
102
103    // Execute pre_rolling_compact hook.
104    // TeinHookContext is None: compact runs from a background task that does not hold a
105    // full ToolCallContext (no per-request VFS caller or registry snapshot). Tein hook
106    // callbacks fire but cannot use call-tool or (harness io). Known limitation — promote
107    // to Some if a full async context is threaded through compact in the future.
108    let tools = tools::load_tools(&app.plugins_dir)?;
109    let hook_data = serde_json::json!({
110        "context_name": context.name,
111        "message_count": context.messages.len(),
112        "non_system_count": non_system_messages.len(),
113        "summary": context.summary,
114    });
115    let _ = tools::execute_hook(
116        &tools,
117        tools::HookPoint::PreRollingCompact,
118        &hook_data,
119        None,
120    );
121
122    // Load tasks and flock goals to guide compaction decisions.
123    let task_metas = crate::state::tasks::collect_tasks(&app.vfs, context_name).await;
124    let task_table = crate::state::tasks::build_summary_table(&task_metas);
125    let flock_contexts = load_flock_contexts(&app.vfs, context_name).unwrap_or_default();
126    let goals = format_flock_sections(&flock_contexts);
127
128    // Build message list in transcript format for LLM to analyze.
129    // For tool messages, include a summary representation.
130    let messages_for_llm: Vec<serde_json::Value> = non_system_messages
131        .iter()
132        .map(|m| {
133            let id = m["_id"].as_str().unwrap_or("");
134            let role = m["role"].as_str().unwrap_or("");
135            let content_repr = if let Some(tool_calls) = m["tool_calls"].as_array() {
136                // Assistant message with tool calls
137                let names: Vec<&str> = tool_calls
138                    .iter()
139                    .filter_map(|tc| tc["function"]["name"].as_str())
140                    .collect();
141                format!("[tool calls: {}]", names.join(", "))
142            } else {
143                let content = m["content"].as_str().unwrap_or("");
144                if content.len() > 500 {
145                    let end = content.floor_char_boundary(500);
146                    format!("{}... [truncated]", &content[..end])
147                } else {
148                    content.to_string()
149                }
150            };
151            serde_json::json!({
152                "id": id,
153                "role": role,
154                "content": content_repr,
155            })
156        })
157        .collect();
158
159    // Calculate target drop count based on config percentage
160    let drop_percentage = resolved_config.rolling_compact_drop_percentage;
161    let target_drop_count = drop_count(non_system_messages.len(), drop_percentage);
162
163    // Ask LLM which messages to drop.
164    // {GOALS} and {TASKS} expand to a labelled section or empty string — the
165    // adjacent placeholders in the template collapse cleanly when both are absent.
166    let decision_prompt = ROLLING_COMPACT_DECISION_TEMPLATE
167        .replace(
168            "{MESSAGES}",
169            &serde_json::to_string_pretty(&messages_for_llm).unwrap_or_default(),
170        )
171        .replace(
172            "{GOALS}",
173            &if goals.is_empty() {
174                String::new()
175            } else {
176                format!("{}\n\n", goals)
177            },
178        )
179        .replace(
180            "{TASKS}",
181            &if task_table.is_empty() {
182                String::new()
183            } else {
184                format!("CURRENT TASKS:\n{}\n\n", task_table)
185            },
186        )
187        .replace(
188            "{SUMMARY}",
189            if context.summary.is_empty() {
190                "(No existing summary)"
191            } else {
192                &context.summary
193            },
194        )
195        .replace("{TARGET_COUNT}", &target_drop_count.to_string());
196
197    // First LLM call: decide what to drop
198    let decision_messages = vec![json!({
199        "role": "user",
200        "content": decision_prompt,
201    })];
202
203    let ids_to_drop: Vec<String> = match gateway::chat(resolved_config, &decision_messages).await {
204        Ok(content) => {
205            // Try to parse as JSON array
206            serde_json::from_str(&content).unwrap_or_else(|_| {
207                // Try to extract JSON array from response if wrapped in other text
208                if let Some(start) = content.find('[') {
209                    if let Some(end) = content.rfind(']') {
210                        serde_json::from_str(&content[start..=end]).unwrap_or_default()
211                    } else {
212                        Vec::new()
213                    }
214                } else {
215                    Vec::new()
216                }
217            })
218        }
219        Err(_) => Vec::new(),
220    };
221
222    // Determine which messages to actually drop.
223    // Tool exchanges (assistant with tool_calls + subsequent tool results) are atomic:
224    // if the assistant message is dropped, drop associated tool results too.
225    let messages_to_drop: Vec<&serde_json::Value> = if ids_to_drop.is_empty() {
226        // Fallback: drop oldest N messages based on percentage
227        sink.emit_event(CommandEvent::RollingCompactionFallback {
228            drop_percentage: drop_percentage as f64,
229        });
230        non_system_messages
231            .iter()
232            .take(target_drop_count)
233            .copied()
234            .collect()
235    } else {
236        sink.emit_event(CommandEvent::RollingCompactionDecision {
237            archived: ids_to_drop.len(),
238        });
239        non_system_messages
240            .iter()
241            .filter(|m| {
242                let id = m["_id"].as_str().unwrap_or("");
243                ids_to_drop.iter().any(|drop_id| drop_id == id)
244            })
245            .copied()
246            .collect()
247    };
248
249    if messages_to_drop.is_empty() {
250        return Ok(());
251    }
252
253    // Build text of messages to summarize
254    let mut stripped_text = String::new();
255    for m in &messages_to_drop {
256        let role = m["role"].as_str().unwrap_or("unknown").to_uppercase();
257        if let Some(tool_calls) = m["tool_calls"].as_array() {
258            let names: Vec<&str> = tool_calls
259                .iter()
260                .filter_map(|tc| tc["function"]["name"].as_str())
261                .collect();
262            stripped_text.push_str(&format!(
263                "[{}]: [called tools: {}]\n\n",
264                role,
265                names.join(", ")
266            ));
267        } else {
268            let content = m["content"].as_str().unwrap_or("");
269            stripped_text.push_str(&format!("[{}]: {}\n\n", role, content));
270        }
271    }
272
273    // Collect IDs of messages to drop for filtering
274    let drop_ids: std::collections::HashSet<String> = messages_to_drop
275        .iter()
276        .filter_map(|m| m["_id"].as_str().map(|s| s.to_string()))
277        .collect();
278
279    // Atomically drop tool results whose assistant message was dropped
280    let dropped_msgs: Vec<serde_json::Value> =
281        messages_to_drop.iter().map(|m| (*m).clone()).collect();
282    let drop_tool_call_ids = collect_tool_call_ids(&dropped_msgs);
283
284    // Second LLM call: update summary with dropped content.
285    // {GOALS} and {TASKS} expand to a labelled section or empty string — the
286    // adjacent placeholders in the template collapse cleanly when both are absent.
287    let update_prompt = ROLLING_COMPACT_UPDATE_TEMPLATE
288        .replace(
289            "{SUMMARY}",
290            if context.summary.is_empty() {
291                "(No existing summary)"
292            } else {
293                &context.summary
294            },
295        )
296        .replace("{ARCHIVED}", &stripped_text)
297        .replace(
298            "{GOALS}",
299            &if goals.is_empty() {
300                String::new()
301            } else {
302                format!("\n{}\n", goals)
303            },
304        )
305        .replace(
306            "{TASKS}",
307            &if task_table.is_empty() {
308                String::new()
309            } else {
310                format!("\nCURRENT TASKS:\n{}\n", task_table)
311            },
312        );
313
314    let summary_messages = vec![json!({
315        "role": "user",
316        "content": update_prompt,
317    })];
318
319    let new_summary = gateway::chat(resolved_config, &summary_messages).await?;
320
321    if new_summary.is_empty() {
322        return Ok(());
323    }
324
325    // Capture count before we drop the borrow
326    let archived_count = messages_to_drop.len();
327
328    // Drop the borrow by ending use of messages_to_drop
329    drop(messages_to_drop);
330
331    // Filter out dropped messages (system messages always preserved)
332    let remaining_messages = filter_messages(&context.messages, &drop_ids, &drop_tool_call_ids);
333
334    // Update context with new summary and remaining messages
335    context.summary = new_summary.clone();
336    context.messages = remaining_messages;
337    context.updated_at = now_timestamp();
338
339    // Finalize compaction: write anchor to transcript and mark dirty
340    app.finalize_compaction(&context.name, &new_summary)?;
341
342    // Save updated context (summary.md, context_meta.json, etc.)
343    app.save_context(&context)?;
344
345    sink.emit_event(CommandEvent::RollingCompactionComplete {
346        archived: archived_count,
347        remaining: context.messages.len(),
348    });
349
350    // Execute post_rolling_compact hook. TeinHookContext: None — see PreRollingCompact above.
351    let hook_data = serde_json::json!({
352        "context_name": context.name,
353        "message_count": context.messages.len(),
354        "messages_archived": archived_count,
355        "summary": context.summary,
356    });
357    let _ = tools::execute_hook(
358        &tools,
359        tools::HookPoint::PostRollingCompact,
360        &hook_data,
361        None,
362    );
363
364    Ok(())
365}

367/// Auto-triggered compaction: delegates to rolling compaction.
368///
369/// Rolling compaction summarises older messages while preserving recent ones,
370/// keeping the context window within budget without discarding everything.
371/// For full compaction (manual `-c` flag), see `compact_context_with_llm_manual`.
372pub async fn compact_context_with_llm(
373    app: &AppState,
374    context_name: &str,
375    resolved_config: &ResolvedConfig,
376    sink: &dyn OutputSink,
377) -> io::Result<()> {
378    // Use rolling compaction for auto-triggered compaction
379    rolling_compact(app, context_name, resolved_config, sink).await
380}

382/// Full compaction: summarizes all messages and starts fresh (manual -c flag)
383pub async fn compact_context_with_llm_manual(
384    app: &AppState,
385    context_name: &str,
386    resolved_config: &ResolvedConfig,
387    sink: &dyn OutputSink,
388) -> io::Result<()> {

392/// Compact a specific context by name (for -Z flag)
393pub async fn compact_context_by_name(
394    app: &AppState,
395    context_name: &str,
396    sink: &dyn OutputSink,
397) -> io::Result<()> {
398    // Load the context
399    let context = app.load_context(context_name)?;
400    let message_count = context.messages.len();
401
402    if message_count <= 2 {
403        return Ok(());
404    }
405
406    // Finalize compaction: write anchor to transcript and mark dirty
407    let simple_summary = format!(
408        "Context compacted. {} messages archived to transcript.",
409        message_count
410    );
411    app.finalize_compaction(context_name, &simple_summary)?;
412
413    // Create fresh context preserving summary
414    let new_context = Context {
415        name: context_name.to_string(),
416        messages: Vec::new(),
417        created_at: context.created_at,
418        updated_at: now_timestamp(),
419        summary: context.summary,
420    };
421
422    app.save_context(&new_context)?;
423
424    sink.emit_event(CommandEvent::CompactionComplete {
425        context: context_name.to_string(),
426        archived: message_count,
427        remaining: 0,
428    });
429
430    Ok(())
431}

433async fn compact_context_with_llm_internal(
434    app: &AppState,
435    context_name: &str,
436    resolved_config: &ResolvedConfig,
437    sink: &dyn OutputSink,
438) -> io::Result<()> {
439    let context = app.get_or_create_context(context_name)?;
440
441    if context.messages.is_empty() || context.messages.len() <= 2 {
442        return Ok(());
443    }
444
445    // Execute pre_compact hook. TeinHookContext: None — compact_context_with_llm_internal
446    // does not carry a full ToolCallContext; tein callbacks fire but cannot use call-tool
447    // or (harness io). Known limitation — see PreRollingCompact in rolling_compact.
448    let tools = tools::load_tools(&app.plugins_dir)?;
449    let hook_data = serde_json::json!({
450        "context_name": context.name,
451        "message_count": context.messages.len(),
452        "summary": context.summary,
453    });
454    let _ = tools::execute_hook(&tools, tools::HookPoint::PreCompact, &hook_data, None);
455
456    sink.emit_event(CommandEvent::CompactionStarted {
457        context: context_name.to_string(),
458        message_count: context.messages.len(),
459    });
460
461    // Load compaction prompt (falls back to compiled-in default)
462    let compaction_prompt = app.load_prompt("compaction")?;
463
464    // Build conversation text for summarization, including tool interactions
465    let mut conversation_text = String::new();
466    for m in &context.messages {
467        let role = m["role"].as_str().unwrap_or("unknown");
468        if role == "system" {
469            continue;
470        }
471        if let Some(tool_calls) = m["tool_calls"].as_array() {
472            let names: Vec<&str> = tool_calls
473                .iter()
474                .filter_map(|tc| tc["function"]["name"].as_str())
475                .collect();
476            conversation_text.push_str(&format!(
477                "[ASSISTANT]: [called tools: {}]\n\n",
478                names.join(", ")
479            ));
480        } else if role == "tool" {
481            let content = m["content"].as_str().unwrap_or("");
482            let preview = if content.len() > 200 {
483                let end = content
484                    .char_indices()
485                    .nth(200)
486                    .map(|(i, _)| i)
487                    .unwrap_or(content.len());
488                format!("{}... [truncated]", &content[..end])
489            } else {
490                content.to_string()
491            };
492            conversation_text.push_str(&format!("[TOOL RESULT]: {}\n\n", preview));
493        } else {
494            let content = m["content"].as_str().unwrap_or("");
495            conversation_text.push_str(&format!("[{}]: {}\n\n", role.to_uppercase(), content));
496        }
497    }
498
499    // Prepare messages for compaction request
500    let compaction_messages = vec![
501        json!({
502            "role": "system",
503            "content": compaction_prompt,
504        }),
505        json!({
506            "role": "user",
507            "content": format!("Please summarize this conversation:\n\n{}", conversation_text),
508        }),
509    ];
510
511    let summary = gateway::chat(resolved_config, &compaction_messages).await?;
512
513    if summary.is_empty() {
514        return Err(io::Error::other(
515            "Empty summary received from LLM. This can happen with free-tier models. Try again or use a different model.",
516        ));
517    }
518
519    // Prepare continuation prompt (falls back to compiled-in default)
520    let continuation_prompt = app.load_prompt("continuation")?;
521
522    // Load system prompt
523    let system_prompt = app.load_system_prompt_for(context_name)?;
524
525    // Create new context with system prompt, continuation instructions, and summary
526    let mut new_context = Context {
527        name: context.name.clone(),
528        messages: Vec::new(),
529        created_at: context.created_at,
530        updated_at: now_timestamp(),
531        summary: summary.clone(),
532    };
533
534    // Add system prompt as first message
535    if !system_prompt.is_empty() {
536        new_context.messages.push(json!({
537            "_id": uuid::Uuid::new_v4().to_string(),
538            "role": "system",
539            "content": system_prompt,
540        }));
541    }
542
543    // Add continuation prompt + summary as user message
544    new_context.messages.push(json!({
545        "_id": uuid::Uuid::new_v4().to_string(),
546        "role": "user",
547        "content": format!("{}\n\n--- SUMMARY ---\n{}", continuation_prompt, summary),
548    }));
549
550    // Finalize compaction: write anchor to transcript and mark dirty
551    app.finalize_compaction(&new_context.name, &summary)?;
552
553    // Save the new context
554    app.save_context(&new_context)?;
555
556    sink.emit_event(CommandEvent::CompactionComplete {
557        context: context_name.to_string(),
558        archived: context.messages.len(),
559        remaining: new_context.messages.len(),
560    });
561
562    // Execute post_compact hook. TeinHookContext: None — see PreCompact above.
563    let hook_data = serde_json::json!({
564        "context_name": new_context.name,
565        "message_count": new_context.messages.len(),
566        "summary": new_context.summary,
567    });
568    let _ = tools::execute_hook(&tools, tools::HookPoint::PostCompact, &hook_data, None);
569
570    Ok(())
571}

854    /// Spin up a minimal OpenAI-compatible stub server that always returns the
855    /// given `response_body` string as the assistant message content.
856    ///
857    /// Returns the base URL (e.g. `"http://127.0.0.1:PORT"`) and a join handle.
858    /// The server shuts down automatically when the handle is dropped.
859    async fn spawn_stub_server(response_body: String) -> (String, tokio::task::JoinHandle<()>) {
860        use tokio::io::{AsyncReadExt, AsyncWriteExt};
861        use tokio::net::TcpListener;
862
863        let listener = TcpListener::bind("127.0.0.1:0").await.unwrap();
864        let port = listener.local_addr().unwrap().port();
865        let url = format!("http://127.0.0.1:{}", port);
866
867        let body = serde_json::json!({
868            "id": "stub-1",
869            "object": "chat.completion",
870            "created": 1700000000,
871            "model": "stub",
872            "choices": [{
873                "index": 0,
874                "message": { "role": "assistant", "content": response_body },
875                "finish_reason": "stop"
876            }],
877            "usage": { "prompt_tokens": 1, "completion_tokens": 1, "total_tokens": 2 }
878        })
879        .to_string();
880
881        let handle = tokio::spawn(async move {
882            for _ in 0..10 {
883                let Ok((mut stream, _)) = listener.accept().await else {
884                    break;
885                };
886                let body = body.clone();
887                tokio::spawn(async move {
888                    // Drain the full HTTP request (read until \r\n\r\n)
889                    let mut buf = vec![0u8; 16384];
890                    let mut total = 0;
891                    loop {
892                        match stream.read(&mut buf[total..]).await {
893                            Ok(0) | Err(_) => break,
894                            Ok(n) => {
895                                total += n;
896                                if buf[..total].windows(4).any(|w| w == b"\r\n\r\n") {
897                                    break;
898                                }
899                            }
900                        }
901                    }
902                    // Connection: close forces reqwest to open a fresh connection per request
903                    let response = format!(
904                        "HTTP/1.1 200 OK\r\nContent-Type: application/json\r\nContent-Length: {}\r\nConnection: close\r\n\r\n{}",
905                        body.len(),
906                        body
907                    );
908                    let _ = stream.write_all(response.as_bytes()).await;
909                });
910            }
911        });
912
913        (url, handle)
914    }

1307    /// Integration stress test: large transcript with repeated rolling compaction.
1308    ///
1309    /// Builds a synthetic 90-message transcript (deterministic, seeded from a fixed
1310    /// message table), then drives `rolling_compact` in a loop until the context
1311    /// stabilises at ≤4 non-system messages.  Per-round invariants:
1312    ///
1313    /// - no orphaned tool results
1314    /// - system messages preserved throughout
1315    /// - summary is non-empty after each round
1316    /// - non-system message count strictly decreases (or has hit the ≤4 floor)
1317    /// - a compaction anchor is written to the transcript each round
1318    ///
1319    /// After full compaction the final context is re-loaded and verified to be
1320    /// coherent (parseable, no corrupt JSON, loadable via `get_or_create_context`).
1321    ///
1322    /// Gated on `CHIBI_API_KEY`.  Free models may be slow; the test allows up to
1323    /// 20 compaction rounds as a safety cap.
1324    ///
1325    /// Skip with: `unset CHIBI_API_KEY`
1326    /// Run with:  `CHIBI_API_KEY=<key> cargo test large_transcript_stress -- --ignored --nocapture`
1327    #[tokio::test]
1328    #[ignore]
1329    async fn large_transcript_stress_repeated_rolling_compaction() {
1330        let Ok(api_key) = std::env::var("CHIBI_API_KEY") else {
1331            eprintln!("CHIBI_API_KEY not set — skipping large transcript stress test");
1332            return;
1333        };
1334
1335        let (app, _tmp) = create_test_app();
1336        let ctx_name = "stress-rolling";
1337
1338        // --- Build a deterministic 90-message synthetic transcript ---
1339        //
1340        // Pattern (repeating slice):
1341        //   user, assistant, user, assistant+tool_call, tool_result, user, assistant
1342        // This gives a realistic mix of plain exchanges and tool interactions.
1343        // IDs are stable strings so the LLM decision path can identify messages.
1344        // If the LLM returns unrecognised IDs the fallback (oldest-N%) activates —
1345        // both paths are valid and all invariants must hold for both.
1346        let topics = [
1347            (
1348                "What is Rust?",
1349                "Rust is a systems language focused on safety and performance.",
1350            ),
1351            (
1352                "Explain ownership.",
1353                "Ownership ensures memory safety without a garbage collector.",
1354            ),
1355            (
1356                "How does borrowing work?",
1357                "Borrowing allows references without transferring ownership.",
1358            ),
1359            (
1360                "What are lifetimes?",
1361                "Lifetimes annotate how long references are valid.",
1362            ),
1363            (
1364                "Describe traits.",
1365                "Traits define shared behaviour across types.",
1366            ),
1367            (
1368                "What is async/await?",
1369                "Async/await enables non-blocking I/O with a cooperative model.",
1370            ),
1371            (
1372                "Explain enums.",
1373                "Enums represent a value that can be one of several variants.",
1374            ),
1375            (
1376                "What are closures?",
1377                "Closures capture their environment and can be stored or passed.",
1378            ),
1379            (
1380                "How do iterators work?",
1381                "Iterators lazily produce a sequence of values.",
1382            ),
1383            (
1384                "What is pattern matching?",
1385                "Pattern matching deconstructs values into their components.",
1386            ),
1387        ];
1388
1389        let mut messages: Vec<serde_json::Value> = vec![
1390            json!({ "role": "system", "_id": "sys-1", "content": "You are a knowledgeable Rust tutor." }),
1391        ];
1392
1393        for (i, (question, answer)) in topics.iter().cycle().take(30).enumerate() {
1394            let base = i * 3;
1395            let uid = format!("u{}", base);
1396            let aid = format!("a{}", base);
1397            messages.push(json!({ "role": "user",      "_id": uid, "content": question }));
1398            messages.push(json!({ "role": "assistant",  "_id": aid, "content": answer  }));
1399
1400            // Every third exchange: inject a tool call + result pair
1401            if i % 3 == 2 {
1402                let tc_id = format!("tc{}", base);
1403                let tcm_id = format!("atc{}", base);
1404                let tr_id = format!("tr{}", base);
1405                messages.push(json!({
1406                    "role": "assistant",
1407                    "_id": tcm_id,
1408                    "tool_calls": [{
1409                        "id": tc_id,
1410                        "function": { "name": "lookup_docs", "arguments": format!("{{\"topic\":\"{}\"}}", i) }
1411                    }]
1412                }));
1413                messages.push(json!({
1414                    "role": "tool",
1415                    "_id": tr_id,
1416                    "tool_call_id": tc_id,
1417                    "content": format!("Documentation for topic {} retrieved successfully.", i)
1418                }));
1419            }
1420        }
1421
1422        // Confirm we built a sufficiently large transcript (should be ~100 messages)
1423        let non_system_initial: usize = messages
1424            .iter()
1425            .filter(|m| m["role"].as_str() != Some("system"))
1426            .count();
1427        assert!(
1428            non_system_initial >= 80,
1429            "expected ≥80 non-system messages to start, got {non_system_initial}"
1430        );
1431        eprintln!("stress test: built transcript with {non_system_initial} non-system messages");
1432
1433        let mut context = app.get_or_create_context(ctx_name).unwrap();
1434        context.messages = messages;
1435        app.save_context(&context).unwrap();
1436
1437        let mut resolved = dummy_resolved_config();
1438        resolved.api_key = Some(api_key);
1439        resolved.model = "ratatoskr:free/agentic".to_string();
1440        resolved.rolling_compact_drop_percentage = 50.0;
1441
1442        let mut prev_non_system_count = non_system_initial;
1443        let max_rounds = 20;
1444
1445        for round in 1..=max_rounds {
1446            let before = app.get_or_create_context(ctx_name).unwrap();
1447            let non_system_before: usize = before
1448                .messages
1449                .iter()
1450                .filter(|m| m["role"].as_str() != Some("system"))
1451                .count();
1452
1453            if non_system_before <= 4 {
1454                eprintln!(
1455                    "stress test: stable at round {round} ({non_system_before} non-system msgs) — done"
1456                );
1457                break;
1458            }
1459
1460            eprintln!("stress test: round {round} — {non_system_before} non-system messages");
1461
1462            let result = rolling_compact(&app, ctx_name, &resolved, &NoopSink).await;
1463            assert!(
1464                result.is_ok(),
1465                "round {round}: rolling_compact failed: {:?}",
1466                result
1467            );
1468
1469            let after = app.get_or_create_context(ctx_name).unwrap();
1470
1471            // --- Per-round invariants ---
1472
1473            // 1. System messages are fully preserved.
1474            let system_after = after
1475                .messages
1476                .iter()
1477                .filter(|m| m["role"].as_str() == Some("system"))
1478                .count();
1479            assert!(
1480                system_after >= 1,
1481                "round {round}: system messages must be preserved (found {system_after})"
1482            );
1483
1484            // 2. No orphaned tool results.
1485            let remaining_tc_ids: HashSet<String> = after
1486                .messages
1487                .iter()
1488                .filter_map(|m| m["tool_calls"].as_array())
1489                .flatten()
1490                .filter_map(|tc| tc["id"].as_str().map(|s| s.to_string()))
1491                .collect();
1492            let orphaned: Vec<_> = after
1493                .messages
1494                .iter()
1495                .filter(|m| m["role"].as_str() == Some("tool"))
1496                .filter(|m| {
1497                    let tc_id = m["tool_call_id"].as_str().unwrap_or("");
1498                    !remaining_tc_ids.contains(tc_id)
1499                })
1500                .collect();
1501            assert!(
1502                orphaned.is_empty(),
1503                "round {round}: found orphaned tool results: {:?}",
1504                orphaned
1505            );
1506
1507            // 3. Summary grows or stays non-empty.
1508            assert!(
1509                !after.summary.is_empty(),
1510                "round {round}: summary must be non-empty after compaction"
1511            );
1512
1513            // 4. Non-system message count strictly decreases (or already at floor).
1514            let non_system_after: usize = after
1515                .messages
1516                .iter()
1517                .filter(|m| m["role"].as_str() != Some("system"))
1518                .count();
1519            if non_system_before > 4 {
1520                assert!(
1521                    non_system_after < prev_non_system_count,
1522                    "round {round}: expected message count to decrease from {prev_non_system_count}, got {non_system_after}"
1523                );
1524            }
1525            prev_non_system_count = non_system_after;
1526
1527            // 5. Compaction anchor written to transcript.
1528            let entries = app.read_transcript_entries(ctx_name).unwrap();
1529            let anchor_count = entries
1530                .iter()
1531                .filter(|e| e.entry_type == crate::context::ENTRY_TYPE_COMPACTION)
1532                .count();
1533            assert!(
1534                anchor_count >= round,
1535                "round {round}: expected at least {round} compaction anchor(s) in transcript, found {anchor_count}"
1536            );
1537
1538            if non_system_after <= 4 {
1539                eprintln!(
1540                    "stress test: floor reached after round {round} ({non_system_after} msgs)"
1541                );
1542                break;
1543            }
1544        }
1545
1546        // --- Final coherence check ---
1547        // Context must be loadable and parseable with no corrupt JSON.
1548        let final_ctx = app.get_or_create_context(ctx_name).unwrap();
1549        assert!(
1550            !final_ctx.summary.is_empty(),
1551            "final context: summary must be non-empty"
1552        );
1553        // Re-serialise and re-parse every message to confirm no corruption.
1554        for (idx, msg) in final_ctx.messages.iter().enumerate() {
1555            let serialised = serde_json::to_string(msg).unwrap_or_else(|e| {
1556                panic!("final context: message {idx} failed to serialise: {e}")
1557            });
1558            serde_json::from_str::<serde_json::Value>(&serialised)
1559                .unwrap_or_else(|e| panic!("final context: message {idx} failed to re-parse: {e}"));
1560        }
1561        eprintln!(
1562            "stress test: final context has {} messages, summary len={}",
1563            final_ctx.messages.len(),
1564            final_ctx.summary.len()
1565        );
1566    }
