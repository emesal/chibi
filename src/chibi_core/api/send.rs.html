35/// Options for controlling prompt execution behaviour.
36#[derive(Debug, Clone)]
37pub struct PromptOptions<'a> {
38    pub use_reflection: bool,
39    pub debug: &'a [DebugKey],
40    pub force_render: bool,
41    /// Optional override for the fallback handoff target.
42    pub fallback_override: Option<crate::tools::HandoffTarget>,
43}

45impl<'a> PromptOptions<'a> {
46    pub fn new(use_reflection: bool, debug: &'a [DebugKey], force_render: bool) -> Self {
47        Self {
48            use_reflection,
49            debug,
50            force_render,
51            fallback_override: None,

53    }

69/// Evaluate permission from pre-computed hook results.
70///
71/// Deny-only protocol: if any plugin returns `"denied": true`, the operation
72/// is blocked. Otherwise, falls through to the permission handler (if set)
73/// or fail-safe deny.
74///
75/// Separated from `check_permission()` for unit testing.
76fn evaluate_permission(
77    hook_results: &[(String, serde_json::Value)],
78    hook_data: &serde_json::Value,
79    permission_handler: Option<&PermissionHandler>,
80) -> io::Result<Result<(), String>> {
81    // Check for explicit denial from any plugin
82    for (_plugin_name, result) in hook_results {
83        if result.get_bool_or("denied", false) {
84            let reason = result.get_str_or("reason", "denied by plugin").to_string();
85            return Ok(Err(reason));
86        }
87    }
88
89    // No plugin denied — delegate to permission handler or fail-safe deny
90    match permission_handler {
91        Some(handler) => {
92            if handler(hook_data)? {
93                Ok(Ok(()))
94            } else {
95                Ok(Err("permission denied".to_string()))
96            }
97        }
98        None => Ok(Err(
99            "no permission handler configured (fail-safe deny)".to_string()
100        )),
101    }
102}

104/// Full permission check: fire the hook, then evaluate results.
105fn check_permission(
106    tools: &[Tool],
107    hook: tools::HookPoint,
108    hook_data: &serde_json::Value,
109    permission_handler: Option<&PermissionHandler>,
110    tein_ctx: Option<&tools::TeinHookContext<'_>>,
111) -> io::Result<Result<(), String>> {
112    let hook_results = tools::execute_hook(tools, hook, hook_data, tein_ctx)?;
113    evaluate_permission(&hook_results, hook_data, permission_handler)
114}

116/// Build tool info list for pre_api_tools hook data
117fn build_tool_info_list(
118    all_tools: &[serde_json::Value],
119    registry: &Arc<RwLock<ToolRegistry>>,
120) -> Vec<serde_json::Value> {
121    let reg = registry.read().unwrap();
122    all_tools
123        .iter()
124        .filter_map(|tool| {
125            let name = tool
126                .get("function")
127                .and_then(|f| f.get("name"))
128                .and_then(|n| n.as_str())?;
129            let category = reg
130                .get(name)
131                .map(|t| t.category.as_str())
132                .unwrap_or("plugin"); // unknown tools default to plugin
133            Some(json!({
134                "name": name,
135                "type": category,
136            }))
137        })
138        .collect()
139}

141/// Annotate the fallback tool's description to indicate it's called automatically.
142fn annotate_fallback_tool(tools: &mut [serde_json::Value], fallback_name: &str) {
143    const FALLBACK_SUFFIX: &str = " Called automatically if no other tool is used.";
144
145    for tool in tools.iter_mut() {
146        let name_matches = tool
147            .get("function")
148            .and_then(|f| f.get("name"))
149            .and_then(|n| n.as_str())
150            .is_some_and(|name| name == fallback_name);
151
152        if name_matches {
153            if let Some(func) = tool.get_mut("function")
154                && let Some(desc) = func.get_mut("description")
155                && let Some(desc_str) = desc.as_str()
156            {
157                *desc = serde_json::Value::String(format!("{}{}", desc_str, FALLBACK_SUFFIX));
158            }
159            break; // Only annotate the first match
160        }
161    }
162}

164/// Filter tools based on config include/exclude/exclude_categories lists
165fn filter_tools_by_config(
166    tools: Vec<serde_json::Value>,
167    config: &ToolsConfig,
168    registry: &Arc<RwLock<ToolRegistry>>,
169) -> Vec<serde_json::Value> {
170    let mut result = tools;
171
172    // Apply include filter first (if set, only these tools are considered)
173    if let Some(ref include) = config.include {
174        result.retain(|tool| {
175            tool.get("function")
176                .and_then(|f| f.get("name"))
177                .and_then(|n| n.as_str())
178                .is_some_and(|name| include.contains(&name.to_string()))
179        });
180    }
181
182    // Apply exclude filter (remove these tools from remaining)
183    if let Some(ref exclude) = config.exclude {
184        result.retain(|tool| {
185            tool.get("function")
186                .and_then(|f| f.get("name"))
187                .and_then(|n| n.as_str())
188                .is_some_and(|name| !exclude.contains(&name.to_string()))
189        });
190    }
191
192    // Apply category exclusion (remove tools whose category is excluded)
193    if let Some(ref categories) = config.exclude_categories {
194        let reg = registry.read().unwrap();
195        result.retain(|tool| {
196            tool.get("function")
197                .and_then(|f| f.get("name"))
198                .and_then(|n| n.as_str())
199                .map(|name| {
200                    let cat = reg
201                        .get(name)
202                        .map(|t| t.category.as_str())
203                        .unwrap_or("plugin"); // unknown tools default to plugin
204                    !categories.contains(&cat.to_string())
205                })
206                .unwrap_or(true)
207        });
208    }
209
210    result
211}

213/// Filter tools based on hook results
214/// Multiple hooks: includes are intersected, excludes are unioned
215fn filter_tools_from_hook_results<S: ResponseSink>(
216    tools: Vec<serde_json::Value>,
217    hook_results: &[(String, serde_json::Value)],
218    sink: &mut S,
219) -> io::Result<Vec<serde_json::Value>> {
220    if hook_results.is_empty() {
221        return Ok(tools);
222    }
223
224    let mut result = tools;
225
226    // Collect all includes and excludes from hook results
227    let mut all_includes: Option<Vec<String>> = None;
228    let mut all_excludes: Vec<String> = Vec::new();
229
230    for (hook_name, hook_result) in hook_results {
231        // Handle include lists (intersection)
232        if let Some(include) = hook_result.get_array("include") {
233            let include_names: Vec<String> = include
234                .iter()
235                .filter_map(|v| v.as_str().map(|s| s.to_string()))
236                .collect();
237
238            sink.handle(ResponseEvent::HookDebug {
239                hook: hook_name.clone(),
240                message: format!(
241                    "[Hook pre_api_tools: {} include filter: {:?}]",
242                    hook_name, include_names
243                ),
244            })?;
245
246            all_includes = Some(match all_includes {
247                Some(existing) => existing
248                    .into_iter()
249                    .filter(|name| include_names.contains(name))
250                    .collect(),
251                None => include_names,
252            });
253        }
254
255        // Handle exclude lists (union)
256        if let Some(exclude) = hook_result.get_array("exclude") {
257            let exclude_names: Vec<String> = exclude
258                .iter()
259                .filter_map(|v| v.as_str().map(|s| s.to_string()))
260                .collect();
261
262            sink.handle(ResponseEvent::HookDebug {
263                hook: hook_name.clone(),
264                message: format!(
265                    "[Hook pre_api_tools: {} exclude filter: {:?}]",
266                    hook_name, exclude_names
267                ),
268            })?;
269
270            for name in exclude_names {
271                if !all_excludes.contains(&name) {
272                    all_excludes.push(name);
273                }
274            }
275        }
276    }
277
278    // Apply collected includes (intersection)
279    if let Some(includes) = all_includes {
280        result.retain(|tool| {
281            tool.get("function")
282                .and_then(|f| f.get("name"))
283                .and_then(|n| n.as_str())
284                .is_some_and(|name| includes.contains(&name.to_string()))
285        });
286    }
287
288    // Apply collected excludes (union)
289    if !all_excludes.is_empty() {
290        result.retain(|tool| {
291            tool.get("function")
292                .and_then(|f| f.get("name"))
293                .and_then(|n| n.as_str())
294                .is_some_and(|name| !all_excludes.contains(&name.to_string()))
295        });
296    }
297
298    Ok(result)
299}

301/// Apply request modifications from pre_api_request hook results
302/// Hook returns are merged (not replaced) into the request body
303fn apply_request_modifications<S: ResponseSink>(
304    mut request_body: serde_json::Value,
305    hook_results: &[(String, serde_json::Value)],
306    sink: &mut S,
307) -> io::Result<serde_json::Value> {
308    for (hook_name, hook_result) in hook_results {
309        if let Some(modifications) = hook_result.get("request_body")
310            && let Some(mods_obj) = modifications.as_object()
311        {
312            sink.handle(ResponseEvent::HookDebug {
313                hook: hook_name.clone(),
314                message: format!(
315                    "[Hook pre_api_request: {} modifying request (keys: {:?})]",
316                    hook_name,
317                    mods_obj.keys().collect::<Vec<_>>()
318                ),
319            })?;
320
321            // Merge modifications into request body
322            if let Some(body_obj) = request_body.as_object_mut() {
323                for (key, value) in mods_obj {
324                    body_obj.insert(key.clone(), value.clone());
325                }
326            }
327        }
328    }
329
330    Ok(request_body)
331}

333/// Apply hook overrides from hook results.
334///
335/// Handles three keys from hook return values:
336/// - `"fallback"`: override the fallback tool (existing behaviour)
337/// - `"fuel"`: absolute fuel override (e.g. from `pre_agentic_loop`); ignored when `fuel_unlimited`
338/// - `"fuel_delta"`: relative fuel adjustment (e.g. from `post_tool_batch`); ignored when `fuel_unlimited`
339///
340/// Note: `"fuel": 0` from a hook means "exhaust immediately" (not unlimited).
341/// Unlimited mode is determined solely by the configured `fuel = 0`, not by runtime hook values.
342fn apply_hook_overrides<S: ResponseSink>(
343    handoff: &mut tools::Handoff,
344    fuel_remaining: &mut usize,
345    fuel_unlimited: bool,
346    hook_results: &[(String, serde_json::Value)],
347    sink: &mut S,
348) -> io::Result<()> {
349    for (hook_name, hook_result) in hook_results {
350        if let Some(fallback_str) = hook_result.get_str("fallback") {
351            // Use builtin metadata since hooks can only override to builtins
352            let meta = tools::flow_tool_metadata(fallback_str);
353            let new_fallback = if meta.ends_turn {
354                tools::HandoffTarget::User {
355                    message: String::new(),
356                }
357            } else {
358                tools::HandoffTarget::Agent {
359                    prompt: String::new(),
360                }
361            };
362            handoff.set_fallback(new_fallback);
363            sink.handle(ResponseEvent::HookDebug {
364                hook: hook_name.clone(),
365                message: format!("[Hook {} set fallback to {}]", hook_name, fallback_str),
366            })?;
367        }
368        if !fuel_unlimited {
369            if let Some(fuel) = hook_result.get("fuel").and_then(|v| v.as_u64()) {
370                *fuel_remaining = fuel as usize;
371                sink.handle(ResponseEvent::HookDebug {
372                    hook: hook_name.clone(),
373                    message: format!("[Hook {} set fuel to {}]", hook_name, fuel),
374                })?;
375            }
376            if let Some(delta) = hook_result.get("fuel_delta").and_then(|v| v.as_i64()) {
377                if delta < 0 {
378                    *fuel_remaining = fuel_remaining.saturating_sub((-delta) as usize);
379                } else {
380                    *fuel_remaining = fuel_remaining.saturating_add(delta as usize);
381                }
382                sink.handle(ResponseEvent::HookDebug {
383                    hook: hook_name.clone(),
384                    message: format!("[Hook {} adjusted fuel by {}]", hook_name, delta),
385                })?;
386            }
387        }
388    }
389    Ok(())
390}

396/// Build the full system prompt with all components.
397///
398/// Handles: loading base prompt, goals, summary, reflection prompt,
399/// pre/post system_prompt hooks, and username injection.
400#[allow(clippy::too_many_arguments)]
401fn build_full_system_prompt<S: ResponseSink>(
402    app: &AppState,
403    context_name: &str,
404    summary: &str,
405    use_reflection: bool,
406    tools: &[Tool],
407    resolved_config: &ResolvedConfig,
408    sink: &mut S,
409    home_dir: &Path,
410    project_root: &Path,
411    tein_ctx: Option<&tools::TeinHookContext<'_>>,
412) -> io::Result<String> {
413    // Load base prompts
414    let system_prompt = app.load_system_prompt_for(context_name)?;
415    let reflection_prompt = if use_reflection {
416        app.load_reflection_prompt()?
417    } else {
418        String::new()
419    };
420
421    // Load context-specific state
422    let flock_contexts = load_flock_contexts(&app.vfs, context_name)?;
423
424    // Execute pre_system_prompt hook - can inject content before system prompt sections
425    let pre_sys_hook_data = serde_json::json!({
426        "context_name": context_name,
427        "summary": summary,
428        "flock_goals": flock_contexts.iter()
429            .filter_map(|fc| fc.goals.as_ref().map(|g| serde_json::json!({
430                "flock": fc.flock_name,
431                "goals": g,
432            })))
433            .collect::<Vec<_>>(),
434    });
435    let pre_sys_hook_results = tools::execute_hook(
436        tools,
437        tools::HookPoint::PreSystemPrompt,
438        &pre_sys_hook_data,
439        tein_ctx,
440    )?;
441
442    // Build full system prompt with all components, anchoring identity first
443    let mut full_system_prompt = format!(
444        "You are \"{}\", a chibi agent.\n\n{}",
445        context_name, system_prompt
446    );
447
448    // Prepend any content from pre_system_prompt hooks
449    for (hook_tool_name, result) in &pre_sys_hook_results {
450        if let Some(inject) = result.get_str("inject")
451            && !inject.is_empty()
452        {
453            sink.handle(ResponseEvent::HookDebug {
454                hook: hook_tool_name.clone(),
455                message: format!(
456                    "[Hook pre_system_prompt: {} injected content]",
457                    hook_tool_name
458                ),
459            })?;
460            full_system_prompt = format!("{}\n\n{}", inject, full_system_prompt);
461        }
462    }
463
464    // Load AGENTS.md instructions from standard locations
465    let agents_md = crate::agents_md::load_agents_md(
466        home_dir,
467        &app.chibi_dir,
468        project_root,
469        &std::env::current_dir().unwrap_or_default(),
470    );
471    if !agents_md.is_empty() {
472        full_system_prompt.push_str("\n\n--- AGENT INSTRUCTIONS ---\n");
473        full_system_prompt.push_str(&agents_md);
474    }
475
476    // Add username info at the start if not "user"
477    if resolved_config.username != "user" {
478        full_system_prompt.push_str(&format!(
479            "\n\nThe user speaking to you is called: {}",
480            resolved_config.username
481        ));
482    }
483
484    // Add summary if present
485    if !summary.is_empty() {
486        full_system_prompt.push_str("\n\n--- CONVERSATION SUMMARY ---\n");
487        full_system_prompt.push_str(summary);
488    }
489
490    // Add flock goals/prompts if present
491    let flock_sections = format_flock_sections(&flock_contexts);
492    if !flock_sections.is_empty() {
493        full_system_prompt.push_str(&flock_sections);
494    }
495
496    // Add fuel notice when operating under a fuel budget
497    if resolved_config.fuel > 0 {
498        full_system_prompt.push_str("\n\nUse call_user to return control to the user. You operate within a fuel budget. When fuel runs out control is returned automatically. Fuel refills with each prompt.");
499    }
500
501    // Add reflection prompt last (personality layer)
502    if !reflection_prompt.is_empty() {
503        full_system_prompt.push_str("\n\n");
504        full_system_prompt.push_str(&reflection_prompt);
505    }
506
507    // Execute post_system_prompt hook - can inject content after all system prompt sections
508    let post_sys_hook_data = serde_json::json!({
509        "context_name": context_name,
510        "summary": summary,
511        "flock_goals": flock_contexts.iter()
512            .filter_map(|fc| fc.goals.as_ref().map(|g| serde_json::json!({
513                "flock": fc.flock_name,
514                "goals": g,
515            })))
516            .collect::<Vec<_>>(),
517    });
518    let post_sys_hook_results = tools::execute_hook(
519        tools,
520        tools::HookPoint::PostSystemPrompt,
521        &post_sys_hook_data,
522        tein_ctx,
523    )?;
524
525    // Append any content from post_system_prompt hooks
526    for (hook_tool_name, result) in &post_sys_hook_results {
527        if let Some(inject) = result.get_str("inject")
528            && !inject.is_empty()
529        {
530            sink.handle(ResponseEvent::HookDebug {
531                hook: hook_tool_name.clone(),
532                message: format!(
533                    "[Hook post_system_prompt: {} injected content]",
534                    hook_tool_name
535                ),
536            })?;
537            full_system_prompt.push_str("\n\n");
538            full_system_prompt.push_str(inject);
539        }
540    }
541
542    // Store combined prompt in context_meta for API request reconstruction
543    if !full_system_prompt.is_empty() {
544        app.save_combined_system_prompt(context_name, &full_system_prompt)?;
545    }
546
547    Ok(full_system_prompt)
548}

550/// Result of collecting a streaming response from the LLM.
551struct StreamingResponse {
552    /// The full text response accumulated from chunks.
553    full_response: String,
554    /// Tool calls extracted from the response.
555    tool_calls: Vec<ratatoskr::ToolCall>,
556    /// Response metadata (usage stats, model info).
557    response_meta: Option<serde_json::Value>,
558}
559
560/// Collect a streaming response from the LLM API via ratatoskr.
561///
562/// Handles: SSE parsing, content accumulation, tool call reconstruction.
563/// Pure I/O - no hooks or side effects beyond streaming to sink.
564///
565/// `options` is the canonical source of API parameters and must be derived
566/// from (and reflect any hook modifications to) the request body JSON.
567async fn collect_streaming_response<S: ResponseSink>(
568    resolved_config: &ResolvedConfig,
569    options: &ChatOptions,
570    messages: &[serde_json::Value],
571    all_tools: &[serde_json::Value],
572    sink: &mut S,
573) -> io::Result<StreamingResponse> {
574    let gateway = build_gateway(resolved_config)?;
575
576    // Convert messages
577    let ratatoskr_messages: Vec<_> = messages
578        .iter()
579        .map(to_ratatoskr_message)
580        .collect::<io::Result<Vec<_>>>()?;
581
582    // Convert tools
583    let tool_defs: Vec<_> = all_tools
584        .iter()
585        .filter_map(|t| json_tool_to_definition(t).ok())
586        .collect();
587    let tools_opt = if tool_defs.is_empty() {
588        None
589    } else {
590        Some(tool_defs.as_slice())
591    };
592
593    // Get streaming response
594    let mut stream = gateway
595        .chat_stream(&ratatoskr_messages, tools_opt, options)
596        .await
597        .map_err(|e| io::Error::other(format!("Gateway error: {}", e)))?;
598
599    let mut full_response = String::new();
600    let mut tool_calls: Vec<ratatoskr::ToolCall> = Vec::new();
601    let mut response_meta: Option<serde_json::Value> = None;
602    let mut is_first_content = true;
603
604    while let Some(event_result) = stream.next().await {
605        let event = event_result.map_err(|e| io::Error::other(format!("Stream error: {}", e)))?;
606
607        match event {
608            ChatEvent::Content(chunk) => {
609                // Handle first chunk newline stripping (matches old behavior)
610                let text = if is_first_content {
611                    is_first_content = false;
612                    if let Some(remaining) = chunk.strip_prefix('\n') {
613                        if remaining.is_empty() {
614                            continue;
615                        }
616                        remaining.to_string()
617                    } else {
618                        chunk
619                    }
620                } else {
621                    chunk
622                };
623
624                full_response.push_str(&text);
625                sink.handle(ResponseEvent::TextChunk(&text))?;
626            }
627            ChatEvent::Reasoning(chunk) => {
628                // Reasoning is ephemeral thinking — always forward to sink so
629                // both streaming and JSON consumers can access it
630                sink.handle(ResponseEvent::Reasoning(&chunk))?;
631            }
632            ChatEvent::ToolCallStart { index, id, name } => {
633                // Prevent memory exhaustion
634                if index >= MAX_TOOL_CALLS {
635                    eprintln!(
636                        "[WARN] Tool call index {} exceeds limit {}, skipping",
637                        index, MAX_TOOL_CALLS
638                    );
639                    continue;
640                }
641
642                while tool_calls.len() <= index {
643                    tool_calls.push(ratatoskr::ToolCall::default());
644                }
645
646                tool_calls[index].id = id;
647                tool_calls[index].name = name;
648            }
649            ChatEvent::ToolCallDelta { index, arguments } => {
650                if index < tool_calls.len() {
651                    tool_calls[index].arguments.push_str(&arguments);
652                }
653            }
654            ChatEvent::Usage(usage) => {
655                response_meta = Some(json!({
656                    "usage": {
657                        "prompt_tokens": usage.prompt_tokens,
658                        "completion_tokens": usage.completion_tokens,
659                        "total_tokens": usage.total_tokens
660                    }
661                }));
662            }
663            ChatEvent::ToolCallEnd { .. } => {
664                // Tool call argument streaming complete; nothing to do
665            }
666            ChatEvent::Done => break,
667            _ => {} // forward-compatible with future ChatEvent variants
668        }
669    }
670
671    Ok(StreamingResponse {
672        full_response,
673        tool_calls,
674        response_meta,
675    })
676}

698/// Detects when the same tool call produces the same result consecutively.
699/// Used to inject a warning and charge fuel to break infinite loops.
700struct LoopDetector {
701    last_tool_name: String,
702    last_args: String,
703    last_result: String,
704    /// Whether the last recorded call had a cached result.
705    last_was_cached: bool,
706    /// How many times this call has repeated consecutively.
707    /// Starts at 0 (no previous call). Becomes 1 on first call, 2 on first repeat, etc.
708    pub consecutive_count: u32,
709}
710
711impl LoopDetector {
712    fn new() -> Self {
713        Self {
714            last_tool_name: String::new(),
715            last_args: String::new(),
716            last_result: String::new(),
717            last_was_cached: false,
718            consecutive_count: 0,
719        }
720    }
721
722    /// Record a tool call result. Returns `true` if this is a duplicate (loop detected).
723    ///
724    /// When both the current and previous calls were cached, compares `(tool_name, args)`
725    /// only — cached stubs contain timestamp-based VFS URIs that differ every time,
726    /// defeating result-based comparison (#207).
727    fn check_and_update(
728        &mut self,
729        tool_name: &str,
730        args: &str,
731        result: &str,
732        was_cached: bool,
733    ) -> bool {
734        let is_repeat = tool_name == self.last_tool_name
735            && args == self.last_args
736            && if was_cached && self.last_was_cached {
737                true // both cached: (tool, args) match is sufficient
738            } else {
739                result == self.last_result
740            };
741
742        if is_repeat {
743            self.consecutive_count += 1;
744            true
745        } else {
746            self.last_tool_name = tool_name.to_string();
747            self.last_args = args.to_string();
748            self.last_result = result.to_string();
749            self.last_was_cached = was_cached;
750            self.consecutive_count = 1;
751            false
752        }
753    }
754}

837/// Process PreTool hook results: check for block signals and argument modifications.
838fn apply_pre_tool_results(
839    hook_results: Vec<(String, serde_json::Value)>,
840    tool_name: &str,
841    args: serde_json::Value,
842) -> PreToolResult {
843    let mut result = PreToolResult {
844        blocked: false,
845        block_message: String::new(),
846        args,
847        diagnostics: Vec::new(),
848    };
849
850    for (hook_tool_name, hook_value) in hook_results {
851        if hook_value.get_bool_or("block", false) {
852            result.blocked = true;
853            result.block_message = hook_value
854                .get_str_or("message", "Tool call blocked by hook")
855                .to_string();
856            result.diagnostics.push(format!(
857                "[Hook pre_tool: {} blocked {} - {}]",
858                hook_tool_name, tool_name, result.block_message
859            ));
860            break;
861        }
862
863        if let Some(modified_args) = hook_value.get("arguments") {
864            result.diagnostics.push(format!(
865                "[Hook pre_tool: {} modified arguments for {}]",
866                hook_tool_name, tool_name
867            ));
868            result.args = modified_args.clone();
869        }
870    }
871
872    result
873}

875/// Process PreToolOutput hook results: check for block/replace signals and output modifications.
876/// Returns the (possibly modified) tool output and any diagnostics.
877fn apply_pre_tool_output_results(
878    hook_results: Vec<(String, serde_json::Value)>,
879    tool_name: &str,
880    tool_result: String,
881) -> (String, Vec<String>) {
882    let mut output = tool_result;
883    let mut diagnostics = Vec::new();
884
885    for (hook_tool_name, result) in hook_results {
886        if result.get_bool_or("block", false) {
887            let replacement = result
888                .get_str_or("message", "Output blocked by hook")
889                .to_string();
890            diagnostics.push(format!(
891                "[Hook pre_tool_output: {} blocked output from {}]",
892                hook_tool_name, tool_name
893            ));
894            output = replacement;
895            break;
896        }
897
898        if let Some(modified_output) = result.get_str("output") {
899            diagnostics.push(format!(
900                "[Hook pre_tool_output: {} modified output from {}]",
901                hook_tool_name, tool_name
902            ));
903            output = modified_output.to_string();
904        }
905    }
906
907    (output, diagnostics)
908}

910/// Sink-free execution core for a single tool call.
911///
912/// Performs all tool execution logic (hooks, permission middleware, dispatch via
913/// registry, caching, output hooks) without touching the sink or handoff state.
914/// Collects verbose diagnostics into `ToolExecutionResult::diagnostics` for the
915/// caller to emit.
916///
917/// This enables concurrent execution via `join_all` since it doesn't require
918/// `&mut` access to shared state.
919#[allow(clippy::too_many_arguments)]
920async fn execute_tool_pure(
921    app: &AppState,
922    context_name: &str,
923    tool_call: &ratatoskr::ToolCall,
924    plugin_tools: &[Tool],
925    registry: &Arc<RwLock<ToolRegistry>>,
926    use_reflection: bool,
927    resolved_config: &ResolvedConfig,
928    permission_handler: Option<&PermissionHandler>,
929    project_root: &Path,
930    tein_ctx: Option<&tools::TeinHookContext<'_>>,
931) -> io::Result<ToolExecutionResult> {
932    let mut args: serde_json::Value =
933        serde_json::from_str(&tool_call.arguments).unwrap_or(serde_json::json!({}));
934    let mut diagnostics = Vec::new();
935
936    // Look up tool category and metadata once. Drop the lock before any .await.
937    let (tool_category, tool_metadata) = {
938        let reg = registry.read().unwrap();
939        let tool = reg.get(&tool_call.name);
940        let category = tool.map(|t| t.category).unwrap_or(ToolCategory::Plugin);
941        let metadata = tool
942            .map(|t| t.metadata.clone())
943            .unwrap_or_else(|| tools::flow_tool_metadata(&tool_call.name));
944        (category, metadata)
945    };
946
947    // Execute pre_tool hooks (can modify arguments OR block execution)
948    let pre_hook_data = serde_json::json!({
949        "tool_name": tool_call.name,
950        "arguments": args,
951    });
952    let pre_hook_results = tools::execute_hook(
953        plugin_tools,
954        tools::HookPoint::PreTool,
955        &pre_hook_data,
956        tein_ctx,
957    )?;
958
959    let pre_tool = apply_pre_tool_results(pre_hook_results, &tool_call.name, args);
960    let blocked = pre_tool.blocked;
961    let block_message = pre_tool.block_message;
962    args = pre_tool.args;
963    diagnostics.extend(pre_tool.diagnostics);
964
965    // If blocked, skip execution and use block message as result
966    let tool_result = if blocked {
967        block_message
968    } else if tool_metadata.flow_control {
969        // Handoff tools don't execute — they just produce a result message.
970        // The caller applies the actual handoff mutation.
971        if tool_metadata.ends_turn {
972            let message = args.get_str_or("message", "");
973            if message.is_empty() {
974                "Returning to user".to_string()
975            } else {
976                message.to_string()
977            }
978        } else {
979            let prompt = args.get_str_or("prompt", "");
980            if prompt.is_empty() {
981                "Continuing processing".to_string()
982            } else {
983                format!("Continuing with: {}", prompt)
984            }
985        }
986    } else if tool_call.name == tools::REFLECTION_TOOL_NAME && !use_reflection {
987        "Error: Reflection tool is not enabled".to_string()
988    } else if tool_call.name == tools::SEND_MESSAGE_TOOL_NAME {
989        // send_message is intercepted here before registry dispatch — it uses
990        // plugin hooks and async inbox delivery not expressible as a plain handler.
991        execute_send_message_pure(
992            app,
993            context_name,
994            plugin_tools,
995            &args,
996            &mut diagnostics,
997            tein_ctx,
998        )?
999    } else if tool_call.name == tools::MODEL_INFO_TOOL_NAME {
1000        // model_info requires an async gateway call not available at registration time.
1001        match args.get_str("model") {
1002            Some(model) => {
1003                let gateway = build_gateway(resolved_config)?;
1004                match crate::model_info::fetch_metadata(&gateway, model).await {
1005                    Ok(metadata) => {
1006                        let json = crate::model_info::format_model_json(&metadata);
1007                        serde_json::to_string_pretty(&json)
1008                            .unwrap_or_else(|e| format!("Error serialising metadata: {}", e))
1009                    }
1010                    Err(e) => format!("Error: {}", e),
1011                }
1012            }
1013            None => "Error: missing required 'model' parameter".to_string(),
1014        }
1015    } else {
1016        // Permission middleware: gate on category, then dispatch via registry.
1017        // Category was looked up from registry above; unknown tools default to Plugin.
1018        let permission_denied: Option<String> = match tool_category {
1019            ToolCategory::FsRead => {
1020                // VFS paths bypass OS permission gating; zone-based permissions enforced inside handler.
1021                let raw_path = args.get_str("path").unwrap_or("");
1022                if VfsPath::is_vfs_uri(raw_path) {
1023                    None // bypass: let registry dispatch handle it
1024                } else {
1025                    let resolved_path_str =
1026                        if !raw_path.is_empty() && std::path::Path::new(raw_path).is_relative() {
1027                            project_root.join(raw_path).to_string_lossy().to_string()
1028                        } else {
1029                            raw_path.to_string()
1030                        };
1031                    if !resolved_path_str.is_empty() {
1032                        match tools::classify_file_path(&resolved_path_str, resolved_config) {
1033                            Ok(tools::FilePathAccess::Allowed(_)) => None,
1034                            Ok(tools::FilePathAccess::NeedsPermission(_)) => {
1035                                let hook_data = serde_json::json!({
1036                                    "tool_name": tool_call.name,
1037                                    "path": resolved_path_str,
1038                                });
1039                                check_permission(
1040                                    plugin_tools,
1041                                    tools::HookPoint::PreFileRead,
1042                                    &hook_data,
1043                                    permission_handler,
1044                                    tein_ctx,
1045                                )?
1046                                .err()
1047                                .map(|r| format!("Permission denied: {r}"))
1048                            }
1049                            Err(e) => Some(format!("Permission denied: {e}")),
1050                        }
1051                    } else {
1052                        None // cache_id / no-path access — always allowed
1053                    }
1054                }
1055            }
1056            ToolCategory::FsWrite => {
1057                // VFS paths: fire PreVfsWrite hook (advisory, non-blocking).
1058                // Zone-based permissions are enforced inside the handler.
1059                // OS paths go through the PreFileWrite permission gate.
1060                let raw_path = args.get_str("path").unwrap_or("");
1061                if VfsPath::is_vfs_uri(raw_path) {
1062                    let hook_data = serde_json::json!({
1063                        "tool_name": tool_call.name,
1064                        "path": raw_path,
1065                        "content": args.get_str("content"),
1066                        "caller": context_name,
1067                    });
1068                    let _ = tools::execute_hook(
1069                        plugin_tools,
1070                        tools::HookPoint::PreVfsWrite,
1071                        &hook_data,
1072                        tein_ctx,
1073                    );
1074                    None // never blocks — VFS zone permissions enforced in handler
1075                } else {
1076                    let hook_data = if tool_call.name == tools::FILE_EDIT_TOOL_NAME {
1077                        serde_json::json!({
1078                            "tool_name": tool_call.name,
1079                            "path": raw_path,
1080                            "operation": args.get_str("operation").unwrap_or(""),
1081                            "content": args.get_str("content"),
1082                        })
1083                    } else {
1084                        serde_json::json!({
1085                            "tool_name": tool_call.name,
1086                            "path": raw_path,
1087                            "content": args.get_str("content"),
1088                        })
1089                    };
1090                    check_permission(
1091                        plugin_tools,
1092                        tools::HookPoint::PreFileWrite,
1093                        &hook_data,
1094                        permission_handler,
1095                        tein_ctx,
1096                    )?
1097                    .err()
1098                    .map(|r| format!("Permission denied: {r}"))
1099                }
1100            }
1101            ToolCategory::Shell => {
1102                let hook_data = serde_json::json!({
1103                    "tool_name": tool_call.name,
1104                    "command": args.get_str("command").unwrap_or(""),
1105                });
1106                check_permission(
1107                    plugin_tools,
1108                    tools::HookPoint::PreShellExec,
1109                    &hook_data,
1110                    permission_handler,
1111                    tein_ctx,
1112                )?
1113                .err()
1114            }
1115            ToolCategory::Network => {
1116                let url = args.get_str("url").unwrap_or("");
1117                if url.is_empty() {
1118                    // No URL parameter — use summary_params for the permission prompt.
1119                    let summary = tools::tool_call_summary(
1120                        &registry.read().unwrap(),
1121                        &tool_call.name,
1122                        &tool_call.arguments,
1123                    )
1124                    .unwrap_or_default();
1125                    let hook_data = json!({
1126                        "tool_name": tool_call.name,
1127                        "summary": summary,
1128                        "safety": "no_url",
1129                    });
1130                    check_permission(
1131                        plugin_tools,
1132                        tools::HookPoint::PreFetchUrl,
1133                        &hook_data,
1134                        permission_handler,
1135                        tein_ctx,
1136                    )?
1137                    .err()
1138                    .map(|r| format!("Permission denied: {}", r))
1139                } else {
1140                    let safety = tools::classify_url(url);
1141                    if let Some(ref policy) = resolved_config.url_policy {
1142                        if tools::evaluate_url_policy(url, &safety, policy)
1143                            == tools::UrlAction::Deny
1144                        {
1145                            let reason = match &safety {
1146                                tools::UrlSafety::Sensitive(cat) => cat.to_string(),
1147                                tools::UrlSafety::Safe => "denied by URL policy".to_string(),
1148                            };
1149                            Some(format!("Permission denied: {}", reason))
1150                        } else {
1151                            None
1152                        }
1153                    } else if let tools::UrlSafety::Sensitive(category) = &safety {
1154                        let hook_data = json!({
1155                            "tool_name": tool_call.name,
1156                            "url": url,
1157                            "safety": "sensitive",
1158                            "reason": category.to_string(),
1159                        });
1160                        check_permission(
1161                            plugin_tools,
1162                            tools::HookPoint::PreFetchUrl,
1163                            &hook_data,
1164                            permission_handler,
1165                            tein_ctx,
1166                        )?
1167                        .err()
1168                        .map(|r| format!("Permission denied: {}", r))
1169                    } else {
1170                        None
1171                    }
1172                }
1173            }
1174            ToolCategory::Flow => {
1175                // summarize_content URL check (the only flow tool with a URL policy gate)
1176                if tool_call.name == tools::SUMMARIZE_CONTENT_TOOL_NAME
1177                    && let Some(source) = args.get_str("source")
1178                    && tools::is_url(source)
1179                {
1180                    let safety = tools::classify_url(source);
1181                    if let Some(ref policy) = resolved_config.url_policy {
1182                        if tools::evaluate_url_policy(source, &safety, policy)
1183                            == tools::UrlAction::Deny
1184                        {
1185                            let reason = match &safety {
1186                                tools::UrlSafety::Sensitive(cat) => cat.to_string(),
1187                                tools::UrlSafety::Safe => "denied by URL policy".to_string(),
1188                            };
1189                            let msg = format!("Permission denied: {}", reason);
1190                            return Ok(ToolExecutionResult {
1191                                final_result: msg.clone(),
1192                                original_result: msg,
1193                                was_cached: false,
1194                                diagnostics,
1195                            });
1196                        }
1197                    } else if let tools::UrlSafety::Sensitive(category) = &safety {
1198                        let hook_data = json!({
1199                            "tool_name": tool_call.name,
1200                            "url": source,
1201                            "safety": "sensitive",
1202                            "reason": category.to_string(),
1203                        });
1204                        match check_permission(
1205                            plugin_tools,
1206                            tools::HookPoint::PreFetchUrl,
1207                            &hook_data,
1208                            permission_handler,
1209                            tein_ctx,
1210                        )? {
1211                            Ok(()) => {}
1212                            Err(reason) => {
1213                                let msg = format!("Permission denied: {}", reason);
1214                                return Ok(ToolExecutionResult {
1215                                    final_result: msg.clone(),
1216                                    original_result: msg,
1217                                    was_cached: false,
1218                                    diagnostics,
1219                                });
1220                            }
1221                        }
1222                    }
1223                }
1224                None
1225            }
1226            _ => None,
1227        };
1228
1229        if let Some(reason) = permission_denied {
1230            reason
1231        } else {
1232            // Dispatch via registry.
1233            // Clone ToolImpl while holding the read lock, then drop the guard
1234            // before the .await so no lock is held across an async suspension.
1235            let tool_impl = {
1236                let reg = registry.read().unwrap();
1237                reg.get(&tool_call.name).map(|t| t.r#impl.clone())
1238            };
1239            let call_ctx = tools::ToolCallContext {
1240                app,
1241                context_name,
1242                config: resolved_config,
1243                project_root,
1244                vfs: &app.vfs,
1245                vfs_caller: crate::vfs::VfsCaller::Context(context_name),
1246            };
1247            match tool_impl {
1248                Some(ti) => {
1249                    match ToolRegistry::dispatch_impl(ti, &tool_call.name, &args, &call_ctx).await {
1250                        Ok(r) => r,
1251                        Err(e) => format!("Error: {}", e),
1252                    }
1253                }
1254                None => format!("Error: unknown tool: {}", tool_call.name),
1255            }
1256        }
1257    };
1258
1259    // Execute pre_tool_output hooks (can modify or replace output)
1260    let pre_output_hook_data = serde_json::json!({
1261        "tool_name": tool_call.name,
1262        "arguments": args,
1263        "output": tool_result,
1264    });
1265    let pre_output_hook_results = tools::execute_hook(
1266        plugin_tools,
1267        tools::HookPoint::PreToolOutput,
1268        &pre_output_hook_data,
1269        tein_ctx,
1270    )?;
1271
1272    let (tool_result, output_diagnostics) =
1273        apply_pre_tool_output_results(pre_output_hook_results, &tool_call.name, tool_result);
1274    diagnostics.extend(output_diagnostics);
1275
1276    // Check if output should be cached
1277    let (final_result, was_cached) = if !tool_result.starts_with("Error:")
1278        && crate::vfs_cache::should_cache(&tool_result, resolved_config.tool_output_cache_threshold)
1279    {
1280        // Fire pre_cache_output hook (can block caching)
1281        let pre_cache_data = serde_json::json!({
1282            "tool_name": tool_call.name,
1283            "output_size": tool_result.len(),
1284            "arguments": args,
1285        });
1286        let pre_cache_results = tools::execute_hook(
1287            plugin_tools,
1288            tools::HookPoint::PreCacheOutput,
1289            &pre_cache_data,
1290            tein_ctx,
1291        )?;
1292        let cache_blocked = pre_cache_results
1293            .iter()
1294            .any(|(_, r)| r.get_bool_or("block", false));
1295
1296        if cache_blocked {
1297            diagnostics.push(format!(
1298                "[Caching blocked by pre_cache_output hook for {}]",
1299                tool_call.name
1300            ));
1301            (tool_result.clone(), false)
1302        } else {
1303            let cache_id = crate::vfs_cache::generate_cache_id(&tool_call.name, &args);
1304            let vfs_path_str = crate::vfs_cache::vfs_path_for(context_name, &cache_id);
1305            let vfs_uri = crate::vfs_cache::vfs_uri_for(context_name, &cache_id);
1306            let vfs_path = crate::vfs::VfsPath::new(&vfs_path_str).map_err(|e| {
1307                std::io::Error::new(std::io::ErrorKind::InvalidInput, e.to_string())
1308            })?;
1309
1310            match app
1311                .vfs
1312                .write(
1313                    crate::vfs::VfsCaller::System,
1314                    &vfs_path,
1315                    tool_result.as_bytes(),
1316                )
1317                .await
1318            {
1319                Ok(()) => {
1320                    let truncated = crate::vfs_cache::truncated_message(
1321                        &vfs_uri,
1322                        &tool_call.name,
1323                        &tool_result,
1324                        resolved_config.tool_cache_preview_chars,
1325                    );
1326
1327                    diagnostics.push(format!(
1328                        "[Cached {} chars from {} at {}]",
1329                        tool_result.len(),
1330                        tool_call.name,
1331                        vfs_uri,
1332                    ));
1333
1334                    // Fire post_cache_output hook (notification only)
1335                    let post_cache_data = serde_json::json!({
1336                        "tool_name": tool_call.name,
1337                        "cache_id": cache_id,
1338                        "output_size": tool_result.len(),
1339                        "preview_size": truncated.len(),
1340                    });
1341                    let _ = tools::execute_hook(
1342                        plugin_tools,
1343                        tools::HookPoint::PostCacheOutput,
1344                        &post_cache_data,
1345                        tein_ctx,
1346                    );
1347
1348                    (truncated, true)
1349                }
1350                Err(e) => {
1351                    diagnostics.push(format!("[Failed to cache output: {}]", e));
1352                    (tool_result.clone(), false)
1353                }
1354            }
1355        }
1356    } else {
1357        (tool_result.clone(), false)
1358    };
1359
1360    // Execute post_tool_output hooks
1361    let post_output_hook_data = serde_json::json!({
1362        "tool_name": tool_call.name,
1363        "arguments": args,
1364        "output": tool_result,
1365        "final_output": final_result,
1366        "cached": was_cached,
1367    });
1368    let _ = tools::execute_hook(
1369        plugin_tools,
1370        tools::HookPoint::PostToolOutput,
1371        &post_output_hook_data,
1372        tein_ctx,
1373    );
1374
1375    Ok(ToolExecutionResult {
1376        final_result,
1377        original_result: tool_result,
1378        was_cached,
1379        diagnostics,
1380    })
1381}

1383/// Execute a single tool call with all hooks (sequential path).
1384///
1385/// Thin wrapper over `execute_tool_pure` that also handles handoff mutation
1386/// and emits diagnostics to the sink.
1387#[allow(clippy::too_many_arguments)]
1388async fn execute_single_tool<S: ResponseSink>(
1389    app: &AppState,
1390    context_name: &str,
1391    tool_call: &ratatoskr::ToolCall,
1392    plugin_tools: &[Tool],
1393    registry: &Arc<RwLock<ToolRegistry>>,
1394    handoff: &mut tools::Handoff,
1395    use_reflection: bool,
1396    resolved_config: &ResolvedConfig,
1397    permission_handler: Option<&PermissionHandler>,
1398    sink: &mut S,
1399    project_root: &Path,
1400    tein_ctx: Option<&tools::TeinHookContext<'_>>,
1401) -> io::Result<ToolExecutionResult> {
1402    // Apply handoff if this is a flow control tool
1403    let args: serde_json::Value =
1404        serde_json::from_str(&tool_call.arguments).unwrap_or(serde_json::json!({}));
1405    let tool_metadata = {
1406        let reg = registry.read().unwrap();
1407        tools::get_tool_metadata(&reg, &tool_call.name)
1408    };
1409    if tool_metadata.flow_control {
1410        if tool_metadata.ends_turn {
1411            handoff.set_user(args.get_str_or("message", "").to_string());
1412        } else {
1413            handoff.set_agent(args.get_str_or("prompt", "").to_string());
1414        }
1415    }
1416
1417    let result = execute_tool_pure(
1418        app,
1419        context_name,
1420        tool_call,
1421        plugin_tools,
1422        registry,
1423        use_reflection,
1424        resolved_config,
1425        permission_handler,
1426        project_root,
1427        tein_ctx,
1428    )
1429    .await?;
1430
1431    // Emit collected diagnostics to sink
1432    for diag in &result.diagnostics {
1433        sink.handle(ResponseEvent::ToolDiagnostic {
1434            tool: tool_call.name.clone(),
1435            message: diag.clone(),
1436        })?;
1437    }
1438
1439    Ok(result)
1440}

1442/// Sink-free send_message execution. Collects diagnostics into the provided vec.
1443#[allow(clippy::too_many_arguments)]
1444fn execute_send_message_pure(
1445    app: &AppState,
1446    context_name: &str,
1447    tools: &[Tool],
1448    args: &serde_json::Value,
1449    diagnostics: &mut Vec<String>,
1450    tein_ctx: Option<&tools::TeinHookContext<'_>>,
1451) -> io::Result<String> {
1452    let to = args.get_str_or("to", "");
1453    let content = args.get_str_or("content", "");
1454    let from = args.get_str("from").unwrap_or(context_name);
1455
1456    if to.is_empty() {
1457        return Ok("Error: 'to' field is required".to_string());
1458    }
1459    if content.is_empty() {
1460        return Ok("Error: 'content' field is required".to_string());
1461    }
1462
1463    // Execute pre_send_message hooks
1464    let pre_hook_data = serde_json::json!({
1465        "from": from,
1466        "to": to,
1467        "content": content,
1468        "context_name": context_name,
1469    });
1470    let pre_hook_results = tools::execute_hook(
1471        tools,
1472        tools::HookPoint::PreSendMessage,
1473        &pre_hook_data,
1474        tein_ctx,
1475    )?;
1476
1477    // Check if any hook claimed delivery
1478    let mut delivered_via: Option<String> = None;
1479    for (hook_tool_name, hook_result) in &pre_hook_results {
1480        if hook_result.get_bool_or("delivered", false) {
1481            let via = hook_result.get_str_or("via", hook_tool_name);
1482            delivered_via = Some(via.to_string());
1483            diagnostics.push(format!(
1484                "[Hook pre_send_message: {} intercepted delivery]",
1485                hook_tool_name
1486            ));
1487            break;
1488        }
1489    }
1490
1491    let delivery_result = if let Some(via) = delivered_via {
1492        format!("Message delivered to '{}' via {}", to, via)
1493    } else {
1494        let entry = InboxEntry {
1495            id: Uuid::new_v4().to_string(),
1496            timestamp: now_timestamp(),
1497            from: from.to_string(),
1498            to: to.to_string(),
1499            content: content.to_string(),
1500        };
1501        match app.append_to_inbox(to, &entry) {
1502            Ok(()) => format!("Message delivered to '{}' via local inbox", to),
1503            Err(e) => format!("Error delivering message: {}", e),
1504        }
1505    };
1506
1507    // Execute post_send_message hooks
1508    let post_hook_data = serde_json::json!({
1509        "from": from,
1510        "to": to,
1511        "content": content,
1512        "context_name": context_name,
1513        "delivery_result": delivery_result,
1514    });
1515    let _ = tools::execute_hook(
1516        tools,
1517        tools::HookPoint::PostSendMessage,
1518        &post_hook_data,
1519        tein_ctx,
1520    );
1521
1522    Ok(delivery_result)
1523}

1525/// Process all tool calls from a response.
1526///
1527/// Parallel-safe tools (ToolMetadata::parallel == true) run concurrently via
1528/// `join_all`. Sequential tools (flow_control, parallel == false) run after
1529/// the parallel batch completes. Results are emitted to the sink and transcript
1530/// in the original tool_call order regardless of execution order.
1531#[allow(clippy::too_many_arguments)]
1532async fn process_tool_calls<S: ResponseSink>(
1533    app: &AppState,
1534    context_name: &str,
1535    username: &str,
1536    tool_calls: &[ratatoskr::ToolCall],
1537    messages: &mut Vec<serde_json::Value>,
1538    plugin_tools: &[Tool],
1539    registry: &Arc<RwLock<ToolRegistry>>,
1540    handoff: &mut tools::Handoff,
1541    use_reflection: bool,
1542    resolved_config: &ResolvedConfig,
1543    fuel_remaining: &mut usize,
1544    fuel_total: usize,
1545    fuel_unlimited: bool,
1546    sink: &mut S,
1547    permission_handler: Option<&PermissionHandler>,
1548    project_root: &Path,
1549    loop_detector: &mut LoopDetector,
1550    tein_ctx: Option<&tools::TeinHookContext<'_>>,
1551) -> io::Result<()> {
1552    // Convert tool calls to JSON format for the assistant message
1553    let tool_calls_json: Vec<serde_json::Value> = tool_calls
1554        .iter()
1555        .map(|tc| {
1556            serde_json::json!({
1557                "id": tc.id,
1558                "type": "function",
1559                "function": {
1560                    "name": tc.name,
1561                    "arguments": tc.arguments,
1562                }
1563            })
1564        })
1565        .collect();
1566
1567    messages.push(serde_json::json!({
1568        "role": "assistant",
1569        "tool_calls": tool_calls_json,
1570    }));
1571
1572    // Partition tool calls into parallel-safe and sequential batches.
1573    // We store (original_index, tool_call) to preserve ordering in results.
1574    let mut parallel_batch: Vec<(usize, &ratatoskr::ToolCall)> = Vec::new();
1575    let mut sequential_batch: Vec<(usize, &ratatoskr::ToolCall)> = Vec::new();
1576
1577    for (i, tc) in tool_calls.iter().enumerate() {
1578        let metadata = tools::get_tool_metadata(&registry.read().unwrap(), &tc.name);
1579        if metadata.parallel && !metadata.flow_control {
1580            parallel_batch.push((i, tc));
1581        } else {
1582            sequential_batch.push((i, tc));
1583        }
1584    }
1585
1586    // Results indexed by original position
1587    let mut results: Vec<Option<ToolExecutionResult>> =
1588        (0..tool_calls.len()).map(|_| None).collect();
1589
1590    // Execute parallel batch concurrently via join_all.
1591    // These futures run on the current task (no spawn), interleaving at .await
1592    // points — safe with !Send types like AppState's RefCell.
1593    if !parallel_batch.is_empty() {
1594        let parallel_futures: Vec<_> = parallel_batch
1595            .iter()
1596            .map(|(_idx, tc)| {
1597                execute_tool_pure(
1598                    app,
1599                    context_name,
1600                    tc,
1601                    plugin_tools,
1602                    registry,
1603                    use_reflection,
1604                    resolved_config,
1605                    permission_handler,
1606                    project_root,
1607                    tein_ctx,
1608                )
1609            })
1610            .collect();
1611
1612        let parallel_results = futures_util::future::join_all(parallel_futures).await;
1613
1614        for ((idx, _tc), result) in parallel_batch.iter().zip(parallel_results) {
1615            results[*idx] = Some(result?);
1616        }
1617    }
1618
1619    let parallel_indices: std::collections::HashSet<usize> =
1620        parallel_batch.iter().map(|(idx, _)| *idx).collect();
1621
1622    // Execute sequential batch one at a time (these may mutate handoff)
1623    for (idx, tc) in &sequential_batch {
1624        let result = execute_single_tool(
1625            app,
1626            context_name,
1627            tc,
1628            plugin_tools,
1629            registry,
1630            handoff,
1631            use_reflection,
1632            resolved_config,
1633            permission_handler,
1634            sink,
1635            project_root,
1636            tein_ctx,
1637        )
1638        .await?;
1639        results[*idx] = Some(result);
1640    }
1641
1642    // Write all tool_call entries to transcript first (matches API message order:
1643    // one assistant message with tool_calls[], then individual tool result messages).
1644    // Flow control tools produce message + control_transfer entries instead.
1645    for (i, tc) in tool_calls.iter().enumerate() {
1646        let metadata = tools::get_tool_metadata(&registry.read().unwrap(), &tc.name);
1647        if metadata.flow_control {
1648            // Flow control tools produce message + control_transfer entries later (in the
1649            // results loop). No tool_call entry written for them.
1650        } else {
1651            let tool_call_entry =
1652                create_tool_call_entry(context_name, &tc.name, &tc.arguments, &tc.id);
1653            app.append_to_transcript_and_context(context_name, &tool_call_entry)?;
1654            sink.handle(ResponseEvent::TranscriptEntry(tool_call_entry))?;
1655        }
1656
1657        // Pre-log diagnostics for parallel-executed tools only.
1658        // Sequential tools have already emitted their diagnostics in execute_single_tool.
1659        if parallel_indices.contains(&i)
1660            && let Some(result) = &results[i]
1661        {
1662            for diag in &result.diagnostics {
1663                sink.handle(ResponseEvent::ToolDiagnostic {
1664                    tool: tc.name.clone(),
1665                    message: diag.clone(),
1666                })?;
1667            }
1668        }
1669    }
1670
1671    // Emit sink events and write tool_result entries in original order
1672    for (i, tc) in tool_calls.iter().enumerate() {
1673        let result = results[i]
1674            .take()
1675            .expect("all tool results should be populated");
1676
1677        sink.handle(ResponseEvent::ToolDiagnostic {
1678            tool: tc.name.clone(),
1679            message: format!("[Tool: {}]", tc.name),
1680        })?;
1681
1682        let summary = tools::tool_call_summary(&registry.read().unwrap(), &tc.name, &tc.arguments);
1683        sink.handle(ResponseEvent::ToolStart {
1684            name: tc.name.clone(),
1685            summary,
1686        })?;
1687
1688        // Log tool result to transcript.
1689        // Flow control tools produce message + control_transfer entries instead of tool_call/result.
1690        let metadata = tools::get_tool_metadata(&registry.read().unwrap(), &tc.name);
1691
1692        if metadata.flow_control {
1693            // call_user: write the message entry + control_transfer entry
1694            if metadata.ends_turn {
1695                let args: serde_json::Value =
1696                    serde_json::from_str(&tc.arguments).unwrap_or(serde_json::json!({}));
1697                let message = args.get_str_or("message", "").to_string();
1698
1699                if !message.is_empty() {
1700                    let msg_entry = create_flow_control_message_entry(
1701                        context_name,
1702                        username,
1703                        &message,
1704                        "agent",
1705                    );
1706                    app.append_to_transcript_and_context(context_name, &msg_entry)?;
1707                    sink.handle(ResponseEvent::TranscriptEntry(msg_entry))?;
1708                }
1709
1710                // Write control_transfer entry
1711                let ct_entry = create_control_transfer_entry(context_name, username);
1712                app.append_to_transcript_and_context(context_name, &ct_entry)?;
1713                sink.handle(ResponseEvent::TranscriptEntry(ct_entry))?;
1714
1715                // Display the message to the user
1716                if !message.is_empty() {
1717                    sink.handle(ResponseEvent::StartResponse)?;
1718                    sink.handle(ResponseEvent::TextChunk(&message))?;
1719                    sink.handle(ResponseEvent::Newline)?;
1720                    sink.handle(ResponseEvent::Finished)?;
1721                }
1722            }
1723            // Note: call_agent is disabled as an LLM tool (not in FLOW_TOOL_DEFS).
1724            // If it somehow appears (e.g., from hooks/fallback), the existing Handoff
1725            // mechanism handles continuation — no transcript entries needed here.
1726            // call_agent will be repurposed for future inter-agent control transfer.
1727
1728            // Don't push tool result to messages for flow control tools.
1729            // call_user exits immediately; the Handoff mechanism handles call_agent.
1730            continue;
1731        }
1732
1733        let logged_result = if result.was_cached {
1734            &result.final_result
1735        } else {
1736            &result.original_result
1737        };
1738        let tool_result_entry =
1739            create_tool_result_entry(context_name, &tc.name, logged_result, &tc.id);
1740        app.append_to_transcript_and_context(context_name, &tool_result_entry)?;
1741        sink.handle(ResponseEvent::TranscriptEntry(tool_result_entry))?;
1742
1743        sink.handle(ResponseEvent::ToolResult {
1744            name: tc.name.clone(),
1745            result: result.final_result.clone(),
1746            cached: result.was_cached,
1747        })?;
1748
1749        // Show full content of goals updates
1750        if matches!(tc.name.as_str(), "update_goals")
1751            && let Ok(args) = serde_json::from_str::<serde_json::Value>(&tc.arguments)
1752            && let Some(content) = args["content"].as_str()
1753        {
1754            sink.handle(ResponseEvent::ToolDiagnostic {
1755                tool: tc.name.clone(),
1756                message: format!("[{}]\n{}", tc.name, content),
1757            })?;
1758        }
1759
1760        // Execute post_tool hooks
1761        let args: serde_json::Value =
1762            serde_json::from_str(&tc.arguments).unwrap_or(serde_json::json!({}));
1763        let post_hook_data = serde_json::json!({
1764            "tool_name": tc.name,
1765            "arguments": args,
1766            "result": result.original_result,
1767            "cached": result.was_cached,
1768        });
1769        let _ = tools::execute_hook(
1770            plugin_tools,
1771            tools::HookPoint::PostTool,
1772            &post_hook_data,
1773            tein_ctx,
1774        );
1775
1776        // Fire PostVfsWrite for successful VFS writes (advisory, non-blocking).
1777        {
1778            let raw_path = args.get_str("path").unwrap_or("");
1779            let is_vfs_write = {
1780                let reg = registry.read().unwrap();
1781                reg.get(&tc.name)
1782                    .map(|t| t.category == tools::ToolCategory::FsWrite)
1783                    .unwrap_or(false)
1784            };
1785            if is_vfs_write
1786                && VfsPath::is_vfs_uri(raw_path)
1787                && !result.original_result.starts_with("Error")
1788            {
1789                let vfs_post_hook_data = serde_json::json!({
1790                    "tool_name": tc.name,
1791                    "path": raw_path,
1792                    "caller": context_name,
1793                });
1794                let _ = tools::execute_hook(
1795                    plugin_tools,
1796                    tools::HookPoint::PostVfsWrite,
1797                    &vfs_post_hook_data,
1798                    tein_ctx,
1799                );
1800            }
1801        }
1802
1803        // Apply handoff for parallel-executed flow control tools
1804        // (sequential ones already applied via execute_single_tool)
1805        if metadata.flow_control && metadata.parallel {
1806            // This shouldn't happen (flow_control tools are always sequential),
1807            // but handle it defensively
1808            if metadata.ends_turn {
1809                handoff.set_user(args.get_str_or("message", "").to_string());
1810            } else {
1811                handoff.set_agent(args.get_str_or("prompt", "").to_string());
1812            }
1813        }
1814
1815        // Loop detection: warn and charge fuel if same (tool, args, result) repeats.
1816        let is_loop = loop_detector.check_and_update(
1817            &tc.name,
1818            &tc.arguments,
1819            &result.final_result,
1820            result.was_cached,
1821        );
1822
1823        messages.push(serde_json::json!({
1824            "role": "tool",
1825            "tool_call_id": tc.id,
1826            "content": result.final_result,
1827        }));
1828
1829        if is_loop {
1830            if !fuel_unlimited {
1831                *fuel_remaining =
1832                    fuel_remaining.saturating_sub(resolved_config.fuel_empty_response_cost);
1833                sink.handle(ResponseEvent::FuelStatus {
1834                    event: crate::api::sink::FuelEvent::EmptyResponse,
1835                    remaining: *fuel_remaining,
1836                    total: fuel_total,
1837                })?;
1838                if *fuel_remaining == 0 {
1839                    sink.handle(ResponseEvent::FuelExhausted { total: fuel_total })?;
1840                    return Ok(());
1841                }
1842            }
1843            let warning = format!(
1844                "[Loop detected] You have called {}({}) {} time(s) in a row and received the same result. \
1845                 This is not making progress. Try a different approach, use a different tool, \
1846                 or ask the user for help.",
1847                tc.name, tc.arguments, loop_detector.consecutive_count,
1848            );
1849            messages.push(serde_json::json!({
1850                "role": "tool",
1851                "tool_call_id": tc.id,
1852                "content": warning,
1853            }));
1854        }
1855    }
1856
1857    // Execute post_tool_batch hook — allows plugins to override fallback after seeing tool results
1858    let tool_batch_info: Vec<serde_json::Value> = tool_calls
1859        .iter()
1860        .map(|tc| {
1861            json!({
1862                "name": tc.name,
1863                "arguments": serde_json::from_str::<serde_json::Value>(&tc.arguments).unwrap_or(json!({})),
1864            })
1865        })
1866        .collect();
1867    let mut hook_data = json!({
1868        "context_name": context_name,
1869        "current_fallback": resolved_config.fallback_tool,
1870        "tool_calls": tool_batch_info,
1871    });
1872    if !fuel_unlimited {
1873        hook_data["fuel_remaining"] = json!(*fuel_remaining);
1874        hook_data["fuel_total"] = json!(fuel_total);
1875    }
1876    let hook_results = tools::execute_hook(
1877        plugin_tools,
1878        tools::HookPoint::PostToolBatch,
1879        &hook_data,
1880        tein_ctx,
1881    )?;
1882    apply_hook_overrides(handoff, fuel_remaining, fuel_unlimited, &hook_results, sink)?;
1883
1884    Ok(())
1885}

1887/// Result of handling a final (non-tool-call) response.
1888enum FinalResponseAction {

1890    ReturnToUser,
1891    /// Continue with another prompt (agent loop).

1895/// Handle the final text response from the LLM.
1896///
1897/// Handles: Empty response detection/retry, save assistant message,
1898/// post_message hook, handoff decision (return vs recurse).
1899#[allow(clippy::too_many_arguments)]
1900fn handle_final_response<S: ResponseSink>(
1901    app: &AppState,
1902    context_name: &str,
1903    full_response: &str,
1904    final_prompt: &str,
1905    mut handoff: tools::Handoff,
1906    plugin_tools: &[Tool],
1907    resolved_config: &ResolvedConfig,
1908    sink: &mut S,
1909    tein_ctx: Option<&tools::TeinHookContext<'_>>,
1910) -> io::Result<FinalResponseAction> {
1911    // Get or create context to add the message
1912    let mut context = app.get_or_create_context(context_name)?;
1913
1914    // Save the assistant's response
1915    app.add_message(
1916        &mut context,
1917        "assistant".to_string(),
1918        full_response.to_string(),
1919    );
1920
1921    let assistant_entry =
1922        create_assistant_message_entry(context_name, full_response, &resolved_config.username);
1923    app.append_to_transcript_and_context(context_name, &assistant_entry)?;
1924    sink.handle(ResponseEvent::TranscriptEntry(assistant_entry))?;
1925
1926    // Execute post_message hooks
1927    let hook_data = serde_json::json!({
1928        "prompt": final_prompt,
1929        "response": full_response,
1930        "context_name": context_name,
1931    });
1932    let _ = tools::execute_hook(
1933        plugin_tools,
1934        tools::HookPoint::PostMessage,
1935        &hook_data,
1936        tein_ctx,
1937    );
1938
1939    if app.should_warn(&context.messages) {
1940        let remaining = app.remaining_tokens(&context.messages);
1941        sink.handle(ResponseEvent::ContextWarning {
1942            tokens_remaining: remaining,
1943        })?;
1944    }
1945
1946    sink.handle(ResponseEvent::Newline)?;
1947
1948    // Determine next action based on handoff.
1949    // Fuel exhaustion is the caller's responsibility — we just report the action.
1950    match handoff.take() {
1951        tools::HandoffTarget::User { .. } => {
1952            // call_user message already displayed and logged in process_tool_calls.
1953            Ok(FinalResponseAction::ReturnToUser)
1954        }
1955        // call_agent disabled as LLM tool. HandoffTarget::Agent retained for the
1956        // fallback tool mechanism and hook overrides. Will be repurposed for
1957        // inter-agent control transfer.
1958        tools::HandoffTarget::Agent { prompt } => {
1959            Ok(FinalResponseAction::ContinueWithPrompt(prompt))
1960        }
1961    }
1962}

1964/// Send a prompt to the LLM with streaming response via ResponseSink.
1965///
1966/// This is the main entry point for sending prompts. It handles:
1967/// - Hook execution (pre_message, post_message, etc.)
1968/// - Inbox message injection
1969/// - Tool execution loop with fuel budget
1970/// - Context management
1971/// - Auto-compaction
1972///
1973/// # Arguments
1974///
1975/// * `context_name` - The name of the context to use for this prompt
1976#[allow(clippy::too_many_arguments)]
1977pub async fn send_prompt<S: ResponseSink>(
1978    app: &AppState,
1979    context_name: &str,
1980    initial_prompt: String,
1981    registry: Arc<RwLock<ToolRegistry>>,
1982    resolved_config: &ResolvedConfig,
1983    options: &PromptOptions<'_>,
1984    sink: &mut S,
1985    permission_handler: Option<&PermissionHandler>,
1986    home_dir: &Path,
1987    project_root: &Path,
1988) -> io::Result<()> {
1989    let mut resolved_config = resolved_config.clone();
1990    tools::ensure_project_root_allowed(&mut resolved_config, project_root);
1991    let resolved_config = resolved_config; // re-bind as immutable
1992
1993    // Context for tein hook dispatch — enables call-tool and (harness io) from
1994    // tein hook callbacks in the send path.
1995    // When synthesised-tools feature is off, tein_hook_ctx_ref is always None.
1996    #[cfg(feature = "synthesised-tools")]
1997    let tein_hook_ctx = tools::TeinHookContext {
1998        app,
1999        context_name,
2000        config: &resolved_config,
2001        project_root,
2002        vfs: &app.vfs,
2003        registry: Arc::clone(&registry),
2004    };
2005    #[cfg(feature = "synthesised-tools")]
2006    let tein_hook_ctx_ref: Option<&tools::TeinHookContext<'_>> = Some(&tein_hook_ctx);
2007    #[cfg(not(feature = "synthesised-tools"))]
2008    let tein_hook_ctx_ref: Option<&tools::TeinHookContext<'_>> = None;
2009
2010    let fuel_total = resolved_config.fuel;
2011    let mut fuel_remaining = fuel_total;
2012    let fuel_unlimited = fuel_total == 0;
2013    let mut current_prompt = initial_prompt;
2014
2015    // Tools eligible for hook dispatch: see Tool::is_hook_eligible.
2016    let plugin_tools: Vec<Tool> = registry
2017        .read()
2018        .unwrap()
2019        .filter(|t| t.is_hook_eligible())
2020        .into_iter()
2021        .cloned()
2022        .collect();
2023
2024    // Outer loop: each iteration is a full setup + agentic exchange.
2025    // First iteration is the user's turn (free); continuations cost 1 fuel.
2026    loop {
2027        // === Validation & Setup ===
2028        if current_prompt.trim().is_empty() {
2029            return Err(io::Error::new(
2030                ErrorKind::InvalidInput,
2031                "Prompt cannot be empty",
2032            ));
2033        }
2034
2035        app.validate_config(&resolved_config, &registry.read().unwrap())?;
2036
2037        if !fuel_unlimited {
2038            sink.handle(ResponseEvent::FuelStatus {
2039                remaining: fuel_remaining,
2040                total: fuel_total,
2041                event: crate::api::sink::FuelEvent::EnteringTurn,
2042            })?;
2043        }
2044        let use_reflection = options.use_reflection;
2045        let debug = options.debug;
2046
2047        let mut context = app.get_or_create_context(context_name)?;
2048
2049        // === Pre-message Hooks & Inbox ===
2050        let mut final_prompt = current_prompt.clone();
2051        let hook_data = serde_json::json!({
2052            "prompt": current_prompt,
2053            "context_name": context.name,
2054            "summary": context.summary,
2055        });
2056        let hook_results = tools::execute_hook(
2057            &plugin_tools,
2058            tools::HookPoint::PreMessage,
2059            &hook_data,
2060            tein_hook_ctx_ref,
2061        )?;
2062        for (tool_name, result) in hook_results {
2063            if let Some(modified) = result.get_str("prompt") {
2064                sink.handle(ResponseEvent::HookDebug {
2065                    hook: tool_name.clone(),
2066                    message: format!("[Hook pre_message: {} modified prompt]", tool_name),
2067                })?;
2068                final_prompt = modified.to_string();
2069            }
2070        }
2071
2072        // Check inbox and inject messages before the user prompt
2073        let inbox_messages = app.load_and_clear_inbox(context_name)?;
2074        if !inbox_messages.is_empty() {
2075            let mut inbox_content = String::from("--- INBOX MESSAGES ---\n");
2076            for msg in &inbox_messages {
2077                inbox_content.push_str(&format!("[From: {}] {}\n", msg.from, msg.content));
2078            }
2079            inbox_content.push_str("--- END INBOX ---\n\n");
2080            final_prompt = format!("{}{}", inbox_content, final_prompt);
2081            sink.handle(ResponseEvent::InboxInjected {
2082                count: inbox_messages.len(),
2083            })?;
2084        }
2085
2086        // Add datetime prefix to user message
2087        let datetime_prefix = chrono::Local::now().format("%Y%m%d-%H%M%z").to_string();
2088        let prefixed_prompt = format!("[{}] {}", datetime_prefix, final_prompt);
2089
2090        // Add user message to context and transcript
2091        app.add_message(&mut context, "user".to_string(), prefixed_prompt.clone());
2092        let user_entry =
2093            create_user_message_entry(context_name, &prefixed_prompt, &resolved_config.username);
2094        app.append_to_transcript_and_context(context_name, &user_entry)?;
2095        sink.handle(ResponseEvent::TranscriptEntry(user_entry))?;
2096
2097        // Mark control transfer from user to agent
2098        let ct_entry = create_control_transfer_entry(&resolved_config.username, context_name);
2099        app.append_to_transcript_and_context(context_name, &ct_entry)?;
2100        sink.handle(ResponseEvent::TranscriptEntry(ct_entry))?;
2101
2102        // Context window warning
2103        if app.should_warn(&context.messages) {
2104            let remaining = app.remaining_tokens(&context.messages);
2105            sink.handle(ResponseEvent::ContextWarning {
2106                tokens_remaining: remaining,
2107            })?;
2108        }
2109
2110        // === Auto-compaction Check ===
2111        if app.should_auto_compact(&context, &resolved_config) {
2112            return compact_context_with_llm(app, context_name, &resolved_config, &NoopSink).await;
2113        }
2114
2115        // === Build System Prompt ===
2116        let full_system_prompt = build_full_system_prompt(
2117            app,
2118            context_name,
2119            &context.summary,
2120            use_reflection,
2121            &plugin_tools,
2122            &resolved_config,
2123            sink,
2124            home_dir,
2125            project_root,
2126            tein_hook_ctx_ref,
2127        )?;
2128
2129        // === Prepare Messages ===
2130        let mut messages: Vec<serde_json::Value> = if !full_system_prompt.is_empty() {
2131            vec![serde_json::json!({
2132                "role": "system",
2133                "content": full_system_prompt,
2134            })]
2135        } else {
2136            Vec::new()
2137        };
2138
2139        // Add conversation messages (skip system messages, strip internal _id field)
2140        for m in &context.messages {
2141            let role = m["role"].as_str().unwrap_or("");
2142            if role == "system" {
2143                continue;
2144            }
2145            let mut msg = m.clone();
2146            if let Some(obj) = msg.as_object_mut() {
2147                obj.remove("_id");
2148            }
2149            messages.push(msg);
2150        }
2151
2152        // === Ephemeral Task Injection ===
2153        // Collect task metadata from VFS at request time and inject a summary
2154        // table as a system message before the current user turn. Never persisted
2155        // to transcript — cache-friendly.
2156        {
2157            let tasks = crate::state::tasks::collect_tasks(&app.vfs, context_name).await;
2158            let summary = crate::state::tasks::build_summary_table(&tasks);
2159            crate::state::tasks::inject_before_last_user(&mut messages, summary);
2160        }
2161
2162        // === Prepare Tools ===
2163        // Resolve available preset capability names for the current cost tier so the
2164        // LLM sees valid values in the spawn_agent tool description.
2165        let preset_capabilities: Vec<String> = {
2166            use ratatoskr::ModelGateway;
2167            match build_gateway(&resolved_config) {
2168                Ok(gw) => gw
2169                    .list_presets()
2170                    .get(&resolved_config.subagent_cost_tier)
2171                    .map(|caps| caps.iter().cloned().collect())
2172                    .unwrap_or_default(),
2173                Err(_) => vec![],
2174            }
2175        };
2176        let spawn_agent_preset_desc = tools::spawn_agent_preset_description(&preset_capabilities);
2177
2178        // Build all tools from the registry in a single pass. Flow tools are
2179        // included here; spawn_agent's `preset` param description is patched
2180        // inline with the dynamic preset list.
2181        let mut all_tools: Vec<serde_json::Value> = {
2182            let reg = registry.read().unwrap();
2183            reg.all()
2184                .filter(|t| t.name != tools::REFLECTION_TOOL_NAME || use_reflection)
2185                .map(|t| {
2186                    let mut json = t.to_api_format();
2187                    if t.name == tools::SPAWN_AGENT_TOOL_NAME {
2188                        json["function"]["parameters"]["properties"]["preset"]["description"] =
2189                            serde_json::Value::String(spawn_agent_preset_desc.clone());
2190                    }
2191                    json
2192                })
2193                .collect()
2194        };
2195        annotate_fallback_tool(&mut all_tools, &resolved_config.fallback_tool);
2196        all_tools = filter_tools_by_config(all_tools, &resolved_config.tools, &registry);
2197
2198        // filter synthesised tools by visibility (zone-based scoping)
2199        #[cfg(feature = "synthesised-tools")]
2200        {
2201            let context_flocks =
2202                tools::vfs_block_on(app.vfs.flock_list_for(&context.name)).unwrap_or_default();
2203            let registry_guard = registry.read().unwrap();
2204            all_tools.retain(|tool| {
2205                let name = tool
2206                    .get("function")
2207                    .and_then(|f| f.get("name"))
2208                    .and_then(|n| n.as_str())
2209                    .unwrap_or("");
2210                registry_guard.is_tool_visible(name, &context.name, &context_flocks)
2211            });
2212        }
2213
2214        // Execute pre_api_tools hook
2215        let tool_info = build_tool_info_list(&all_tools, &registry);
2216        let mut hook_data = json!({
2217            "context_name": context.name,
2218            "tools": tool_info,
2219        });
2220        if !fuel_unlimited {
2221            hook_data["fuel_remaining"] = json!(fuel_remaining);
2222            hook_data["fuel_total"] = json!(fuel_total);
2223        }
2224        let hook_results = tools::execute_hook(
2225            &plugin_tools,
2226            tools::HookPoint::PreApiTools,
2227            &hook_data,
2228            tein_hook_ctx_ref,
2229        )?;
2230        all_tools = filter_tools_from_hook_results(all_tools, &hook_results, sink)?;
2231
2232        // === Build Request ===
2233        // Derive ChatOptions from config (single source of truth for API params).
2234        // Serialise to JSON for hook inspection/modification, then deserialise
2235        // back so hook changes are honoured in the actual API call.
2236        let tools_for_request = if resolved_config.no_tool_calls {
2237            None
2238        } else {
2239            Some(all_tools.as_slice())
2240        };
2241        let base_options = to_chat_options(&resolved_config);
2242        let mut request_body = serde_json::to_value(&base_options).unwrap_or_else(|_| json!({}));
2243        request_body["messages"] = json!(messages);
2244        request_body["stream"] = json!(true);
2245        if let Some(tools) = tools_for_request
2246            && !tools.is_empty()
2247        {
2248            request_body["tools"] = json!(tools);
2249        }
2250
2251        // Execute pre_api_request hook
2252        let mut hook_data = json!({
2253            "context_name": context.name,
2254            "request_body": request_body,
2255        });
2256        if !fuel_unlimited {
2257            hook_data["fuel_remaining"] = json!(fuel_remaining);
2258            hook_data["fuel_total"] = json!(fuel_total);
2259        }
2260        let hook_results = tools::execute_hook(
2261            &plugin_tools,
2262            tools::HookPoint::PreApiRequest,
2263            &hook_data,
2264            tein_hook_ctx_ref,
2265        )?;
2266        request_body = apply_request_modifications(request_body, &hook_results, sink)?;
2267
2268        // Deserialise ChatOptions from the (potentially hook-modified) request body
2269        // so API parameter overrides from hooks are honoured in the actual call.
2270        let chat_options: ChatOptions =
2271            serde_json::from_value(request_body.clone()).unwrap_or(base_options);
2272
2273        // === Initialize Handoff ===
2274        let fallback = options.fallback_override.clone().unwrap_or_else(|| {
2275            let meta =
2276                tools::get_tool_metadata(&registry.read().unwrap(), &resolved_config.fallback_tool);
2277            if meta.ends_turn {
2278                tools::HandoffTarget::User {
2279                    message: String::new(),
2280                }
2281            } else {
2282                tools::HandoffTarget::Agent {
2283                    prompt: String::new(),
2284                }
2285            }
2286        });
2287        let mut handoff = tools::Handoff::new(fallback);
2288
2289        // Execute pre_agentic_loop hook
2290        let mut hook_data = json!({
2291            "context_name": context.name,
2292            "current_fallback": resolved_config.fallback_tool,
2293            "message": final_prompt,
2294        });
2295        if !fuel_unlimited {
2296            hook_data["fuel_remaining"] = json!(fuel_remaining);
2297            hook_data["fuel_total"] = json!(fuel_total);
2298        }
2299        let hook_results = tools::execute_hook(
2300            &plugin_tools,
2301            tools::HookPoint::PreAgenticLoop,
2302            &hook_data,
2303            tein_hook_ctx_ref,
2304        )?;
2305        apply_hook_overrides(
2306            &mut handoff,
2307            &mut fuel_remaining,
2308            fuel_unlimited,
2309            &hook_results,
2310            sink,
2311        )?;
2312
2313        // Loop detector resets per user-message turn (each outer loop iteration).
2314        let mut loop_detector = LoopDetector::new();
2315
2316        // === Inner Loop: stream responses and process tool calls ===
2317        loop {
2318            sink.handle(ResponseEvent::StartResponse)?;
2319            log_request_if_enabled(app, context_name, debug, &request_body);
2320
2321            let response = collect_streaming_response(
2322                &resolved_config,
2323                &chat_options,
2324                &messages,
2325                &all_tools,
2326                sink,
2327            )
2328            .await?;
2329
2330            // Log response metadata
2331            if let Some(ref meta) = response.response_meta {
2332                log_response_meta_if_enabled(app, context_name, debug, meta);
2333            }
2334
2335            // Signal streaming finished
2336            sink.handle(ResponseEvent::Finished)?;
2337
2338            // Handle tool calls
2339            if !response.tool_calls.is_empty() {
2340                process_tool_calls(
2341                    app,
2342                    context_name,
2343                    &resolved_config.username,
2344                    &response.tool_calls,
2345                    &mut messages,
2346                    &plugin_tools,
2347                    &registry,
2348                    &mut handoff,
2349                    use_reflection,
2350                    &resolved_config,
2351                    &mut fuel_remaining,
2352                    fuel_total,
2353                    fuel_unlimited,
2354                    sink,
2355                    permission_handler,
2356                    project_root,
2357                    &mut loop_detector,
2358                    tein_hook_ctx_ref,
2359                )
2360                .await?;
2361
2362                // Keep request_body in sync for logging
2363                request_body["messages"] = serde_json::json!(messages);
2364
2365                // If call_user was invoked, end the turn immediately — no follow-up API call.
2366                // The message and control_transfer were already written in process_tool_calls.
2367                if handoff.ends_turn_requested() {
2368                    return Ok(());
2369                }
2370
2371                // Tool call round costs 1 fuel
2372                if !fuel_unlimited {
2373                    fuel_remaining = fuel_remaining.saturating_sub(1);
2374                    sink.handle(ResponseEvent::FuelStatus {
2375                        remaining: fuel_remaining,
2376                        total: fuel_total,
2377                        event: crate::api::sink::FuelEvent::AfterToolBatch,
2378                    })?;
2379                    if fuel_remaining == 0 {
2380                        sink.handle(ResponseEvent::FuelExhausted { total: fuel_total })?;
2381                        return Ok(());
2382                    }
2383                }
2384                continue;
2385            }
2386
2387            // Check for empty response
2388            if response.full_response.trim().is_empty() {
2389                if !fuel_unlimited {
2390                    fuel_remaining =
2391                        fuel_remaining.saturating_sub(resolved_config.fuel_empty_response_cost);
2392                    if fuel_remaining == 0 {
2393                        sink.handle(ResponseEvent::FuelExhausted { total: fuel_total })?;
2394                        return Ok(());
2395                    }
2396                    sink.handle(ResponseEvent::FuelStatus {
2397                        remaining: fuel_remaining,
2398                        total: fuel_total,
2399                        event: crate::api::sink::FuelEvent::EmptyResponse,
2400                    })?;
2401                }
2402                continue;
2403            }
2404
2405            // Text response — break inner loop to handle final response
2406            match handle_final_response(
2407                app,
2408                context_name,
2409                &response.full_response,
2410                &final_prompt,
2411                handoff,
2412                &plugin_tools,
2413                &resolved_config,
2414                sink,
2415                tein_hook_ctx_ref,
2416            )? {
2417                FinalResponseAction::ReturnToUser => return Ok(()),
2418                FinalResponseAction::ContinueWithPrompt(continue_prompt) => {
2419                    if !fuel_unlimited {
2420                        fuel_remaining = fuel_remaining.saturating_sub(1);
2421                        if fuel_remaining == 0 {
2422                            sink.handle(ResponseEvent::FuelExhausted { total: fuel_total })?;
2423                            return Ok(());
2424                        }
2425                        let prompt_preview = if continue_prompt.len() > 80 {
2426                            let end = continue_prompt.floor_char_boundary(77);
2427                            format!("{}...", &continue_prompt[..end])
2428                        } else {
2429                            continue_prompt.clone()
2430                        };
2431                        sink.handle(ResponseEvent::FuelStatus {
2432                            remaining: fuel_remaining,
2433                            total: fuel_total,
2434                            event: crate::api::sink::FuelEvent::AfterContinuation {
2435                                prompt_preview,
2436                            },
2437                        })?;
2438                    }
2439                    // Prefix the continuation prompt; omit fuel numbers when unlimited
2440                    current_prompt = if fuel_unlimited {
2441                        format!(
2442                            "[reengaged via {}. call_user(<message>) to end turn.]\n{}",
2443                            resolved_config.fallback_tool, continue_prompt
2444                        )
2445                    } else {
2446                        format!(
2447                            "[reengaged (fuel: {}/{}) via {}. call_user(<message>) to end turn.]\n{}",
2448                            fuel_remaining,
2449                            fuel_total,
2450                            resolved_config.fallback_tool,
2451                            continue_prompt
2452                        )
2453                    };
2454                    break; // break inner, continue outer
2455                }
2456            }
2457        }
2458    }
2459}
