35/// Tool type classification for pre_api_tools hook
36#[derive(Debug, Clone, Copy, PartialEq, Eq)]
37enum ToolType {
38    Builtin,
39    File,
40    Agent,
41    Coding,
42    Plugin,
43}

45impl ToolType {
46    fn as_str(&self) -> &'static str {
47        match self {
48            ToolType::Builtin => "builtin",
49            ToolType::File => "file",
50            ToolType::Agent => "agent",
51            ToolType::Coding => "coding",
52            ToolType::Plugin => "plugin",
53        }
54    }

57/// Classify a tool's type based on its name.
58///
59/// Delegates to the authoritative `is_*_tool()` functions in each tool module,
60/// ensuring classification stays in sync with tool registration automatically.
61fn classify_tool_type(name: &str, plugin_names: &[&str]) -> ToolType {
62    if tools::is_builtin_tool(name) {
63        ToolType::Builtin
64    } else if tools::is_file_tool(name) {
65        ToolType::File
66    } else if tools::is_agent_tool(name) {
67        ToolType::Agent
68    } else if tools::is_coding_tool(name) {
69        ToolType::Coding
70    } else if plugin_names.contains(&name) {
71        ToolType::Plugin
72    } else {
73        // Unknown tools default to plugin type
74        ToolType::Plugin
75    }
76}

82/// Evaluate permission from pre-computed hook results.
83///
84/// Deny-only protocol: if any plugin returns `"denied": true`, the operation
85/// is blocked. Otherwise, falls through to the permission handler (if set)
86/// or fail-safe deny.
87///
88/// Separated from `check_permission()` for unit testing.
89fn evaluate_permission(
90    hook_results: &[(String, serde_json::Value)],
91    hook_data: &serde_json::Value,
92    permission_handler: Option<&PermissionHandler>,
93) -> io::Result<Result<(), String>> {
94    // Check for explicit denial from any plugin
95    for (_plugin_name, result) in hook_results {
96        if result.get_bool_or("denied", false) {
97            let reason = result.get_str_or("reason", "denied by plugin").to_string();
98            return Ok(Err(reason));
99        }
100    }
101
102    // No plugin denied — delegate to permission handler or fail-safe deny
103    match permission_handler {
104        Some(handler) => {
105            if handler(hook_data)? {
106                Ok(Ok(()))
107            } else {
108                Ok(Err("permission denied".to_string()))
109            }
110        }
111        None => Ok(Err(
112            "no permission handler configured (fail-safe deny)".to_string()
113        )),
114    }
115}

117/// Full permission check: fire the hook, then evaluate results.
118fn check_permission(
119    tools: &[Tool],
120    hook: tools::HookPoint,
121    hook_data: &serde_json::Value,
122    permission_handler: Option<&PermissionHandler>,
123) -> io::Result<Result<(), String>> {
124    let hook_results = tools::execute_hook(tools, hook, hook_data)?;
125    evaluate_permission(&hook_results, hook_data, permission_handler)
126}

128/// Build tool info list for pre_api_tools hook data
129fn build_tool_info_list(
130    all_tools: &[serde_json::Value],
131    plugin_tools: &[Tool],
132) -> Vec<serde_json::Value> {
133    let plugin_names: Vec<&str> = plugin_tools.iter().map(|t| t.name.as_str()).collect();
134
135    all_tools
136        .iter()
137        .filter_map(|tool| {
138            let name = tool
139                .get("function")
140                .and_then(|f| f.get("name"))
141                .and_then(|n| n.as_str())?;
142            let tool_type = classify_tool_type(name, &plugin_names);
143            Some(json!({
144                "name": name,
145                "type": tool_type.as_str(),
146            }))
147        })
148        .collect()
149}

151/// Annotate the fallback tool's description to indicate it's called automatically.
152fn annotate_fallback_tool(tools: &mut [serde_json::Value], fallback_name: &str) {
153    const FALLBACK_SUFFIX: &str = " Called automatically if no other tool is used.";
154
155    for tool in tools.iter_mut() {
156        let name_matches = tool
157            .get("function")
158            .and_then(|f| f.get("name"))
159            .and_then(|n| n.as_str())
160            .is_some_and(|name| name == fallback_name);
161
162        if name_matches {
163            if let Some(func) = tool.get_mut("function")
164                && let Some(desc) = func.get_mut("description")
165                && let Some(desc_str) = desc.as_str()
166            {
167                *desc = serde_json::Value::String(format!("{}{}", desc_str, FALLBACK_SUFFIX));
168            }
169            break; // Only annotate the first match
170        }
171    }
172}

174/// Filter tools based on config include/exclude/exclude_categories lists
175fn filter_tools_by_config(
176    tools: Vec<serde_json::Value>,
177    config: &ToolsConfig,
178) -> Vec<serde_json::Value> {
179    let mut result = tools;
180
181    // Apply include filter first (if set, only these tools are considered)
182    if let Some(ref include) = config.include {
183        result.retain(|tool| {
184            tool.get("function")
185                .and_then(|f| f.get("name"))
186                .and_then(|n| n.as_str())
187                .is_some_and(|name| include.contains(&name.to_string()))
188        });
189    }
190
191    // Apply exclude filter (remove these tools from remaining)
192    if let Some(ref exclude) = config.exclude {
193        result.retain(|tool| {
194            tool.get("function")
195                .and_then(|f| f.get("name"))
196                .and_then(|n| n.as_str())
197                .is_some_and(|name| !exclude.contains(&name.to_string()))
198        });
199    }
200
201    // Apply category exclusion (remove tools whose category is excluded)
202    if let Some(ref categories) = config.exclude_categories {
203        result.retain(|tool| {
204            tool.get("function")
205                .and_then(|f| f.get("name"))
206                .and_then(|n| n.as_str())
207                .map(|name| {
208                    let tool_type = classify_tool_type(name, &[]);
209                    !categories.contains(&tool_type.as_str().to_string())
210                })
211                .unwrap_or(true)
212        });
213    }
214
215    result
216}

218/// Filter tools based on hook results
219/// Multiple hooks: includes are intersected, excludes are unioned
220fn filter_tools_from_hook_results<S: ResponseSink>(
221    tools: Vec<serde_json::Value>,
222    hook_results: &[(String, serde_json::Value)],
223    verbose: bool,
224    sink: &mut S,
225) -> io::Result<Vec<serde_json::Value>> {
226    if hook_results.is_empty() {
227        return Ok(tools);
228    }
229
230    let mut result = tools;
231
232    // Collect all includes and excludes from hook results
233    let mut all_includes: Option<Vec<String>> = None;
234    let mut all_excludes: Vec<String> = Vec::new();
235
236    for (hook_name, hook_result) in hook_results {
237        // Handle include lists (intersection)
238        if let Some(include) = hook_result.get_array("include") {
239            let include_names: Vec<String> = include
240                .iter()
241                .filter_map(|v| v.as_str().map(|s| s.to_string()))
242                .collect();
243
244            if verbose {
245                sink.handle(ResponseEvent::Diagnostic {
246                    message: format!(
247                        "[Hook pre_api_tools: {} include filter: {:?}]",
248                        hook_name, include_names
249                    ),
250                    verbose_only: true,
251                })?;
252            }
253
254            all_includes = Some(match all_includes {
255                Some(existing) => existing
256                    .into_iter()
257                    .filter(|name| include_names.contains(name))
258                    .collect(),
259                None => include_names,
260            });
261        }
262
263        // Handle exclude lists (union)
264        if let Some(exclude) = hook_result.get_array("exclude") {
265            let exclude_names: Vec<String> = exclude
266                .iter()
267                .filter_map(|v| v.as_str().map(|s| s.to_string()))
268                .collect();
269
270            if verbose {
271                sink.handle(ResponseEvent::Diagnostic {
272                    message: format!(
273                        "[Hook pre_api_tools: {} exclude filter: {:?}]",
274                        hook_name, exclude_names
275                    ),
276                    verbose_only: true,
277                })?;
278            }
279
280            for name in exclude_names {
281                if !all_excludes.contains(&name) {
282                    all_excludes.push(name);
283                }
284            }
285        }
286    }
287
288    // Apply collected includes (intersection)
289    if let Some(includes) = all_includes {
290        result.retain(|tool| {
291            tool.get("function")
292                .and_then(|f| f.get("name"))
293                .and_then(|n| n.as_str())
294                .is_some_and(|name| includes.contains(&name.to_string()))
295        });
296    }
297
298    // Apply collected excludes (union)
299    if !all_excludes.is_empty() {
300        result.retain(|tool| {
301            tool.get("function")
302                .and_then(|f| f.get("name"))
303                .and_then(|n| n.as_str())
304                .is_some_and(|name| !all_excludes.contains(&name.to_string()))
305        });
306    }
307
308    Ok(result)
309}

311/// Apply request modifications from pre_api_request hook results
312/// Hook returns are merged (not replaced) into the request body
313fn apply_request_modifications<S: ResponseSink>(
314    mut request_body: serde_json::Value,
315    hook_results: &[(String, serde_json::Value)],
316    verbose: bool,
317    sink: &mut S,
318) -> io::Result<serde_json::Value> {
319    for (hook_name, hook_result) in hook_results {
320        if let Some(modifications) = hook_result.get("request_body")
321            && let Some(mods_obj) = modifications.as_object()
322        {
323            if verbose {
324                sink.handle(ResponseEvent::Diagnostic {
325                    message: format!(
326                        "[Hook pre_api_request: {} modifying request (keys: {:?})]",
327                        hook_name,
328                        mods_obj.keys().collect::<Vec<_>>()
329                    ),
330                    verbose_only: true,
331                })?;
332            }
333
334            // Merge modifications into request body
335            if let Some(body_obj) = request_body.as_object_mut() {
336                for (key, value) in mods_obj {
337                    body_obj.insert(key.clone(), value.clone());
338                }
339            }
340        }
341    }
342
343    Ok(request_body)
344}

346/// Apply hook overrides from hook results.
347///
348/// Handles three keys from hook return values:
349/// - `"fallback"`: override the fallback tool (existing behaviour)
350/// - `"fuel"`: absolute fuel override (e.g. from `pre_agentic_loop`)
351/// - `"fuel_delta"`: relative fuel adjustment (e.g. from `post_tool_batch`)
352fn apply_hook_overrides<S: ResponseSink>(
353    handoff: &mut tools::Handoff,
354    fuel_remaining: &mut usize,
355    hook_results: &[(String, serde_json::Value)],
356    verbose: bool,
357    sink: &mut S,
358) -> io::Result<()> {
359    for (hook_name, hook_result) in hook_results {
360        if let Some(fallback_str) = hook_result.get_str("fallback") {
361            // Use builtin metadata since hooks can only override to builtins
362            let meta = tools::builtin_tool_metadata(fallback_str);
363            let new_fallback = if meta.ends_turn {
364                tools::HandoffTarget::User {
365                    message: String::new(),
366                }
367            } else {
368                tools::HandoffTarget::Agent {
369                    prompt: String::new(),
370                }
371            };
372            handoff.set_fallback(new_fallback);
373            if verbose {
374                sink.handle(ResponseEvent::Diagnostic {
375                    message: format!("[Hook {} set fallback to {}]", hook_name, fallback_str),
376                    verbose_only: true,
377                })?;
378            }
379        }
380        if let Some(fuel) = hook_result.get("fuel").and_then(|v| v.as_u64()) {
381            *fuel_remaining = fuel as usize;
382            if verbose {
383                sink.handle(ResponseEvent::Diagnostic {
384                    message: format!("[Hook {} set fuel to {}]", hook_name, fuel),
385                    verbose_only: true,
386                })?;
387            }
388        }
389        if let Some(delta) = hook_result.get("fuel_delta").and_then(|v| v.as_i64()) {
390            if delta < 0 {
391                *fuel_remaining = fuel_remaining.saturating_sub((-delta) as usize);
392            } else {
393                *fuel_remaining = fuel_remaining.saturating_add(delta as usize);
394            }
395            if verbose {
396                sink.handle(ResponseEvent::Diagnostic {
397                    message: format!("[Hook {} adjusted fuel by {}]", hook_name, delta),
398                    verbose_only: true,
399                })?;
400            }
401        }
402    }
403    Ok(())
404}

410/// Build the full system prompt with all components.
411///
412/// Handles: loading base prompt, todos, goals, summary, reflection prompt,
413/// pre/post system_prompt hooks, and username injection.
414#[allow(clippy::too_many_arguments)]
415fn build_full_system_prompt<S: ResponseSink>(
416    app: &AppState,
417    context_name: &str,
418    summary: &str,
419    use_reflection: bool,
420    tools: &[Tool],
421    resolved_config: &ResolvedConfig,
422    verbose: bool,
423    sink: &mut S,
424    home_dir: &Path,
425    project_root: &Path,
426) -> io::Result<String> {
427    // Load base prompts
428    let system_prompt = app.load_system_prompt_for(context_name)?;
429    let reflection_prompt = if use_reflection {
430        app.load_reflection_prompt()?
431    } else {
432        String::new()
433    };
434
435    // Load context-specific state
436    let todos = app.load_todos(context_name)?;
437    let goals = app.load_goals(context_name)?;
438
439    // Execute pre_system_prompt hook - can inject content before system prompt sections
440    let pre_sys_hook_data = serde_json::json!({
441        "context_name": context_name,
442        "summary": summary,
443        "todos": todos,
444        "goals": goals,
445    });
446    let pre_sys_hook_results =
447        tools::execute_hook(tools, tools::HookPoint::PreSystemPrompt, &pre_sys_hook_data)?;
448
449    // Build full system prompt with all components
450    let mut full_system_prompt = system_prompt;
451
452    // Prepend any content from pre_system_prompt hooks
453    for (hook_tool_name, result) in &pre_sys_hook_results {
454        if let Some(inject) = result.get_str("inject")
455            && !inject.is_empty()
456        {
457            if verbose {
458                sink.handle(ResponseEvent::Diagnostic {
459                    message: format!(
460                        "[Hook pre_system_prompt: {} injected content]",
461                        hook_tool_name
462                    ),
463                    verbose_only: true,
464                })?;
465            }
466            full_system_prompt = format!("{}\n\n{}", inject, full_system_prompt);
467        }
468    }
469
470    // Load AGENTS.md instructions from standard locations
471    let agents_md = crate::agents_md::load_agents_md(
472        home_dir,
473        &app.chibi_dir,
474        project_root,
475        &std::env::current_dir().unwrap_or_default(),
476    );
477    if !agents_md.is_empty() {
478        full_system_prompt.push_str("\n\n--- AGENT INSTRUCTIONS ---\n");
479        full_system_prompt.push_str(&agents_md);
480    }
481
482    // Add username info at the start if not "user"
483    if resolved_config.username != "user" {
484        full_system_prompt.push_str(&format!(
485            "\n\nThe user speaking to you is called: {}",
486            resolved_config.username
487        ));
488    }
489
490    // Add context name
491    full_system_prompt.push_str(&format!("\n\nCurrent context: {}", context_name));
492
493    // Add summary if present
494    if !summary.is_empty() {
495        full_system_prompt.push_str("\n\n--- CONVERSATION SUMMARY ---\n");
496        full_system_prompt.push_str(summary);
497    }
498
499    // Add goals if present
500    if !goals.is_empty() {
501        full_system_prompt.push_str("\n\n--- CURRENT GOALS ---\n");
502        full_system_prompt.push_str(&goals);
503    }
504
505    // Add todos if present
506    if !todos.is_empty() {
507        full_system_prompt.push_str("\n\n--- CURRENT TODOS ---\n");
508        full_system_prompt.push_str(&todos);
509    }
510
511    // Add reflection prompt last (personality layer)
512    if !reflection_prompt.is_empty() {
513        full_system_prompt.push_str("\n\n");
514        full_system_prompt.push_str(&reflection_prompt);
515    }
516
517    // Execute post_system_prompt hook - can inject content after all system prompt sections
518    let post_sys_hook_data = serde_json::json!({
519        "context_name": context_name,
520        "summary": summary,
521        "todos": todos,
522        "goals": goals,
523    });
524    let post_sys_hook_results = tools::execute_hook(
525        tools,
526        tools::HookPoint::PostSystemPrompt,
527        &post_sys_hook_data,
528    )?;
529
530    // Append any content from post_system_prompt hooks
531    for (hook_tool_name, result) in &post_sys_hook_results {
532        if let Some(inject) = result.get_str("inject")
533            && !inject.is_empty()
534        {
535            if verbose {
536                sink.handle(ResponseEvent::Diagnostic {
537                    message: format!(
538                        "[Hook post_system_prompt: {} injected content]",
539                        hook_tool_name
540                    ),
541                    verbose_only: true,
542                })?;
543            }
544            full_system_prompt.push_str("\n\n");
545            full_system_prompt.push_str(inject);
546        }
547    }
548
549    // Store combined prompt in context_meta for API request reconstruction
550    if !full_system_prompt.is_empty() {
551        app.save_combined_system_prompt(context_name, &full_system_prompt)?;
552    }
553
554    Ok(full_system_prompt)
555}

557/// Result of collecting a streaming response from the LLM.
558struct StreamingResponse {
559    /// The full text response accumulated from chunks.
560    full_response: String,
561    /// Tool calls extracted from the response.
562    tool_calls: Vec<ratatoskr::ToolCall>,
563    /// Whether any tool calls were present.
564    has_tool_calls: bool,
565    /// Response metadata (usage stats, model info).
566    response_meta: Option<serde_json::Value>,
567}
568
569/// Collect a streaming response from the LLM API via ratatoskr.
570///
571/// Handles: SSE parsing, content accumulation, tool call reconstruction.
572/// Pure I/O - no hooks or side effects beyond streaming to sink.
573async fn collect_streaming_response<S: ResponseSink>(
574    resolved_config: &ResolvedConfig,
575    messages: &[serde_json::Value],
576    all_tools: &[serde_json::Value],
577    verbose: bool,
578    sink: &mut S,
579) -> io::Result<StreamingResponse> {
580    let gateway = build_gateway(resolved_config)?;
581
582    // Convert messages
583    let ratatoskr_messages: Vec<_> = messages
584        .iter()
585        .map(to_ratatoskr_message)
586        .collect::<io::Result<Vec<_>>>()?;
587
588    // Convert tools
589    let tool_defs: Vec<_> = all_tools
590        .iter()
591        .filter_map(|t| json_tool_to_definition(t).ok())
592        .collect();
593    let tools_opt = if tool_defs.is_empty() {
594        None
595    } else {
596        Some(tool_defs.as_slice())
597    };
598
599    // Build options
600    let options = to_chat_options(resolved_config);
601
602    // Get streaming response
603    let mut stream = gateway
604        .chat_stream(&ratatoskr_messages, tools_opt, &options)
605        .await
606        .map_err(|e| io::Error::other(format!("Gateway error: {}", e)))?;
607
608    let mut full_response = String::new();
609    let mut tool_calls: Vec<ratatoskr::ToolCall> = Vec::new();
610    let mut has_tool_calls = false;
611    let mut response_meta: Option<serde_json::Value> = None;
612    let mut is_first_content = true;
613    let json_mode = sink.is_json_mode();
614
615    while let Some(event_result) = stream.next().await {
616        let event = event_result.map_err(|e| io::Error::other(format!("Stream error: {}", e)))?;
617
618        match event {
619            ChatEvent::Content(chunk) => {
620                // Handle first chunk newline stripping (matches old behavior)
621                let text = if is_first_content {
622                    is_first_content = false;
623                    if let Some(remaining) = chunk.strip_prefix('\n') {
624                        if remaining.is_empty() {
625                            continue;
626                        }
627                        remaining.to_string()
628                    } else {
629                        chunk
630                    }
631                } else {
632                    chunk
633                };
634
635                full_response.push_str(&text);
636                if !json_mode {
637                    sink.handle(ResponseEvent::TextChunk(&text))?;
638                }
639            }
640            ChatEvent::Reasoning(chunk) => {
641                // Reasoning is ephemeral thinking — always forward to sink so
642                // both streaming and JSON consumers can access it
643                sink.handle(ResponseEvent::Reasoning(&chunk))?;
644            }
645            ChatEvent::ToolCallStart { index, id, name } => {
646                has_tool_calls = true;
647
648                // Prevent memory exhaustion
649                if index >= MAX_TOOL_CALLS {
650                    if verbose {
651                        eprintln!(
652                            "[WARN] Tool call index {} exceeds limit {}, skipping",
653                            index, MAX_TOOL_CALLS
654                        );
655                    }
656                    continue;
657                }
658
659                while tool_calls.len() <= index {
660                    tool_calls.push(ratatoskr::ToolCall::default());
661                }
662
663                tool_calls[index].id = id;
664                tool_calls[index].name = name;
665            }
666            ChatEvent::ToolCallDelta { index, arguments } => {
667                if index < tool_calls.len() {
668                    tool_calls[index].arguments.push_str(&arguments);
669                }
670            }
671            ChatEvent::Usage(usage) => {
672                response_meta = Some(json!({
673                    "usage": {
674                        "prompt_tokens": usage.prompt_tokens,
675                        "completion_tokens": usage.completion_tokens,
676                        "total_tokens": usage.total_tokens
677                    }
678                }));
679            }
680            ChatEvent::ToolCallEnd { .. } => {
681                // Tool call argument streaming complete; nothing to do
682            }
683            ChatEvent::Done => break,
684            _ => {} // forward-compatible with future ChatEvent variants
685        }
686    }
687
688    Ok(StreamingResponse {
689        full_response,
690        tool_calls,
691        has_tool_calls,
692        response_meta,
693    })
694}

702    /// Whether the result was cached.
703    was_cached: bool,
704    /// Verbose diagnostic messages collected during execution.
705    diagnostics: Vec<String>,
706}

708/// Sink-free execution core for a single tool call.
709///
710/// Performs all tool execution logic (hooks, dispatch, caching, output hooks)
711/// without touching the sink or handoff state. Collects verbose diagnostics
712/// into `ToolExecutionResult::diagnostics` for the caller to emit.
713///
714/// This enables concurrent execution via `join_all` since it doesn't require
715/// `&mut` access to shared state.
716#[allow(clippy::too_many_arguments)]
717async fn execute_tool_pure(
718    app: &AppState,
719    context_name: &str,
720    tool_call: &ratatoskr::ToolCall,
721    tools: &[Tool],
722    use_reflection: bool,
723    resolved_config: &ResolvedConfig,
724    verbose: bool,
725    permission_handler: Option<&PermissionHandler>,
726    project_root: &Path,
727) -> io::Result<ToolExecutionResult> {
728    let mut args: serde_json::Value =
729        serde_json::from_str(&tool_call.arguments).unwrap_or(serde_json::json!({}));
730    let mut diagnostics = Vec::new();
731
732    // Check for control flow tools using metadata — we compute the result message
733    // but leave handoff mutation to the caller
734    let tool_metadata = tools::get_tool_metadata(tools, &tool_call.name);
735
736    // Execute pre_tool hooks (can modify arguments OR block execution)
737    let pre_hook_data = serde_json::json!({
738        "tool_name": tool_call.name,
739        "arguments": args,
740    });
741    let pre_hook_results = tools::execute_hook(tools, tools::HookPoint::PreTool, &pre_hook_data)?;
742
743    let mut blocked = false;
744    let mut block_message = String::new();
745
746    for (hook_tool_name, result) in pre_hook_results {
747        // Check for block signal first
748        if result.get_bool_or("block", false) {
749            blocked = true;
750            block_message = result
751                .get_str_or("message", "Tool call blocked by hook")
752                .to_string();
753            if verbose {
754                diagnostics.push(format!(
755                    "[Hook pre_tool: {} blocked {} - {}]",
756                    hook_tool_name, tool_call.name, block_message
757                ));
758            }
759            break;
760        }
761
762        // Check for argument modification
763        if let Some(modified_args) = result.get("arguments") {
764            if verbose {
765                diagnostics.push(format!(
766                    "[Hook pre_tool: {} modified arguments for {}]",
767                    hook_tool_name, tool_call.name
768                ));
769            }
770            args = modified_args.clone();
771        }
772    }
773
774    // If blocked, skip execution and use block message as result
775    let tool_result = if blocked {
776        block_message
777    } else if tool_metadata.flow_control {
778        // Handoff tools don't execute — they just produce a result message.
779        // The caller applies the actual handoff mutation.
780        if tool_metadata.ends_turn {
781            let message = args.get_str_or("message", "");
782            if message.is_empty() {
783                "Returning to user".to_string()
784            } else {
785                message.to_string()
786            }
787        } else {
788            let prompt = args.get_str_or("prompt", "");
789            if prompt.is_empty() {
790                "Continuing processing".to_string()
791            } else {
792                format!("Continuing with: {}", prompt)
793            }
794        }
795    } else if tool_call.name == tools::REFLECTION_TOOL_NAME && !use_reflection {
796        "Error: Reflection tool is not enabled".to_string()
797    } else if let Some(builtin_result) = tools::execute_builtin_tool(
798        app,
799        context_name,
800        &tool_call.name,
801        &args,
802        Some(resolved_config),
803    ) {
804        match builtin_result {
805            Ok(r) => r,
806            Err(e) => format!("Error: {}", e),
807        }
808    } else if tool_call.name == tools::SEND_MESSAGE_TOOL_NAME {
809        execute_send_message_pure(app, context_name, tools, &args, verbose, &mut diagnostics)?
810    } else if tool_call.name == tools::MODEL_INFO_TOOL_NAME {
811        match args.get_str("model") {
812            Some(model) => {
813                let gateway = build_gateway(resolved_config)?;
814                match crate::model_info::fetch_metadata(&gateway, model).await {
815                    Ok(metadata) => {
816                        let json = crate::model_info::format_model_json(&metadata);
817                        serde_json::to_string_pretty(&json)
818                            .unwrap_or_else(|e| format!("Error serialising metadata: {}", e))
819                    }
820                    Err(e) => format!("Error: {}", e),
821                }
822            }
823            None => "Error: missing required 'model' parameter".to_string(),
824        }
825    } else if tools::is_file_tool(&tool_call.name) {
826        // Write tools need permission via pre_file_write hook
827        if tool_call.name == tools::WRITE_FILE_TOOL_NAME {
828            let hook_data = serde_json::json!({
829                "tool_name": tool_call.name,
830                "path": args.get_str("path").unwrap_or(""),
831                "content": args.get_str("content"),
832            });
833            match check_permission(
834                tools,
835                tools::HookPoint::PreFileWrite,
836                &hook_data,
837                permission_handler,
838            )? {
839                Ok(()) => match tools::execute_file_tool(
840                    app,
841                    context_name,
842                    &tool_call.name,
843                    &args,
844                    resolved_config,
845                ) {
846                    Some(Ok(r)) => r,
847                    Some(Err(e)) => format!("Error: {}", e),
848                    None => format!("Error: Unknown file tool '{}'", tool_call.name),
849                },
850                Err(reason) => format!("Error: {}", reason),
851            }
852        } else {
853            // Read-only file tools: auto-allow inside allowed paths, prompt outside
854            let path = args.get_str("path").unwrap_or("");
855            let permission_denied = if !path.is_empty() {
856                match tools::classify_file_path(path, resolved_config) {
857                    Ok(tools::FilePathAccess::Allowed(_)) => None,
858                    Ok(tools::FilePathAccess::NeedsPermission(_)) => {
859                        let hook_data = serde_json::json!({
860                            "tool_name": tool_call.name,
861                            "path": path,
862                        });
863                        check_permission(
864                            tools,
865                            tools::HookPoint::PreFileRead,
866                            &hook_data,
867                            permission_handler,
868                        )?
869                        .err()
870                    }
871                    Err(e) => Some(e.to_string()),
872                }
873            } else {
874                // cache_id access (no path) — always allowed
875                None
876            };
877
878            if let Some(reason) = permission_denied {
879                format!("Error: {}", reason)
880            } else {
881                match tools::execute_file_tool(
882                    app,
883                    context_name,
884                    &tool_call.name,
885                    &args,
886                    resolved_config,
887                ) {
888                    Some(Ok(r)) => r,
889                    Some(Err(e)) => format!("Error: {}", e),
890                    None => format!("Error: Unknown file tool '{}'", tool_call.name),
891                }
892            }
893        }
894    } else if tools::is_agent_tool(&tool_call.name) {
895        // URL policy / permission check for summarize_content
896        if tool_call.name == tools::SUMMARIZE_CONTENT_TOOL_NAME
897            && let Some(source) = args.get_str("source")
898            && tools::agent_tools::is_url(source)
899        {
900            let safety = tools::classify_url(source);
901
902            if let Some(ref policy) = resolved_config.url_policy {
903                // policy is authoritative — no fallback to permission handler
904                if tools::evaluate_url_policy(source, &safety, policy) == tools::UrlAction::Deny {
905                    let reason = match &safety {
906                        tools::UrlSafety::Sensitive(cat) => cat.to_string(),
907                        tools::UrlSafety::Safe => "denied by URL policy".to_string(),
908                    };
909                    let msg = format!("Permission denied: {}", reason);
910                    return Ok(ToolExecutionResult {
911                        final_result: msg.clone(),
912                        original_result: msg,
913                        was_cached: false,
914                        diagnostics,
915                    });
916                }
917            } else if let tools::UrlSafety::Sensitive(category) = &safety {
918                // no policy — existing behaviour: check permission handler
919                let hook_data = json!({
920                    "tool_name": tool_call.name,
921                    "url": source,
922                    "safety": "sensitive",
923                    "reason": category.to_string(),
924                });
925                match check_permission(
926                    tools,
927                    tools::HookPoint::PreFetchUrl,
928                    &hook_data,
929                    permission_handler,
930                )? {
931                    Ok(()) => {}
932                    Err(reason) => {
933                        let msg = format!("Permission denied: {}", reason);
934                        return Ok(ToolExecutionResult {
935                            final_result: msg.clone(),
936                            original_result: msg,
937                            was_cached: false,
938                            diagnostics,
939                        });
940                    }
941                }
942            }
943        }
944        match tools::execute_agent_tool(resolved_config, &tool_call.name, &args, tools).await {
945            Ok(r) => r,
946            Err(e) => format!("Error: {}", e),
947        }
948    } else if tools::is_coding_tool(&tool_call.name) {
949        // Gated coding tools need permission checks
950        if tool_call.name == tools::SHELL_EXEC_TOOL_NAME {
951            let hook_data = serde_json::json!({
952                "tool_name": tool_call.name,
953                "command": args.get_str("command").unwrap_or(""),
954            });
955            match check_permission(
956                tools,
957                tools::HookPoint::PreShellExec,
958                &hook_data,
959                permission_handler,
960            )? {
961                Ok(()) => {
962                    match tools::execute_coding_tool(&tool_call.name, &args, project_root, tools)
963                        .await
964                    {
965                        Some(Ok(r)) => r,
966                        Some(Err(e)) => format!("Error: {}", e),
967                        None => format!("Error: Unknown coding tool '{}'", tool_call.name),
968                    }
969                }
970                Err(reason) => format!("Error: {}", reason),
971            }
972        } else if tool_call.name == tools::FILE_EDIT_TOOL_NAME {
973            let hook_data = serde_json::json!({
974                "tool_name": tool_call.name,
975                "path": args.get_str("path").unwrap_or(""),
976                "operation": args.get_str("operation").unwrap_or(""),
977                "content": args.get_str("content"),
978            });
979            match check_permission(
980                tools,
981                tools::HookPoint::PreFileWrite,
982                &hook_data,
983                permission_handler,
984            )? {
985                Ok(()) => {
986                    match tools::execute_coding_tool(&tool_call.name, &args, project_root, tools)
987                        .await
988                    {
989                        Some(Ok(r)) => r,
990                        Some(Err(e)) => format!("Error: {}", e),
991                        None => format!("Error: Unknown coding tool '{}'", tool_call.name),
992                    }
993                }
994                Err(reason) => format!("Error: {}", reason),
995            }
996        } else if tool_call.name == tools::FETCH_URL_TOOL_NAME {
997            // URL policy / permission check for fetch_url (mirrors summarize_content gating)
998            let url = args.get_str("url").unwrap_or("");
999            let safety = tools::classify_url(url);
1000
1001            let denied = if let Some(ref policy) = resolved_config.url_policy {
1002                tools::evaluate_url_policy(url, &safety, policy) == tools::UrlAction::Deny
1003            } else {
1004                false
1005            };
1006
1007            if denied {
1008                let reason = match &safety {
1009                    tools::UrlSafety::Sensitive(cat) => cat.to_string(),
1010                    tools::UrlSafety::Safe => "denied by URL policy".to_string(),
1011                };
1012                format!("Permission denied: {}", reason)
1013            } else if let tools::UrlSafety::Sensitive(category) = &safety {
1014                // No policy — fall back to permission handler for sensitive URLs
1015                let hook_data = json!({
1016                    "tool_name": tool_call.name,
1017                    "url": url,
1018                    "safety": "sensitive",
1019                    "reason": category.to_string(),
1020                });
1021                match check_permission(
1022                    tools,
1023                    tools::HookPoint::PreFetchUrl,
1024                    &hook_data,
1025                    permission_handler,
1026                )? {
1027                    Ok(()) => {
1028                        match tools::execute_coding_tool(
1029                            &tool_call.name,
1030                            &args,
1031                            project_root,
1032                            tools,
1033                        )
1034                        .await
1035                        {
1036                            Some(Ok(r)) => r,
1037                            Some(Err(e)) => format!("Error: {}", e),
1038                            None => format!("Error: Unknown coding tool '{}'", tool_call.name),
1039                        }
1040                    }
1041                    Err(reason) => format!("Permission denied: {}", reason),
1042                }
1043            } else {
1044                match tools::execute_coding_tool(&tool_call.name, &args, project_root, tools).await
1045                {
1046                    Some(Ok(r)) => r,
1047                    Some(Err(e)) => format!("Error: {}", e),
1048                    None => format!("Error: Unknown coding tool '{}'", tool_call.name),
1049                }
1050            }
1051        } else {
1052            // Ungated coding tools: dir_list, glob_files, grep_files,
1053            // index_update, index_query, index_status
1054            match tools::execute_coding_tool(&tool_call.name, &args, project_root, tools).await {
1055                Some(Ok(r)) => r,
1056                Some(Err(e)) => format!("Error: {}", e),
1057                None => format!("Error: Unknown coding tool '{}'", tool_call.name),
1058            }
1059        }
1060    } else if let Some(tool) = tools::find_tool(tools, &tool_call.name) {
1061        match tools::execute_tool(tool, &args, verbose) {
1062            Ok(r) => r,
1063            Err(e) => format!("Error: {}", e),
1064        }
1065    } else {
1066        format!("Error: Unknown tool '{}'", tool_call.name)
1067    };
1068
1069    // Execute pre_tool_output hooks (can modify or replace output)
1070    let mut tool_result = tool_result;
1071    let pre_output_hook_data = serde_json::json!({
1072        "tool_name": tool_call.name,
1073        "arguments": args,
1074        "output": tool_result,
1075    });
1076    let pre_output_hook_results = tools::execute_hook(
1077        tools,
1078        tools::HookPoint::PreToolOutput,
1079        &pre_output_hook_data,
1080    )?;
1081
1082    for (hook_tool_name, result) in pre_output_hook_results {
1083        if result.get_bool_or("block", false) {
1084            let replacement = result
1085                .get_str_or("message", "Output blocked by hook")
1086                .to_string();
1087            if verbose {
1088                diagnostics.push(format!(
1089                    "[Hook pre_tool_output: {} blocked output from {}]",
1090                    hook_tool_name, tool_call.name
1091                ));
1092            }
1093            tool_result = replacement;
1094            break;
1095        }
1096
1097        if let Some(modified_output) = result.get_str("output") {
1098            if verbose {
1099                diagnostics.push(format!(
1100                    "[Hook pre_tool_output: {} modified output from {}]",
1101                    hook_tool_name, tool_call.name
1102                ));
1103            }
1104            tool_result = modified_output.to_string();
1105        }
1106    }
1107
1108    // Check if output should be cached
1109    let (final_result, was_cached) = if !tool_result.starts_with("Error:")
1110        && cache::should_cache(&tool_result, resolved_config.tool_output_cache_threshold)
1111    {
1112        // Fire pre_cache_output hook (can block caching)
1113        let pre_cache_data = serde_json::json!({
1114            "tool_name": tool_call.name,
1115            "output_size": tool_result.len(),
1116            "arguments": args,
1117        });
1118        let pre_cache_results =
1119            tools::execute_hook(tools, tools::HookPoint::PreCacheOutput, &pre_cache_data)?;
1120        let cache_blocked = pre_cache_results
1121            .iter()
1122            .any(|(_, r)| r.get_bool_or("block", false));
1123
1124        if cache_blocked {
1125            if verbose {
1126                diagnostics.push(format!(
1127                    "[Caching blocked by pre_cache_output hook for {}]",
1128                    tool_call.name
1129                ));
1130            }
1131            (tool_result.clone(), false)
1132        } else {
1133            let cache_dir = app.tool_cache_dir(context_name);
1134            match cache::cache_output(&cache_dir, &tool_call.name, &tool_result, &args) {
1135                Ok(entry) => {
1136                    match cache::generate_truncated_message(
1137                        &entry,
1138                        resolved_config.tool_cache_preview_chars,
1139                    ) {
1140                        Ok(truncated) => {
1141                            if verbose {
1142                                diagnostics.push(format!(
1143                                    "[Cached {} chars from {} as {}]",
1144                                    tool_result.len(),
1145                                    tool_call.name,
1146                                    entry.metadata.id
1147                                ));
1148                            }
1149
1150                            // Fire post_cache_output hook (notification only)
1151                            let post_cache_data = serde_json::json!({
1152                                "tool_name": tool_call.name,
1153                                "cache_id": entry.metadata.id,
1154                                "output_size": tool_result.len(),
1155                                "preview_size": truncated.len(),
1156                            });
1157                            let _ = tools::execute_hook(
1158                                tools,
1159                                tools::HookPoint::PostCacheOutput,
1160                                &post_cache_data,
1161                            );
1162
1163                            (truncated, true)
1164                        }
1165                        Err(_) => (tool_result.clone(), false),
1166                    }
1167                }
1168                Err(e) => {
1169                    if verbose {
1170                        diagnostics.push(format!("[Failed to cache output: {}]", e));
1171                    }
1172                    (tool_result.clone(), false)
1173                }
1174            }
1175        }
1176    } else {
1177        (tool_result.clone(), false)
1178    };
1179
1180    // Execute post_tool_output hooks
1181    let post_output_hook_data = serde_json::json!({
1182        "tool_name": tool_call.name,
1183        "arguments": args,
1184        "output": tool_result,
1185        "final_output": final_result,
1186        "cached": was_cached,
1187    });
1188    let _ = tools::execute_hook(
1189        tools,
1190        tools::HookPoint::PostToolOutput,
1191        &post_output_hook_data,
1192    );
1193
1194    Ok(ToolExecutionResult {
1195        final_result,
1196        original_result: tool_result,
1197        was_cached,
1198        diagnostics,
1199    })
1200}

1202/// Execute a single tool call with all hooks (sequential path).
1203///
1204/// Thin wrapper over `execute_tool_pure` that also handles handoff mutation
1205/// and emits diagnostics to the sink.
1206#[allow(clippy::too_many_arguments)]
1207async fn execute_single_tool<S: ResponseSink>(
1208    app: &AppState,
1209    context_name: &str,
1210    tool_call: &ratatoskr::ToolCall,
1211    tools: &[Tool],
1212    handoff: &mut tools::Handoff,
1213    use_reflection: bool,
1214    resolved_config: &ResolvedConfig,
1215    permission_handler: Option<&PermissionHandler>,
1216    verbose: bool,
1217    sink: &mut S,
1218    project_root: &Path,
1219) -> io::Result<ToolExecutionResult> {
1220    // Apply handoff if this is a flow control tool
1221    let args: serde_json::Value =
1222        serde_json::from_str(&tool_call.arguments).unwrap_or(serde_json::json!({}));
1223    let tool_metadata = tools::get_tool_metadata(tools, &tool_call.name);
1224    if tool_metadata.flow_control {
1225        if tool_metadata.ends_turn {
1226            handoff.set_user(args.get_str_or("message", "").to_string());
1227        } else {
1228            handoff.set_agent(args.get_str_or("prompt", "").to_string());
1229        }
1230    }
1231
1232    let result = execute_tool_pure(
1233        app,
1234        context_name,
1235        tool_call,
1236        tools,
1237        use_reflection,
1238        resolved_config,
1239        verbose,
1240        permission_handler,
1241        project_root,
1242    )
1243    .await?;
1244
1245    // Emit collected diagnostics to sink
1246    for diag in &result.diagnostics {
1247        sink.handle(ResponseEvent::Diagnostic {
1248            message: diag.clone(),
1249            verbose_only: true,
1250        })?;
1251    }
1252
1253    Ok(result)
1254}

1256/// Sink-free send_message execution. Collects diagnostics into the provided vec.
1257#[allow(clippy::too_many_arguments)]
1258fn execute_send_message_pure(
1259    app: &AppState,
1260    context_name: &str,
1261    tools: &[Tool],
1262    args: &serde_json::Value,
1263    verbose: bool,
1264    diagnostics: &mut Vec<String>,
1265) -> io::Result<String> {
1266    let to = args.get_str_or("to", "");
1267    let content = args.get_str_or("content", "");
1268    let from = args.get_str("from").unwrap_or(context_name);
1269
1270    if to.is_empty() {
1271        return Ok("Error: 'to' field is required".to_string());
1272    }
1273    if content.is_empty() {
1274        return Ok("Error: 'content' field is required".to_string());
1275    }
1276
1277    // Execute pre_send_message hooks
1278    let pre_hook_data = serde_json::json!({
1279        "from": from,
1280        "to": to,
1281        "content": content,
1282        "context_name": context_name,
1283    });
1284    let pre_hook_results =
1285        tools::execute_hook(tools, tools::HookPoint::PreSendMessage, &pre_hook_data)?;
1286
1287    // Check if any hook claimed delivery
1288    let mut delivered_via: Option<String> = None;
1289    for (hook_tool_name, hook_result) in &pre_hook_results {
1290        if hook_result.get_bool_or("delivered", false) {
1291            let via = hook_result.get_str_or("via", hook_tool_name);
1292            delivered_via = Some(via.to_string());
1293            if verbose {
1294                diagnostics.push(format!(
1295                    "[Hook pre_send_message: {} intercepted delivery]",
1296                    hook_tool_name
1297                ));
1298            }
1299            break;
1300        }
1301    }
1302
1303    let delivery_result = if let Some(via) = delivered_via {
1304        format!("Message delivered to '{}' via {}", to, via)
1305    } else {
1306        let entry = InboxEntry {
1307            id: Uuid::new_v4().to_string(),
1308            timestamp: now_timestamp(),
1309            from: from.to_string(),
1310            to: to.to_string(),
1311            content: content.to_string(),
1312        };
1313        match app.append_to_inbox(to, &entry) {
1314            Ok(()) => format!("Message delivered to '{}' via local inbox", to),
1315            Err(e) => format!("Error delivering message: {}", e),
1316        }
1317    };
1318
1319    // Execute post_send_message hooks
1320    let post_hook_data = serde_json::json!({
1321        "from": from,
1322        "to": to,
1323        "content": content,
1324        "context_name": context_name,
1325        "delivery_result": delivery_result,
1326    });
1327    let _ = tools::execute_hook(tools, tools::HookPoint::PostSendMessage, &post_hook_data);
1328
1329    Ok(delivery_result)
1330}

1332/// Process all tool calls from a response.
1333///
1334/// Parallel-safe tools (ToolMetadata::parallel == true) run concurrently via
1335/// `join_all`. Sequential tools (flow_control, parallel == false) run after
1336/// the parallel batch completes. Results are emitted to the sink and transcript
1337/// in the original tool_call order regardless of execution order.
1338#[allow(clippy::too_many_arguments)]
1339async fn process_tool_calls<S: ResponseSink>(
1340    app: &AppState,
1341    context_name: &str,
1342    tool_calls: &[ratatoskr::ToolCall],
1343    messages: &mut Vec<serde_json::Value>,
1344    tools: &[Tool],
1345    handoff: &mut tools::Handoff,
1346    use_reflection: bool,
1347    resolved_config: &ResolvedConfig,
1348    fuel_remaining: &mut usize,
1349    fuel_total: usize,
1350    verbose: bool,
1351    sink: &mut S,
1352    permission_handler: Option<&PermissionHandler>,
1353    project_root: &Path,
1354) -> io::Result<()> {
1355    // Convert tool calls to JSON format for the assistant message
1356    let tool_calls_json: Vec<serde_json::Value> = tool_calls
1357        .iter()
1358        .map(|tc| {
1359            serde_json::json!({
1360                "id": tc.id,
1361                "type": "function",
1362                "function": {
1363                    "name": tc.name,
1364                    "arguments": tc.arguments,
1365                }
1366            })
1367        })
1368        .collect();
1369
1370    messages.push(serde_json::json!({
1371        "role": "assistant",
1372        "tool_calls": tool_calls_json,
1373    }));
1374
1375    // Partition tool calls into parallel-safe and sequential batches.
1376    // We store (original_index, tool_call) to preserve ordering in results.
1377    let mut parallel_batch: Vec<(usize, &ratatoskr::ToolCall)> = Vec::new();
1378    let mut sequential_batch: Vec<(usize, &ratatoskr::ToolCall)> = Vec::new();
1379
1380    for (i, tc) in tool_calls.iter().enumerate() {
1381        let metadata = tools::get_tool_metadata(tools, &tc.name);
1382        if metadata.parallel && !metadata.flow_control {
1383            parallel_batch.push((i, tc));
1384        } else {
1385            sequential_batch.push((i, tc));
1386        }
1387    }
1388
1389    // Results indexed by original position
1390    let mut results: Vec<Option<ToolExecutionResult>> =
1391        (0..tool_calls.len()).map(|_| None).collect();
1392
1393    // Execute parallel batch concurrently via join_all.
1394    // These futures run on the current task (no spawn), interleaving at .await
1395    // points — safe with !Send types like AppState's RefCell.
1396    if !parallel_batch.is_empty() {
1397        let parallel_futures: Vec<_> = parallel_batch
1398            .iter()
1399            .map(|(_idx, tc)| {
1400                execute_tool_pure(
1401                    app,
1402                    context_name,
1403                    tc,
1404                    tools,
1405                    use_reflection,
1406                    resolved_config,
1407                    verbose,
1408                    permission_handler,
1409                    project_root,
1410                )
1411            })
1412            .collect();
1413
1414        let parallel_results = futures_util::future::join_all(parallel_futures).await;
1415
1416        for ((idx, _tc), result) in parallel_batch.iter().zip(parallel_results) {
1417            results[*idx] = Some(result?);
1418        }
1419    }
1420
1421    // Execute sequential batch one at a time (these may mutate handoff)
1422    for (idx, tc) in &sequential_batch {
1423        let result = execute_single_tool(
1424            app,
1425            context_name,
1426            tc,
1427            tools,
1428            handoff,
1429            use_reflection,
1430            resolved_config,
1431            permission_handler,
1432            verbose,
1433            sink,
1434            project_root,
1435        )
1436        .await?;
1437        results[*idx] = Some(result);
1438    }
1439
1440    // Write all tool_call entries to transcript first (matches API message order:
1441    // one assistant message with tool_calls[], then individual tool result messages)
1442    for (i, tc) in tool_calls.iter().enumerate() {
1443        let tool_call_entry = create_tool_call_entry(context_name, &tc.name, &tc.arguments, &tc.id);
1444        app.append_to_transcript_and_context(context_name, &tool_call_entry)?;
1445        sink.handle(ResponseEvent::TranscriptEntry(tool_call_entry))?;
1446
1447        // Pre-log verbose diagnostic while we're iterating
1448        if verbose && let Some(result) = &results[i] {
1449            for diag in &result.diagnostics {
1450                sink.handle(ResponseEvent::Diagnostic {
1451                    message: diag.clone(),
1452                    verbose_only: true,
1453                })?;
1454            }
1455        }
1456    }
1457
1458    // Emit sink events and write tool_result entries in original order
1459    for (i, tc) in tool_calls.iter().enumerate() {
1460        let result = results[i]
1461            .take()
1462            .expect("all tool results should be populated");
1463
1464        if verbose {
1465            sink.handle(ResponseEvent::Diagnostic {
1466                message: format!("[Tool: {}]", tc.name),
1467                verbose_only: true,
1468            })?;
1469        }
1470
1471        // Emit diagnostics collected during parallel execution (if not already emitted)
1472        if !verbose {
1473            for diag in &result.diagnostics {
1474                sink.handle(ResponseEvent::Diagnostic {
1475                    message: diag.clone(),
1476                    verbose_only: true,
1477                })?;
1478            }
1479        }
1480
1481        let summary = tools::tool_call_summary(tools, &tc.name, &tc.arguments);
1482        sink.handle(ResponseEvent::ToolStart {
1483            name: tc.name.clone(),
1484            summary,
1485        })?;
1486
1487        // Log tool result to transcript
1488        let logged_result = if result.was_cached {
1489            &result.final_result
1490        } else {
1491            &result.original_result
1492        };
1493        let tool_result_entry =
1494            create_tool_result_entry(context_name, &tc.name, logged_result, &tc.id);
1495        app.append_to_transcript_and_context(context_name, &tool_result_entry)?;
1496        sink.handle(ResponseEvent::TranscriptEntry(tool_result_entry))?;
1497
1498        sink.handle(ResponseEvent::ToolResult {
1499            name: tc.name.clone(),
1500            result: result.final_result.clone(),
1501            cached: result.was_cached,
1502        })?;
1503
1504        // Show full content of todos/goals updates in verbose mode
1505        if verbose
1506            && matches!(tc.name.as_str(), "update_todos" | "update_goals")
1507            && let Ok(args) = serde_json::from_str::<serde_json::Value>(&tc.arguments)
1508            && let Some(content) = args["content"].as_str()
1509        {
1510            sink.handle(ResponseEvent::Diagnostic {
1511                message: format!("[{}]\n{}", tc.name, content),
1512                verbose_only: true,
1513            })?;
1514        }
1515
1516        // Execute post_tool hooks
1517        let args: serde_json::Value =
1518            serde_json::from_str(&tc.arguments).unwrap_or(serde_json::json!({}));
1519        let post_hook_data = serde_json::json!({
1520            "tool_name": tc.name,
1521            "arguments": args,
1522            "result": result.original_result,
1523            "cached": result.was_cached,
1524        });
1525        let _ = tools::execute_hook(tools, tools::HookPoint::PostTool, &post_hook_data);
1526
1527        // Apply handoff for parallel-executed flow control tools
1528        // (sequential ones already applied via execute_single_tool)
1529        let metadata = tools::get_tool_metadata(tools, &tc.name);
1530        if metadata.flow_control && metadata.parallel {
1531            // This shouldn't happen (flow_control tools are always sequential),
1532            // but handle it defensively
1533            if metadata.ends_turn {
1534                handoff.set_user(args.get_str_or("message", "").to_string());
1535            } else {
1536                handoff.set_agent(args.get_str_or("prompt", "").to_string());
1537            }
1538        }
1539
1540        messages.push(serde_json::json!({
1541            "role": "tool",
1542            "tool_call_id": tc.id,
1543            "content": result.final_result,
1544        }));
1545    }
1546
1547    // Execute post_tool_batch hook — allows plugins to override fallback after seeing tool results
1548    let tool_batch_info: Vec<serde_json::Value> = tool_calls
1549        .iter()
1550        .map(|tc| {
1551            json!({
1552                "name": tc.name,
1553                "arguments": serde_json::from_str::<serde_json::Value>(&tc.arguments).unwrap_or(json!({})),
1554            })
1555        })
1556        .collect();
1557    let hook_data = json!({
1558        "context_name": context_name,
1559        "fuel_remaining": fuel_remaining,
1560        "fuel_total": fuel_total,
1561        "current_fallback": resolved_config.fallback_tool,
1562        "tool_calls": tool_batch_info,
1563    });
1564    let hook_results = tools::execute_hook(tools, tools::HookPoint::PostToolBatch, &hook_data)?;
1565    apply_hook_overrides(handoff, fuel_remaining, &hook_results, verbose, sink)?;
1566
1567    Ok(())
1568}

1570/// Result of handling a final (non-tool-call) response.
1571enum FinalResponseAction {
1572    /// Return to the user (turn ended).
1573    ReturnToUser,
1574    /// Continue with another prompt (agent loop).
1575    ContinueWithPrompt(String),
1576}
1577
1578/// Handle the final text response from the LLM.
1579///
1580/// Handles: Empty response detection/retry, save assistant message,
1581/// post_message hook, handoff decision (return vs recurse).
1582#[allow(clippy::too_many_arguments)]
1583fn handle_final_response<S: ResponseSink>(
1584    app: &AppState,
1585    context_name: &str,
1586    full_response: &str,
1587    final_prompt: &str,
1588    mut handoff: tools::Handoff,
1589    tools: &[Tool],
1590    _resolved_config: &ResolvedConfig,
1591    verbose: bool,
1592    sink: &mut S,
1593) -> io::Result<FinalResponseAction> {
1594    // Get or create context to add the message
1595    let mut context = app.get_or_create_context(context_name)?;
1596
1597    // Save the assistant's response
1598    app.add_message(
1599        &mut context,
1600        "assistant".to_string(),
1601        full_response.to_string(),
1602    );
1603
1604    let assistant_entry = create_assistant_message_entry(context_name, full_response);
1605    app.append_to_transcript_and_context(context_name, &assistant_entry)?;
1606    sink.handle(ResponseEvent::TranscriptEntry(assistant_entry))?;
1607
1608    // Execute post_message hooks
1609    let hook_data = serde_json::json!({
1610        "prompt": final_prompt,
1611        "response": full_response,
1612        "context_name": context_name,
1613    });
1614    let _ = tools::execute_hook(tools, tools::HookPoint::PostMessage, &hook_data);
1615
1616    if app.should_warn(&context.messages) {
1617        let remaining = app.remaining_tokens(&context.messages);
1618        if verbose {
1619            sink.handle(ResponseEvent::Diagnostic {
1620                message: format!("[Context window warning: {} tokens remaining]", remaining),
1621                verbose_only: true,
1622            })?;
1623        }
1624    }
1625
1626    sink.handle(ResponseEvent::Newline)?;
1627
1628    // Determine next action based on handoff.
1629    // Fuel exhaustion is the caller's responsibility — we just report the action.
1630    match handoff.take() {
1631        tools::HandoffTarget::User { message } => {
1632            if !message.is_empty() {
1633                sink.handle(ResponseEvent::TextChunk(&message))?;
1634                sink.handle(ResponseEvent::Newline)?;
1635            }
1636            Ok(FinalResponseAction::ReturnToUser)
1637        }
1638        tools::HandoffTarget::Agent { prompt } => {
1639            Ok(FinalResponseAction::ContinueWithPrompt(prompt))
1640        }
1641    }
1642}

1644/// Send a prompt to the LLM with streaming response via ResponseSink.
1645///
1646/// This is the main entry point for sending prompts. It handles:
1647/// - Hook execution (pre_message, post_message, etc.)
1648/// - Inbox message injection
1649/// - Tool execution loop with fuel budget
1650/// - Context management
1651/// - Auto-compaction
1652///
1653/// # Arguments
1654///
1655/// * `context_name` - The name of the context to use for this prompt
1656#[allow(clippy::too_many_arguments)]
1657pub async fn send_prompt<S: ResponseSink>(
1658    app: &AppState,
1659    context_name: &str,
1660    initial_prompt: String,
1661    tools: &[Tool],
1662    resolved_config: &ResolvedConfig,
1663    options: &PromptOptions<'_>,
1664    sink: &mut S,
1665    permission_handler: Option<&PermissionHandler>,
1666    home_dir: &Path,
1667    project_root: &Path,
1668) -> io::Result<()> {
1669    let fuel_total = resolved_config.fuel;
1670    let mut fuel_remaining = fuel_total;
1671    let mut current_prompt = initial_prompt;
1672
1673    // Outer loop: each iteration is a full setup + agentic exchange.
1674    // First iteration is the user's turn (free); continuations cost 1 fuel.
1675    loop {
1676        // === Validation & Setup ===
1677        if current_prompt.trim().is_empty() {
1678            return Err(io::Error::new(
1679                ErrorKind::InvalidInput,
1680                "Prompt cannot be empty",
1681            ));
1682        }
1683
1684        app.validate_config(resolved_config, tools)?;
1685
1686        let verbose = options.verbose;
1687
1688        if verbose {
1689            sink.handle(ResponseEvent::Diagnostic {
1690                message: format!("[fuel: {}/{} entering turn]", fuel_remaining, fuel_total),
1691                verbose_only: true,
1692            })?;
1693        }
1694        let use_reflection = options.use_reflection;
1695        let debug = options.debug;
1696
1697        let mut context = app.get_or_create_context(context_name)?;
1698
1699        // === Pre-message Hooks & Inbox ===
1700        let mut final_prompt = current_prompt.clone();
1701        let hook_data = serde_json::json!({
1702            "prompt": current_prompt,
1703            "context_name": context.name,
1704            "summary": context.summary,
1705        });
1706        let hook_results = tools::execute_hook(tools, tools::HookPoint::PreMessage, &hook_data)?;
1707        for (tool_name, result) in hook_results {
1708            if let Some(modified) = result.get_str("prompt") {
1709                if verbose {
1710                    eprintln!("[Hook pre_message: {} modified prompt]", tool_name);
1711                }
1712                final_prompt = modified.to_string();
1713            }
1714        }
1715
1716        // Check inbox and inject messages before the user prompt
1717        let inbox_messages = app.load_and_clear_inbox(context_name)?;
1718        if !inbox_messages.is_empty() {
1719            let mut inbox_content = String::from("--- INBOX MESSAGES ---\n");
1720            for msg in &inbox_messages {
1721                inbox_content.push_str(&format!("[From: {}] {}\n", msg.from, msg.content));
1722            }
1723            inbox_content.push_str("--- END INBOX ---\n\n");
1724            final_prompt = format!("{}{}", inbox_content, final_prompt);
1725            if verbose {
1726                sink.handle(ResponseEvent::Diagnostic {
1727                    message: format!("[Inbox: {} message(s) injected]", inbox_messages.len()),
1728                    verbose_only: true,
1729                })?;
1730            }
1731        }
1732
1733        // Add datetime prefix to user message
1734        let datetime_prefix = chrono::Local::now().format("%Y%m%d-%H%M%z").to_string();
1735        let prefixed_prompt = format!("[{}] {}", datetime_prefix, final_prompt);
1736
1737        // Add user message to context and transcript
1738        app.add_message(&mut context, "user".to_string(), prefixed_prompt.clone());
1739        let user_entry =
1740            create_user_message_entry(context_name, &prefixed_prompt, &resolved_config.username);
1741        app.append_to_transcript_and_context(context_name, &user_entry)?;
1742        sink.handle(ResponseEvent::TranscriptEntry(user_entry))?;
1743
1744        // Context window warning
1745        if app.should_warn(&context.messages) {
1746            let remaining = app.remaining_tokens(&context.messages);
1747            if verbose {
1748                sink.handle(ResponseEvent::Diagnostic {
1749                    message: format!("[Context window warning: {} tokens remaining]", remaining),
1750                    verbose_only: true,
1751                })?;
1752            }
1753        }
1754
1755        // === Auto-compaction Check ===
1756        if app.should_auto_compact(&context, resolved_config) {
1757            return compact_context_with_llm(app, context_name, resolved_config, verbose).await;
1758        }
1759
1760        // === Build System Prompt ===
1761        let full_system_prompt = build_full_system_prompt(
1762            app,
1763            context_name,
1764            &context.summary,
1765            use_reflection,
1766            tools,
1767            resolved_config,
1768            verbose,
1769            sink,
1770            home_dir,
1771            project_root,
1772        )?;
1773
1774        // === Prepare Messages ===
1775        let mut messages: Vec<serde_json::Value> = if !full_system_prompt.is_empty() {
1776            vec![serde_json::json!({
1777                "role": "system",
1778                "content": full_system_prompt,
1779            })]
1780        } else {
1781            Vec::new()
1782        };
1783
1784        // Add conversation messages (skip system messages, strip internal _id field)
1785        for m in &context.messages {
1786            let role = m["role"].as_str().unwrap_or("");
1787            if role == "system" {
1788                continue;
1789            }
1790            let mut msg = m.clone();
1791            if let Some(obj) = msg.as_object_mut() {
1792                obj.remove("_id");
1793            }
1794            messages.push(msg);
1795        }
1796
1797        // === Prepare Tools ===
1798        let mut all_tools = tools::tools_to_api_format(tools);
1799        all_tools.extend(tools::builtin_tools_to_api_format(use_reflection));
1800        all_tools.extend(tools::all_file_tools_to_api_format());
1801        all_tools.extend(tools::all_agent_tools_to_api_format());
1802        all_tools.extend(tools::all_coding_tools_to_api_format());
1803        annotate_fallback_tool(&mut all_tools, &resolved_config.fallback_tool);
1804        all_tools = filter_tools_by_config(all_tools, &resolved_config.tools);
1805
1806        // Execute pre_api_tools hook
1807        let tool_info = build_tool_info_list(&all_tools, tools);
1808        let hook_data = json!({
1809            "context_name": context.name,
1810            "tools": tool_info,
1811            "fuel_remaining": fuel_remaining,
1812            "fuel_total": fuel_total,
1813        });
1814        let hook_results = tools::execute_hook(tools, tools::HookPoint::PreApiTools, &hook_data)?;
1815        all_tools = filter_tools_from_hook_results(all_tools, &hook_results, verbose, sink)?;
1816
1817        // === Build Request ===
1818        let tools_for_request = if resolved_config.no_tool_calls {
1819            None
1820        } else {
1821            Some(all_tools.as_slice())
1822        };
1823        let mut request_body =
1824            build_request_body(resolved_config, &messages, tools_for_request, true);
1825
1826        // Execute pre_api_request hook
1827        let hook_data = json!({
1828            "context_name": context.name,
1829            "request_body": request_body,
1830            "fuel_remaining": fuel_remaining,
1831            "fuel_total": fuel_total,
1832        });
1833        let hook_results = tools::execute_hook(tools, tools::HookPoint::PreApiRequest, &hook_data)?;
1834        request_body = apply_request_modifications(request_body, &hook_results, verbose, sink)?;
1835
1836        // === Initialize Handoff ===
1837        let fallback = options.fallback_override.clone().unwrap_or_else(|| {
1838            let meta = tools::get_tool_metadata(tools, &resolved_config.fallback_tool);
1839            if meta.ends_turn {
1840                tools::HandoffTarget::User {
1841                    message: String::new(),
1842                }
1843            } else {
1844                tools::HandoffTarget::Agent {
1845                    prompt: String::new(),
1846                }
1847            }
1848        });
1849        let mut handoff = tools::Handoff::new(fallback);
1850
1851        // Execute pre_agentic_loop hook
1852        let hook_data = json!({
1853            "context_name": context.name,
1854            "fuel_remaining": fuel_remaining,
1855            "fuel_total": fuel_total,
1856            "current_fallback": resolved_config.fallback_tool,
1857            "message": final_prompt,
1858        });
1859        let hook_results =
1860            tools::execute_hook(tools, tools::HookPoint::PreAgenticLoop, &hook_data)?;
1861        apply_hook_overrides(
1862            &mut handoff,
1863            &mut fuel_remaining,
1864            &hook_results,
1865            verbose,
1866            sink,
1867        )?;
1868
1869        // === Inner Loop: stream responses and process tool calls ===
1870        loop {
1871            sink.handle(ResponseEvent::StartResponse)?;
1872            log_request_if_enabled(app, context_name, debug, &request_body);
1873
1874            let response =
1875                collect_streaming_response(resolved_config, &messages, &all_tools, verbose, sink)
1876                    .await?;
1877
1878            // Log response metadata
1879            if let Some(ref meta) = response.response_meta {
1880                log_response_meta_if_enabled(app, context_name, debug, meta);
1881            }
1882
1883            // Signal streaming finished
1884            if !sink.is_json_mode() {
1885                sink.handle(ResponseEvent::Finished)?;
1886            }
1887
1888            // Handle tool calls
1889            if response.has_tool_calls && !response.tool_calls.is_empty() {
1890                process_tool_calls(
1891                    app,
1892                    context_name,
1893                    &response.tool_calls,
1894                    &mut messages,
1895                    tools,
1896                    &mut handoff,
1897                    use_reflection,
1898                    resolved_config,
1899                    &mut fuel_remaining,
1900                    fuel_total,
1901                    verbose,
1902                    sink,
1903                    permission_handler,
1904                    project_root,
1905                )
1906                .await?;
1907
1908                // Keep request_body in sync for logging
1909                request_body["messages"] = serde_json::json!(messages);
1910
1911                // Tool call round costs 1 fuel
1912                fuel_remaining = fuel_remaining.saturating_sub(1);
1913                if verbose {
1914                    sink.handle(ResponseEvent::Diagnostic {
1915                        message: format!(
1916                            "[fuel: {}/{} after tool batch]",
1917                            fuel_remaining, fuel_total
1918                        ),
1919                        verbose_only: true,
1920                    })?;
1921                }
1922                if fuel_remaining == 0 {
1923                    sink.handle(ResponseEvent::Diagnostic {
1924                        message: format!(
1925                            "[fuel exhausted (0/{}), returning control to user]",
1926                            fuel_total
1927                        ),
1928                        verbose_only: false,
1929                    })?;
1930                    return Ok(());
1931                }
1932                continue;
1933            }
1934
1935            // Check for empty response
1936            if response.full_response.trim().is_empty() {
1937                fuel_remaining =
1938                    fuel_remaining.saturating_sub(resolved_config.fuel_empty_response_cost);
1939                if fuel_remaining == 0 {
1940                    sink.handle(ResponseEvent::Diagnostic {
1941                        message: format!(
1942                            "[fuel exhausted (0/{}), returning control to user]",
1943                            fuel_total
1944                        ),
1945                        verbose_only: false,
1946                    })?;
1947                    return Ok(());
1948                }
1949                if verbose {
1950                    sink.handle(ResponseEvent::Diagnostic {
1951                        message: format!(
1952                            "[empty response, fuel: {}/{}]",
1953                            fuel_remaining, fuel_total
1954                        ),
1955                        verbose_only: true,
1956                    })?;
1957                }
1958                continue;
1959            }
1960
1961            // Text response — break inner loop to handle final response
1962            match handle_final_response(
1963                app,
1964                context_name,
1965                &response.full_response,
1966                &final_prompt,
1967                handoff,
1968                tools,
1969                resolved_config,
1970                verbose,
1971                sink,
1972            )? {
1973                FinalResponseAction::ReturnToUser => return Ok(()),
1974                FinalResponseAction::ContinueWithPrompt(continue_prompt) => {
1975                    fuel_remaining = fuel_remaining.saturating_sub(1);
1976                    if fuel_remaining == 0 {
1977                        sink.handle(ResponseEvent::Diagnostic {
1978                            message: format!(
1979                                "[fuel exhausted (0/{}), returning control to user]",
1980                                fuel_total
1981                            ),
1982                            verbose_only: false,
1983                        })?;
1984                        return Ok(());
1985                    }
1986                    if verbose {
1987                        sink.handle(ResponseEvent::Diagnostic {
1988                            message: format!(
