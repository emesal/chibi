34/// Tool type classification for pre_api_tools hook
35#[derive(Debug, Clone, Copy, PartialEq, Eq)]
36enum ToolType {
37    Builtin,
38    File,
39    Plugin,
40}
41
42impl ToolType {
43    fn as_str(&self) -> &'static str {
44        match self {
45            ToolType::Builtin => "builtin",
46            ToolType::File => "file",
47            ToolType::Plugin => "plugin",
48        }
49    }
50}

52/// Built-in tool names (todos, goals, reflection, send_message)
53const BUILTIN_TOOL_NAMES: &[&str] = &[
54    "update_todos",
55    "update_goals",
56    "update_reflection",
57    "send_message",
58];

60/// File tool names (file_head, file_tail, file_lines, file_grep, cache_list)
61const FILE_TOOL_NAMES: &[&str] = &[
62    "file_head",
63    "file_tail",
64    "file_lines",
65    "file_grep",
66    "cache_list",

69/// Classify a tool's type based on its name
70fn classify_tool_type(name: &str, plugin_names: &[&str]) -> ToolType {
71    if BUILTIN_TOOL_NAMES.contains(&name) {
72        ToolType::Builtin
73    } else if FILE_TOOL_NAMES.contains(&name) {
74        ToolType::File
75    } else if plugin_names.contains(&name) {
76        ToolType::Plugin
77    } else {
78        // Unknown tools default to plugin type
79        ToolType::Plugin
80    }
81}

83/// Build tool info list for pre_api_tools hook data
84fn build_tool_info_list(
85    all_tools: &[serde_json::Value],
86    plugin_tools: &[Tool],
87) -> Vec<serde_json::Value> {
88    let plugin_names: Vec<&str> = plugin_tools.iter().map(|t| t.name.as_str()).collect();
89
90    all_tools
91        .iter()
92        .filter_map(|tool| {
93            let name = tool
94                .get("function")
95                .and_then(|f| f.get("name"))
96                .and_then(|n| n.as_str())?;
97            let tool_type = classify_tool_type(name, &plugin_names);
98            Some(json!({
99                "name": name,
100                "type": tool_type.as_str(),
101            }))
102        })
103        .collect()
104}

106/// Annotate the fallback tool's description to indicate it's called automatically.
107fn annotate_fallback_tool(tools: &mut [serde_json::Value], fallback_name: &str) {
108    const FALLBACK_SUFFIX: &str = " Called automatically if no other tool is used.";
109
110    for tool in tools.iter_mut() {
111        let name_matches = tool
112            .get("function")
113            .and_then(|f| f.get("name"))
114            .and_then(|n| n.as_str())
115            .is_some_and(|name| name == fallback_name);
116
117        if name_matches {
118            if let Some(func) = tool.get_mut("function")
119                && let Some(desc) = func.get_mut("description")
120                && let Some(desc_str) = desc.as_str()
121            {
122                *desc = serde_json::Value::String(format!("{}{}", desc_str, FALLBACK_SUFFIX));
123            }
124            break; // Only annotate the first match
125        }
126    }
127}

129/// Filter tools based on config include/exclude lists
130fn filter_tools_by_config(
131    tools: Vec<serde_json::Value>,
132    config: &ToolsConfig,
133) -> Vec<serde_json::Value> {
134    let mut result = tools;
135
136    // Apply include filter first (if set, only these tools are considered)
137    if let Some(ref include) = config.include {
138        result.retain(|tool| {
139            tool.get("function")
140                .and_then(|f| f.get("name"))
141                .and_then(|n| n.as_str())
142                .is_some_and(|name| include.contains(&name.to_string()))
143        });
144    }
145
146    // Apply exclude filter (remove these tools from remaining)
147    if let Some(ref exclude) = config.exclude {
148        result.retain(|tool| {
149            tool.get("function")
150                .and_then(|f| f.get("name"))
151                .and_then(|n| n.as_str())
152                .is_some_and(|name| !exclude.contains(&name.to_string()))
153        });
154    }
155
156    result
157}

159/// Filter tools based on hook results
160/// Multiple hooks: includes are intersected, excludes are unioned
161fn filter_tools_from_hook_results<S: ResponseSink>(
162    tools: Vec<serde_json::Value>,
163    hook_results: &[(String, serde_json::Value)],
164    verbose: bool,
165    sink: &mut S,
166) -> io::Result<Vec<serde_json::Value>> {
167    if hook_results.is_empty() {
168        return Ok(tools);
169    }
170
171    let mut result = tools;
172
173    // Collect all includes and excludes from hook results
174    let mut all_includes: Option<Vec<String>> = None;
175    let mut all_excludes: Vec<String> = Vec::new();
176
177    for (hook_name, hook_result) in hook_results {
178        // Handle include lists (intersection)
179        if let Some(include) = hook_result.get_array("include") {
180            let include_names: Vec<String> = include
181                .iter()
182                .filter_map(|v| v.as_str().map(|s| s.to_string()))
183                .collect();
184
185            if verbose {
186                sink.handle(ResponseEvent::Diagnostic {
187                    message: format!(
188                        "[Hook pre_api_tools: {} include filter: {:?}]",
189                        hook_name, include_names
190                    ),
191                    verbose_only: true,
192                })?;
193            }
194
195            all_includes = Some(match all_includes {
196                Some(existing) => existing
197                    .into_iter()
198                    .filter(|name| include_names.contains(name))
199                    .collect(),
200                None => include_names,
201            });
202        }
203
204        // Handle exclude lists (union)
205        if let Some(exclude) = hook_result.get_array("exclude") {
206            let exclude_names: Vec<String> = exclude
207                .iter()
208                .filter_map(|v| v.as_str().map(|s| s.to_string()))
209                .collect();
210
211            if verbose {
212                sink.handle(ResponseEvent::Diagnostic {
213                    message: format!(
214                        "[Hook pre_api_tools: {} exclude filter: {:?}]",
215                        hook_name, exclude_names
216                    ),
217                    verbose_only: true,
218                })?;
219            }
220
221            for name in exclude_names {
222                if !all_excludes.contains(&name) {
223                    all_excludes.push(name);
224                }
225            }
226        }
227    }
228
229    // Apply collected includes (intersection)
230    if let Some(includes) = all_includes {
231        result.retain(|tool| {
232            tool.get("function")
233                .and_then(|f| f.get("name"))
234                .and_then(|n| n.as_str())
235                .is_some_and(|name| includes.contains(&name.to_string()))
236        });
237    }
238
239    // Apply collected excludes (union)
240    if !all_excludes.is_empty() {
241        result.retain(|tool| {
242            tool.get("function")
243                .and_then(|f| f.get("name"))
244                .and_then(|n| n.as_str())
245                .is_some_and(|name| !all_excludes.contains(&name.to_string()))
246        });
247    }
248
249    Ok(result)
250}

252/// Apply request modifications from pre_api_request hook results
253/// Hook returns are merged (not replaced) into the request body
254fn apply_request_modifications<S: ResponseSink>(
255    mut request_body: serde_json::Value,
256    hook_results: &[(String, serde_json::Value)],
257    verbose: bool,
258    sink: &mut S,
259) -> io::Result<serde_json::Value> {
260    for (hook_name, hook_result) in hook_results {
261        if let Some(modifications) = hook_result.get("request_body")
262            && let Some(mods_obj) = modifications.as_object()
263        {
264            if verbose {
265                sink.handle(ResponseEvent::Diagnostic {
266                    message: format!(
267                        "[Hook pre_api_request: {} modifying request (keys: {:?})]",
268                        hook_name,
269                        mods_obj.keys().collect::<Vec<_>>()
270                    ),
271                    verbose_only: true,
272                })?;
273            }
274
275            // Merge modifications into request body
276            if let Some(body_obj) = request_body.as_object_mut() {
277                for (key, value) in mods_obj {
278                    body_obj.insert(key.clone(), value.clone());
279                }
280            }
281        }
282    }
283
284    Ok(request_body)
285}

287/// Apply fallback override from hook results.
288/// Hooks can return `{"fallback": "<tool_name>"}` to override the fallback tool.
289/// The tool must be a flow_control tool (validated elsewhere).
290fn apply_fallback_override<S: ResponseSink>(
291    handoff: &mut tools::Handoff,
292    hook_results: &[(String, serde_json::Value)],
293    verbose: bool,
294    sink: &mut S,
295) -> io::Result<()> {
296    for (hook_name, hook_result) in hook_results {
297        if let Some(fallback_str) = hook_result.get_str("fallback") {
298            // Use builtin metadata since hooks can only override to builtins
299            let meta = tools::builtin_tool_metadata(fallback_str);
300            let new_fallback = if meta.ends_turn {
301                tools::HandoffTarget::User {
302                    message: String::new(),
303                }
304            } else {
305                tools::HandoffTarget::Agent {
306                    prompt: String::new(),
307                }
308            };
309            handoff.set_fallback(new_fallback);
310            if verbose {
311                sink.handle(ResponseEvent::Diagnostic {
312                    message: format!("[Hook {} set fallback to {}]", hook_name, fallback_str),
313                    verbose_only: true,
314                })?;
315            }
316        }
317    }
318    Ok(())
319}

359/// Build the full system prompt with all components.
360///
361/// Handles: loading base prompt, todos, goals, summary, reflection prompt,
362/// pre/post system_prompt hooks, and username injection.
363#[allow(clippy::too_many_arguments)]
364fn build_full_system_prompt<S: ResponseSink>(
365    app: &AppState,
366    context_name: &str,
367    summary: &str,
368    use_reflection: bool,
369    tools: &[Tool],
370    resolved_config: &ResolvedConfig,
371    verbose: bool,
372    sink: &mut S,
373) -> io::Result<String> {
374    // Load base prompts
375    let system_prompt = app.load_system_prompt_for(context_name)?;
376    let reflection_prompt = if use_reflection {
377        app.load_reflection_prompt()?
378    } else {
379        String::new()
380    };
381
382    // Load context-specific state
383    let todos = app.load_todos(context_name)?;
384    let goals = app.load_goals(context_name)?;
385
386    // Execute pre_system_prompt hook - can inject content before system prompt sections
387    let pre_sys_hook_data = serde_json::json!({
388        "context_name": context_name,
389        "summary": summary,
390        "todos": todos,
391        "goals": goals,
392    });
393    let pre_sys_hook_results =
394        tools::execute_hook(tools, tools::HookPoint::PreSystemPrompt, &pre_sys_hook_data)?;
395
396    // Build full system prompt with all components
397    let mut full_system_prompt = system_prompt;
398
399    // Prepend any content from pre_system_prompt hooks
400    for (hook_tool_name, result) in &pre_sys_hook_results {
401        if let Some(inject) = result.get_str("inject")
402            && !inject.is_empty()
403        {
404            if verbose {
405                sink.handle(ResponseEvent::Diagnostic {
406                    message: format!(
407                        "[Hook pre_system_prompt: {} injected content]",
408                        hook_tool_name
409                    ),
410                    verbose_only: true,
411                })?;
412            }
413            full_system_prompt = format!("{}\n\n{}", inject, full_system_prompt);
414        }
415    }
416
417    // Add username info at the start if not "user"
418    if resolved_config.username != "user" {
419        full_system_prompt.push_str(&format!(
420            "\n\nThe user speaking to you is called: {}",
421            resolved_config.username
422        ));
423    }
424
425    // Add context name
426    full_system_prompt.push_str(&format!("\n\nCurrent context: {}", context_name));
427
428    // Add summary if present
429    if !summary.is_empty() {
430        full_system_prompt.push_str("\n\n--- CONVERSATION SUMMARY ---\n");
431        full_system_prompt.push_str(summary);
432    }
433
434    // Add goals if present
435    if !goals.is_empty() {
436        full_system_prompt.push_str("\n\n--- CURRENT GOALS ---\n");
437        full_system_prompt.push_str(&goals);
438    }
439
440    // Add todos if present
441    if !todos.is_empty() {
442        full_system_prompt.push_str("\n\n--- CURRENT TODOS ---\n");
443        full_system_prompt.push_str(&todos);
444    }
445
446    // Add reflection prompt last (personality layer)
447    if !reflection_prompt.is_empty() {
448        full_system_prompt.push_str("\n\n");
449        full_system_prompt.push_str(&reflection_prompt);
450    }
451
452    // Execute post_system_prompt hook - can inject content after all system prompt sections
453    let post_sys_hook_data = serde_json::json!({
454        "context_name": context_name,
455        "summary": summary,
456        "todos": todos,
457        "goals": goals,
458    });
459    let post_sys_hook_results = tools::execute_hook(
460        tools,
461        tools::HookPoint::PostSystemPrompt,
462        &post_sys_hook_data,
463    )?;
464
465    // Append any content from post_system_prompt hooks
466    for (hook_tool_name, result) in &post_sys_hook_results {
467        if let Some(inject) = result.get_str("inject")
468            && !inject.is_empty()
469        {
470            if verbose {
471                sink.handle(ResponseEvent::Diagnostic {
472                    message: format!(
473                        "[Hook post_system_prompt: {} injected content]",
474                        hook_tool_name
475                    ),
476                    verbose_only: true,
477                })?;
478            }
479            full_system_prompt.push_str("\n\n");
480            full_system_prompt.push_str(inject);
481        }
482    }
483
484    // Store combined prompt in context_meta for API request reconstruction
485    if !full_system_prompt.is_empty() {
486        app.save_combined_system_prompt(context_name, &full_system_prompt)?;
487    }
488
489    Ok(full_system_prompt)
490}

492/// Result of collecting a streaming response from the LLM.
493struct StreamingResponse {
494    /// The full text response accumulated from chunks.
495    full_response: String,
496    /// Tool calls extracted from the response.
497    tool_calls: Vec<llm::ToolCallAccumulator>,
498    /// Whether any tool calls were present.
499    has_tool_calls: bool,
500    /// Response metadata (usage stats, model info).
501    response_meta: Option<serde_json::Value>,
502}
503
504/// Collect a streaming response from the LLM API via ratatoskr.
505///
506/// Handles: SSE parsing, content accumulation, tool call reconstruction.
507/// Pure I/O - no hooks or side effects beyond streaming to sink.
508async fn collect_streaming_response<S: ResponseSink>(
509    resolved_config: &ResolvedConfig,
510    messages: &[serde_json::Value],
511    all_tools: &[serde_json::Value],
512    verbose: bool,
513    sink: &mut S,
514) -> io::Result<StreamingResponse> {
515    let gateway = build_gateway(resolved_config)?;
516
517    // Convert messages
518    let ratatoskr_messages: Vec<_> = messages
519        .iter()
520        .map(to_ratatoskr_message)
521        .collect::<io::Result<Vec<_>>>()?;
522
523    // Convert tools
524    let tool_defs: Vec<_> = all_tools
525        .iter()
526        .filter_map(|t| json_tool_to_definition(t).ok())
527        .collect();
528    let tools_opt = if tool_defs.is_empty() {
529        None
530    } else {
531        Some(tool_defs.as_slice())
532    };
533
534    // Build options
535    let options = to_chat_options(resolved_config);
536
537    // Get streaming response
538    let mut stream = gateway
539        .chat_stream(&ratatoskr_messages, tools_opt, &options)
540        .await
541        .map_err(|e| io::Error::other(format!("Gateway error: {}", e)))?;
542
543    let mut full_response = String::new();
544    let mut tool_calls: Vec<llm::ToolCallAccumulator> = Vec::new();
545    let mut has_tool_calls = false;
546    let mut response_meta: Option<serde_json::Value> = None;
547    let mut is_first_content = true;
548    let json_mode = sink.is_json_mode();
549
550    while let Some(event_result) = stream.next().await {
551        let event = event_result.map_err(|e| io::Error::other(format!("Stream error: {}", e)))?;
552
553        match event {
554            ChatEvent::Content(chunk) => {
555                // Handle first chunk newline stripping (matches old behavior)
556                let text = if is_first_content {
557                    is_first_content = false;
558                    if let Some(remaining) = chunk.strip_prefix('\n') {
559                        if remaining.is_empty() {
560                            continue;
561                        }
562                        remaining.to_string()
563                    } else {
564                        chunk
565                    }
566                } else {
567                    chunk
568                };
569
570                full_response.push_str(&text);
571                if !json_mode {
572                    sink.handle(ResponseEvent::TextChunk(&text))?;
573                }
574            }
575            ChatEvent::Reasoning(chunk) => {
576                // Reasoning content - could log in verbose mode or ignore
577                if verbose {
578                    eprintln!("[Reasoning] {}", chunk);
579                }
580            }
581            ChatEvent::ToolCallStart { index, id, name } => {
582                has_tool_calls = true;
583
584                // Prevent memory exhaustion
585                if index >= MAX_TOOL_CALLS {
586                    if verbose {
587                        eprintln!(
588                            "[WARN] Tool call index {} exceeds limit {}, skipping",
589                            index, MAX_TOOL_CALLS
590                        );
591                    }
592                    continue;
593                }
594
595                while tool_calls.len() <= index {
596                    tool_calls.push(llm::ToolCallAccumulator::default());
597                }
598
599                tool_calls[index].id = id;
600                tool_calls[index].name = name;
601            }
602            ChatEvent::ToolCallDelta { index, arguments } => {
603                if index < tool_calls.len() {
604                    tool_calls[index].arguments.push_str(&arguments);
605                }
606            }
607            ChatEvent::Usage(usage) => {
608                response_meta = Some(json!({
609                    "usage": {
610                        "prompt_tokens": usage.prompt_tokens,
611                        "completion_tokens": usage.completion_tokens,
612                        "total_tokens": usage.total_tokens
613                    }
614                }));
615            }
616            ChatEvent::Done => break,
617        }
618    }
619
620    Ok(StreamingResponse {
621        full_response,
622        tool_calls,
623        has_tool_calls,
624        response_meta,
625    })
626}

632    /// The original untruncated result.
633    original_result: String,
634    /// Whether the result was cached.

638/// Execute a single tool call with all hooks.
639///
640/// Handles: pre_tool hooks, tool dispatch (handoff/builtin/file/plugin),
641/// caching, pre/post tool_output hooks.
642#[allow(clippy::too_many_arguments)]
643fn execute_single_tool<S: ResponseSink>(
644    app: &AppState,
645    context_name: &str,
646    tool_call: &llm::ToolCallAccumulator,
647    tools: &[Tool],
648    handoff: &mut tools::Handoff,
649    use_reflection: bool,
650    resolved_config: &ResolvedConfig,
651    verbose: bool,
652    sink: &mut S,
653) -> io::Result<ToolExecutionResult> {
654    let mut args: serde_json::Value =
655        serde_json::from_str(&tool_call.arguments).unwrap_or(serde_json::json!({}));
656
657    // Check for control flow tools using metadata
658    let tool_metadata = tools::get_tool_metadata(tools, &tool_call.name);
659    let is_handoff_tool = if tool_metadata.flow_control {
660        if tool_metadata.ends_turn {
661            let message = args.get_str_or("message", "").to_string();
662            handoff.set_user(message);
663        } else {
664            let prompt = args.get_str_or("prompt", "").to_string();
665            handoff.set_agent(prompt);
666        }
667        true
668    } else {
669        false
670    };
671
672    // Execute pre_tool hooks (can modify arguments OR block execution)
673    let pre_hook_data = serde_json::json!({
674        "tool_name": tool_call.name,
675        "arguments": args,
676    });
677    let pre_hook_results = tools::execute_hook(tools, tools::HookPoint::PreTool, &pre_hook_data)?;
678
679    let mut blocked = false;
680    let mut block_message = String::new();
681
682    for (hook_tool_name, result) in pre_hook_results {
683        // Check for block signal first
684        if result.get_bool_or("block", false) {
685            blocked = true;
686            block_message = result
687                .get_str_or("message", "Tool call blocked by hook")
688                .to_string();
689            if verbose {
690                sink.handle(ResponseEvent::Diagnostic {
691                    message: format!(
692                        "[Hook pre_tool: {} blocked {} - {}]",
693                        hook_tool_name, tool_call.name, block_message
694                    ),
695                    verbose_only: true,
696                })?;
697            }
698            break;
699        }
700
701        // Check for argument modification
702        if let Some(modified_args) = result.get("arguments") {
703            if verbose {
704                sink.handle(ResponseEvent::Diagnostic {
705                    message: format!(
706                        "[Hook pre_tool: {} modified arguments for {}]",
707                        hook_tool_name, tool_call.name
708                    ),
709                    verbose_only: true,
710                })?;
711            }
712            args = modified_args.clone();
713        }
714    }
715
716    // If blocked, skip execution and use block message as result
717    let tool_result = if blocked {
718        block_message
719    } else if is_handoff_tool {
720        // Handoff tools don't execute - they just set the handoff target
721        // Use metadata to determine the result message
722        if tool_metadata.ends_turn {
723            let message = args.get_str_or("message", "");
724            if message.is_empty() {
725                "Returning to user".to_string()
726            } else {
727                message.to_string()
728            }
729        } else {
730            let prompt = args.get_str_or("prompt", "");
731            if prompt.is_empty() {
732                "Continuing processing".to_string()
733            } else {
734                format!("Continuing with: {}", prompt)
735            }
736        }
737    } else if tool_call.name == tools::REFLECTION_TOOL_NAME && !use_reflection {
738        "Error: Reflection tool is not enabled".to_string()
739    } else if let Some(builtin_result) =
740        tools::execute_builtin_tool(app, context_name, &tool_call.name, &args)
741    {
742        match builtin_result {
743            Ok(r) => r,
744            Err(e) => format!("Error: {}", e),
745        }
746    } else if tool_call.name == tools::SEND_MESSAGE_TOOL_NAME {
747        // Handle built-in send_message tool
748        execute_send_message_tool(app, context_name, tools, &args, verbose, sink)?
749    } else if tools::is_file_tool(&tool_call.name) {
750        // For write tools, check permission via pre_file_write hook first
751        if tool_call.name == tools::WRITE_FILE_TOOL_NAME
752            || tool_call.name == tools::PATCH_FILE_TOOL_NAME
753        {
754            let hook_data = serde_json::json!({
755                "tool_name": tool_call.name,
756                "path": args.get_str("path").unwrap_or(""),
757                "content": args.get_str("content"),
758                "find": args.get_str("find"),
759                "replace": args.get_str("replace"),
760            });
761            let hook_results =
762                tools::execute_hook(tools, tools::HookPoint::PreFileWrite, &hook_data)?;
763
764            // Check if any hook denied the operation
765            let mut denied = false;
766            let mut deny_reason = String::new();
767            for (_hook_name, result) in &hook_results {
768                if !result.get_bool_or("approved", false) {
769                    denied = true;
770                    deny_reason = result
771                        .get_str_or("reason", "Permission denied by hook")
772                        .to_string();
773                    break;
774                }
775            }
776
777            if denied {
778                format!("Error: {}", deny_reason)
779            } else if hook_results.is_empty() {
780                // No hooks registered = fail-safe deny
781                "Error: No permission handler configured. File write tools require a pre_file_write hook plugin.".to_string()
782            } else {
783                // Permission granted, execute the tool
784                match tools::execute_file_tool(
785                    app,
786                    context_name,
787                    &tool_call.name,
788                    &args,
789                    resolved_config,
790                ) {
791                    Some(Ok(r)) => r,
792                    Some(Err(e)) => format!("Error: {}", e),
793                    None => format!("Error: Unknown file tool '{}'", tool_call.name),
794                }
795            }
796        } else {
797            // Regular file tools (read-only) don't need permission
798            match tools::execute_file_tool(
799                app,
800                context_name,
801                &tool_call.name,
802                &args,
803                resolved_config,
804            ) {
805                Some(Ok(r)) => r,
806                Some(Err(e)) => format!("Error: {}", e),
807                None => format!("Error: Unknown file tool '{}'", tool_call.name),
808            }
809        }
810    } else if let Some(tool) = tools::find_tool(tools, &tool_call.name) {
811        match tools::execute_tool(tool, &args, verbose) {
812            Ok(r) => r,
813            Err(e) => format!("Error: {}", e),
814        }
815    } else {
816        format!("Error: Unknown tool '{}'", tool_call.name)
817    };
818
819    // Execute pre_tool_output hooks (can modify or replace output)
820    let mut tool_result = tool_result;
821    let pre_output_hook_data = serde_json::json!({
822        "tool_name": tool_call.name,
823        "arguments": args,
824        "output": tool_result,
825    });
826    let pre_output_hook_results = tools::execute_hook(
827        tools,
828        tools::HookPoint::PreToolOutput,
829        &pre_output_hook_data,
830    )?;
831
832    for (hook_tool_name, result) in pre_output_hook_results {
833        if result.get_bool_or("block", false) {
834            let replacement = result
835                .get_str_or("message", "Output blocked by hook")
836                .to_string();
837            if verbose {
838                sink.handle(ResponseEvent::Diagnostic {
839                    message: format!(
840                        "[Hook pre_tool_output: {} blocked output from {}]",
841                        hook_tool_name, tool_call.name
842                    ),
843                    verbose_only: true,
844                })?;
845            }
846            tool_result = replacement;
847            break;
848        }
849
850        if let Some(modified_output) = result.get_str("output") {
851            if verbose {
852                sink.handle(ResponseEvent::Diagnostic {
853                    message: format!(
854                        "[Hook pre_tool_output: {} modified output from {}]",
855                        hook_tool_name, tool_call.name
856                    ),
857                    verbose_only: true,
858                })?;
859            }
860            tool_result = modified_output.to_string();
861        }
862    }
863
864    // Check if output should be cached
865    let (final_result, was_cached) = if !tool_result.starts_with("Error:")
866        && cache::should_cache(&tool_result, resolved_config.tool_output_cache_threshold)
867    {
868        let cache_dir = app.tool_cache_dir(context_name);
869        match cache::cache_output(&cache_dir, &tool_call.name, &tool_result, &args) {
870            Ok(entry) => {
871                match cache::generate_truncated_message(
872                    &entry,
873                    resolved_config.tool_cache_preview_chars,
874                ) {
875                    Ok(truncated) => {
876                        if verbose {
877                            sink.handle(ResponseEvent::Diagnostic {
878                                message: format!(
879                                    "[Cached {} chars from {} as {}]",
880                                    tool_result.len(),
881                                    tool_call.name,
882                                    entry.metadata.id
883                                ),
884                                verbose_only: true,
885                            })?;
886                        }
887                        (truncated, true)
888                    }
889                    Err(_) => (tool_result.clone(), false),
890                }
891            }
892            Err(e) => {
893                if verbose {
894                    sink.handle(ResponseEvent::Diagnostic {
895                        message: format!("[Failed to cache output: {}]", e),
896                        verbose_only: true,
897                    })?;
898                }
899                (tool_result.clone(), false)
900            }
901        }
902    } else {
903        (tool_result.clone(), false)
904    };
905
906    // Execute post_tool_output hooks
907    let post_output_hook_data = serde_json::json!({
908        "tool_name": tool_call.name,
909        "arguments": args,
910        "output": tool_result,
911        "final_output": final_result,
912        "cached": was_cached,
913    });
914    let _ = tools::execute_hook(
915        tools,
916        tools::HookPoint::PostToolOutput,
917        &post_output_hook_data,
918    );
919
920    Ok(ToolExecutionResult {
921        final_result,
922        original_result: tool_result,
923        was_cached,
924    })
925}

927/// Execute the send_message built-in tool.
928#[allow(clippy::too_many_arguments)]
929fn execute_send_message_tool<S: ResponseSink>(
930    app: &AppState,
931    context_name: &str,
932    tools: &[Tool],
933    args: &serde_json::Value,
934    verbose: bool,
935    sink: &mut S,
936) -> io::Result<String> {
937    let to = args.get_str_or("to", "");
938    let content = args.get_str_or("content", "");
939    let from = args.get_str("from").unwrap_or(context_name);
940
941    if to.is_empty() {
942        return Ok("Error: 'to' field is required".to_string());
943    }
944    if content.is_empty() {
945        return Ok("Error: 'content' field is required".to_string());
946    }
947
948    // Execute pre_send_message hooks
949    let pre_hook_data = serde_json::json!({
950        "from": from,
951        "to": to,
952        "content": content,
953        "context_name": context_name,
954    });
955    let pre_hook_results =
956        tools::execute_hook(tools, tools::HookPoint::PreSendMessage, &pre_hook_data)?;
957
958    // Check if any hook claimed delivery
959    let mut delivered_via: Option<String> = None;
960    for (hook_tool_name, hook_result) in &pre_hook_results {
961        if hook_result.get_bool_or("delivered", false) {
962            let via = hook_result.get_str_or("via", hook_tool_name);
963            delivered_via = Some(via.to_string());
964            if verbose {
965                sink.handle(ResponseEvent::Diagnostic {
966                    message: format!(
967                        "[Hook pre_send_message: {} intercepted delivery]",
968                        hook_tool_name
969                    ),
970                    verbose_only: true,
971                })?;
972            }
973            break;
974        }
975    }
976
977    let delivery_result = if let Some(via) = delivered_via {
978        format!("Message delivered to '{}' via {}", to, via)
979    } else {
980        let entry = InboxEntry {
981            id: Uuid::new_v4().to_string(),
982            timestamp: now_timestamp(),
983            from: from.to_string(),
984            to: to.to_string(),
985            content: content.to_string(),
986        };
987        match app.append_to_inbox(to, &entry) {
988            Ok(()) => format!("Message delivered to '{}' via local inbox", to),
989            Err(e) => format!("Error delivering message: {}", e),
990        }
991    };
992
993    // Execute post_send_message hooks
994    let post_hook_data = serde_json::json!({
995        "from": from,
996        "to": to,
997        "content": content,
998        "context_name": context_name,
999        "delivery_result": delivery_result,
1000    });
1001    let _ = tools::execute_hook(tools, tools::HookPoint::PostSendMessage, &post_hook_data);
1002
1003    Ok(delivery_result)
1004}

1006/// Process all tool calls from a response.
1007///
1008/// Handles: Tool call JSON conversion, loop over execute_single_tool,
1009/// transcript entries, post_tool_batch hook.
1010#[allow(clippy::too_many_arguments)]
1011fn process_tool_calls<S: ResponseSink>(
1012    app: &AppState,
1013    context_name: &str,
1014    tool_calls: &[llm::ToolCallAccumulator],
1015    messages: &mut Vec<serde_json::Value>,
1016    tools: &[Tool],
1017    handoff: &mut tools::Handoff,
1018    use_reflection: bool,
1019    resolved_config: &ResolvedConfig,
1020    recursion_depth: usize,
1021    verbose: bool,
1022    sink: &mut S,
1023) -> io::Result<()> {
1024    // Convert tool calls to JSON format for the assistant message
1025    let tool_calls_json: Vec<serde_json::Value> = tool_calls
1026        .iter()
1027        .map(|tc| {
1028            serde_json::json!({
1029                "id": tc.id,
1030                "type": "function",
1031                "function": {
1032                    "name": tc.name,
1033                    "arguments": tc.arguments,
1034                }
1035            })
1036        })
1037        .collect();
1038
1039    messages.push(serde_json::json!({
1040        "role": "assistant",
1041        "tool_calls": tool_calls_json,
1042    }));
1043
1044    // Execute each tool and add results
1045    for tc in tool_calls {
1046        if verbose {
1047            sink.handle(ResponseEvent::Diagnostic {
1048                message: format!("[Tool: {}]", tc.name),
1049                verbose_only: true,
1050            })?;
1051        }
1052
1053        sink.handle(ResponseEvent::ToolStart {
1054            name: tc.name.clone(),
1055        })?;
1056
1057        let result = execute_single_tool(
1058            app,
1059            context_name,
1060            tc,
1061            tools,
1062            handoff,
1063            use_reflection,
1064            resolved_config,
1065            verbose,
1066            sink,
1067        )?;
1068
1069        // Log tool call and result
1070        let tool_call_entry = create_tool_call_entry(context_name, &tc.name, &tc.arguments);
1071        app.append_to_transcript_and_context(context_name, &tool_call_entry)?;
1072        sink.handle(ResponseEvent::TranscriptEntry(tool_call_entry))?;
1073
1074        let logged_result = if result.was_cached {
1075            &result.final_result
1076        } else {
1077            &result.original_result
1078        };
1079        let tool_result_entry = create_tool_result_entry(context_name, &tc.name, logged_result);
1080        app.append_to_transcript_and_context(context_name, &tool_result_entry)?;
1081        sink.handle(ResponseEvent::TranscriptEntry(tool_result_entry))?;
1082
1083        sink.handle(ResponseEvent::ToolResult {
1084            name: tc.name.clone(),
1085            result: result.final_result.clone(),
1086            cached: result.was_cached,
1087        })?;
1088
1089        // Execute post_tool hooks
1090        let args: serde_json::Value =
1091            serde_json::from_str(&tc.arguments).unwrap_or(serde_json::json!({}));
1092        let post_hook_data = serde_json::json!({
1093            "tool_name": tc.name,
1094            "arguments": args,
1095            "result": result.original_result,
1096            "cached": result.was_cached,
1097        });
1098        let _ = tools::execute_hook(tools, tools::HookPoint::PostTool, &post_hook_data);
1099
1100        messages.push(serde_json::json!({
1101            "role": "tool",
1102            "tool_call_id": tc.id,
1103            "content": result.final_result,
1104        }));
1105    }
1106
1107    // Execute post_tool_batch hook - allows plugins to override fallback after seeing tool results
1108    let tool_batch_info: Vec<serde_json::Value> = tool_calls
1109        .iter()
1110        .map(|tc| {
1111            json!({
1112                "name": tc.name,
1113                "arguments": serde_json::from_str::<serde_json::Value>(&tc.arguments).unwrap_or(json!({})),
1114            })
1115        })
1116        .collect();
1117    let hook_data = json!({
1118        "context_name": context_name,
1119        "recursion_depth": recursion_depth,
1120        "current_fallback": resolved_config.fallback_tool,
1121        "tool_calls": tool_batch_info,
1122    });
1123    let hook_results = tools::execute_hook(tools, tools::HookPoint::PostToolBatch, &hook_data)?;
1124    apply_fallback_override(handoff, &hook_results, verbose, sink)?;
1125
1126    Ok(())
1127}

1129/// Result of handling a final (non-tool-call) response.
1130enum FinalResponseAction {
1131    /// Return to the user (turn ended).
1132    ReturnToUser,
1133    /// Continue with another prompt (agent loop).
1134    ContinueWithPrompt(String),
1135}
1136
1137/// Handle the final text response from the LLM.
1138///
1139/// Handles: Empty response detection/retry, save assistant message,
1140/// post_message hook, handoff decision (return vs recurse).
1141#[allow(clippy::too_many_arguments)]
1142fn handle_final_response<S: ResponseSink>(
1143    app: &AppState,
1144    context_name: &str,
1145    full_response: &str,
1146    final_prompt: &str,
1147    mut handoff: tools::Handoff,
1148    tools: &[Tool],
1149    recursion_depth: usize,
1150    resolved_config: &ResolvedConfig,
1151    verbose: bool,
1152    sink: &mut S,
1153) -> io::Result<FinalResponseAction> {
1154    // Get or create context to add the message
1155    let mut context = app.get_or_create_context(context_name)?;
1156
1157    // Save the assistant's response
1158    app.add_message(
1159        &mut context,
1160        "assistant".to_string(),
1161        full_response.to_string(),
1162    );
1163
1164    let assistant_entry = create_assistant_message_entry(context_name, full_response);
1165    app.append_to_transcript_and_context(context_name, &assistant_entry)?;
1166    sink.handle(ResponseEvent::TranscriptEntry(assistant_entry))?;
1167
1168    // Execute post_message hooks
1169    let hook_data = serde_json::json!({
1170        "prompt": final_prompt,
1171        "response": full_response,
1172        "context_name": context_name,
1173    });
1174    let _ = tools::execute_hook(tools, tools::HookPoint::PostMessage, &hook_data);
1175
1176    if app.should_warn(&context.messages) {
1177        let remaining = app.remaining_tokens(&context.messages);
1178        if verbose {
1179            sink.handle(ResponseEvent::Diagnostic {
1180                message: format!("[Context window warning: {} tokens remaining]", remaining),
1181                verbose_only: true,
1182            })?;
1183        }
1184    }
1185
1186    sink.handle(ResponseEvent::Newline)?;
1187
1188    // Determine next action based on handoff
1189    match handoff.take() {
1190        tools::HandoffTarget::User { message } => {
1191            if !message.is_empty() {
1192                // Output the message as final text
1193                sink.handle(ResponseEvent::TextChunk(&message))?;
1194                sink.handle(ResponseEvent::Newline)?;
1195            }
1196            Ok(FinalResponseAction::ReturnToUser)
1197        }
1198        tools::HandoffTarget::Agent { prompt } => {
1199            let new_depth = recursion_depth + 1;
1200            if new_depth >= app.config.max_recursion_depth {
1201                sink.handle(ResponseEvent::Diagnostic {
1202                    message: format!(
1203                        "[Max recursion depth ({}) reached, stopping]",
1204                        app.config.max_recursion_depth
1205                    ),
1206                    verbose_only: false,
1207                })?;
1208                return Ok(FinalResponseAction::ReturnToUser);
1209            }
1210            if verbose {
1211                sink.handle(ResponseEvent::Diagnostic {
1212                    message: format!(
1213                        "[Continuing processing ({}/{}): {}]",
1214                        new_depth,
1215                        app.config.max_recursion_depth,
1216                        if prompt.is_empty() {
1217                            "(no prompt)"
1218                        } else {
1219                            &prompt
1220                        }
1221                    ),
1222                    verbose_only: true,
1223                })?;
1224            }
1225            let continue_prompt = if prompt.is_empty() {
1226                format!(
1227                    "[Reengaged ({}/{}) via {} tool. call_user(<message>) to end turn.]",
1228                    new_depth, app.config.max_recursion_depth, resolved_config.fallback_tool
1229                )
1230            } else {
1231                format!(
1232                    "[Reengaged ({}/{}) via {} tool: {}]",
1233                    new_depth,
1234                    app.config.max_recursion_depth,
1235                    resolved_config.fallback_tool,
1236                    prompt
1237                )
1238            };
1239            Ok(FinalResponseAction::ContinueWithPrompt(continue_prompt))
1240        }
1241    }
1242}

1244#[allow(clippy::too_many_arguments)]
1245async fn send_prompt_with_depth<S: ResponseSink>(
1246    app: &AppState,
1247    context_name: &str,
1248    prompt: String,
1249    tools: &[Tool],
1250    recursion_depth: usize,
1251    resolved_config: &ResolvedConfig,
1252    options: &PromptOptions<'_>,
1253    sink: &mut S,
1254) -> io::Result<()> {
1255    // === Validation & Setup ===
1256    if prompt.trim().is_empty() {
1257        return Err(io::Error::new(
1258            ErrorKind::InvalidInput,
1259            "Prompt cannot be empty",
1260        ));
1261    }
1262
1263    app.validate_config(resolved_config, tools)?;
1264
1265    let verbose = options.verbose;
1266    let use_reflection = options.use_reflection;
1267    let debug = options.debug;
1268
1269    let mut context = app.get_or_create_context(context_name)?;
1270
1271    // === Pre-message Hooks & Inbox ===
1272    let mut final_prompt = prompt.clone();
1273    let hook_data = serde_json::json!({
1274        "prompt": prompt,
1275        "context_name": context.name,
1276        "summary": context.summary,
1277    });
1278    let hook_results = tools::execute_hook(tools, tools::HookPoint::PreMessage, &hook_data)?;
1279    for (tool_name, result) in hook_results {
1280        if let Some(modified) = result.get_str("prompt") {
1281            if verbose {
1282                eprintln!("[Hook pre_message: {} modified prompt]", tool_name);
1283            }
1284            final_prompt = modified.to_string();
1285        }
1286    }
1287
1288    // Check inbox and inject messages before the user prompt
1289    let inbox_messages = app.load_and_clear_inbox(context_name)?;
1290    if !inbox_messages.is_empty() {
1291        let mut inbox_content = String::from("--- INBOX MESSAGES ---\n");
1292        for msg in &inbox_messages {
1293            inbox_content.push_str(&format!("[From: {}] {}\n", msg.from, msg.content));
1294        }
1295        inbox_content.push_str("--- END INBOX ---\n\n");
1296        final_prompt = format!("{}{}", inbox_content, final_prompt);
1297        if verbose {
1298            sink.handle(ResponseEvent::Diagnostic {
1299                message: format!("[Inbox: {} message(s) injected]", inbox_messages.len()),
1300                verbose_only: true,
1301            })?;
1302        }
1303    }
1304
1305    // Add datetime prefix to user message
1306    let datetime_prefix = chrono::Local::now().format("%Y%m%d-%H%M%z").to_string();
1307    let prefixed_prompt = format!("[{}] {}", datetime_prefix, final_prompt);
1308
1309    // Add user message to context and transcript
1310    app.add_message(&mut context, "user".to_string(), prefixed_prompt.clone());
1311    let user_entry =
1312        create_user_message_entry(context_name, &prefixed_prompt, &resolved_config.username);
1313    app.append_to_transcript_and_context(context_name, &user_entry)?;
1314    sink.handle(ResponseEvent::TranscriptEntry(user_entry))?;
1315
1316    // Context window warning
1317    if app.should_warn(&context.messages) {
1318        let remaining = app.remaining_tokens(&context.messages);
1319        if verbose {
1320            sink.handle(ResponseEvent::Diagnostic {
1321                message: format!("[Context window warning: {} tokens remaining]", remaining),
1322                verbose_only: true,
1323            })?;
1324        }
1325    }
1326
1327    // === Auto-compaction Check ===
1328    if app.should_auto_compact(&context, resolved_config) {
1329        return compact_context_with_llm(app, context_name, resolved_config, verbose).await;
1330    }
1331
1332    // === Build System Prompt ===
1333    let full_system_prompt = build_full_system_prompt(
1334        app,
1335        context_name,
1336        &context.summary,
1337        use_reflection,
1338        tools,
1339        resolved_config,
1340        verbose,
1341        sink,
1342    )?;
1343
1344    // === Prepare Messages ===
1345    let mut messages: Vec<serde_json::Value> = if !full_system_prompt.is_empty() {
1346        vec![serde_json::json!({
1347            "role": "system",
1348            "content": full_system_prompt,
1349        })]
1350    } else {
1351        Vec::new()
1352    };
1353
1354    // Add conversation messages (skip system messages)
1355    for m in &context.messages {
1356        if m.role == "system" {
1357            continue;
1358        }
1359        messages.push(serde_json::json!({
1360            "role": m.role,
1361            "content": m.content,
1362        }));
1363    }
1364
1365    // === Prepare Tools ===
1366    let mut all_tools = tools::tools_to_api_format(tools);
1367    all_tools.extend(tools::builtin_tools_to_api_format(use_reflection));
1368    all_tools.extend(tools::all_file_tools_to_api_format());
1369    annotate_fallback_tool(&mut all_tools, &resolved_config.fallback_tool);
1370    all_tools = filter_tools_by_config(all_tools, &resolved_config.tools);
1371
1372    // Execute pre_api_tools hook
1373    let tool_info = build_tool_info_list(&all_tools, tools);
1374    let hook_data = json!({
1375        "context_name": context.name,
1376        "tools": tool_info,
1377        "recursion_depth": recursion_depth,
1378    });
1379    let hook_results = tools::execute_hook(tools, tools::HookPoint::PreApiTools, &hook_data)?;
1380    all_tools = filter_tools_from_hook_results(all_tools, &hook_results, verbose, sink)?;
1381
1382    // === Build Request ===
1383    let mut request_body = build_request_body(resolved_config, &messages, Some(&all_tools), true);
1384
1385    // Execute pre_api_request hook
1386    let hook_data = json!({
1387        "context_name": context.name,
1388        "request_body": request_body,
1389        "recursion_depth": recursion_depth,
1390    });
1391    let hook_results = tools::execute_hook(tools, tools::HookPoint::PreApiRequest, &hook_data)?;
1392    request_body = apply_request_modifications(request_body, &hook_results, verbose, sink)?;
1393
1394    // === Initialize Handoff ===
1395    let fallback = options.fallback_override.clone().unwrap_or_else(|| {
1396        let meta = tools::get_tool_metadata(tools, &resolved_config.fallback_tool);
1397        if meta.ends_turn {
1398            tools::HandoffTarget::User {
1399                message: String::new(),
1400            }
1401        } else {
1402            tools::HandoffTarget::Agent {
1403                prompt: String::new(),
1404            }
1405        }
1406    });
1407    let mut handoff = tools::Handoff::new(fallback);
1408
1409    // Execute pre_agentic_loop hook
1410    let hook_data = json!({
1411        "context_name": context.name,
1412        "recursion_depth": recursion_depth,
1413        "current_fallback": resolved_config.fallback_tool,
1414        "message": final_prompt,
1415    });
1416    let hook_results = tools::execute_hook(tools, tools::HookPoint::PreAgenticLoop, &hook_data)?;
1417    apply_fallback_override(&mut handoff, &hook_results, verbose, sink)?;
1418
1419    // === Main Loop ===
1420    let mut consecutive_empty_responses = 0usize;
1421    loop {
1422        sink.handle(ResponseEvent::StartResponse)?;
1423        log_request_if_enabled(app, context_name, debug, &request_body);
1424
1425        let response =
1426            collect_streaming_response(resolved_config, &messages, &all_tools, verbose, sink)
1427                .await?;
1428
1429        // Log response metadata
1430        if let Some(ref meta) = response.response_meta {
1431            log_response_meta_if_enabled(app, context_name, debug, meta);
1432        }
1433
1434        // Signal streaming finished
1435        if !sink.is_json_mode() {
1436            sink.handle(ResponseEvent::Finished)?;
1437        }
1438
1439        // Handle tool calls
1440        if response.has_tool_calls && !response.tool_calls.is_empty() {
1441            process_tool_calls(
1442                app,
1443                context_name,
1444                &response.tool_calls,
1445                &mut messages,
1446                tools,
1447                &mut handoff,
1448                use_reflection,
1449                resolved_config,
1450                recursion_depth,
1451                verbose,
1452                sink,
1453            )?;
1454
1455            // Keep request_body in sync for logging
1456            request_body["messages"] = serde_json::json!(messages);
1457            consecutive_empty_responses = 0;
1458            continue;
1459        }
1460
1461        // Check for empty response
1462        if response.full_response.trim().is_empty() {
1463            consecutive_empty_responses += 1;
1464            if consecutive_empty_responses >= app.config.max_empty_responses {
1465                sink.handle(ResponseEvent::Diagnostic {
1466                    message: format!(
1467                        "[Max consecutive empty responses ({}) reached, stopping]",
1468                        app.config.max_empty_responses
1469                    ),
1470                    verbose_only: false,
1471                })?;
1472                return Ok(());
1473            }
1474            if verbose {
1475                sink.handle(ResponseEvent::Diagnostic {
1476                    message: format!(
1477                        "[Empty response {}/{}, retrying]",
1478                        consecutive_empty_responses, app.config.max_empty_responses
1479                    ),
1480                    verbose_only: true,
1481                })?;
1482            }
1483            continue;
1484        }
1485
1486        // Handle final response
1487        match handle_final_response(
1488            app,
1489            context_name,
1490            &response.full_response,
1491            &final_prompt,
1492            handoff,
1493            tools,
1494            recursion_depth,
1495            resolved_config,
1496            verbose,
1497            sink,
1498        )? {
1499            FinalResponseAction::ReturnToUser => return Ok(()),
1500            FinalResponseAction::ContinueWithPrompt(continue_prompt) => {
1501                return Box::pin(send_prompt_with_depth(
1502                    app,
1503                    context_name,
1504                    continue_prompt,
1505                    tools,
1506                    recursion_depth + 1,
1507                    resolved_config,
1508                    options,
1509                    sink,
1510                ))
1511                .await;
1512            }
1513        }
1514    }
1515}
