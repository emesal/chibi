1//! Context CRUD operations for AppState.
2//!
3//! Methods for creating, loading, saving, clearing, destroying, renaming, and listing contexts.

5use crate::context::{Context, ContextEntry, now_timestamp, validate_context_name};
6use std::fs;
7use std::io::{self, ErrorKind};

9use super::{AppState, StatePaths, create_archival_anchor};

11impl AppState {
12    /// Get or create a context by name.
13    ///
14    /// Returns an existing context if found, or creates a new empty one.
15    /// This is the parameterized version of `get_current_context`.
16    pub fn get_or_create_context(&self, name: &str) -> io::Result<Context> {
17        self.load_context(name).or_else(|e| {
18            if e.kind() == ErrorKind::NotFound {
19                // Return empty context if it doesn't exist yet
20                Ok(Context::new(name.to_string()))
21            } else {
22                Err(e)
23            }
24        })
25    }
26
27    pub fn save_and_register_context(&self, context: &Context) -> io::Result<()> {
28        self.save_context(context)?;
29
30        // Ensure the context is tracked in state.
31        // Important: Read state from disk to get the authoritative list of contexts,
32        // rather than using in-memory state which may be stale.
33        if !self.state.contexts.iter().any(|e| e.name == context.name) {
34            let disk_state = if self.state_path.exists() {
35                let content = fs::read_to_string(&self.state_path)?;
36                serde_json::from_str(&content).unwrap_or_else(|_| self.state.clone())
37            } else {
38                self.state.clone()
39            };
40            let mut new_state = disk_state;
41            if !new_state.contexts.iter().any(|e| e.name == context.name) {
42                new_state.contexts.push(ContextEntry::with_created_at(
43                    context.name.clone(),
44                    context.created_at,
45                ));
46            }
47            new_state.save(&self.state_path)?;
48        }
49
50        Ok(())
51    }

53    pub fn add_message(&self, context: &mut Context, role: String, content: String) {
54        context.messages.push(serde_json::json!({
55            "_id": uuid::Uuid::new_v4().to_string(),
56            "role": role,
57            "content": content,
58        }));
59        context.updated_at = now_timestamp();
60    }

62    /// Clear the current context (archive history).
63    ///
64    /// # Concurrency
65    ///
66    /// This function is safe to call without holding a `ContextLock` because:
67    /// - Transcript writes use `FileLock` via `append_to_transcript`
68    /// - Context saves use atomic writes via `save_context`
69    ///
70    /// If another process is running `send_prompt`, the transcript operations
71    /// will serialize correctly via their respective locks.
72    pub fn clear_context(&self, context_name: &str) -> io::Result<()> {
73        let context = self.get_or_create_context(context_name)?;
74
75        // Don't clear if already empty
76        if context.messages.is_empty() {
77            return Ok(());
78        }
79
80        // Write archival anchor to transcript.jsonl
81        let archival_anchor = create_archival_anchor(&context.name);
82        self.append_to_transcript(&context.name, &archival_anchor)?;
83
84        // Mark context dirty so it rebuilds with new anchor on next load
85        self.mark_context_dirty(&context.name)?;
86
87        // Create fresh context (preserving nothing - full clear)
88        let new_context = Context::new(context_name.to_string());
89        self.save_context(&new_context)?;
90
91        // Invalidate active state cache after all writes are complete,
92        // so the next append re-scans fresh state from disk
93        self.active_state_cache.borrow_mut().remove(&context.name);
94        Ok(())
95    }
96
97    /// Destroy a context and its directory.
98    ///
99    /// Returns `true` if the context existed and was destroyed, `false` if it didn't exist.
100    ///
101    /// Note: Session state (what was current/previous) is managed by CLI. This method
102    /// just deletes the context. Caller is responsible for updating session if needed.
103    pub fn destroy_context(&mut self, name: &str) -> io::Result<bool> {
104        let dir = self.context_dir(name);
105        if !dir.exists() {
106            return Ok(false);
107        }
108
109        // Remove the directory
110        fs::remove_dir_all(&dir)?;
111
112        // Invalidate active state cache for the destroyed context
113        self.active_state_cache.borrow_mut().remove(name);
114
115        // Update state - remove from contexts list and save
116        self.state.contexts.retain(|e| e.name != name);
117        self.state.save(&self.state_path)?;
118
119        Ok(true)
120    }

122    pub fn rename_context(&mut self, old_name: &str, new_name: &str) -> io::Result<()> {
123        validate_context_name(new_name)?;
124
125        let old_dir = self.context_dir(old_name);
126        let new_dir = self.context_dir(new_name);
127
128        if !old_dir.exists() {
129            return Err(io::Error::new(
130                ErrorKind::NotFound,
131                format!("Context '{}' does not exist", old_name),
132            ));
133        }
134
135        if new_dir.exists() {
136            return Err(io::Error::new(
137                ErrorKind::AlreadyExists,
138                format!("Context '{}' already exists", new_name),
139            ));
140        }
141
142        // Rename the directory (context name is derived from directory, no file updates needed)
143        fs::rename(&old_dir, &new_dir)?;
144
145        // Update state: preserve created_at from old entry
146        let created_at = self
147            .state
148            .contexts
149            .iter()
150            .find(|e| e.name == old_name)
151            .map(|e| e.created_at)
152            .unwrap_or_else(now_timestamp);
153
154        self.state.contexts.retain(|e| e.name != old_name);
155        if !self.state.contexts.iter().any(|e| e.name == new_name) {
156            self.state
157                .contexts
158                .push(ContextEntry::with_created_at(new_name, created_at));
159        }
160        self.state.save(&self.state_path)?;
161
162        // Note: If CLI's session.current_context was renamed, CLI must update it.
163        Ok(())
164    }
165
166    pub fn list_contexts(&self) -> Vec<String> {
167        // state.json is the single source of truth (synced with filesystem on startup)
168        self.state.contexts.iter().map(|e| e.name.clone()).collect()
169    }
