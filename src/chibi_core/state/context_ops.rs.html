12    /// Get or create a context by name.
13    ///
14    /// Returns an existing context if found, or creates a new empty one.
15    /// This is the parameterized version of `get_current_context`.
16    pub fn get_or_create_context(&self, name: &str) -> io::Result<Context> {
17        self.load_context(name).or_else(|e| {
18            if e.kind() == ErrorKind::NotFound {
19                // Return empty context if it doesn't exist yet
20                Ok(Context::new(name.to_string()))
21            } else {
22                Err(e)
23            }
24        })
25    }

27    /// Save a context and ensure it is registered in state.json.
28    ///
29    /// Note: this updates disk state only — `self.state.contexts` is not mutated
30    /// (AppState is `&self`). Callers that need the updated context list should
31    /// use `sync_state_with_filesystem()` to reload from disk.
32    pub fn save_and_register_context(&self, context: &Context) -> io::Result<()> {
33        self.save_context(context)?;
34
35        // Ensure the context is tracked in state.
36        // Read state from disk to get the authoritative list of contexts,
37        // rather than using in-memory state which may be stale.
38        let already_known = {
39            let state = self.state.read().unwrap();
40            state.contexts.iter().any(|e| e.name == context.name)
41        };
42        if !already_known {
43            let disk_state = if self.state_path.exists() {
44                let content = fs::read_to_string(&self.state_path)?;
45                let state = self.state.read().unwrap();
46                serde_json::from_str(&content).unwrap_or_else(|_| state.clone())
47            } else {
48                let state = self.state.read().unwrap();
49                state.clone()
50            };
51            let mut new_state = disk_state;
52            if !new_state.contexts.iter().any(|e| e.name == context.name) {
53                new_state.contexts.push(ContextEntry::with_created_at(
54                    context.name.clone(),
55                    context.created_at,
56                ));
57            }
58            new_state.save(&self.state_path)?;
59        }
60
61        Ok(())
62    }

64    pub fn add_message(&self, context: &mut Context, role: String, content: String) {
65        context.messages.push(serde_json::json!({
66            "_id": uuid::Uuid::new_v4().to_string(),
67            "role": role,
68            "content": content,
69        }));
70        context.updated_at = now_timestamp();
71    }

73    /// Clear the current context (archive history).
74    ///
75    /// # Concurrency
76    ///
77    /// This function is safe to call without holding a `ContextLock` because:
78    /// - Transcript writes use `FileLock` via `append_to_transcript`
79    /// - Context saves use atomic writes via `save_context`
80    ///
81    /// If another process is running `send_prompt`, the transcript operations
82    /// will serialize correctly via their respective locks.
83    pub fn clear_context(&self, context_name: &str) -> io::Result<()> {
84        let context = self.get_or_create_context(context_name)?;
85
86        // Don't clear if already empty
87        if context.messages.is_empty() {
88            return Ok(());
89        }
90
91        // Write archival anchor to transcript.jsonl
92        let archival_anchor = create_archival_anchor(&context.name);
93        self.append_to_transcript(&context.name, &archival_anchor)?;
94
95        // Mark context dirty so it rebuilds with new anchor on next load
96        self.mark_context_dirty(&context.name)?;
97
98        // Create fresh context (preserving nothing - full clear)
99        let new_context = Context::new(context_name.to_string());
100        self.save_context(&new_context)?;
101
102        // Evict cached scheme_eval tein session so the next eval call gets a
103        // fresh interpreter with the current prelude (prevents stale C-level
104        // state from surviving across context clears).
105        crate::tools::evict_eval_context(context_name);
106
107        // Invalidate active state cache after all writes are complete,
108        // so the next append re-scans fresh state from disk
109        self.active_state_cache.borrow_mut().remove(&context.name);
110        Ok(())
111    }

113    /// Destroy a context and its directory.
114    ///
115    /// Returns `true` if the context existed and was destroyed, `false` if it didn't exist.
116    ///
117    /// Note: Session state (what was current/previous) is managed by CLI. This method
118    /// just deletes the context. Caller is responsible for updating session if needed.
119    pub fn destroy_context(&mut self, name: &str) -> io::Result<bool> {
120        let dir = self.context_dir(name);
121        if !dir.exists() {
122            return Ok(false);
123        }
124
125        // Remove the directory
126        fs::remove_dir_all(&dir)?;
127
128        // Evict cached scheme_eval tein session (see clear_context).
129        crate::tools::evict_eval_context(name);
130
131        // Invalidate active state cache for the destroyed context
132        self.active_state_cache.borrow_mut().remove(name);
133
134        // Update state - remove from contexts list and save
135        {
136            let mut state = self.state.write().unwrap();
137            state.contexts.retain(|e| e.name != name);
138        }
139        self.save()?;
140
141        Ok(true)
142    }

144    pub fn rename_context(&mut self, old_name: &str, new_name: &str) -> io::Result<()> {
145        validate_context_name(new_name)?;
146
147        let old_dir = self.context_dir(old_name);
148        let new_dir = self.context_dir(new_name);
149
150        if !old_dir.exists() {
151            return Err(io::Error::new(
152                ErrorKind::NotFound,
153                format!("Context '{}' does not exist", old_name),
154            ));
155        }
156
157        if new_dir.exists() {
158            return Err(io::Error::new(
159                ErrorKind::AlreadyExists,
160                format!("Context '{}' already exists", new_name),
161            ));
162        }
163
164        // Rename the directory (context name is derived from directory, no file updates needed)
165        fs::rename(&old_dir, &new_dir)?;
166
167        // Evict cached scheme_eval tein session for old name (see clear_context).
168        // Don't insert under new name — let it lazily recreate on next eval.
169        crate::tools::evict_eval_context(old_name);
170
171        // Update state: preserve created_at from old entry.
172        // Drop write guard before file I/O (atomic_write_json) to avoid
173        // blocking the tokio executor while holding a global write lock.
174        {
175            let mut state = self.state.write().unwrap();
176            let created_at = state
177                .contexts
178                .iter()
179                .find(|e| e.name == old_name)
180                .map(|e| e.created_at)
181                .unwrap_or_else(now_timestamp);
182
183            state.contexts.retain(|e| e.name != old_name);
184            if !state.contexts.iter().any(|e| e.name == new_name) {
185                state
186                    .contexts
187                    .push(ContextEntry::with_created_at(new_name, created_at));
188            }
189        } // write guard dropped here
190        self.save()?;
191
192        // Note: If CLI's session.current_context was renamed, CLI must update it.
193        Ok(())
194    }

197        // state.json is the single source of truth (synced with filesystem on startup)
198        let state = self.state.read().unwrap();
199        state.contexts.iter().map(|e| e.name.clone()).collect()
