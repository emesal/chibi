12/// Compiled-in default prompts, embedded from crates/chibi-core/prompts/.
13/// Users override these by placing a file at ~/.chibi/prompts/<name>.md.
14const DEFAULT_PROMPT_CHIBI: &str = include_str!("../../prompts/chibi.md");
15const DEFAULT_PROMPT_COMPACTION: &str = include_str!("../../prompts/compaction.md");
16const DEFAULT_PROMPT_CONTINUATION: &str = include_str!("../../prompts/continuation.md");
17
18/// Return the compiled-in default for a named prompt, or empty string if none exists.
19fn default_prompt(name: &str) -> &'static str {
20    match name {
21        "chibi" => DEFAULT_PROMPT_CHIBI,
22        "compaction" => DEFAULT_PROMPT_COMPACTION,
23        "continuation" => DEFAULT_PROMPT_CONTINUATION,
24        _ => "",
25    }
26}

29    /// Load a named prompt from ~/.chibi/prompts/<name>.md, falling back to the
30    /// compiled-in default if no override is installed.
31    pub fn load_prompt(&self, name: &str) -> io::Result<String> {
32        let prompt_path = self.prompts_dir.join(format!("{}.md", name));
33        if prompt_path.exists() {
34            return fs::read_to_string(&prompt_path);
35        }
36        Ok(default_prompt(name).to_string())
37    }

42    /// Load the system prompt for a specific context.
43    pub fn load_system_prompt_for(&self, context_name: &str) -> io::Result<String> {
44        let context_prompt_path = self.context_prompt_file(context_name);
45        if context_prompt_path.exists() {
46            fs::read_to_string(&context_prompt_path)
47        } else {
48            // Fall back to default prompt
49            self.load_prompt("chibi")
50        }
51    }

53    /// Set a custom system prompt for a specific context.
54    /// This also logs a system_prompt_changed event (with full content) to transcript,
55    /// updates the mtime in context_meta, and invalidates the prefix.
56    pub fn set_system_prompt_for(&self, context_name: &str, content: &str) -> io::Result<()> {
57        self.ensure_context_dir(context_name)?;
58        let prompt_path = self.context_prompt_file(context_name);
59
60        // Check if content actually changed
61        let old_content = if prompt_path.exists() {
62            fs::read_to_string(&prompt_path).ok()
63        } else {
64            None
65        };
66
67        crate::safe_io::atomic_write_text(&prompt_path, content)?;
68
69        // Only log change and invalidate if content actually changed
70        if old_content.as_deref() != Some(content) {
71            // Log system_prompt_changed event to transcript with full raw prompt content
72            let entry = TranscriptEntry {
73                id: Uuid::new_v4().to_string(),
74                timestamp: now_timestamp(),
75                from: "system".to_string(),
76                to: context_name.to_string(),
77                content: content.to_string(), // Full raw prompt for history
78                entry_type: ENTRY_TYPE_SYSTEM_PROMPT_CHANGED.to_string(),
79                metadata: None,
80                tool_call_id: None,
81            };
82            self.append_to_transcript(context_name, &entry)?;
83
84            // Update mtime in context_meta
85            self.update_system_prompt_mtime(context_name)?;
86
87            // Invalidate prefix so context.jsonl will be rebuilt on next load
88            self.mark_context_dirty(context_name)?;
89        }
90
91        Ok(())
92    }

94    /// Update system_prompt_md_mtime in context_meta to current file mtime
95    fn update_system_prompt_mtime(&self, context_name: &str) -> io::Result<()> {
96        let prompt_path = self.context_prompt_file(context_name);
97        let mtime = self.get_file_mtime(&prompt_path);
98
99        let mut meta = self.load_context_meta(context_name).unwrap_or_default();
100        meta.system_prompt_md_mtime = mtime;
101        self.save_context_meta(context_name, &meta)
102    }

104    /// Get file mtime as unix timestamp, or None if file doesn't exist
105    fn get_file_mtime(&self, path: &std::path::Path) -> Option<u64> {
106        if path.exists() {
107            fs::metadata(path)
108                .and_then(|m| m.modified())
109                .ok()
110                .and_then(|t| t.duration_since(std::time::UNIX_EPOCH).ok())
111                .map(|d| d.as_secs())
112        } else {
113            None
114        }
115    }

117    /// Check if system_prompt.md was modified externally since last tracked mtime.
118    /// If so, log the new prompt to transcript, update mtime, and mark context dirty.
119    pub(crate) fn check_system_prompt_mtime_change(&self, context_name: &str) -> io::Result<()> {
120        let prompt_path = self.context_prompt_file(context_name);
121        let current_mtime = self.get_file_mtime(&prompt_path);
122
123        // Load stored mtime from context_meta
124        let meta = self.load_context_meta(context_name).unwrap_or_default();
125        let stored_mtime = meta.system_prompt_md_mtime;
126
127        // If mtimes differ (including None -> Some or Some -> None), prompt changed
128        if current_mtime != stored_mtime {
129            // Only log if the file actually exists (has content to log)
130            if prompt_path.exists() {
131                let content = fs::read_to_string(&prompt_path)?;
132
133                // Log system_prompt_changed event to transcript with full raw prompt
134                let entry = TranscriptEntry {
135                    id: Uuid::new_v4().to_string(),
136                    timestamp: now_timestamp(),
137                    from: "system".to_string(),
138                    to: context_name.to_string(),
139                    content, // Full raw prompt for history
140                    entry_type: ENTRY_TYPE_SYSTEM_PROMPT_CHANGED.to_string(),
141                    metadata: None,
142                    tool_call_id: None,
143                };
144                self.append_to_transcript(context_name, &entry)?;
145            }
146
147            // Update mtime in context_meta
148            self.update_system_prompt_mtime(context_name)?;
149
150            // Mark context dirty so it rebuilds
151            self.mark_context_dirty(context_name)?;
152        }
153
154        Ok(())
155    }

157    /// Store the combined system prompt (with hook injections) in context_meta.
158    /// This allows reconstructing the full API request from context.jsonl + context_meta.json.
159    pub fn save_combined_system_prompt(
160        &self,
161        context_name: &str,
162        combined_prompt: &str,
163    ) -> io::Result<()> {
164        let mut meta = self.load_context_meta(context_name).unwrap_or_default();
165        meta.last_combined_prompt = Some(combined_prompt.to_string());
166        self.save_context_meta(context_name, &meta)
167    }

187        let reflection_path = self.prompts_dir.join("reflection.md");
