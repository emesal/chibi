13impl AppState {
14    /// Load local config for a context (returns default if doesn't exist)
15    pub fn load_local_config(&self, context_name: &str) -> io::Result<LocalConfig> {
16        let path = self.local_config_file(context_name);
17        if path.exists() {
18            let content = fs::read_to_string(&path)?;
19            toml::from_str(&content).map_err(|e| {
20                io::Error::new(
21                    ErrorKind::InvalidData,
22                    format!("Failed to parse local.toml: {}", e),
23                )
24            })
25        } else {
26            Ok(LocalConfig::default())
27        }
28    }
29
30    /// Save local config for a context (atomic write)
31    pub fn save_local_config(
32        &self,
33        context_name: &str,
34        local_config: &LocalConfig,
35    ) -> io::Result<()> {
36        self.ensure_context_dir(context_name)?;
37        let path = self.local_config_file(context_name);
38        let content = toml::to_string_pretty(local_config)
39            .map_err(|e| io::Error::other(format!("Failed to serialize local.toml: {}", e)))?;
40        crate::safe_io::atomic_write_text(&path, &content)
41    }

43    /// Resolve model name using models.toml aliases
44    /// If the model is an alias defined in models.toml, return the full model name
45    /// Otherwise return the original model name
46    pub fn resolve_model_name(&self, model: &str) -> String {
47        if self.models_config.models.contains_key(model) {
48            // The model name itself is a key in models.toml, use it as-is
49            // (models.toml maps alias -> metadata, not alias -> full name)
50            model.to_string()
51        } else {
52            model.to_string()
53        }
54    }

56    /// Resolve the full configuration, applying overrides in order:
57    /// 1. Runtime override (passed as parameter, highest priority)
58    /// 2. Context-local config (local.toml)
59    /// 3. Models.toml (per-model API params)
60    /// 4. Environment variables (`CHIBI_API_KEY`, `CHIBI_MODEL`)
61    /// 5. Global config (config.toml)
62    /// 6. Defaults
63    pub fn resolve_config(
64        &self,
65        context_name: &str,
66        username_override: Option<&str>,
67    ) -> io::Result<ResolvedConfig> {
68        let local = self.load_local_config(context_name)?;
69
70        // Start with defaults, then merge global config
71        let mut api_params = ApiParams::defaults();
72        api_params = api_params.merge_with(&self.config.api);
73
74        // Start with global config values, applying defaults for optional fields
75        let mut resolved = ResolvedConfig {
76            api_key: self.config.api_key.clone(),
77            model: self
78                .config
79                .model
80                .clone()
81                .unwrap_or_else(|| ConfigDefaults::MODEL.to_string()),
82            context_window_limit: self
83                .config
84                .context_window_limit
85                .unwrap_or(ConfigDefaults::CONTEXT_WINDOW_LIMIT),
86            warn_threshold_percent: self.config.warn_threshold_percent,
87            no_tool_calls: self.config.no_tool_calls,
88            auto_compact: self.config.auto_compact,
89            auto_compact_threshold: self.config.auto_compact_threshold,
90            fuel: self.config.fuel,
91            fuel_empty_response_cost: self.config.fuel_empty_response_cost,
92            username: self.config.username.clone(),
93            reflection_enabled: self.config.reflection_enabled,
94            reflection_character_limit: self.config.reflection_character_limit,
95            rolling_compact_drop_percentage: self.config.rolling_compact_drop_percentage,
96            tool_output_cache_threshold: self.config.tool_output_cache_threshold,
97            tool_cache_max_age_days: self.config.tool_cache_max_age_days,
98            auto_cleanup_cache: self.config.auto_cleanup_cache,
99            tool_cache_preview_chars: self.config.tool_cache_preview_chars,
100            file_tools_allowed_paths: self.config.file_tools_allowed_paths.clone(),
101            api: api_params,
102            tools: self.config.tools.clone(),
103            fallback_tool: self.config.fallback_tool.clone(),
104            storage: self.config.storage.clone(),
105            url_policy: self.config.url_policy.clone(),
106            subagent_cost_tier: self.config.subagent_cost_tier.clone(),
107            extra: BTreeMap::new(),
108        };
109
110        // Apply environment variable overrides (between global config and local.toml)
111        apply_env_overrides(&mut resolved);
112
113        // Apply local config overrides (simple fields via macro, see LocalConfig::apply_overrides)
114        local.apply_overrides(&mut resolved);
115
116        // Apply context-level storage config overrides
117        resolved.storage = resolved.storage.merge(&local.storage);
118
119        // Apply context-level API params (Layer 3)
120        if let Some(ref local_api) = local.api {
121            resolved.api = resolved.api.merge_with(local_api);
122        }
123
124        // Apply context-level tool filtering config (merge local on top of global)
125        if let Some(ref local_tools) = local.tools {
126            resolved.tools = resolved.tools.merge_local(local_tools);
127        }
128
129        // Apply runtime username override (highest priority)
130        if let Some(username) = username_override {
131            resolved.username = username.to_string();
132        }
133
134        // Default file_tools_allowed_paths to cwd when empty (after all overrides).
135        // This ensures project files are readable without explicit config.
136        if resolved.file_tools_allowed_paths.is_empty()
137            && let Ok(cwd) = std::env::current_dir()
138        {
139            resolved.file_tools_allowed_paths = vec![cwd.to_string_lossy().to_string()];
140        }
141
142        // Resolve model name and potentially override context window + API params
143        resolved.model = self.resolve_model_name(&resolved.model);
144        if let Some(model_meta) = self.models_config.models.get(&resolved.model) {
145            // Apply model-level API params (Layer 2 - after global, before context)
146            // Note: We merge model params before context params because context should override model
147            // But we do this after context-level override for the rest of config, so we need to
148            // re-merge context params on top
149            let model_api = resolved.api.merge_with(&model_meta.api);
150            // Re-apply context-level API params on top of model params
151            resolved.api = if let Some(ref local_api) = local.api {
152                model_api.merge_with(local_api)
153            } else {
154                model_api
155            };
156        }
157
158        Ok(resolved)
159    }

161    /// Validate resolved config against loaded tools
162    ///
163    /// Checks that fallback_tool exists and has flow_control=true metadata.
164    pub fn validate_config(
165        &self,
166        resolved: &ResolvedConfig,
167        tools: &[crate::tools::Tool],
168    ) -> io::Result<()> {
169        let fallback = &resolved.fallback_tool;
170
171        // Get metadata (checks plugins then builtins)
172        let meta = crate::tools::get_tool_metadata(tools, fallback);
173
174        // Verify tool exists: must be in plugins OR be a known builtin
175        let is_builtin = matches!(
176            fallback.as_str(),
177            crate::tools::CALL_AGENT_TOOL_NAME | crate::tools::CALL_USER_TOOL_NAME
178        );
179        let in_plugins = tools.iter().any(|t| t.name == *fallback);
180
181        if !is_builtin && !in_plugins {
182            return Err(io::Error::new(
183                ErrorKind::InvalidInput,
184                format!("fallback_tool '{}' not found", fallback),
185            ));
186        }
187
188        // Enforce: fallback must be a flow_control tool
189        if !meta.flow_control {
190            return Err(io::Error::new(
191                ErrorKind::InvalidInput,
192                format!(
193                    "fallback_tool '{}' must have flow_control=true metadata",
194                    fallback
195                ),
196            ));
197        }
198
199        Ok(())
200    }

203/// Environment variable names for config overrides.
204pub const ENV_API_KEY: &str = "CHIBI_API_KEY";
205pub const ENV_MODEL: &str = "CHIBI_MODEL";
206
207/// Apply environment variable overrides onto a resolved config.
208///
209/// Priority: global config.toml < **env vars** < context local.toml.
210fn apply_env_overrides(resolved: &mut ResolvedConfig) {
211    if let Ok(key) = env::var(ENV_API_KEY) {
212        resolved.api_key = Some(key);
213    }
214    if let Ok(model) = env::var(ENV_MODEL) {
215        resolved.model = model;
216    }
