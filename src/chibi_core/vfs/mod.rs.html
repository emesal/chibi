1//! Virtual file system: sandboxed, shared storage for contexts.
2//!
3//! The VFS provides a permission-enforced file namespace that contexts can
4//! read and write without engaging the OS-level permission system. Paths use
5//! a `vfs://` URI scheme and never leak OS `PathBuf` values.
6//!
7//! # Architecture (Approach A — thin trait, fat router)
8//!
9//! ```text
10//! tool code  ->  Vfs (permissions + path validation)  ->  VfsBackend (dumb storage)
11//! ```
12//!
13//! The `Vfs` struct enforces zone-based permissions and delegates to a
14//! `VfsBackend` trait implementation. Backends are trivially simple — just
15//! storage, no permission logic.
16//!
17//! # Future evolution
18//!
19//! - **Multi-backend mounting**: `Vfs` maps path prefixes to different backends
20//!   (e.g. `/shared/` on disk, `/remote/` on XMPP). Longest-prefix match.
21//! - **Middleware layers**: Composable tower-style layers (logging, caching)
22//!   wrapping backends (approach C in the design doc). Refactor when needed.
23

39#[cfg(test)]
40mod integration_tests {
41    use super::*;
42    use std::io::ErrorKind;
43    use tempfile::TempDir;
44
45    fn setup() -> (TempDir, Vfs) {
46        let dir = TempDir::new().unwrap();
47        let backend = LocalBackend::new(dir.path().to_path_buf());
48        (dir, Vfs::new(Box::new(backend)))
49    }
50
51    #[tokio::test]
52    async fn test_vfs_multi_context_integration() {
53        let (_dir, vfs) = setup();
54
55        // --- SYSTEM writes to /sys/ ---
56        let sys_config = VfsPath::new("/sys/config.json").unwrap();
57        vfs.write(SYSTEM_CALLER, &sys_config, b"{\"version\":1}")
58            .await
59            .unwrap();
60        assert!(vfs.exists(SYSTEM_CALLER, &sys_config).await.unwrap());
61        assert_eq!(
62            vfs.read(SYSTEM_CALLER, &sys_config).await.unwrap(),
63            b"{\"version\":1}"
64        );
65
66        // --- alice operations ---
67        let shared_tasks = VfsPath::new("/shared/tasks.md").unwrap();
68        let alice_notes = VfsPath::new("/home/alice/notes.md").unwrap();
69
70        vfs.write("alice", &shared_tasks, b"# tasks\n- implement vfs")
71            .await
72            .unwrap();
73        vfs.write("alice", &alice_notes, b"alice's private notes")
74            .await
75            .unwrap();
76
77        // alice cannot write to /sys/
78        let err = vfs
79            .write("alice", &VfsPath::new("/sys/data").unwrap(), b"nope")
80            .await
81            .unwrap_err();
82        assert_eq!(err.kind(), ErrorKind::PermissionDenied);
83
84        // alice cannot write to bob's home
85        let err = vfs
86            .write(
87                "alice",
88                &VfsPath::new("/home/bob/secret.md").unwrap(),
89                b"nope",
90            )
91            .await
92            .unwrap_err();
93        assert_eq!(err.kind(), ErrorKind::PermissionDenied);
94
95        // --- bob operations ---
96        // bob can read alice's home (read is always allowed)
97        let data = vfs.read("bob", &alice_notes).await.unwrap();
98        assert_eq!(data, b"alice's private notes");
99
100        // bob cannot write to alice's home
101        let err = vfs.write("bob", &alice_notes, b"hacked").await.unwrap_err();
102        assert_eq!(err.kind(), ErrorKind::PermissionDenied);
103
104        // bob can write to /shared/
105        let bob_file = VfsPath::new("/shared/bob-file.txt").unwrap();
106        vfs.write("bob", &bob_file, b"bob's contribution")
107            .await
108            .unwrap();
109
110        // bob can write to own home
111        let bob_private = VfsPath::new("/home/bob/private.md").unwrap();
112        vfs.write("bob", &bob_private, b"bob's secret")
113            .await
114            .unwrap();
115
116        // --- copy/move with permissions ---
117        let backup = VfsPath::new("/shared/backup.md").unwrap();
118        vfs.copy("alice", &shared_tasks, &backup).await.unwrap();
119        assert_eq!(
120            vfs.read("alice", &backup).await.unwrap(),
121            b"# tasks\n- implement vfs"
122        );
123
124        // alice cannot copy to bob's home
125        let err = vfs
126            .copy(
127                "alice",
128                &shared_tasks,
129                &VfsPath::new("/home/bob/stolen.md").unwrap(),
130            )
131            .await
132            .unwrap_err();
133        assert_eq!(err.kind(), ErrorKind::PermissionDenied);
134
135        // bob moves his file in /shared/
136        let renamed = VfsPath::new("/shared/renamed.txt").unwrap();
137        vfs.rename("bob", &bob_file, &renamed).await.unwrap();
138        assert!(!vfs.exists("bob", &bob_file).await.unwrap());
139        assert_eq!(
140            vfs.read("bob", &renamed).await.unwrap(),
141            b"bob's contribution"
142        );
143
144        // --- delete with permissions ---
145        // bob cannot delete alice's file
146        let err = vfs.delete("bob", &alice_notes).await.unwrap_err();
147        assert_eq!(err.kind(), ErrorKind::PermissionDenied);
148
149        // alice deletes her own file
150        vfs.delete("alice", &alice_notes).await.unwrap();
151        assert!(!vfs.exists("alice", &alice_notes).await.unwrap());
152
153        // --- list and metadata ---
154        let mut shared_entries = vfs
155            .list("alice", &VfsPath::new("/shared").unwrap())
156            .await
157            .unwrap();
158        shared_entries.sort_by(|a, b| a.name.cmp(&b.name));
159        let names: Vec<&str> = shared_entries.iter().map(|e| e.name.as_str()).collect();
160        assert!(names.contains(&"backup.md"));
161        assert!(names.contains(&"renamed.txt"));
162        assert!(names.contains(&"tasks.md"));
163
164        let meta = vfs.metadata("alice", &backup).await.unwrap();
165        assert_eq!(meta.kind, VfsEntryKind::File);
166        assert!(meta.size > 0);
167
168        let mut home_entries = vfs
169            .list("alice", &VfsPath::new("/home").unwrap())
170            .await
171            .unwrap();
172        home_entries.sort_by(|a, b| a.name.cmp(&b.name));
173        let home_names: Vec<&str> = home_entries.iter().map(|e| e.name.as_str()).collect();
174        assert!(home_names.contains(&"alice"));
175        assert!(home_names.contains(&"bob"));
176
177        // --- mkdir ---
178        let subdir = VfsPath::new("/shared/subdir").unwrap();
179        vfs.mkdir("alice", &subdir).await.unwrap();
180        assert!(vfs.exists("alice", &subdir).await.unwrap());
181        let subdir_meta = vfs.metadata("alice", &subdir).await.unwrap();
182        assert_eq!(subdir_meta.kind, VfsEntryKind::Directory);
183
184        // alice cannot mkdir in /sys/
185        let err = vfs
186            .mkdir("alice", &VfsPath::new("/sys/forbidden").unwrap())
187            .await
188            .unwrap_err();
189        assert_eq!(err.kind(), ErrorKind::PermissionDenied);
190    }
