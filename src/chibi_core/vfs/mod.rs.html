1//! Virtual file system: sandboxed, shared storage for contexts.
2//!
3//! The VFS provides a permission-enforced file namespace that contexts can
4//! read and write without engaging the OS-level permission system. Paths use
5//! a `vfs://` URI scheme and never leak OS `PathBuf` values.
6//!
7//! # Architecture (Approach A — thin trait, fat router)
8//!
9//! ```text
10//! tool code  ->  Vfs (permissions + path validation)  ->  VfsBackend (dumb storage)
11//! ```
12//!
13//! The `Vfs` struct enforces zone-based permissions and delegates to a
14//! `VfsBackend` trait implementation. Backends are trivially simple — just
15//! storage, no permission logic.
16//!
17//! # Future evolution
18//!
19//! - **Multi-backend mounting**: `Vfs` maps path prefixes to different backends
20//!   (e.g. `/shared/` on disk, `/remote/` on XMPP). Longest-prefix match.
21//! - **Middleware layers**: Composable tower-style layers (logging, caching)
22//!   wrapping backends (approach C in the design doc). Refactor when needed.

46#[cfg(test)]
47mod integration_tests {
48    use super::*;
49    use std::io::ErrorKind;
50    use tempfile::TempDir;
51
52    fn setup() -> (TempDir, Vfs) {
53        let dir = TempDir::new().unwrap();
54        let backend = LocalBackend::new(dir.path().to_path_buf());
55        (dir, Vfs::new(Box::new(backend), "test-site-0000"))
56    }
57
58    #[tokio::test]
59    async fn test_vfs_multi_context_integration() {
60        let (_dir, vfs) = setup();
61
62        // --- SYSTEM writes to /sys/ ---
63        let sys_config = VfsPath::new("/sys/config.json").unwrap();
64        vfs.write(VfsCaller::System, &sys_config, b"{\"version\":1}")
65            .await
66            .unwrap();
67        assert!(vfs.exists(VfsCaller::System, &sys_config).await.unwrap());
68        assert_eq!(
69            vfs.read(VfsCaller::System, &sys_config).await.unwrap(),
70            b"{\"version\":1}"
71        );
72
73        // --- alice operations ---
74        let shared_tasks = VfsPath::new("/shared/tasks.md").unwrap();
75        let alice_notes = VfsPath::new("/home/alice/notes.md").unwrap();
76
77        vfs.write(
78            VfsCaller::Context("alice"),
79            &shared_tasks,
80            b"# tasks\n- implement vfs",
81        )
82        .await
83        .unwrap();
84        vfs.write(
85            VfsCaller::Context("alice"),
86            &alice_notes,
87            b"alice's private notes",
88        )
89        .await
90        .unwrap();
91
92        // alice cannot write to /sys/
93        let err = vfs
94            .write(
95                VfsCaller::Context("alice"),
96                &VfsPath::new("/sys/data").unwrap(),
97                b"nope",
98            )
99            .await
100            .unwrap_err();
101        assert_eq!(err.kind(), ErrorKind::PermissionDenied);
102
103        // alice cannot write to bob's home
104        let err = vfs
105            .write(
106                VfsCaller::Context("alice"),
107                &VfsPath::new("/home/bob/secret.md").unwrap(),
108                b"nope",
109            )
110            .await
111            .unwrap_err();
112        assert_eq!(err.kind(), ErrorKind::PermissionDenied);
113
114        // --- bob operations ---
115        // bob can read alice's home (read is always allowed)
116        let data = vfs
117            .read(VfsCaller::Context("bob"), &alice_notes)
118            .await
119            .unwrap();
120        assert_eq!(data, b"alice's private notes");
121
122        // bob cannot write to alice's home
123        let err = vfs
124            .write(VfsCaller::Context("bob"), &alice_notes, b"hacked")
125            .await
126            .unwrap_err();
127        assert_eq!(err.kind(), ErrorKind::PermissionDenied);
128
129        // bob can write to /shared/
130        let bob_file = VfsPath::new("/shared/bob-file.txt").unwrap();
131        vfs.write(VfsCaller::Context("bob"), &bob_file, b"bob's contribution")
132            .await
133            .unwrap();
134
135        // bob can write to own home
136        let bob_private = VfsPath::new("/home/bob/private.md").unwrap();
137        vfs.write(VfsCaller::Context("bob"), &bob_private, b"bob's secret")
138            .await
139            .unwrap();
140
141        // --- copy/move with permissions ---
142        let backup = VfsPath::new("/shared/backup.md").unwrap();
143        vfs.copy(VfsCaller::Context("alice"), &shared_tasks, &backup)
144            .await
145            .unwrap();
146        assert_eq!(
147            vfs.read(VfsCaller::Context("alice"), &backup)
148                .await
149                .unwrap(),
150            b"# tasks\n- implement vfs"
151        );
152
153        // alice cannot copy to bob's home
154        let err = vfs
155            .copy(
156                VfsCaller::Context("alice"),
157                &shared_tasks,
158                &VfsPath::new("/home/bob/stolen.md").unwrap(),
159            )
160            .await
161            .unwrap_err();
162        assert_eq!(err.kind(), ErrorKind::PermissionDenied);
163
164        // bob moves his file in /shared/
165        let renamed = VfsPath::new("/shared/renamed.txt").unwrap();
166        vfs.rename(VfsCaller::Context("bob"), &bob_file, &renamed)
167            .await
168            .unwrap();
169        assert!(
170            !vfs.exists(VfsCaller::Context("bob"), &bob_file)
171                .await
172                .unwrap()
173        );
174        assert_eq!(
175            vfs.read(VfsCaller::Context("bob"), &renamed).await.unwrap(),
176            b"bob's contribution"
177        );
178
179        // --- delete with permissions ---
180        // bob cannot delete alice's file
181        let err = vfs
182            .delete(VfsCaller::Context("bob"), &alice_notes)
183            .await
184            .unwrap_err();
185        assert_eq!(err.kind(), ErrorKind::PermissionDenied);
186
187        // alice deletes her own file
188        vfs.delete(VfsCaller::Context("alice"), &alice_notes)
189            .await
190            .unwrap();
191        assert!(
192            !vfs.exists(VfsCaller::Context("alice"), &alice_notes)
193                .await
194                .unwrap()
195        );
196
197        // --- list and metadata ---
198        let mut shared_entries = vfs
199            .list(
200                VfsCaller::Context("alice"),
201                &VfsPath::new("/shared").unwrap(),
202            )
203            .await
204            .unwrap();
205        shared_entries.sort_by(|a, b| a.name.cmp(&b.name));
206        let names: Vec<&str> = shared_entries.iter().map(|e| e.name.as_str()).collect();
207        assert!(names.contains(&"backup.md"));
208        assert!(names.contains(&"renamed.txt"));
209        assert!(names.contains(&"tasks.md"));
210
211        let meta = vfs
212            .metadata(VfsCaller::Context("alice"), &backup)
213            .await
214            .unwrap();
215        assert_eq!(meta.kind, VfsEntryKind::File);
216        assert!(meta.size > 0);
217
218        let mut home_entries = vfs
219            .list(VfsCaller::Context("alice"), &VfsPath::new("/home").unwrap())
220            .await
221            .unwrap();
222        home_entries.sort_by(|a, b| a.name.cmp(&b.name));
223        let home_names: Vec<&str> = home_entries.iter().map(|e| e.name.as_str()).collect();
224        assert!(home_names.contains(&"alice"));
225        assert!(home_names.contains(&"bob"));
226
227        // --- mkdir ---
228        let subdir = VfsPath::new("/shared/subdir").unwrap();
229        vfs.mkdir(VfsCaller::Context("alice"), &subdir)
230            .await
231            .unwrap();
232        assert!(
233            vfs.exists(VfsCaller::Context("alice"), &subdir)
234                .await
235                .unwrap()
236        );
237        let subdir_meta = vfs
238            .metadata(VfsCaller::Context("alice"), &subdir)
239            .await
240            .unwrap();
241        assert_eq!(subdir_meta.kind, VfsEntryKind::Directory);
242
243        // alice cannot mkdir in /sys/
244        let err = vfs
245            .mkdir(
246                VfsCaller::Context("alice"),
247                &VfsPath::new("/sys/forbidden").unwrap(),
248            )
249            .await
250            .unwrap_err();
251        assert_eq!(err.kind(), ErrorKind::PermissionDenied);
252    }
