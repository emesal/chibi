1//! VFS backend trait: the storage abstraction.
2//!
3//! Backends are intentionally simple — just storage, no permission logic.
4//! The `Vfs` struct handles permissions and delegates to the backend.
5//!
6//! # Implementing a backend
7//!
8//! Implement all methods on `VfsBackend`. The `VfsPath` values you receive
9//! are already validated; you only need to map them to your storage.
10//!
11//! # Future: middleware layers
12//!
13//! The current design (approach A) has `Vfs` call the backend directly.
14//! A future evolution (approach C) would wrap backends in composable
15//! middleware layers (logging, caching, etc.) a la tower. The trait
16//! signature is designed to be compatible with that transition.
17//!
18//! # Dyn-compatibility
19//!
20//! Methods return `Pin<Box<dyn Future>>` instead of `impl Future` so that
21//! `Box<dyn VfsBackend>` works. This enables multi-backend mounting where
22//! `Vfs` selects a backend at runtime via longest-prefix match.
23//!
24//! All input references share a single lifetime `'a` so the returned
25//! future can borrow from both `&self` and any path arguments.
26
27use std::future::Future;
28use std::io;
29use std::pin::Pin;
30
31use super::path::VfsPath;
32use super::types::{VfsEntry, VfsMetadata};
33
34/// Boxed, Send future — the return type for all backend methods.
35type BoxFuture<'a, T> = Pin<Box<dyn Future<Output = T> + Send + 'a>>;
36
37/// Storage backend for the virtual file system.
38///
39/// All methods receive validated `VfsPath` values. Backends translate these
40/// to their native addressing (file paths, database keys, XMPP nodes, etc.).
41///
42/// All methods are async (returning boxed futures) to accommodate
43/// network-backed implementations and dyn-compatibility.
44pub trait VfsBackend: Send + Sync {
45    /// Read the full contents of a file.
46    fn read<'a>(&'a self, path: &'a VfsPath) -> BoxFuture<'a, io::Result<Vec<u8>>>;
47
48    /// Write (create or overwrite) a file with the given contents.
49    fn write<'a>(&'a self, path: &'a VfsPath, data: &'a [u8]) -> BoxFuture<'a, io::Result<()>>;
50
51    /// Append data to an existing file, creating it if it doesn't exist.
52    fn append<'a>(&'a self, path: &'a VfsPath, data: &'a [u8]) -> BoxFuture<'a, io::Result<()>>;
53
54    /// Delete a file. Returns `NotFound` if the path doesn't exist.
55    fn delete<'a>(&'a self, path: &'a VfsPath) -> BoxFuture<'a, io::Result<()>>;
56
57    /// List entries in a directory. Returns empty vec if path doesn't exist.
58    fn list<'a>(&'a self, path: &'a VfsPath) -> BoxFuture<'a, io::Result<Vec<VfsEntry>>>;
59
60    /// Check whether a path exists.
61    fn exists<'a>(&'a self, path: &'a VfsPath) -> BoxFuture<'a, io::Result<bool>>;
62
63    /// Create a directory (and parents if needed).
64    fn mkdir<'a>(&'a self, path: &'a VfsPath) -> BoxFuture<'a, io::Result<()>>;
65
66    /// Copy a file from src to dst. Both paths are within this backend.
67    fn copy<'a>(&'a self, src: &'a VfsPath, dst: &'a VfsPath) -> BoxFuture<'a, io::Result<()>>;
68
69    /// Rename (move) a file from src to dst. Both paths are within this backend.
70    fn rename<'a>(&'a self, src: &'a VfsPath, dst: &'a VfsPath) -> BoxFuture<'a, io::Result<()>>;
71
72    /// Get metadata for a path.
73    fn metadata<'a>(&'a self, path: &'a VfsPath) -> BoxFuture<'a, io::Result<VfsMetadata>>;
74}
