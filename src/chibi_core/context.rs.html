57/// Metadata entry for a context in state.json
58#[derive(Debug, Serialize, Deserialize, Clone, PartialEq)]
59pub struct ContextEntry {
60    pub name: String,
61    pub created_at: u64,
62    /// Last time this context was used (0 = never updated)
63    #[serde(default)]
64    pub last_activity_at: u64,
65    /// Auto-destroy after this many seconds of inactivity (0 = disabled)
66    #[serde(default)]
67    pub destroy_after_seconds_inactive: u64,
68    /// Auto-destroy at this timestamp (0 = disabled)
69    #[serde(default)]
70    pub destroy_at: u64,
71}
72
73impl ContextEntry {
74    pub fn with_created_at(name: impl Into<String>, created_at: u64) -> Self {
75        Self {
76            name: name.into(),
77            created_at,
78            last_activity_at: 0,
79            destroy_after_seconds_inactive: 0,
80            destroy_at: 0,
81        }
82    }
83
84    /// Update last_activity_at to the current time
85    pub fn touch(&mut self) {
86        self.last_activity_at = now_timestamp();
87    }
88
89    /// Check if this context should be auto-destroyed
90    pub fn should_auto_destroy(&self) -> bool {
91        let now = now_timestamp();
92
93        // Check destroy_at (absolute timestamp)
94        if self.destroy_at >= 1 && now > self.destroy_at {
95            return true;
96        }
97
98        // Check inactivity timeout
99        if self.destroy_after_seconds_inactive >= 1 && self.last_activity_at >= 1 {
100            let deadline = self.last_activity_at + self.destroy_after_seconds_inactive;
101            if now > deadline {
102                return true;
103            }
104        }
105
106        false
107    }
108}

126pub fn is_valid_context_name(name: &str) -> bool {
127    // Reject reserved names
128    if name == "-" || crate::vfs::is_reserved_caller_name(name) {
129        return false;
130    }
131    !name.is_empty()
132        && name
133            .chars()
134            .all(|c| c.is_ascii_alphanumeric() || c == '-' || c == '_')
135}
136
137pub fn validate_context_name(name: &str) -> io::Result<()> {
138    if name == "-" {
139        return Err(io::Error::new(
140            ErrorKind::InvalidInput,
141            "Invalid context name '-'. This is a reserved name used to reference the previous context.",
142        ));
143    }
144    if !is_valid_context_name(name) {
145        return Err(io::Error::new(
146            ErrorKind::InvalidInput,
147            format!(
148                "Invalid context name '{}'. Names must be alphanumeric with dashes and underscores only.",
149                name
150            ),
151        ));
152    }
153    Ok(())
154}

156pub fn now_timestamp() -> u64 {
157    std::time::SystemTime::now()
158        .duration_since(std::time::UNIX_EPOCH)
159        .unwrap_or(std::time::Duration::ZERO)
160        .as_secs()
161}

207/// Builder for creating TranscriptEntry instances with a fluent API.
208/// All fields have sensible defaults: auto-generated ID, current timestamp,
209/// empty strings for from/to/content, and "message" for entry_type.
210#[derive(Default)]
211pub struct TranscriptEntryBuilder {
212    from: Option<String>,
213    to: Option<String>,
214    content: Option<String>,
215    entry_type: Option<String>,
216    metadata: Option<EntryMetadata>,
217    tool_call_id: Option<String>,
218}
219
220impl TranscriptEntryBuilder {
221    /// Set the source/sender of the entry
222    pub fn from(mut self, from: impl Into<String>) -> Self {
223        self.from = Some(from.into());
224        self
225    }
226
227    /// Set the destination/recipient of the entry
228    pub fn to(mut self, to: impl Into<String>) -> Self {
229        self.to = Some(to.into());
230        self
231    }
232
233    /// Set the content of the entry
234    pub fn content(mut self, content: impl Into<String>) -> Self {
235        self.content = Some(content.into());
236        self
237    }
238
239    /// Set the entry type (e.g., "message", "tool_call", "tool_result")
240    pub fn entry_type(mut self, entry_type: impl Into<String>) -> Self {
241        self.entry_type = Some(entry_type.into());
242        self
243    }
244
245    /// Set metadata for the entry
246    pub fn metadata(mut self, metadata: EntryMetadata) -> Self {
247        self.metadata = Some(metadata);
248        self
249    }
250
251    /// Set the tool call ID for correlating tool_call/tool_result pairs
252    pub fn tool_call_id(mut self, id: impl Into<String>) -> Self {
253        self.tool_call_id = Some(id.into());
254        self
255    }
256
257    /// Build the TranscriptEntry with auto-generated ID and current timestamp
258    pub fn build(self) -> TranscriptEntry {
259        TranscriptEntry {
260            id: uuid::Uuid::new_v4().to_string(),
261            timestamp: now_timestamp(),
262            from: self.from.unwrap_or_default(),
263            to: self.to.unwrap_or_default(),
264            content: self.content.unwrap_or_default(),
265            entry_type: self
266                .entry_type
267                .unwrap_or_else(|| ENTRY_TYPE_MESSAGE.to_string()),
268            metadata: self.metadata,
269            tool_call_id: self.tool_call_id,
270        }
271    }
272}

746            })
747            .collect();

751        }

755        let parsed: ContextState = serde_json::from_str(&content).unwrap();
756        assert!(parsed.contexts[0].name.starts_with("ctx-"));
