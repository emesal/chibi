57/// Metadata entry for a context in state.json
58#[derive(Debug, Serialize, Deserialize, Clone, PartialEq)]
59pub struct ContextEntry {
60    pub name: String,
61    pub created_at: u64,
62    /// Last time this context was used (0 = never updated)
63    #[serde(default)]
64    pub last_activity_at: u64,
65    /// Auto-destroy after this many seconds of inactivity (0 = disabled)
66    #[serde(default)]
67    pub destroy_after_seconds_inactive: u64,
68    /// Auto-destroy at this timestamp (0 = disabled)
69    #[serde(default)]
70    pub destroy_at: u64,
71    /// Working directory at context creation time. Used to resolve relative file
72    /// paths consistently across sessions. None for contexts created before this
73    /// field was added (falls back to std::env::current_dir()).
74    #[serde(default)]
75    pub cwd: Option<String>,
76}
77
78impl ContextEntry {
79    pub fn with_created_at(name: impl Into<String>, created_at: u64) -> Self {
80        Self {
81            name: name.into(),
82            created_at,
83            last_activity_at: 0,
84            destroy_after_seconds_inactive: 0,
85            destroy_at: 0,
86            cwd: std::env::current_dir()
87                .ok()
88                .map(|p| p.to_string_lossy().into_owned()),
89        }
90    }
91
92    /// Update last_activity_at to the current time
93    pub fn touch(&mut self) {
94        self.last_activity_at = now_timestamp();
95    }
96
97    /// Check if this context should be auto-destroyed
98    pub fn should_auto_destroy(&self) -> bool {
99        let now = now_timestamp();
100
101        // Check destroy_at (absolute timestamp)
102        if self.destroy_at >= 1 && now > self.destroy_at {
103            return true;
104        }
105
106        // Check inactivity timeout
107        if self.destroy_after_seconds_inactive >= 1 && self.last_activity_at >= 1 {
108            let deadline = self.last_activity_at + self.destroy_after_seconds_inactive;
109            if now > deadline {
110                return true;
111            }
112        }
113
114        false
115    }
116}

134pub fn is_valid_context_name(name: &str) -> bool {
135    // Reject reserved names and names starting with '-' (clashes with CLI flag syntax).
136    if name.starts_with('-') || crate::vfs::is_reserved_caller_name(name) {
137        return false;
138    }
139    !name.is_empty()
140        && name
141            .chars()
142            .all(|c| c.is_ascii_alphanumeric() || c == '-' || c == '_')
143}
144
145pub fn validate_context_name(name: &str) -> io::Result<()> {
146    if name.starts_with('-') {
147        return Err(io::Error::new(
148            ErrorKind::InvalidInput,
149            format!(
150                "Invalid context name '{name}'. Context names cannot start with a dash (reserved for CLI flags).",
151            ),
152        ));
153    }
154    if !is_valid_context_name(name) {
155        return Err(io::Error::new(
156            ErrorKind::InvalidInput,
157            format!(
158                "Invalid context name '{}'. Names must be alphanumeric with dashes and underscores only.",
159                name
160            ),
161        ));
162    }
163    Ok(())
164}

166pub fn now_timestamp() -> u64 {
167    std::time::SystemTime::now()
168        .duration_since(std::time::UNIX_EPOCH)
169        .unwrap_or(std::time::Duration::ZERO)
170        .as_secs()
171}

195    pub timestamp: u64,
196    pub from: String,
197    pub to: String,
198    pub content: String,
199    pub entry_type: String,
200    /// Optional metadata for anchor entries (summary, hash, etc.)

202    pub metadata: Option<EntryMetadata>,
203    /// Tool call ID for correlating tool_call and tool_result entries.
204    /// Present on tool_call and tool_result entry types; absent on messages.
205    #[serde(default, skip_serializing_if = "Option::is_none")]

208    /// Used by entries_to_messages() instead of inferring from from/to.
209    /// None for old entries and non-message types (tool_call, tool_result, anchors).
210    #[serde(default, skip_serializing_if = "Option::is_none")]
211    pub role: Option<String>,
212    /// Semantic marker for flow control events (user prompts, call_user, control_transfer).
213    /// NOT used as a filter criterion — purely informational.
214    #[serde(default, skip_serializing_if = "std::ops::Not::not")]
215    pub flow_control: bool,
216}

289    /// Build the TranscriptEntry with auto-generated ID and current timestamp
290    pub fn build(self) -> TranscriptEntry {
291        TranscriptEntry {

293            timestamp: now_timestamp(),
294            from: self.from.unwrap_or_default(),
295            to: self.to.unwrap_or_default(),
296            content: self.content.unwrap_or_default(),
297            entry_type: self
298                .entry_type
299                .unwrap_or_else(|| ENTRY_TYPE_MESSAGE.to_string()),
300            metadata: self.metadata,
301            tool_call_id: self.tool_call_id,
302            role: self.role,
303            flow_control: self.flow_control,
304        }
305    }

760        use tempfile::TempDir;
761

763        let state_path = Arc::new(temp_dir.path().join("state.json"));
764
765        let handles: Vec<_> = (0..10)
766            .map(|i| {
767                let path = Arc::clone(&state_path);

778            .collect();

787        assert!(parsed.contexts[0].name.starts_with("ctx-"));
