17/// Apply `Option`-field overrides from a source struct to a target struct.
18///
19/// For each field name, if `$src.field` is `Some(v)`, sets `$dst.field = v`
20/// (cloning as needed). This eliminates the repetitive `if let Some` blocks
21/// in config resolution.
22macro_rules! apply_option_overrides {
23    ($src:expr, $dst:expr, $($field:ident),+ $(,)?) => {
24        $(
25            if let Some(ref v) = $src.$field {
26                $dst.$field = v.clone();
27            }
28        )+
29    };
30}
31
32/// Early-return from `get_field` for simple config fields.
33///
34/// Call at the top of `get_field` — handles standard fields and returns,
35/// falling through for special cases handled by the caller's match block.
36///
37/// Display modes:
38/// - `display`: uses `to_string()` (bool, usize, u64)
39/// - `clone`: returns `.clone()` (String fields)
40/// - `int`: casts f32 to i32 before display (no decimals)
41/// - `fmt`: uses `format!("{}", v)` (f32 with decimals)
42macro_rules! config_get_field {
43    ($self:expr, $path:expr,
44     $(display: $($d_field:ident),* ;)?
45     $(clone: $($c_field:ident),* ;)?
46     $(int: $($i_field:ident),* ;)?
47     $(fmt: $($f_field:ident),* ;)?
48    ) => {
49        match $path {
50            $($(
51                stringify!($d_field) => return Some($self.$d_field.to_string()),
52            )*)?
53            $($(
54                stringify!($c_field) => return Some($self.$c_field.clone()),
55            )*)?
56            $($(
57                stringify!($i_field) => return Some(format!("{}", $self.$i_field as i32)),
58            )*)?
59            $($(
60                stringify!($f_field) => return Some(format!("{}", $self.$f_field)),
61            )*)?
62            _ => {} // fall through to caller's match
63        }
64    };
65}
66
67/// Early-return from `set_field` for simple config fields.
68///
69/// Parses `$value` into the appropriate type and assigns it, returning `Ok(())`.
70/// Falls through for fields not listed here (handled by the caller's match block).
71///
72/// Parse modes:
73/// - `bool`: parses "true"/"false"
74/// - `usize`: parses via `str::parse::<usize>()`
75/// - `u64`: parses via `str::parse::<u64>()`
76/// - `f32`: parses via `str::parse::<f32>()`
77/// - `string`: clones directly
78macro_rules! config_set_field {
79    ($self:expr, $path:expr, $value:expr,
80     $(bool: $($b_field:ident),* ;)?
81     $(usize: $($u_field:ident),* ;)?
82     $(u64: $($u64_field:ident),* ;)?
83     $(f32: $($f_field:ident),* ;)?
84     $(string: $($s_field:ident),* ;)?
85    ) => {
86        match $path {
87            $($( stringify!($b_field) => {
88                $self.$b_field = $value.parse::<bool>()
89                    .map_err(|_| format!("invalid bool for '{}': {}", $path, $value))?;
90                return Ok(());
91            }, )*)?
92            $($( stringify!($u_field) => {
93                $self.$u_field = $value.parse::<usize>()
94                    .map_err(|_| format!("invalid usize for '{}': {}", $path, $value))?;
95                return Ok(());
96            }, )*)?
97            $($( stringify!($u64_field) => {
98                $self.$u64_field = $value.parse::<u64>()
99                    .map_err(|_| format!("invalid u64 for '{}': {}", $path, $value))?;
100                return Ok(());
101            }, )*)?
102            $($( stringify!($f_field) => {
103                $self.$f_field = $value.parse::<f32>()
104                    .map_err(|_| format!("invalid f32 for '{}': {}", $path, $value))?;
105                return Ok(());
106            }, )*)?
107            $($( stringify!($s_field) => {
108                $self.$s_field = $value.to_string();
109                return Ok(());
110            }, )*)?
111            _ => {} // fall through to caller's match
112        }
113    };
114}

146/// Reasoning configuration for models that support extended thinking
147/// Either `effort` OR `max_tokens` should be set, not both (mutually exclusive).
148/// If both are provided during deserialization, `max_tokens` wins and `effort` is cleared.
149#[derive(Debug, Clone, Default, Serialize)]
150#[serde(into = "ReasoningConfigRaw")]
151pub struct ReasoningConfig {
152    /// Effort level (mutually exclusive with max_tokens)
153    /// Supported by: OpenAI o1/o3/GPT-5 series, Grok models
154    pub effort: Option<ReasoningEffort>,
155
156    /// Maximum tokens for reasoning (mutually exclusive with effort)
157    /// Supported by: Gemini thinking models, Anthropic, some Qwen models
158    /// Anthropic: min 1024, max 128000
159    pub max_tokens: Option<usize>,
160
161    /// Exclude reasoning from response (model still reasons internally)
162    /// Default: false
163    pub exclude: Option<bool>,
164
165    /// Explicitly enable reasoning (defaults to medium effort if true)
166    pub enabled: Option<bool>,
167}
168
169/// Raw deserialization target for ReasoningConfig.
170/// Mutual exclusivity is enforced during From conversion.
171#[derive(Deserialize, Serialize, JsonSchema)]
172struct ReasoningConfigRaw {
173    #[serde(default, skip_serializing_if = "Option::is_none")]
174    effort: Option<ReasoningEffort>,
175    #[serde(default, skip_serializing_if = "Option::is_none")]
176    max_tokens: Option<usize>,
177    #[serde(default, skip_serializing_if = "Option::is_none")]
178    exclude: Option<bool>,
179    #[serde(default, skip_serializing_if = "Option::is_none")]
180    enabled: Option<bool>,
181}
182
183impl From<ReasoningConfigRaw> for ReasoningConfig {
184    fn from(raw: ReasoningConfigRaw) -> Self {
185        let mut config = ReasoningConfig {
186            effort: raw.effort,
187            max_tokens: raw.max_tokens,
188            exclude: raw.exclude,
189            enabled: raw.enabled,
190        };
191        if config.enforce_mutual_exclusivity() {
192            eprintln!("[WARN] reasoning: both effort and max_tokens set, using max_tokens");
193        }
194        config
195    }
196}

198impl From<ReasoningConfig> for ReasoningConfigRaw {
199    fn from(config: ReasoningConfig) -> Self {
200        ReasoningConfigRaw {
201            effort: config.effort,
202            max_tokens: config.max_tokens,
203            exclude: config.exclude,
204            enabled: config.enabled,
205        }
206    }
207}

216    }

219impl ReasoningConfig {
220    /// Check if this config has any values set
221    pub fn is_empty(&self) -> bool {
222        self.effort.is_none()
223            && self.max_tokens.is_none()
224            && self.exclude.is_none()
225            && self.enabled.is_none()
226    }
227
228    /// Enforce mutual exclusivity: effort and max_tokens cannot both be set.
229    /// When both are present, max_tokens wins (it's the more explicit/specific setting).
230    /// Returns true if a conflict was resolved.
231    pub fn enforce_mutual_exclusivity(&mut self) -> bool {
232        if self.effort.is_some() && self.max_tokens.is_some() {
233            self.effort = None;
234            true
235        } else {
236            false
237        }
238    }
239
240    /// Merge with another ReasoningConfig, where `other` takes precedence
241    pub fn merge_with(&self, other: &ReasoningConfig) -> ReasoningConfig {
242        let effort = other.effort.or(self.effort);
243        let max_tokens = other.max_tokens.or(self.max_tokens);
244
245        let mut merged = ReasoningConfig {
246            effort,
247            max_tokens,
248            exclude: other.exclude.or(self.exclude),
249            enabled: other.enabled.or(self.enabled),
250        };
251
252        // If the override explicitly sets one side, the other must be cleared
253        if other.effort.is_some() {
254            merged.max_tokens = None;
255        } else if other.max_tokens.is_some() {
256            merged.effort = None;
257        } else {
258            // Neither side was explicitly overridden — if self somehow had both,
259            // enforce the invariant
260            merged.enforce_mutual_exclusivity();
261        }
262
263        merged
264    }
265}

347    /// Allow parallel tool calls (default: true)

366}

407            response_format: other
408                .response_format
409                .clone()
410                .or_else(|| self.response_format.clone()),
411        }
412    }

455/// Merge two optional string vecs: append `local` to `global`, deduplicating entries.
456fn merge_option_vecs(
457    global: &Option<Vec<String>>,
458    local: &Option<Vec<String>>,
459) -> Option<Vec<String>> {
460    match (global, local) {
461        (Some(g), Some(l)) => {
462            let mut merged = g.clone();
463            for item in l {
464                if !merged.contains(item) {
465                    merged.push(item.clone());
466                }
467            }
468            Some(merged)
469        }
470        (None, Some(l)) => Some(l.clone()),
471        (Some(g), None) => Some(g.clone()),
472        (None, None) => None,
473    }
474}

476impl ToolsConfig {
477    /// Merge a local (per-context) tools config on top of this (global) config.
478    ///
479    /// - `include`: local overrides global entirely if set
480    /// - `exclude`: local appends to global (deduplicated)
481    /// - `exclude_categories`: local appends to global (deduplicated)
482    pub fn merge_local(&self, local: &ToolsConfig) -> ToolsConfig {
483        let include = if local.include.is_some() {
484            local.include.clone()
485        } else {
486            self.include.clone()
487        };
488
489        ToolsConfig {
490            include,
491            exclude: merge_option_vecs(&self.exclude, &local.exclude),
492            exclude_categories: merge_option_vecs(
493                &self.exclude_categories,
494                &local.exclude_categories,
495            ),
496        }
497    }
498}

723impl LocalConfig {
724    /// Apply all simple-override fields from this local config onto a resolved config.
725    ///
726    /// Fields with custom merge semantics (api, storage, tools) are NOT handled here —
727    /// those are applied separately in `resolve_config`.
728    ///
729    /// ## Adding a new config field
730    ///
731    /// 1. Add the field to `Config` (with `#[serde(default = "...")]`)
732    /// 2. Add `Option<T>` field to `LocalConfig`
733    /// 3. Add the concrete field to `ResolvedConfig`
734    /// 4. Add the field name to the `apply_option_overrides!` list below
735    /// 5. Add to `ResolvedConfig::get_field()` (macro or hand-written match arm)
736    /// 6. Add to `ResolvedConfig::list_fields()`
737    /// 7. Initialise from `self.config` in `resolve_config()` struct literal
738    pub fn apply_overrides(&self, resolved: &mut ResolvedConfig) {
739        // api_key is Option<String> on both sides, needs special handling
740        if let Some(ref api_key) = self.api_key {
741            resolved.api_key = Some(api_key.clone());
742        }
743        // All other simple fields: local Some(v) overrides resolved value
744        apply_option_overrides!(
745            self,
746            resolved,
747            model,
748            username,
749            fallback_tool,
750            file_tools_allowed_paths,
751            no_tool_calls,
752            auto_compact,
753            auto_compact_threshold,
754            fuel,
755            fuel_empty_response_cost,
756            warn_threshold_percent,
757            context_window_limit,
758            reflection_enabled,
759            reflection_character_limit,
760            rolling_compact_drop_percentage,
761            tool_output_cache_threshold,
762            tool_cache_max_age_days,
763            auto_cleanup_cache,
764            tool_cache_preview_chars,
765            subagent_cost_tier,
766        );
767        // url_policy: whole-object override (not merge)
768        if self.url_policy.is_some() {
769            resolved.url_policy = self.url_policy.clone();
770        }
771    }
772}

1129    /// Apply a sequence of key-value overrides, short-circuiting on the first error.
1130    pub fn apply_overrides_from_pairs(&mut self, pairs: &[(String, String)]) -> Result<(), String> {
1131        for (key, value) in pairs {
1132            self.set_field(key, value)
1133                .map_err(|e| format!("override '{}={}': {}", key, value, e))?;
1134        }
1135        Ok(())
