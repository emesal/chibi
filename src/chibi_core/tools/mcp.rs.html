1//! MCP bridge client — connects chibi-core to the chibi-mcp-bridge daemon.
2//!
3//! MCP tools are identified by virtual `mcp://server/tool` paths and appear
4//! as regular `Tool` structs in the tools vec. Communication with the bridge
5//! daemon uses JSON-over-TCP via a lockfile-discovered address.
6
7use std::io;
8use std::net::SocketAddr;
9use std::path::{Path, PathBuf};
10
11use super::{Tool, ToolMetadata};

13/// Check if a tool has an MCP virtual path.
14pub fn is_mcp_tool(tool: &Tool) -> bool {
15    tool.path.to_str().is_some_and(|p| p.starts_with("mcp://"))
16}
17
18/// Parse server and tool name from an `mcp://server/tool` path.
19pub fn parse_mcp_path(path: &Path) -> Option<(&str, &str)> {
20    let s = path.to_str()?;
21    let rest = s.strip_prefix("mcp://")?;
22    rest.split_once('/')
23}

25/// Convert bridge tool info into a chibi `Tool`.
26pub fn mcp_tool_from_info(
27    server: &str,
28    name: &str,
29    description: &str,
30    parameters: serde_json::Value,
31) -> Tool {
32    Tool {
33        name: format!("{server}_{name}"),
34        description: description.to_string(),
35        parameters,
36        path: PathBuf::from(format!("mcp://{server}/{name}")),
37        hooks: vec![],
38        metadata: ToolMetadata::new(),
39        summary_params: vec![],
40    }
41}

43/// Lockfile content from the bridge daemon.
44#[derive(serde::Deserialize)]
45struct LockContent {
46    #[allow(dead_code)]
47    pid: u32,
48    address: String,
49    #[allow(dead_code)]
50    started: u64,
51    #[serde(default = "default_heartbeat_secs")]
52    heartbeat_secs: u64,
53    #[serde(default)]
54    timestamp: u64,
55}
56
57fn default_heartbeat_secs() -> u64 {
58    30
59}
60
61/// Check if a bridge lockfile is stale.
62///
63/// Returns true if the PID in the lockfile is no longer running, or if the
64/// heartbeat timestamp is older than 1.5x the heartbeat interval.
65fn is_lockfile_stale(lock: &LockContent) -> bool {
66    // PID liveness: instant detection of a crashed bridge without waiting for
67    // the heartbeat to expire. On Linux we probe /proc/<pid>; on other unix
68    // variants we fall back to timestamp-only staleness.
69    #[cfg(target_os = "linux")]
70    {
71        if !std::path::Path::new(&format!("/proc/{}", lock.pid)).exists() {
72            return true;
73        }
74    }
75
76    if lock.timestamp == 0 {
77        return false; // Legacy lockfile without timestamp — assume alive
78    }
79    let now = std::time::SystemTime::now()
80        .duration_since(std::time::UNIX_EPOCH)
81        .unwrap_or(std::time::Duration::ZERO)
82        .as_secs();
83    let stale_threshold = (lock.heartbeat_secs as f64 * 1.5) as u64;
84    now.saturating_sub(lock.timestamp) > stale_threshold
85}

87/// Read the bridge address from the lockfile, verifying PID liveness.
88pub fn read_bridge_address(home: &Path) -> io::Result<SocketAddr> {
89    let lock_path = home.join("mcp-bridge.lock");
90    let content = std::fs::read_to_string(&lock_path)?;
91    let lock: LockContent = serde_json::from_str(&content).map_err(|e| {
92        io::Error::new(io::ErrorKind::InvalidData, format!("invalid lockfile: {e}"))
93    })?;
94
95    // Check heartbeat staleness
96    if is_lockfile_stale(&lock) {
97        let _ = std::fs::remove_file(&lock_path);
98        return Err(io::Error::new(
99            io::ErrorKind::NotConnected,
100            "bridge process not running (stale lockfile)",
101        ));
102    }
103
104    lock.address.parse().map_err(|e| {
105        io::Error::new(
106            io::ErrorKind::InvalidData,
107            format!("invalid address in lockfile: {e}"),
108        )
109    })
110}

112/// Ensure the bridge daemon is running, spawning it if necessary.
113///
114/// Uses a spawn-mutex file (`mcp-bridge-spawning.lock`) to prevent concurrent
115/// callers from each spawning their own bridge instance. The mutex is held only
116/// during the spawn+poll window and removed once the bridge lockfile appears.
117///
118/// Flow:
119/// 1. Fast path: bridge lockfile exists and is fresh — return its address.
120/// 2. Acquire spawn-mutex (O_CREAT | O_EXCL). If another process holds it,
121///    skip spawning and just poll for the bridge lockfile.
122/// 3. Spawn bridge, poll for lockfile (up to 10s), release spawn-mutex.
123pub fn ensure_bridge_running(home: &Path) -> io::Result<SocketAddr> {
124    // Fast path: bridge is already running.
125    if let Ok(addr) = read_bridge_address(home) {
126        return Ok(addr);
127    }
128
129    let spawn_mutex = home.join("mcp-bridge-spawning.lock");
130
131    // Try to acquire the spawn-mutex exclusively.
132    let we_spawn = std::fs::OpenOptions::new()
133        .write(true)
134        .create_new(true)
135        .open(&spawn_mutex)
136        .is_ok();
137
138    if we_spawn {
139        // We won the race — spawn the bridge.
140        let result = spawn_bridge(home);
141        // Release mutex before polling so other waiters can proceed.
142        let _ = std::fs::remove_file(&spawn_mutex);
143        result?;
144    }
145    // Whether we spawned or another process did, poll until the bridge is up.
146    poll_for_bridge(home)
147}

149/// Spawn `chibi-mcp-bridge` as a detached background process.
150fn spawn_bridge(home: &Path) -> io::Result<()> {
151    let bridge_bin = which_bridge()?;
152    let mut cmd = std::process::Command::new(&bridge_bin);
153    if let Some(home_str) = home.to_str() {
154        cmd.env("CHIBI_HOME", home_str);
155    }
156    cmd.stdin(std::process::Stdio::null())
157        .stdout(std::process::Stdio::null())
158        .stderr(std::process::Stdio::null());
159    cmd.spawn()
160        .map_err(|e| io::Error::other(format!("failed to spawn chibi-mcp-bridge: {e}")))?;
161    Ok(())
162}

164/// Poll for the bridge lockfile to appear (up to 10s).
165fn poll_for_bridge(home: &Path) -> io::Result<SocketAddr> {
166    let deadline = std::time::Instant::now() + std::time::Duration::from_secs(10);
167    loop {
168        std::thread::sleep(std::time::Duration::from_millis(100));
169        if let Ok(addr) = read_bridge_address(home) {
170            return Ok(addr);
171        }
172        if std::time::Instant::now() >= deadline {
173            return Err(io::Error::new(
174                io::ErrorKind::TimedOut,
175                "bridge daemon did not start within 10 seconds",
176            ));
177        }
178    }
179}

181/// Locate the `chibi-mcp-bridge` binary.
182///
183/// Checks next to the current executable first, then falls back to PATH.
184fn which_bridge() -> io::Result<PathBuf> {
185    // Check next to the current executable first
186    if let Ok(exe) = std::env::current_exe() {
187        let sibling = exe.with_file_name("chibi-mcp-bridge");
188        if sibling.exists() {
189            return Ok(sibling);
190        }
191    }
192    // Fall back to PATH — Command::new will resolve it
193    Ok(PathBuf::from("chibi-mcp-bridge"))
194}

196/// Send a JSON request to the bridge and read the response.
197pub fn send_request(addr: SocketAddr, request: &str) -> io::Result<String> {
198    use std::io::{Read, Write};
199    use std::net::TcpStream;
200
201    let mut stream = TcpStream::connect(addr)?;
202    stream.set_read_timeout(Some(std::time::Duration::from_secs(30)))?;
203    stream.write_all(request.as_bytes())?;
204    stream.shutdown(std::net::Shutdown::Write)?;
205
206    let mut response = String::new();
207    stream.read_to_string(&mut response)?;
208    Ok(response)
209}

240/// Load MCP tools from the bridge daemon.
241///
242/// Returns an empty vec if the bridge is not running and cannot be started
243/// (e.g., no config file or binary not found).
244pub fn load_mcp_tools(home: &Path) -> io::Result<Vec<Tool>> {
245    // Only attempt if config file exists
246    if !home.join("mcp-bridge.toml").exists() {
247        return Ok(vec![]);
248    }
249
250    let addr = ensure_bridge_running(home)?;
251    let response = send_request(addr, r#"{"op":"list_tools"}"#)?;
252    let parsed: ListToolsResponse = serde_json::from_str(&response).map_err(|e| {
253        io::Error::new(
254            io::ErrorKind::InvalidData,
255            format!("invalid bridge response: {e}"),
256        )
257    })?;
258
259    if !parsed.ok {
260        return Err(io::Error::other(
261            parsed.error.unwrap_or_else(|| "bridge error".into()),
262        ));
263    }
264
265    Ok(parsed
266        .tools
267        .into_iter()
268        .map(|t| mcp_tool_from_info(&t.server, &t.name, &t.description, t.parameters))
269        .collect())
270}

272/// Execute an MCP tool via the bridge daemon.
273pub fn execute_mcp_tool(tool: &Tool, args: &serde_json::Value, home: &Path) -> io::Result<String> {
274    let (server, tool_name) = parse_mcp_path(&tool.path).ok_or_else(|| {
275        io::Error::new(
276            io::ErrorKind::InvalidInput,
277            format!("not an MCP tool path: {:?}", tool.path),
278        )
279    })?;
280
281    let addr = read_bridge_address(home).or_else(|_| ensure_bridge_running(home))?;
282
283    let request = serde_json::json!({
284        "op": "call_tool",
285        "server": server,
286        "tool": tool_name,
287        "args": args,
288    });
289
290    let response = send_request(addr, &request.to_string())?;
291    let parsed: CallToolResponse = serde_json::from_str(&response).map_err(|e| {
292        io::Error::new(
293            io::ErrorKind::InvalidData,
294            format!("invalid bridge response: {e}"),
295        )
296    })?;
297
298    if !parsed.ok {
299        return Err(io::Error::other(
300            parsed.error.unwrap_or_else(|| "MCP tool error".into()),
301        ));
302    }
303
304    Ok(parsed.result.unwrap_or_default())
305}
