24/// Async future type for tool handlers.
25///
26/// Not `Send` — tool dispatch runs on a single tokio task via `join_all` (see
27/// `send.rs`), so futures never cross thread boundaries. `AppState` and `Vfs`
28/// contain `RefCell` fields that are `!Sync`, which would prevent `Send` anyway.
29pub type BoxFuture<'a, T> = Pin<Box<dyn Future<Output = T> + 'a>>;
30
31/// Uniform async handler. Captures its own state at registration time.
32/// All runtime values come through `ToolCall.context`.
33///
34/// `Send + Sync` on the closure itself allows the `Arc<ToolHandler>` to be
35/// shared across threads (e.g. during registry reads). The *future* returned
36/// is intentionally `!Send` — see `BoxFuture`.
37pub type ToolHandler = Arc<dyn Fn(ToolCall<'_>) -> BoxFuture<'_, io::Result<String>> + Send + Sync>;
38
39/// Runtime context passed per-call. Carries values not known at registration time.
40pub struct ToolCallContext<'a> {
41    pub app: &'a AppState,
42    pub context_name: &'a str,
43    pub config: &'a ResolvedConfig,
44    pub project_root: &'a Path,
45    pub vfs: &'a Vfs,
46    pub vfs_caller: VfsCaller<'a>,
47}
48
49/// Input to a tool handler.
50pub struct ToolCall<'a> {
51    pub name: &'a str,
52    pub args: &'a Value,
53    pub context: &'a ToolCallContext<'a>,
54}

56/// How a tool is implemented — the registry's dispatch discriminant.
57///
58/// Replaces `Tool.path: PathBuf`. Each variant carries only the info needed
59/// to execute that tool type.
60pub enum ToolImpl {
61    /// Built-in Rust handler. The closure captures its own dependencies.
62    Builtin(ToolHandler),
63    /// OS-path plugin executable (spawned as subprocess).
64    Plugin(PathBuf),
65    /// MCP bridge tool (JSON-over-TCP to mcp-bridge daemon).
66    Mcp { server: String, tool_name: String },
67    /// Scheme tool loaded from VFS source via tein. Context is shared across
68    /// all tools in the same `.scm` file (Arc). `exec_binding` names the
69    /// scheme binding to call: `"tool-execute"` for single-tool convention
70    /// format, `"%tool-execute-{name}%"` for `define-tool` multi-tool files.
71    /// `registry` is the owning registry, passed to `execute_synthesised` so
72    /// `call-tool` uses a per-call registry instead of a global static.
73    ///
74    /// Mutation site: if exec_binding format changes, update `extract_single_tool`
75    /// and `extract_multi_tools` in synthesised.rs.
76    #[cfg(feature = "synthesised-tools")]
77    Synthesised {
78        vfs_path: crate::vfs::VfsPath,
79        exec_binding: String,
80        context: std::sync::Arc<super::synthesised::TeinSession>,
81        registry: Arc<RwLock<ToolRegistry>>,
82        /// The tein worker thread's `ThreadId`, captured at context init time.
83        /// Used as the key in `BRIDGE_CALL_CTX` so concurrent synthesised tool
84        /// calls from different tein contexts never overwrite each other's entry.
85        worker_thread_id: std::thread::ThreadId,
86        /// Maps hook points to scheme binding names for hook callbacks.
87        /// Populated from `%hook-registry%` during tool loading.
88        hook_bindings: std::collections::HashMap<super::hooks::HookPoint, String>,
89    },
90}

92impl Clone for ToolImpl {
93    fn clone(&self) -> Self {
94        match self {
95            ToolImpl::Builtin(h) => ToolImpl::Builtin(h.clone()),
96            ToolImpl::Plugin(p) => ToolImpl::Plugin(p.clone()),
97            ToolImpl::Mcp { server, tool_name } => ToolImpl::Mcp {
98                server: server.clone(),
99                tool_name: tool_name.clone(),
100            },
101            #[cfg(feature = "synthesised-tools")]
102            ToolImpl::Synthesised {
103                vfs_path,
104                exec_binding,
105                context,
106                registry,
107                worker_thread_id,
108                hook_bindings,
109            } => ToolImpl::Synthesised {
110                vfs_path: vfs_path.clone(),
111                exec_binding: exec_binding.clone(),
112                context: context.clone(),
113                registry: Arc::clone(registry),
114                worker_thread_id: *worker_thread_id,
115                hook_bindings: hook_bindings.clone(),
116            },
117        }
118    }
119}

161/// Single source of truth for all tools at runtime.
162///
163/// - `IndexMap` — O(1) lookup, preserves insertion order for deterministic
164///   tool lists sent to the LLM.
165/// - Pure dispatch: `dispatch_with_context` finds the tool and calls its handler.
166///   Hooks, permissions, and caching stay in `send.rs` as middleware.
167pub struct ToolRegistry {
168    tools: IndexMap<String, Tool>,
169}
170
171impl ToolRegistry {
172    pub fn new() -> Self {
173        Self {
174            tools: IndexMap::new(),
175        }
176    }
177
178    /// Register a tool. Replaces any existing tool with the same name (hot-reload).
179    pub fn register(&mut self, tool: Tool) {
180        self.tools.insert(tool.name.clone(), tool);
181    }
182
183    /// Remove a tool by name. Uses `shift_remove` (not `swap_remove`) to keep
184    /// insertion order stable for remaining tools — order is deterministic for
185    /// the LLM tool list.
186    pub fn unregister(&mut self, name: &str) -> Option<Tool> {
187        self.tools.shift_remove(name)
188    }
189
190    /// Find all synthesised tool names whose VFS path matches `path`.
191    ///
192    /// Returns a `Vec<String>` of tool names so the caller can unregister or
193    /// replace them. Handles multi-tool files (multiple tools per `.scm` file).
194    #[cfg(feature = "synthesised-tools")]
195    pub fn find_all_by_vfs_path(&self, path: &crate::vfs::VfsPath) -> Vec<String> {
196        self.tools
197            .values()
198            .filter(
199                |t| matches!(&t.r#impl, ToolImpl::Synthesised { vfs_path, .. } if vfs_path == path),
200            )
201            .map(|t| t.name.clone())
202            .collect()
203    }

205    /// Check if a synthesised tool is visible to the given context.
206    ///
207    /// Visibility rules (non-synthesised tools always visible):
208    /// - `/tools/shared/*` — visible to all contexts
209    /// - `/tools/home/<ctx>/*` — visible only to context `<ctx>`
210    /// - `/tools/flocks/<flock>/*` — visible to contexts that are members of `<flock>`
211    /// - Unknown zone — visible by default (forward-compat)
212    ///
213    /// `flock_memberships` is the list of flock names the context belongs to.
214    /// Returns `true` if tool is not found (caller's filter can then skip it).
215    #[cfg(feature = "synthesised-tools")]
216    pub fn is_tool_visible(
217        &self,
218        tool_name: &str,
219        context_name: &str,
220        flock_memberships: &[String],
221    ) -> bool {
222        let tool = match self.get(tool_name) {
223            Some(t) => t,
224            None => return true, // not found — let other filters handle it
225        };
226        match &tool.r#impl {
227            ToolImpl::Synthesised { vfs_path, .. } => {
228                let path = vfs_path.as_str();
229                if path.starts_with("/tools/shared/") {
230                    true
231                } else if let Some(rest) = path.strip_prefix("/tools/home/") {
232                    // /tools/home/alice/foo.scm → owner is "alice"
233                    rest.split('/').next() == Some(context_name)
234                } else if let Some(rest) = path.strip_prefix("/tools/flocks/") {
235                    // /tools/flocks/dev-team/foo.scm → flock is "dev-team"
236                    rest.split('/')
237                        .next()
238                        .map(|flock| flock_memberships.iter().any(|f| f == flock))
239                        .unwrap_or(false)
240                } else {
241                    true // unknown zone — visible by default
242                }
243            }
244            _ => true, // non-synthesised tools always visible
245        }
246    }

248    /// Look up by name. O(1).
249    pub fn get(&self, name: &str) -> Option<&Tool> {
250        self.tools.get(name)
251    }
252
253    /// All tools in registration order.
254    pub fn all(&self) -> impl Iterator<Item = &Tool> {
255        self.tools.values()
256    }
257
258    /// All tools matching a predicate.
259    pub fn filter(&self, pred: impl Fn(&Tool) -> bool) -> Vec<&Tool> {
260        self.tools.values().filter(|t| pred(t)).collect()
261    }

263    /// Dispatch an already-cloned `ToolImpl` without holding `&self`.
264    ///
265    /// Use this when the caller needs to release an `RwLockReadGuard` before
266    /// the `.await` — clone the `ToolImpl` while holding the lock, drop the
267    /// guard, then call `dispatch_impl`.
268    pub async fn dispatch_impl(
269        tool_impl: ToolImpl,
270        name: &str,
271        args: &Value,
272        ctx: &ToolCallContext<'_>,
273    ) -> io::Result<String> {
274        let call = ToolCall {
275            name,
276            args,
277            context: ctx,
278        };
279        match tool_impl {
280            ToolImpl::Builtin(handler) => handler(call).await,
281            ToolImpl::Plugin(path) => super::plugins::execute_tool_by_path(&path, name, args),
282            ToolImpl::Mcp { server, tool_name } => {
283                let home = ctx.app.chibi_dir.clone();
284                super::mcp::execute_mcp_call(&server, &tool_name, args, &home)
285            }
286            #[cfg(feature = "synthesised-tools")]
287            ToolImpl::Synthesised {
288                context,
289                exec_binding,
290                registry,
291                worker_thread_id,
292                ..
293            } => {
294                super::synthesised::execute_synthesised(
295                    context.as_ref(),
296                    &exec_binding,
297                    &call,
298                    registry,
299                    worker_thread_id,
300                )
301                .await
302            }
303        }
304    }
305
306    /// Dispatch a tool call with runtime context. Pure dispatch — no hooks,
307    /// no permissions. Policy stays in `send.rs` as middleware.
308    ///
309    /// Clones `ToolImpl` before any `.await` so no borrow of `self` crosses
310    /// an async suspension point, then delegates to `dispatch_impl`.
311    pub async fn dispatch_with_context(
312        &self,
313        name: &str,
314        args: &Value,
315        ctx: &ToolCallContext<'_>,
316    ) -> io::Result<String> {
317        let tool_impl = self
318            .get(name)
319            .ok_or_else(|| {
320                io::Error::new(io::ErrorKind::NotFound, format!("unknown tool: {name}"))
321            })?
322            .r#impl
323            .clone();
324        Self::dispatch_impl(tool_impl, name, args, ctx).await
325    }
326}

489    fn make_synth_tool(name: &str, vfs_path: &str) -> Tool {
490        // minimal synthesised tool stub for visibility tests
491        // uses a fresh context that evaluates to a trivial single-tool
492        let path = crate::vfs::VfsPath::new(vfs_path).unwrap();

497(define tool-description "stub")
498(define tool-parameters '())
499(define (tool-execute args) "stub")"#
500        );
501        crate::tools::synthesised::load_tool_from_source(&source, &path, &registry).unwrap()
502    }
