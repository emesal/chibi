98/// Register all fs_write tools into the registry.
99pub fn register_fs_write_tools(registry: &mut super::registry::ToolRegistry) {
100    use super::Tool;
101    use super::registry::{ToolCategory, ToolHandler};
102    use std::sync::Arc;
103
104    let handler: ToolHandler = Arc::new(|call| {
105        // execute_fs_write_tool is sync — extract result before the async block so
106        // no !Sync references (&Vfs) cross an .await point.
107        let ctx = call.context;
108        let result = execute_fs_write_tool(
109            call.name,
110            call.args,
111            ctx.project_root,
112            ctx.config,
113            ctx.vfs,
114            ctx.vfs_caller,
115        )
116        .unwrap_or_else(|| {
117            Err(io::Error::new(
118                io::ErrorKind::NotFound,
119                format!("unknown fs_write tool: {}", call.name),
120            ))
121        });
122        Box::pin(async move { result })
123    });
124
125    for def in FS_WRITE_TOOL_DEFS {
126        registry.register(Tool::from_builtin_def(
127            def,
128            handler.clone(),
129            ToolCategory::FsWrite,
130        ));
131    }
132}

136/// Execute an fs_write tool by name.
137///
138/// Returns `Some(result)` when the tool name is recognised, `None` otherwise.
139/// Note: permission gating (PreFileWrite hook) must be applied by the caller.
140pub fn execute_fs_write_tool(
141    tool_name: &str,
142    args: &serde_json::Value,
143    project_root: &Path,
144    config: &ResolvedConfig,
145    vfs: &Vfs,
146    caller: VfsCaller<'_>,
147) -> Option<io::Result<String>> {
148    match tool_name {
149        WRITE_FILE_TOOL_NAME => {
150            let path = require_str_param(args, "path");
151            let content = require_str_param(args, "content");
152            match (path, content) {
153                (Ok(p), Ok(c)) => Some(execute_write_file(&p, &c, Some((vfs, caller)))),
154                (Err(e), _) | (_, Err(e)) => Some(Err(e)),
155            }
156        }
157        FILE_EDIT_TOOL_NAME => Some(execute_file_edit(args, project_root, config, vfs, caller)),
158        _ => None,
159    }
160}

164/// Execute write_file: write content to a file or VFS path.
165///
166/// Note: Permission check via pre_file_write hook happens in send.rs before this is called.
167///
168/// When `path` starts with `vfs:///`, the write is routed through the VFS.
169/// The `vfs` parameter must be `Some((vfs, caller))` for VFS writes.
170pub fn execute_write_file(
171    path: &str,
172    content: &str,
173    vfs: Option<(&Vfs, VfsCaller<'_>)>,
174) -> io::Result<String> {
175    if VfsPath::is_vfs_uri(path) {
176        let vfs_path = VfsPath::from_uri(path)?;
177        let (vfs, caller) =
178            vfs.ok_or_else(|| io::Error::other("VFS not available for vfs:// path"))?;
179        vfs_block_on(vfs.write(caller, &vfs_path, content.as_bytes()))?;
180        return Ok(format!(
181            "File written successfully: {} ({} bytes)",
182            vfs_path,
183            content.len()
184        ));
185    }
186
187    let path = PathBuf::from(path);
188
189    // Create parent directories if needed
190    if let Some(parent) = path.parent()
191        && !parent.as_os_str().is_empty()
192    {
193        std::fs::create_dir_all(parent)?;
194    }
195
196    crate::safe_io::atomic_write_text(&path, content)?;
197
198    Ok(format!(
199        "File written successfully: {} ({} bytes)",
200        path.display(),
201        content.len()
202    ))
203}

207/// Edit operation variants
208enum EditOperation {
209    ReplaceLines {
210        start: usize,
211        end: usize,
212        content: String,
213    },
214    InsertBefore {
215        line: usize,
216        content: String,
217    },
218    InsertAfter {
219        line: usize,
220        content: String,
221    },
222    DeleteLines {
223        start: usize,
224        end: usize,
225    },
226    ReplaceString {
227        find: String,
228        replace: String,
229    },
230}

232/// Execute file_edit: apply a structured edit to a file atomically.
233///
234/// For VFS paths (`vfs:///...`), reads via `vfs.read()`, applies the edit
235/// in memory, and writes back via `vfs.write()`.
236pub fn execute_file_edit(
237    args: &serde_json::Value,
238    project_root: &Path,
239    config: &ResolvedConfig,
240    vfs: &Vfs,
241    caller: VfsCaller<'_>,
242) -> io::Result<String> {
243    let path_str = require_str_param(args, "path")?;
244    let operation_str = require_str_param(args, "operation")?;
245    let op = parse_edit_operation(&operation_str, args)?;
246
247    let resolved = resolve_tool_path(&path_str, project_root, config)?;
248
249    match resolved {
250        ResolvedPath::Os(file_path) => execute_file_edit_os(&file_path, op),
251        ResolvedPath::Vfs(vfs_path) => execute_file_edit_vfs(vfs, caller, &vfs_path, op),
252    }
253}
254
255/// Apply an edit operation to an OS file.
256fn execute_file_edit_os(file_path: &Path, op: EditOperation) -> io::Result<String> {
257    let content = std::fs::read_to_string(file_path).map_err(|e| {
258        io::Error::new(
259            e.kind(),
260            format!("Failed to read '{}': {}", file_path.display(), e),
261        )
262    })?;
263
264    let (new_content, summary) = apply_edit(op, &content, &file_path.display().to_string())?;
265    crate::safe_io::atomic_write_text(file_path, &new_content)?;
266    Ok(summary)
267}

269/// Apply an edit operation to a VFS file.
270///
271/// Read is world-readable by design (all zones can read all zones — see
272/// `Vfs::check_read`). Write is zone-checked: only the owning zone or
273/// callers with explicit cross-zone write access may write. This asymmetry
274/// is intentional and documented in `docs/vfs.md`.
275fn execute_file_edit_vfs(
276    vfs: &Vfs,
277    caller: VfsCaller<'_>,
278    vfs_path: &VfsPath,
279    op: EditOperation,
280) -> io::Result<String> {
281    let data = vfs_block_on(vfs.read(caller, vfs_path))?;
282    let content = String::from_utf8_lossy(&data).into_owned();
283
284    let (new_content, summary) = apply_edit(op, &content, vfs_path.as_str())?;
285    vfs_block_on(vfs.write(caller, vfs_path, new_content.as_bytes()))?;
286    Ok(summary)
287}

289/// Apply an edit operation to content, returning (new_content, summary).
290///
291/// Shared by both OS and VFS edit paths — the edit logic is identical,
292/// only read/write differs.
293fn apply_edit(
294    op: EditOperation,
295    content: &str,
296    display_path: &str,
297) -> io::Result<(String, String)> {
298    let mut lines: Vec<String> = content.lines().map(String::from).collect();
299    let had_trailing_newline = content.ends_with('\n');
300
301    match op {
302        EditOperation::ReplaceLines {
303            start,
304            end,
305            content: new_content,
306        } => {
307            validate_line_range(start, end, lines.len())?;
308            let new_lines: Vec<String> = new_content.lines().map(String::from).collect();
309            lines.splice((start - 1)..end, new_lines);
310        }
311        EditOperation::InsertBefore {
312            line,
313            content: new_content,
314        } => {
315            validate_line_number(line, lines.len())?;
316            let new_lines: Vec<String> = new_content.lines().map(String::from).collect();
317            let insert_at = line - 1;
318            for (i, l) in new_lines.into_iter().enumerate() {
319                lines.insert(insert_at + i, l);
320            }
321        }
322        EditOperation::InsertAfter {
323            line,
324            content: new_content,
325        } => {
326            validate_line_number(line, lines.len())?;
327            let new_lines: Vec<String> = new_content.lines().map(String::from).collect();
328            let insert_at = line;
329            for (i, l) in new_lines.into_iter().enumerate() {
330                lines.insert(insert_at + i, l);
331            }
332        }
333        EditOperation::DeleteLines { start, end } => {
334            validate_line_range(start, end, lines.len())?;
335            lines.drain((start - 1)..end);
336        }
337        EditOperation::ReplaceString { find, replace } => {
338            if !content.contains(&find) {
339                return Err(io::Error::new(
340                    ErrorKind::NotFound,
341                    format!("String not found in file: {:?}", find),
342                ));
343            }
344            // `replacen` operates on the raw string, so the trailing newline is preserved
345            // naturally if the original had one. Early return is correct here; the line-based
346            // newline restoration below is not needed.
347            let new_content = content.replacen(&find, &replace, 1);
348            return Ok((
349                new_content,
350                format!(
351                    "Replaced string in {} ({} → {} bytes)",
352                    display_path,
353                    find.len(),
354                    replace.len()
355                ),
356            ));
357        }
358    }
359
360    // Reassemble lines and restore trailing newline
361    let line_count = lines.len();
362    let mut new_content = lines.join("\n");
363    if had_trailing_newline || !new_content.is_empty() {
364        new_content.push('\n');
365    }
366
367    Ok((
368        new_content,
369        format!("Edited {} ({} lines)", display_path, line_count),
370    ))
371}

373/// Parse the operation string and required parameters into an `EditOperation`.
374fn parse_edit_operation(op: &str, args: &serde_json::Value) -> io::Result<EditOperation> {
375    match op {
376        "replace_lines" => {
377            let start = require_line_param(args, "line_start")?;
378            let end = require_line_param(args, "line_end")?;
379            let content = require_str_param(args, "content")?;
380            Ok(EditOperation::ReplaceLines {
381                start,
382                end,
383                content,
384            })
385        }
386        "insert_before" => {
387            let line = require_line_param(args, "line_start")?;
388            let content = require_str_param(args, "content")?;
389            Ok(EditOperation::InsertBefore { line, content })
390        }
391        "insert_after" => {
392            let line = require_line_param(args, "line_start")?;
393            let content = require_str_param(args, "content")?;
394            Ok(EditOperation::InsertAfter { line, content })
395        }
396        "delete_lines" => {
397            let start = require_line_param(args, "line_start")?;
398            let end = require_line_param(args, "line_end")?;
399            Ok(EditOperation::DeleteLines { start, end })
400        }
401        "replace_string" => {
402            let find = require_str_param(args, "find")?;
403            let replace = require_str_param(args, "replace")?;
404            Ok(EditOperation::ReplaceString { find, replace })
405        }
406        other => Err(io::Error::new(
407            ErrorKind::InvalidInput,
408            format!(
409                "Unknown operation '{}'. Valid: replace_lines, insert_before, insert_after, delete_lines, replace_string",
410                other
411            ),
412        )),
413    }
414}

416/// Extract a required 1-indexed line number parameter.
417fn require_line_param(args: &serde_json::Value, name: &str) -> io::Result<usize> {
418    use crate::json_ext::JsonExt;
419    let n = args.get_u64(name).ok_or_else(|| {
420        io::Error::new(
421            ErrorKind::InvalidInput,
422            format!("Missing '{}' parameter", name),
423        )
424    })? as usize;
425    if n == 0 {
426        return Err(io::Error::new(
427            ErrorKind::InvalidInput,
428            format!("'{}' must be >= 1 (line numbers are 1-indexed)", name),
429        ));
430    }
431    Ok(n)
432}

434/// Validate that a single 1-indexed line number is within the file.
435fn validate_line_number(line: usize, total: usize) -> io::Result<()> {
436    if line > total {
437        return Err(io::Error::new(
438            ErrorKind::InvalidInput,
439            format!("Line {} is out of bounds (file has {} lines)", line, total),
440        ));
441    }
442    Ok(())
443}
444
445/// Validate that a 1-indexed inclusive line range is valid for the given file.
446fn validate_line_range(start: usize, end: usize, total: usize) -> io::Result<()> {
447    if end < start {
448        return Err(io::Error::new(
449            ErrorKind::InvalidInput,
450            format!("line_end ({}) must be >= line_start ({})", end, start),
451        ));
452    }
453    if end > total {
454        return Err(io::Error::new(
455            ErrorKind::InvalidInput,
456            format!(
457                "line_end {} is out of bounds (file has {} lines)",
458                end, total
459            ),
460        ));
461    }
462    Ok(())
463}
