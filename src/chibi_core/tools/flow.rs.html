171/// Target for control handoff after tool execution
172#[derive(Debug, Clone)]
173pub enum HandoffTarget {
174    /// Continue with LLM processing
175    Agent { prompt: String },
176    /// Return control to user
177    User { message: String },
178}
179
180impl Default for HandoffTarget {
181    fn default() -> Self {
182        Self::Agent {
183            prompt: String::new(),
184        }
185    }
186}
187
188/// Tracks handoff decision during tool execution.
189/// Last explicit call wins; falls back to configured default.
190#[derive(Debug)]
191pub struct Handoff {
192    next: Option<HandoffTarget>,
193    fallback: HandoffTarget,
194}
195
196impl Handoff {
197    pub fn new(fallback: HandoffTarget) -> Self {
198        Self {
199            next: None,
200            fallback,
201        }
202    }
203
204    pub fn set_agent(&mut self, prompt: String) {
205        self.next = Some(HandoffTarget::Agent { prompt });
206    }
207
208    pub fn set_user(&mut self, message: String) {
209        self.next = Some(HandoffTarget::User { message });
210    }
211
212    /// Take the handoff decision, resetting to fallback for next use
213    pub fn take(&mut self) -> HandoffTarget {
214        self.next.take().unwrap_or_else(|| self.fallback.clone())
215    }
216
217    /// Override the fallback target (used by hooks)
218    pub fn set_fallback(&mut self, target: HandoffTarget) {
219        self.fallback = target;
220    }
221
222    /// Check if an explicit end-turn (call_user) has been requested
223    pub fn ends_turn_requested(&self) -> bool {
224        matches!(self.next, Some(HandoffTarget::User { .. }))
225    }
226}

230/// Options for sub-agent spawning.
231/// Designed to grow as ratatoskr gains model metadata and presets.
232#[derive(Debug, Clone, Default)]
233pub struct SpawnOptions {
234    /// Model override (uses parent's model if None)
235    pub model: Option<String>,
236    /// Temperature override
237    pub temperature: Option<f32>,
238    /// Max tokens override
239    pub max_tokens: Option<usize>,
240    /// Preset capability name (e.g. "fast", "reasoning").
241    /// Resolved against `config.subagent_cost_tier`. Explicit model/temperature/max_tokens win over preset defaults.
242    pub preset: Option<String>,
243}
244
245impl SpawnOptions {
246    /// Parse spawn options from tool arguments.
247    pub fn from_args(args: &serde_json::Value) -> Self {
248        Self {
249            model: args.get_str("model").map(String::from),
250            temperature: args
251                .get("temperature")
252                .and_then(|v| v.as_f64())
253                .map(|f| f as f32),
254            max_tokens: args
255                .get("max_tokens")
256                .and_then(|v| v.as_u64())
257                .map(|n| n as usize),
258            preset: args.get_str("preset").map(String::from),
259        }
260    }
261}

263/// Apply `PresetParameters` as defaults to `ApiParams`.
264/// Fills `None` fields only — never overwrites `Some` values set by the caller.
265fn apply_preset_defaults(params: &ratatoskr::PresetParameters, api: &mut crate::config::ApiParams) {
266    macro_rules! fill {
267        ($field:ident) => {
268            if api.$field.is_none() {
269                api.$field = params.$field.clone();
270            }
271        };
272    }
273    fill!(temperature);
274    fill!(top_p);
275    fill!(max_tokens);
276    fill!(frequency_penalty);
277    fill!(presence_penalty);
278    fill!(seed);
279    fill!(stop);
280    fill!(parallel_tool_calls);
281    // Note: top_k, reasoning, tool_choice, response_format, cache_prompt,
282    // raw_provider_options are in PresetParameters but not in ApiParams — skip.
283}

285/// Apply spawn options to a cloned config, returning the effective config.
286/// `gateway` is used for preset resolution when `opts.preset` is set.
287fn apply_spawn_options(
288    config: &ResolvedConfig,
289    opts: &SpawnOptions,
290    gateway: Option<&ratatoskr::EmbeddedGateway>,
291) -> ResolvedConfig {
292    let mut c = config.clone();
293
294    // Resolve preset first (explicit opts override preset defaults)
295    if let (Some(capability), Some(gw)) = (opts.preset.as_deref(), gateway) {
296        use ratatoskr::ModelGateway;
297        match gw.resolve_preset(&config.subagent_cost_tier, capability) {
298            Some(resolution) => {
299                c.model = resolution.model;
300                if let Some(params) = resolution.parameters {
301                    apply_preset_defaults(&params, &mut c.api);
302                }
303            }
304            None => {
305                eprintln!(
306                    "[WARN] spawn_agent: no preset for tier '{}' / capability '{}' — using parent model",
307                    config.subagent_cost_tier, capability
308                );
309            }
310        }
311    }
312
313    // Explicit overrides win over preset defaults
314    if let Some(ref model) = opts.model {
315        c.model = model.clone();
316    }
317    if let Some(temp) = opts.temperature {
318        c.api.temperature = Some(temp);
319    }
320    if let Some(max) = opts.max_tokens {
321        c.api.max_tokens = Some(max);
322    }
323    c
324}

347/// Reusable async primitive — both tool-facing and internal.
348/// Fires pre/post_spawn_agent hooks.
349pub async fn spawn_agent(
350    config: &ResolvedConfig,
351    system_prompt: &str,
352    input: &str,
353    options: &SpawnOptions,
354    tools: &[Tool],
355) -> io::Result<String> {
356    let gateway = gateway::build_gateway(config).ok();
357    let effective_config = apply_spawn_options(config, options, gateway.as_ref());
358
359    let hook_data = json!({
360        "system_prompt": system_prompt,
361        "input": input,
362        "model": effective_config.model,
363        "temperature": effective_config.api.temperature,
364        "max_tokens": effective_config.api.max_tokens,
365    });
366    let hook_results = execute_hook(tools, HookPoint::PreSpawnAgent, &hook_data, None)?;
367
368    for (_hook_name, result) in &hook_results {
369        if let Some(response) = result.get_str("response") {
370            return Ok(response.to_string());
371        }
372        if result.get_bool_or("block", false) {
373            let message = result
374                .get_str_or("message", "Sub-agent call blocked by hook")
375                .to_string();
376            return Ok(message);
377        }
378    }
379
380    let messages = vec![
381        json!({ "role": "system", "content": system_prompt }),
382        json!({ "role": "user", "content": input }),
383    ];
384
385    let response = gateway::chat(&effective_config, &messages).await?;
386
387    let post_hook_data = json!({
388        "system_prompt": system_prompt,
389        "input": input,
390        "model": effective_config.model,
391        "response": response,
392    });
393    let _ = execute_hook(tools, HookPoint::PostSpawnAgent, &post_hook_data, None);
394
395    Ok(response)
396}

398/// Reads file or fetches URL, then delegates to spawn_agent for processing.
399pub async fn summarize_content(
400    config: &ResolvedConfig,
401    source: &str,
402    instructions: &str,
403    options: &SpawnOptions,
404    tools: &[Tool],
405) -> io::Result<String> {
406    let content = if is_url(source) {
407        fetch_url(source).await?
408    } else {
409        read_file(source, config)?
410    };
411
412    if content.is_empty() {
413        return Ok(format!("Source '{}' is empty.", source));
414    }
415
416    let system_prompt = SUMMARIZE_CONTENT_SYSTEM_PROMPT.trim();
417    let input = format!(
418        "## Instructions\n{}\n\n## Content from: {}\n{}",
419        instructions, source, content
420    );
421
422    spawn_agent(config, system_prompt, &input, options, tools).await
423}

427/// Get metadata for flow tools.
428pub fn flow_tool_metadata(name: &str) -> ToolMetadata {
429    match name {
430        // call_agent: disabled as LLM tool but metadata retained for fallback mechanism
431        // and hook overrides. See FLOW_TOOL_DEFS comment for full rationale.
432        CALL_AGENT_TOOL_NAME => ToolMetadata {
433            parallel: false,
434            flow_control: true,
435            ends_turn: false,
436        },
437        CALL_USER_TOOL_NAME => ToolMetadata {
438            parallel: false,
439            flow_control: true,
440            ends_turn: true,
441        },
442        SPAWN_AGENT_TOOL_NAME => ToolMetadata {
443            parallel: false,
444            flow_control: false,
445            ends_turn: false,
446        },
447        _ => ToolMetadata::new(),
448    }
449}

451/// Execute an async flow tool (spawn_agent, summarize_content).
452///
453/// Returns `Ok(Some(result))` if handled, `Ok(None)` if not a flow tool.
454/// Note: call_agent, call_user, send_message, model_info are handled specially
455/// by the dispatcher in `api/send.rs` due to their async/hook requirements.
456pub async fn execute_flow_tool(
457    config: &ResolvedConfig,
458    tool_name: &str,
459    args: &serde_json::Value,
460    tools: &[Tool],
461) -> io::Result<Option<String>> {
462    let options = SpawnOptions::from_args(args);
463    match tool_name {
464        SPAWN_AGENT_TOOL_NAME => {
465            let system_prompt = args.get_str("system_prompt").ok_or_else(|| {
466                io::Error::new(ErrorKind::InvalidInput, "Missing 'system_prompt' parameter")
467            })?;
468            let input = args.get_str("input").ok_or_else(|| {
469                io::Error::new(ErrorKind::InvalidInput, "Missing 'input' parameter")
470            })?;
471            spawn_agent(config, system_prompt, input, &options, tools)
472                .await
473                .map(Some)
474        }
475        SUMMARIZE_CONTENT_TOOL_NAME => {
476            let source = args.get_str("source").ok_or_else(|| {
477                io::Error::new(ErrorKind::InvalidInput, "Missing 'source' parameter")
478            })?;
479            let instructions = args.get_str("instructions").ok_or_else(|| {
480                io::Error::new(ErrorKind::InvalidInput, "Missing 'instructions' parameter")
481            })?;
482            summarize_content(config, source, instructions, &options, tools)
483                .await
484                .map(Some)
485        }
486        _ => Ok(None),
487    }
488}

490/// Register all flow tools into the registry with per-tool metadata overrides.
491///
492/// Flow tools that are intercepted by `send.rs` middleware (send_message,
493/// call_user, model_info) are registered for metadata/lookup purposes but their
494/// handler returns an error if directly dispatched — `send.rs` must intercept
495/// them before calling `registry.dispatch_with_context`.
496pub fn register_flow_tools(registry: &mut super::registry::ToolRegistry) {
497    use super::Tool;
498    use super::registry::{ToolCategory, ToolHandler};
499    use std::sync::Arc;
500
501    for def in FLOW_TOOL_DEFS {
502        let name = def.name;
503        // Per-tool metadata: flow tools have non-default ToolMetadata values.
504        let metadata = flow_tool_metadata(name);
505
506        let handler: ToolHandler = Arc::new(move |call| {
507            let ctx = call.context;
508            let config = ctx.config;
509            let tool_name = call.name;
510            let args = call.args;
511            Box::pin(async move {
512                execute_flow_tool(config, tool_name, args, &[])
513                    .await
514                    // io::Result<Option<String>> -> io::Result<String>
515                    .and_then(|opt| {
516                        opt.ok_or_else(|| {
517                            io::Error::other(format!(
518                                "flow tool '{tool_name}' is handled by send.rs middleware \
519                                 and must not be dispatched through the registry directly"
520                            ))
521                        })
522                    })
523            })
524        });
525
526        // Construct with per-tool metadata override (from_builtin_def uses ToolMetadata::new()).
527        let tool = Tool {
528            name: def.name.to_string(),
529            description: def.description.to_string(),
530            parameters: def.to_json_schema(),
531            hooks: vec![],
532            metadata,
533            summary_params: def.summary_params.iter().map(|s| s.to_string()).collect(),
534            r#impl: super::registry::ToolImpl::Builtin(handler),
535            category: ToolCategory::Flow,
536        };
537        registry.register(tool);
538    }
539}

541/// Build the dynamic description for `spawn_agent`'s `preset` parameter.
542///
543/// `preset_capabilities`: capability names available via the configured cost tier.
544/// If empty, the description notes no presets are configured.
545///
546/// Used by `send.rs` to patch the description inline while iterating the registry,
547/// keeping tool-list construction in a single registry pass.
548pub fn spawn_agent_preset_description(preset_capabilities: &[String]) -> String {
549    if preset_capabilities.is_empty() {
550        "Preset capability name (no presets configured for this tier)".to_string()
551    } else {
552        format!(
553            "Preset capability name — one of: {} (cost tier set by config). \
554             Sets the model and default parameters for the sub-agent. \
555             Explicit model/temperature/max_tokens override preset defaults.",
556            preset_capabilities.join(", ")
557        )
558    }
559}

561/// Convert all flow tools to API format.
562///
563/// `preset_capabilities`: capability names available via the configured cost tier.
564/// If empty, the `preset` param description notes no presets are configured.
565///
566/// Kept for use in tests. Production code uses registry iteration + inline patching.
567#[cfg(test)]
568pub fn all_flow_tools_to_api_format(preset_capabilities: &[&str]) -> Vec<serde_json::Value> {
569    let preset_desc = spawn_agent_preset_description(
570        &preset_capabilities
571            .iter()
572            .map(|s| s.to_string())
573            .collect::<Vec<_>>(),
574    );
575
576    FLOW_TOOL_DEFS
577        .iter()
578        .map(|def| {
579            let mut json = def.to_api_format();
580            if def.name == SPAWN_AGENT_TOOL_NAME {
581                json["function"]["parameters"]["properties"]["preset"]["description"] =
582                    serde_json::Value::String(preset_desc.clone());
583            }
584            json
585        })
586        .collect()
587}
