1//! Canonical path resolver for all file-accessing builtin tools.
2//!
3//! Every tool that touches the filesystem or VFS goes through
4//! [`resolve_tool_path`] — there is no other path resolver.
5
6use crate::config::ResolvedConfig;
7use crate::vfs::VfsPath;
8use std::io::{self, ErrorKind};
9use std::path::{Path, PathBuf};
10
11/// A resolved tool path — either an OS path or a VFS path.
12///
13/// Returned by [`resolve_tool_path`] so callers can branch on OS vs VFS
14/// without re-parsing the URI.
15#[derive(Debug)]
16pub(crate) enum ResolvedPath {
17    Os(PathBuf),
18    Vfs(VfsPath),
19}
20
21impl ResolvedPath {
22    /// Extract the OS path, returning an error for VFS paths.
23    ///
24    /// Used by tools that operate exclusively on the OS filesystem
25    /// (e.g. `dir_list`, `glob_files`, `grep_files`).
26    pub(crate) fn require_os(self, tool_name: &str) -> io::Result<PathBuf> {
27        match self {
28            Self::Os(p) => Ok(p),
29            Self::Vfs(_) => Err(io::Error::new(
30                ErrorKind::InvalidInput,
31                format!("{} does not support vfs:// paths", tool_name),
32            )),
33        }
34    }
35}
36
37/// Resolve a tool path string into a validated [`ResolvedPath`].
38///
39/// - `vfs:///` URIs → [`ResolvedPath::Vfs`]
40/// - relative paths → joined with `project_root`, then security-validated
41/// - absolute paths → security-validated directly
42///
43/// All OS paths are checked against `config.file_tools_allowed_paths` via
44/// [`security::validate_file_path`](super::security::validate_file_path).
45pub(crate) fn resolve_tool_path(
46    path_str: &str,
47    project_root: &Path,
48    config: &ResolvedConfig,
49) -> io::Result<ResolvedPath> {
50    if VfsPath::is_vfs_uri(path_str) {
51        let vfs_path = VfsPath::from_uri(path_str)?;
52        return Ok(ResolvedPath::Vfs(vfs_path));
53    }
54
55    let resolved = if Path::new(path_str).is_relative() {
56        project_root.join(path_str).to_string_lossy().to_string()
57    } else {
58        path_str.to_string()
59    };
60    let validated = super::security::validate_file_path(&resolved, config)?;
61    Ok(ResolvedPath::Os(validated))
62}

64#[cfg(test)]
65mod tests {
66    use super::*;
67    use crate::config::ResolvedConfig;
68    use std::fs;
69    use tempfile::TempDir;
70
71    fn make_config(allowed: &[&str]) -> ResolvedConfig {
72        ResolvedConfig {
73            file_tools_allowed_paths: allowed.iter().map(|s| s.to_string()).collect(),
74            ..Default::default()
75        }
76    }
77
78    #[test]
79    fn relative_path_resolves_against_project_root() {
80        let dir = TempDir::new().unwrap();
81        let docs = dir.path().join("docs");
82        fs::create_dir_all(&docs).unwrap();
83        fs::write(docs.join("readme.md"), "hi").unwrap();
84
85        let config = make_config(&[dir.path().to_str().unwrap()]);
86        let result = resolve_tool_path("docs/readme.md", dir.path(), &config);
87        assert!(result.is_ok(), "relative path should resolve: {:?}", result);
88        match result.unwrap() {
89            ResolvedPath::Os(p) => {
90                assert!(p.ends_with("docs/readme.md"));
91            }
92            ResolvedPath::Vfs(_) => panic!("expected Os path"),
93        }
94    }
95
96    #[test]
97    fn absolute_path_ignores_project_root() {
98        let dir = TempDir::new().unwrap();
99        let file = dir.path().join("abs.txt");
100        fs::write(&file, "content").unwrap();
101
102        let config = make_config(&[dir.path().to_str().unwrap()]);
103        let result = resolve_tool_path(file.to_str().unwrap(), dir.path(), &config);
104        assert!(
105            result.is_ok(),
106            "absolute allowed path should work: {:?}",
107            result
108        );
109    }
110
111    #[test]
112    fn path_outside_allowed_dirs_is_rejected() {
113        let allowed_dir = TempDir::new().unwrap();
114        let other_dir = TempDir::new().unwrap();
115        let file = other_dir.path().join("secret.txt");
116        fs::write(&file, "secret").unwrap();
117
118        let config = make_config(&[allowed_dir.path().to_str().unwrap()]);
119        let result = resolve_tool_path(file.to_str().unwrap(), allowed_dir.path(), &config);
120        assert!(result.is_err(), "path outside allowed dirs should fail");
121    }
122
123    #[test]
124    fn relative_path_outside_allowed_dirs_is_rejected() {
125        let project_dir = TempDir::new().unwrap();
126        let allowed_dir = TempDir::new().unwrap();
127        // project_root differs from the allowed dir
128        fs::write(project_dir.path().join("secret.txt"), "x").unwrap();
129
130        let config = make_config(&[allowed_dir.path().to_str().unwrap()]);
131        let result = resolve_tool_path("secret.txt", project_dir.path(), &config);
132        assert!(
133            result.is_err(),
134            "relative path outside allowed dirs should fail"

138    #[test]
139    fn vfs_uri_returns_vfs_path() {
140        let config = make_config(&[]);
141        let result = resolve_tool_path("vfs:///shared/notes.txt", Path::new("/unused"), &config);
142        assert!(result.is_ok());
143        match result.unwrap() {
144            ResolvedPath::Vfs(p) => assert_eq!(p.as_str(), "/shared/notes.txt"),
145            ResolvedPath::Os(_) => panic!("expected Vfs path"),
146        }
147    }

149    #[test]
150    fn invalid_vfs_uri_errors() {
151        let config = make_config(&[]);
152        let root = Path::new("/some/root");
153        assert!(resolve_tool_path("vfs:////etc", root, &config).is_err());
154        assert!(resolve_tool_path("vfs:///../etc", root, &config).is_err());
155    }

157    #[test]
158    fn require_os_extracts_os_path() {
159        let p = PathBuf::from("/some/path");
160        let resolved = ResolvedPath::Os(p.clone());
161        assert_eq!(resolved.require_os("test").unwrap(), p);
162    }
