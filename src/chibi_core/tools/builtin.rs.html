36/// Property definition for a tool parameter
37pub struct ToolPropertyDef {
38    pub name: &'static str,
39    pub prop_type: &'static str,
40    pub description: &'static str,
41    /// Optional default value (for integer defaults only, as used by file tools)
42    pub default: Option<i64>,
43}
44
45/// Built-in tool definition for declarative registry
46pub struct BuiltinToolDef {
47    pub name: &'static str,
48    pub description: &'static str,
49    pub properties: &'static [ToolPropertyDef],
50    pub required: &'static [&'static str],
51}
52
53impl BuiltinToolDef {
54    /// Convert this tool definition to API format
55    pub fn to_api_format(&self) -> serde_json::Value {
56        let mut props = serde_json::Map::new();
57        for prop in self.properties {
58            let mut prop_obj = serde_json::json!({
59                "type": prop.prop_type,
60                "description": prop.description,
61            });
62            if let Some(default) = prop.default {
63                prop_obj["default"] = serde_json::json!(default);
64            }
65            props.insert(prop.name.to_string(), prop_obj);
66        }
67
68        serde_json::json!({
69            "type": "function",
70            "function": {
71                "name": self.name,
72                "description": self.description,
73                "parameters": {
74                    "type": "object",
75                    "properties": props,
76                    "required": self.required,
77                }
78            }
79        })
80    }
81}

167/// Look up a specific builtin tool definition by name.
168/// Returns None if not found. Use this for testing or conditional tool access.
169pub fn get_builtin_tool_def(name: &str) -> Option<&'static BuiltinToolDef> {
170    BUILTIN_TOOL_DEFS.iter().find(|def| def.name == name)
171}

173/// Convert all built-in tools to API format
174pub fn all_builtin_tools_to_api_format() -> Vec<serde_json::Value> {
175    BUILTIN_TOOL_DEFS
176        .iter()
177        .map(|def| def.to_api_format())
178        .collect()
179}
180
181/// Convert built-in tools to API format, optionally excluding reflection tool
182pub fn builtin_tools_to_api_format(include_reflection: bool) -> Vec<serde_json::Value> {
183    BUILTIN_TOOL_DEFS
184        .iter()
185        .filter(|def| include_reflection || def.name != REFLECTION_TOOL_NAME)
186        .map(|def| def.to_api_format())
187        .collect()
188}

211/// Target for control handoff after tool execution
212#[derive(Debug, Clone)]
213pub enum HandoffTarget {
214    /// Continue with LLM processing
215    Agent { prompt: String },
216    /// Return control to user
217    User { message: String },
218}
219
220impl Default for HandoffTarget {
221    fn default() -> Self {
222        Self::Agent {
223            prompt: String::new(),
224        }
225    }
226}
227
228/// Tracks handoff decision during tool execution.
229/// Last explicit call wins; falls back to configured default.
230#[derive(Debug)]
231pub struct Handoff {
232    next: Option<HandoffTarget>,
233    fallback: HandoffTarget,
234}
235
236impl Handoff {
237    pub fn new(fallback: HandoffTarget) -> Self {
238        Self {
239            next: None,
240            fallback,
241        }
242    }
243
244    pub fn set_agent(&mut self, prompt: String) {
245        self.next = Some(HandoffTarget::Agent { prompt });
246    }
247
248    pub fn set_user(&mut self, message: String) {
249        self.next = Some(HandoffTarget::User { message });
250    }
251
252    /// Take the handoff decision, resetting to fallback for next use
253    pub fn take(&mut self) -> HandoffTarget {
254        self.next.take().unwrap_or_else(|| self.fallback.clone())
255    }
256
257    /// Override the fallback target (used by hooks)
258    pub fn set_fallback(&mut self, target: HandoffTarget) {
259        self.fallback = target;
260    }
261}

265/// Execute the built-in update_reflection tool
266pub fn execute_reflection_tool(
267    prompts_dir: &Path,
268    arguments: &serde_json::Value,
269    character_limit: usize,
270) -> io::Result<String> {
271    let content = arguments["content"]
272        .as_str()
273        .ok_or_else(|| io::Error::new(ErrorKind::InvalidInput, "Missing 'content' parameter"))?;
274
275    // Check character limit
276    if content.len() > character_limit {
277        return Ok(format!(
278            "Error: Content exceeds the {} character limit ({} characters provided). Please shorten your reflection.",
279            character_limit,
280            content.len()
281        ));
282    }
283
284    let reflection_path = prompts_dir.join("reflection.md");
285    crate::safe_io::atomic_write_text(&reflection_path, content)?;
286
287    Ok(format!(
288        "Reflection updated successfully ({} characters).",
289        content.len()
290    ))
291}

293/// Execute a built-in tool by name.
294/// Returns Some(result) if the tool exists and was executed, None if not a built-in tool.
295///
296/// Note: When called from api.rs, send_message is handled separately to support
297/// hook integration. This function provides a basic send_message implementation
298/// for CLI usage (main.rs).
299pub fn execute_builtin_tool(
300    app: &AppState,
301    context_name: &str,
302    tool_name: &str,
303    args: &serde_json::Value,
304) -> Option<io::Result<String>> {
305    match tool_name {
306        TODOS_TOOL_NAME => {
307            let content = args.get("content").and_then(|v| v.as_str())?;
308            Some(
309                app.save_todos(context_name, content)
310                    .map(|_| format!("Todos updated ({} characters).", content.len())),
311            )
312        }
313        GOALS_TOOL_NAME => {
314            let content = args.get("content").and_then(|v| v.as_str())?;
315            Some(
316                app.save_goals(context_name, content)
317                    .map(|_| format!("Goals updated ({} characters).", content.len())),
318            )
319        }
320        REFLECTION_TOOL_NAME => Some(execute_reflection_tool(
321            &app.prompts_dir,
322            args,
323            app.config.reflection_character_limit,
324        )),
325        SEND_MESSAGE_TOOL_NAME => {
326            let to = args.get("to").and_then(|v| v.as_str())?;
327            let content = args.get("content").and_then(|v| v.as_str())?;
328            Some(
329                app.send_inbox_message_from(context_name, to, content)
330                    .map(|_| format!("Message sent to '{}'", to)),
331            )
332        }
333        _ => None,
334    }
335}
