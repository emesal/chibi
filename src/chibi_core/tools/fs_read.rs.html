222/// Register all fs_read tools into the registry.
223pub fn register_fs_read_tools(registry: &mut super::registry::ToolRegistry) {
224    use super::Tool;
225    use super::registry::{ToolCategory, ToolHandler};
226    use std::sync::Arc;
227
228    let handler: ToolHandler = Arc::new(|call| {
229        // execute_fs_read_tool is sync — extract result before the async block so
230        // no !Sync references (&AppState, &Vfs) cross an .await point.
231        let ctx = call.context;
232        let result = execute_fs_read_tool(
233            ctx.app,
234            ctx.context_name,
235            call.name,
236            call.args,
237            ctx.config,
238            ctx.project_root,
239        )
240        .unwrap_or_else(|| {
241            Err(io::Error::new(
242                io::ErrorKind::NotFound,
243                format!("unknown fs_read tool: {}", call.name),
244            ))
245        });
246        Box::pin(async move { result })
247    });
248
249    for def in FS_READ_TOOL_DEFS {
250        registry.register(Tool::from_builtin_def(
251            def,
252            handler.clone(),
253            ToolCategory::FsRead,
254        ));
255    }
256}

307// === VFS bridge ===
308/// Bridge an async VFS future into synchronous tool dispatch.
309///
310/// Uses `block_in_place` + `block_on` so that it works from sync code inside a
311/// tokio multi-thread runtime (e.g. tool dispatch in `execute_tool_pure`). The
312/// `block_in_place` call tells the runtime scheduler to move other tasks off
313/// this thread while we block.
314///
315/// **Runtime requirement:** `block_in_place` panics on `current_thread`
316/// runtimes. Any test that calls VFS tools must use:
317/// `#[tokio::test(flavor = "multi_thread", worker_threads = 2)]`
318///
319/// If called outside of any tokio runtime (e.g. plain `#[test]`), spins up a
320/// temporary current-thread runtime.
321pub(crate) fn vfs_block_on<F: std::future::Future>(f: F) -> F::Output {
322    match tokio::runtime::Handle::try_current() {
323        Ok(handle) => tokio::task::block_in_place(|| handle.block_on(f)),
324        Err(_) => {
325            // No runtime active — spin up a temporary one.
326            tokio::runtime::Builder::new_current_thread()
327                .enable_all()
328                .build()
329                .expect("failed to build temporary tokio runtime")
330                .block_on(f)
331        }
332    }
333}

335// === file_head / file_tail ===
336
337/// Direction for head/tail reading
338enum ReadDirection {
339    Head,
340    Tail,
341}
342
343/// Shared implementation for file_head and file_tail.
344fn execute_file_head_or_tail(
345    app: &AppState,
346    context_name: &str,
347    args: &serde_json::Value,
348    config: &ResolvedConfig,
349    direction: ReadDirection,
350    project_root: &Path,
351) -> io::Result<String> {
352    let path_str = args
353        .get_str("path")
354        .ok_or_else(|| io::Error::new(ErrorKind::InvalidInput, "Must provide path"))?;
355    let path = resolve_tool_path(path_str, project_root, config)?;
356    let n = args.get_u64_or("lines", 50) as usize;
357
358    match path {
359        ResolvedPath::Os(p) => match direction {
360            ReadDirection::Head => {
361                let file = std::fs::File::open(&p)?;
362                let reader = std::io::BufReader::new(file);
363                let lines: Vec<String> = reader.lines().take(n).collect::<Result<_, _>>()?;
364                Ok(lines.join("\n"))
365            }
366            ReadDirection::Tail => {
367                let content = std::fs::read_to_string(&p)?;
368                let lines: Vec<&str> = content.lines().collect();
369                let start = lines.len().saturating_sub(n);
370                Ok(lines[start..].join("\n"))
371            }
372        },
373        ResolvedPath::Vfs(vfs_path) => {
374            let data = vfs_block_on(app.vfs.read(VfsCaller::Context(context_name), &vfs_path))?;
375            let content = String::from_utf8_lossy(&data);
376            let all_lines: Vec<&str> = content.lines().collect();
377            let selected = match direction {
378                ReadDirection::Head => &all_lines[..all_lines.len().min(n)],
379                ReadDirection::Tail => {
380                    let start = all_lines.len().saturating_sub(n);
381                    &all_lines[start..]
382                }
383            };
384            Ok(selected.join("\n"))
385        }
386    }
387}

437/// Execute file_lines tool
438pub fn execute_file_lines(
439    app: &AppState,
440    context_name: &str,
441    args: &serde_json::Value,
442    config: &ResolvedConfig,
443    project_root: &Path,
444) -> io::Result<String> {
445    let path_str = args
446        .get_str("path")
447        .ok_or_else(|| io::Error::new(ErrorKind::InvalidInput, "Must provide path"))?;
448    let path = resolve_tool_path(path_str, project_root, config)?;
449    let start = require_u64_param(args, "start")? as usize;
450    let end = require_u64_param(args, "end")? as usize;
451
452    if start == 0 {
453        return Err(io::Error::new(
454            ErrorKind::InvalidInput,
455            "Line numbers are 1-indexed, start must be >= 1",
456        ));
457    }
458    if end < start {
459        return Err(io::Error::new(
460            ErrorKind::InvalidInput,
461            "End line must be >= start line",
462        ));
463    }
464
465    match path {
466        ResolvedPath::Os(p) => {
467            let file = std::fs::File::open(&p)?;
468            let reader = std::io::BufReader::new(file);
469            let lines: Vec<String> = reader
470                .lines()
471                .enumerate()
472                .filter(|(i, _)| *i >= start.saturating_sub(1) && *i < end)
473                .map(|(_, line)| line)
474                .collect::<Result<_, _>>()?;
475            Ok(lines.join("\n"))
476        }
477        ResolvedPath::Vfs(vfs_path) => {
478            let data = vfs_block_on(app.vfs.read(VfsCaller::Context(context_name), &vfs_path))?;
479            let content = String::from_utf8_lossy(&data);
480            let lines: Vec<&str> = content
481                .lines()
482                .enumerate()
483                .filter(|(i, _)| *i >= start.saturating_sub(1) && *i < end)
484                .map(|(_, line)| line)
485                .collect();
486            Ok(lines.join("\n"))
487        }
488    }
489}

493/// Execute file_grep tool
494pub fn execute_file_grep(
495    app: &AppState,
496    context_name: &str,
497    args: &serde_json::Value,
498    config: &ResolvedConfig,
499    project_root: &Path,
500) -> io::Result<String> {
501    let path_str = args
502        .get_str("path")
503        .ok_or_else(|| io::Error::new(ErrorKind::InvalidInput, "Must provide path"))?;
504    let path = resolve_tool_path(path_str, project_root, config)?;
505    let pattern = require_str_param(args, "pattern")?;
506    let context_before = args.get_u64_or("context_before", 2) as usize;
507    let context_after = args.get_u64_or("context_after", 2) as usize;
508
509    let result = match path {
510        ResolvedPath::Os(p) => {
511            let content = std::fs::read_to_string(&p)?;
512            grep_in_memory(&content, &pattern, context_before, context_after)?
513        }
514        ResolvedPath::Vfs(vfs_path) => {
515            let data = vfs_block_on(app.vfs.read(VfsCaller::Context(context_name), &vfs_path))?;
516            let content = String::from_utf8_lossy(&data).into_owned();
517            grep_in_memory(&content, &pattern, context_before, context_after)?
518        }
519    };
520
521    if result.is_empty() {
522        Ok(format!("No matches found for pattern: {}", pattern))
523    } else {
524        Ok(result)
525    }
526}

528/// In-memory grep with context lines.
529fn grep_in_memory(
530    content: &str,
531    pattern: &str,
532    context_before: usize,
533    context_after: usize,
534) -> io::Result<String> {
535    let lines: Vec<&str> = content.lines().collect();
536    let regex = regex::Regex::new(pattern)
537        .map_err(|e| io::Error::new(ErrorKind::InvalidInput, format!("Invalid regex: {}", e)))?;
538
539    let mut result = Vec::new();
540    let mut last_end = 0;
541
542    for (i, line) in lines.iter().enumerate() {
543        if regex.is_match(line) {
544            let start = i.saturating_sub(context_before);
545            let end = (i + context_after + 1).min(lines.len());
546
547            if start > last_end && !result.is_empty() {
548                result.push("--".to_string());
549            }
550
551            let range_start = start.max(last_end);
552            for (j, line) in lines.iter().enumerate().take(end).skip(range_start) {
553                let prefix = if j == i { ">" } else { " " };
554                result.push(format!("{}{}:{}", prefix, j + 1, line));
555            }
556
557            last_end = end;
558        }
559    }
560
561    Ok(result.join("\n"))
562}

566/// Execute dir_list: produce an indented directory tree with sizes and type tags.
567fn execute_dir_list(
568    args: &serde_json::Value,
569    project_root: &Path,
570    config: &ResolvedConfig,
571) -> io::Result<String> {
572    let path_str = args.get_str_or("path", ".");
573    let depth = args.get_u64_or("depth", 1) as usize;
574    let show_hidden = args.get_str_or("show_hidden", "false") == "true";
575
576    let root = resolve_tool_path(path_str, project_root, config)?.require_os("dir_list")?;
577
578    if !root.exists() {
579        return Err(io::Error::new(
580            ErrorKind::NotFound,
581            format!("Path not found: {}", root.display()),
582        ));
583    }
584    if !root.is_dir() {
585        return Err(io::Error::new(
586            ErrorKind::InvalidInput,
587            format!("Path is not a directory: {}", root.display()),
588        ));
589    }
590
591    let mut output = format!("{}/\n", root.display());
592    collect_dir_tree(&root, &mut output, 0, depth, show_hidden, "")?;
593    Ok(output)
594}
595
596/// Recursively collect a directory tree into `output`.
597fn collect_dir_tree(
598    dir: &Path,
599    output: &mut String,
600    current_depth: usize,
601    max_depth: usize,
602    show_hidden: bool,
603    prefix: &str,
604) -> io::Result<()> {
605    if current_depth >= max_depth {
606        return Ok(());
607    }
608
609    let mut entries: Vec<_> = std::fs::read_dir(dir)?
610        .filter_map(|e| e.ok())
611        .filter(|e| {
612            if show_hidden {
613                return true;
614            }
615            e.file_name()
616                .to_str()
617                .map(|n| !n.starts_with('.'))
618                .unwrap_or(true)
619        })
620        .collect();
621
622    // Directories first, then files; each group sorted by name
623    entries.sort_by(|a, b| {
624        let a_is_dir = a.file_type().map(|t| t.is_dir()).unwrap_or(false);
625        let b_is_dir = b.file_type().map(|t| t.is_dir()).unwrap_or(false);
626        match (a_is_dir, b_is_dir) {
627            (true, false) => std::cmp::Ordering::Less,
628            (false, true) => std::cmp::Ordering::Greater,
629            _ => a.file_name().cmp(&b.file_name()),
630        }
631    });
632
633    let count = entries.len();
634    for (i, entry) in entries.iter().enumerate() {
635        let is_last = i + 1 == count;
636        let connector = if is_last { "└── " } else { "├── " };
637        let child_prefix = format!("{}{}", prefix, if is_last { "    " } else { "│   " });
638
639        let file_type = entry.file_type()?;
640        let name = entry.file_name();
641        let name_str = name.to_string_lossy();
642
643        if file_type.is_dir() {
644            output.push_str(&format!("{}{}{}/\n", prefix, connector, name_str));
645            collect_dir_tree(
646                &entry.path(),
647                output,
648                current_depth + 1,
649                max_depth,
650                show_hidden,
651                &child_prefix,
652            )?;
653        } else {
654            let size = entry.metadata().map(|m| m.len()).unwrap_or(0);
655            output.push_str(&format!(
656                "{}{}{} ({})\n",
657                prefix,
658                connector,
659                name_str,
660                format_size(size),
661            ));
662        }
663    }
664    Ok(())
665}

667/// Format a byte count into a human-readable string.
668fn format_size(bytes: u64) -> String {
669    if bytes < 1024 {
670        format!("{} B", bytes)
671    } else if bytes < 1024 * 1024 {
672        format!("{:.1} KB", bytes as f64 / 1024.0)
673    } else {
674        format!("{:.1} MB", bytes as f64 / (1024.0 * 1024.0))
675    }
676}

692
693    let root = resolve_tool_path(path_str, project_root, config)?.require_os("glob_files")?;
694
695    if !root.exists() {
696        return Err(io::Error::new(
697            ErrorKind::NotFound,
698            format!("Path not found: {}", root.display()),
699        ));
700    }
701
702    let mut overrides = OverrideBuilder::new(&root);
703    overrides.add(&pattern).map_err(|e| {
704        io::Error::new(
705            ErrorKind::InvalidInput,
706            format!("Invalid glob pattern '{}': {}", pattern, e),
707        )
708    })?;
709    let overrides = overrides.build().map_err(|e| {
710        io::Error::new(
711            ErrorKind::InvalidInput,
712            format!("Failed to build glob matcher: {}", e),
713        )
714    })?;
715
716    let walker = WalkBuilder::new(&root).overrides(overrides).build();
717
718    let mut matches = Vec::new();
719    for entry in walker {
720        if matches.len() >= max_results {
721            break;
722        }
723        let entry = match entry {
724            Ok(e) => e,
725            Err(_) => continue,
726        };
727        if entry.file_type().map(|t| t.is_dir()).unwrap_or(false) {
728            continue;
729        }
730        let path = entry.path();
731        let rel = path.strip_prefix(&root).unwrap_or(path);
732        matches.push(rel.to_string_lossy().into_owned());
733    }
734
735    if matches.is_empty() {
736        Ok(format!("No files found matching pattern: {}", pattern))
737    } else {
738        Ok(matches.join("\n"))
739    }
740}

744/// Execute grep_files: regex-search files under a directory with optional context.
745fn execute_grep_files(
746    args: &serde_json::Value,
747    project_root: &Path,
748    config: &ResolvedConfig,
749) -> io::Result<String> {
750    use ignore::WalkBuilder;
751    use ignore::overrides::OverrideBuilder;
752    use regex::Regex;
753
754    let pattern = require_str_param(args, "pattern")?;
755    let path_str = args.get_str_or("path", ".");
756    let file_pattern = args.get_str("file_pattern").map(String::from);
757    let context_lines = args.get_u64_or("context_lines", 2) as usize;
758    let max_results = args.get_u64_or("max_results", 50) as usize;
759
760    let regex = Regex::new(&pattern).map_err(|e| {
761        io::Error::new(
762            ErrorKind::InvalidInput,
763            format!("Invalid regex pattern '{}': {}", pattern, e),
764        )
765    })?;
766
767    let root = resolve_tool_path(path_str, project_root, config)?.require_os("grep_files")?;
768
769    if !root.exists() {
770        return Err(io::Error::new(
771            ErrorKind::NotFound,
772            format!("Path not found: {}", root.display()),
773        ));
774    }
775
776    let mut builder = WalkBuilder::new(&root);
777    if let Some(ref fp) = file_pattern {
778        let mut overrides = OverrideBuilder::new(&root);
779        overrides.add(fp).map_err(|e| {
780            io::Error::new(
781                ErrorKind::InvalidInput,
782                format!("Invalid file pattern '{}': {}", fp, e),
783            )
784        })?;
785        let ov = overrides.build().map_err(|e| {
786            io::Error::new(
787                ErrorKind::InvalidInput,
788                format!("Failed to build file pattern matcher: {}", e),
789            )
790        })?;
791        builder.overrides(ov);
792    }
793
794    let mut output = String::new();
795    let mut total_matches = 0;
796
797    'walk: for entry in builder.build() {
798        let entry = match entry {
799            Ok(e) => e,
800            Err(_) => continue,
801        };
802        if entry.file_type().map(|t| t.is_dir()).unwrap_or(true) {
803            continue;
804        }
805
806        let file_path = entry.path();
807        let rel_path = file_path
808            .strip_prefix(&root)
809            .unwrap_or(file_path)
810            .to_string_lossy()
811            .into_owned();
812
813        let file = match std::fs::File::open(file_path) {
814            Ok(f) => f,
815            Err(_) => continue,
816        };
817        let reader = std::io::BufReader::new(file);
818        let lines: Vec<String> = reader.lines().map(|l| l.unwrap_or_default()).collect();
819
820        let match_indices: Vec<usize> = lines
821            .iter()
822            .enumerate()
823            .filter(|(_, line)| regex.is_match(line))
824            .map(|(i, _)| i)
825            .collect();
826
827        if match_indices.is_empty() {
828            continue;
829        }
830
831        let mut ranges: Vec<(usize, usize)> = Vec::new();
832        for &idx in &match_indices {
833            let start = idx.saturating_sub(context_lines);
834            let end = (idx + context_lines + 1).min(lines.len());
835            if let Some(last) = ranges.last_mut()
836                && start <= last.1
837            {
838                last.1 = last.1.max(end);
839                continue;
840            }
841            ranges.push((start, end));
842        }
843
844        let match_set: std::collections::HashSet<usize> = match_indices.into_iter().collect();
845
846        for (start, end) in ranges {
847            #[allow(clippy::needless_range_loop)]
848            for i in start..end {
849                let line_num = i + 1;
850                if match_set.contains(&i) {
851                    output.push_str(&format!("{}:{}: {}\n", rel_path, line_num, lines[i]));
852                    total_matches += 1;
853                    if total_matches >= max_results {
854                        output.push_str(&format!(
855                            "\n[Truncated: {} results limit reached]\n",
856                            max_results
857                        ));
858                        break 'walk;
859                    }
860                } else {
861                    output.push_str(&format!("{}:{}-{}\n", rel_path, line_num, lines[i]));
862                }
863            }
864            output.push('\n');
865        }
866    }
867
868    if output.is_empty() {
869        Ok(format!("No matches found for pattern: {}", pattern))
870    } else {
871        Ok(output.trim_end().to_string())
872    }
873}
