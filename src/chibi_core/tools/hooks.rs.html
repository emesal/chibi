1//! Hook system for plugin lifecycle events.
2//!
3//! Hooks allow plugins to be notified at specific points during chibi's execution,
4//! such as before/after messages, tool calls, context switches, and compaction.

18// Tracks which hook points are currently being dispatched to tein callbacks.
19// Prevents re-entrancy: if a tein hook callback triggers an action that fires
20// the same hook point, tein callbacks are skipped on the recursive call.
21// Subprocess hooks still fire normally regardless.
22#[cfg(feature = "synthesised-tools")]
23thread_local! {
24    static TEIN_HOOK_GUARD: RefCell<HashSet<HookPoint>> = RefCell::new(HashSet::new());
25}

27/// Hook points where tools can register to be called
28#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, EnumString, EnumIter, AsRefStr)]
29#[strum(serialize_all = "snake_case")]
30pub enum HookPoint {
31    PreMessage,
32    PostMessage,
33    PreTool,
34    PostTool,
35    PreToolOutput,  // Before tool output is processed (can modify/block output)
36    PostToolOutput, // After tool output is processed (observe only)
37    PreClear,
38    PostClear,
39    PreCompact,
40    PostCompact,
41    PreRollingCompact,
42    PostRollingCompact,
43    OnStart,
44    OnEnd,
45    PreSystemPrompt,  // Can inject content before system prompt sections
46    PostSystemPrompt, // Can inject content after all system prompt sections
47    PreSendMessage,   // Can intercept delivery (return {"delivered": true, "via": "..."})
48    PostSendMessage,  // Observe delivery (read-only)
49    PreCacheOutput,   // Before caching large tool output (can provide custom summary)
50    PostCacheOutput,  // After output is cached (notification only)
51    PreApiTools,      // Before tools are sent to API (can filter tools)
52    PreApiRequest,    // Before API request is sent (can modify full request body)
53    PreAgenticLoop,   // Before entering the tool loop (can override fallback)
54    PostToolBatch,    // After processing a batch of tool calls (can override fallback)
55    PreFileRead, // Before reading a file outside allowed paths (can approve/deny, fail-safe deny)
56    PreFileWrite, // Before file write/patch (can approve/deny/modify operation)
57    PreShellExec, // Before shell command execution (can approve/deny, fail-safe deny)
58    PreFetchUrl, // Before fetching a sensitive URL (can approve/deny, fail-safe deny)
59    PreSpawnAgent, // Before sub-agent call (can intercept/replace with {"response": "..."} or block)
60    PostSpawnAgent, // After sub-agent call (observe only)
61    PostIndexFile, // After a file is indexed (observe: path, lang, symbol_count, ref_count)
62    PreVfsWrite,   // Before a VFS file write (advisory, non-blocking; observe-and-snapshot)
63    PostVfsWrite,  // After a successful VFS file write (observe only)
64}

68/// Describes one field in a hook's payload or return value.
69#[cfg(feature = "synthesised-tools")]
70pub(crate) struct FieldMeta {
71    pub name: &'static str,
72    /// Schematic type string: "string", "number", "bool", "object", "array"
73    pub typ: &'static str,
74    pub description: &'static str,
75}
76
77/// Metadata for one hook point — the canonical single source of truth for
78/// hook contracts. Used to generate `hooks-docs` (scheme alist) and
79/// `docs/hooks.md` (markdown reference section).
80#[cfg(feature = "synthesised-tools")]
81pub(crate) struct HookMeta {
82    pub point: HookPoint,
83    /// Grouping category: "session", "message", "system_prompt", "tool",
84    /// "api", "agentic", "file_permission", "url_security", "agent",
85    /// "cache", "message_delivery", "index", "vfs_write", "context"
86    pub category: &'static str,
87    /// One-line description of when the hook fires.
88    pub description: &'static str,
89    /// Whether the hook callback can return values that modify behaviour.
90    pub can_modify: bool,
91    pub payload_fields: &'static [FieldMeta],
92    /// Empty slice for observe-only hooks.
93    pub return_fields: &'static [FieldMeta],
94    /// Extra context or caveats; empty string if none.
95    pub notes: &'static str,
96}

1315                        _ => "null".to_string(),
1316                    };
1317                    let comma = if i < last { "," } else { "" };
1318                    format!(
1319                        "  \"{}\": {}{}  // {}",
1320                        f.name, example, comma, f.description
1321                    )
1322                })
1323                .collect();

1325        }
1326
1327        // Payload JSON block
1328        if meta.payload_fields.is_empty() {
1329            out.push_str("Payload: (empty)\n");
1330        } else {
1331            out.push_str(&json_block(meta.payload_fields));
1332        }
1333
1334        if !meta.return_fields.is_empty() {
1335            out.push_str("\n**Can return:**\n");
1336            out.push_str(&json_block(meta.return_fields));
1337        }
1338
1339        if !meta.notes.is_empty() {
1340            out.push_str(&format!("\n> **Note:** {}\n", meta.notes));
1341        }
1342    }
1343

1375/// Execute a hook on all tools that registered for it
1376/// Returns a vector of (tool_name, result) for tools that returned non-empty output
1377///
1378/// Hook data is passed via stdin (JSON). The CHIBI_HOOK env var identifies which hook is firing.
1379///
1380/// `tein_ctx` (synthesised-tools feature only): when `Some`, sets `BRIDGE_CALL_CTX` per tein tool
1381/// during dispatch, enabling `call-tool` and `(harness io)` from tein hook callbacks.
1382/// Pass `None` from sync contexts or tests that lack a tokio runtime.
1383pub fn execute_hook(
1384    tools: &[Tool],
1385    hook: HookPoint,
1386    data: &serde_json::Value,
1387    _tein_ctx: Option<&TeinHookContext<'_>>,
1388) -> io::Result<Vec<(String, serde_json::Value)>> {
1389    let mut results = Vec::new();
1390    let data_str = data.to_string();
1391
1392    for tool in tools {
1393        if !tool.hooks.contains(&hook) {
1394            continue;
1395        }
1396
1397        // Only plugin tools can register hooks; extract the executable path.
1398        let plugin_path = match &tool.r#impl {
1399            super::ToolImpl::Plugin(p) => p.clone(),
1400            _ => continue, // non-plugin tools cannot spawn hooks
1401        };
1402        let mut child = Command::new(&plugin_path)
1403            .env("CHIBI_HOOK", hook.as_ref())
1404            .stdin(Stdio::piped())
1405            .stdout(Stdio::piped())
1406            .stderr(Stdio::inherit())
1407            .spawn()
1408            .map_err(|e| {
1409                io::Error::other(format!(
1410                    "Failed to spawn hook {} on {}: {}",
1411                    hook.as_ref(),
1412                    tool.name,
1413                    e
1414                ))
1415            })?;
1416
1417        // Write hook data to stdin (ignore BrokenPipe — child may exit before reading)
1418        if let Some(mut stdin) = child.stdin.take() {
1419            match stdin.write_all(data_str.as_bytes()) {
1420                Ok(()) => {}
1421                Err(e) if e.kind() == io::ErrorKind::BrokenPipe => {}
1422                Err(e) => return Err(e),
1423            }
1424            // stdin is dropped here, closing the pipe and signaling EOF
1425        }
1426
1427        let timeout = std::time::Duration::from_secs(super::PLUGIN_TIMEOUT_SECS);
1428        let context = format!("hook {} on {}", hook.as_ref(), tool.name);
1429        let output = super::wait_with_timeout(child, timeout, &context).map_err(|e| {
1430            io::Error::other(format!(
1431                "Failed to execute hook {} on {}: {}",
1432                hook.as_ref(),
1433                tool.name,
1434                e
1435            ))
1436        })?;
1437
1438        if !output.status.success() {
1439            continue;
1440        }
1441
1442        let stdout = String::from_utf8_lossy(&output.stdout);
1443        let trimmed = stdout.trim();
1444
1445        if trimmed.is_empty() {
1446            continue;
1447        }
1448
1449        // Try to parse as JSON, otherwise wrap as string
1450        let value: serde_json::Value = serde_json::from_str(trimmed)
1451            .unwrap_or_else(|_| serde_json::Value::String(trimmed.to_string()));
1452
1453        results.push((tool.name.clone(), value));
1454    }
1455
1456    // --- synthesised tein hooks ---
1457    #[cfg(feature = "synthesised-tools")]
1458    {
1459        let should_dispatch = TEIN_HOOK_GUARD.with(|guard| !guard.borrow().contains(&hook));
1460
1461        if should_dispatch {
1462            TEIN_HOOK_GUARD.with(|guard| {
1463                guard.borrow_mut().insert(hook);
1464            });
1465
1466            // Deduplicate: tools from the same .scm file share a tein context
1467            // and produce the same (worker_thread_id, binding) pair. Only fire
1468            // each unique (context instance, binding) once per hook event.
1469            let mut dispatched: std::collections::HashSet<(std::thread::ThreadId, String)> =
1470                std::collections::HashSet::new();
1471
1472            for tool in tools {
1473                if !tool.hooks.contains(&hook) {
1474                    continue;
1475                }
1476                let (context, hook_bindings, worker_thread_id) = match &tool.r#impl {
1477                    super::ToolImpl::Synthesised {
1478                        context,
1479                        hook_bindings,
1480                        worker_thread_id,
1481                        ..
1482                    } => (context, hook_bindings, *worker_thread_id),
1483                    _ => continue,
1484                };
1485
1486                // Set call context guard if tein_ctx available — enables call-tool
1487                // and (harness io) from tein hook callbacks.
1488                // Guard drops at end of each loop iteration, clearing the bridge context.
1489                let _bridge_guard = _tein_ctx.map(|ctx| {
1490                    super::synthesised::CallContextGuard::set_from_hook_ctx(ctx, worker_thread_id)
1491                });
1492
1493                let Some(binding) = hook_bindings.get(&hook) else {
1494                    continue;
1495                };
1496
1497                // Skip if this (context, binding) was already dispatched this event.
1498                if !dispatched.insert((worker_thread_id, binding.clone())) {
1499                    continue;
1500                }
1501
1502                let payload = match super::synthesised::json_args_to_scheme_alist(data) {
1503                    Ok(v) => v,
1504                    Err(e) => {
1505                        eprintln!(
1506                            "[WARN] tein hook {}: payload conversion: {e}",
1507                            hook.as_ref()
1508                        );
1509                        continue;
1510                    }
1511                };
1512
1513                let hook_fn = match context.evaluate(binding) {
1514                    Ok(v) => v,
1515                    Err(e) => {
1516                        eprintln!(
1517                            "[WARN] tein hook {} on {}: resolve {binding}: {e}",
1518                            hook.as_ref(),
1519                            tool.name
1520                        );
1521                        continue;
1522                    }
1523                };
1524
1525                let result = match context.call(&hook_fn, &[payload]) {
1526                    Ok(v) => v,
1527                    Err(e) => {
1528                        eprintln!("[WARN] tein hook {} on {}: {e}", hook.as_ref(), tool.name);
1529                        continue;
1530                    }
1531                };
1532
1533                // empty list or nil → no-op, don't push a result
1534                if result.is_nil() {
1535                    continue;
1536                }
1537                if matches!(&result, tein::Value::List(items) if items.is_empty()) {
1538                    continue;
1539                }
1540
1541                match super::synthesised::scheme_value_to_json(&result) {
1542                    Ok(value) => results.push((tool.name.clone(), value)),
1543                    Err(e) => {
1544                        eprintln!(
1545                            "[WARN] tein hook {} on {}: result conversion: {e}",
1546                            hook.as_ref(),
1547                            tool.name
1548                        );
1549                    }
1550                }
1551            }
1552
1553            TEIN_HOOK_GUARD.with(|guard| {
1554                guard.borrow_mut().remove(&hook);
1555            });
1556        }
1557    }
1558
1559    Ok(results)
1560}

1777    fn execute_hook_with_retry(
1778        tools: &[Tool],
1779        hook: HookPoint,
1780        data: &serde_json::Value,
1781    ) -> io::Result<Vec<(String, serde_json::Value)>> {
1782        for attempt in 0..5 {
1783            match execute_hook(tools, hook, data, None) {
1784                Ok(result) => return Ok(result),
1785                Err(e) if e.to_string().contains("Text file busy") && attempt < 4 => {
1786                    std::thread::sleep(std::time::Duration::from_millis(10 * (attempt + 1) as u64));
1787                    continue;
1788                }
1789                Err(e) => return Err(e),
1790            }
1791        }
1792        unreachable!()
1793    }
