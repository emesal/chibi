1//! Synthesised tool loader and executor for tein-backed scheme tools.
2//!
3//! A synthesised tool is a scheme source file (loaded from VFS) that defines
4//! five bindings:
5//!
6//! ```scheme
7//! (import (scheme base))   ; required — sandbox starts with null env
8//! (define tool-name        "my_tool")
9//! (define tool-description "what the tool does")
10//! (define tool-parameters
11//!   '((param-name . ((type . "string") (description . "what it is")))))
12//! (define (tool-execute args)
13//!   (cdr (assoc "param-name" args)))
14//! ```
15//!
16//! Alternatively, use `define-tool` for multi-tool files:
17//!
18//! ```scheme
19//! (import (scheme base))
20//! (import (harness tools))
21//!
22//! (define-tool greet
23//!   (description "greets someone")
24//!   (parameters '((name . ((type . "string") (description . "who to greet")))))
25//!   (execute (lambda (args)
26//!     (string-append "hello " (cdr (assoc "name" args))))))
27//! ```
28//!
29//! **Sandbox note:** synthesised tools run in a null environment — only `import`
30//! is available without an explicit import. You must `(import (scheme base))` to
31//! access `assoc`, `cons`, `car`, `cdr`, `number->string`, etc. Import additional
32//! modules (`(scheme list)`, `(scheme write)`, etc.) as needed.
33//!
34//! `tool-parameters` is a scheme alist mapping parameter names (symbols) to
35//! inner alists describing the JSON Schema properties (`type`, `description`,
36//! `required` — optional, defaults to true).
37//!
38//! `tool-execute` receives an alist built from the JSON tool-call args:
39//! `{"key": "val"}` → `(("key" . val) ...)`. Use `(assoc "key" args)` to
40//! extract values. Keys are scheme strings (not symbols).
41//!
42//! ## `(harness tools)` module
43//!
44//! Every synthesised tool context has `(harness tools)` available. It exports:
45//!
46//! - `call-tool` — invoke another tool by name: `(call-tool "name" args-alist)`
47//! - `define-tool` — macro for declaring multiple tools in one file
48//!
49//! ## `call-tool` bridge
50//!
51//! `call-tool` bridges sync tein → async tokio dispatch via
52//! `Handle::current().block_on()`. The registry and call context are stashed in
53//! `BRIDGE_CALL_CTX` before each invocation and cleared after via a guard.
54//! The registry is embedded in `ToolImpl::Synthesised` and flows through
55//! `execute_synthesised` → `CallContextGuard`, so concurrent calls never
56//! overwrite each other's registry.
57//!
58//! Each synthesised tool gets its own sandboxed tein context, shared via
59//! `Arc` for concurrent dispatch. All access goes through `ThreadLocalContext`,
60//! which is `Send + Sync`.

76/// Shared write buffer for capturing scheme output port data.
77/// Cloned `Arc` handles go into tein's custom output ports; the same
78/// `Arc` is held by `TeinSession` for reading after evaluation.
79#[cfg(feature = "synthesised-tools")]
80struct SharedWriter(Arc<Mutex<Vec<u8>>>);
81
82#[cfg(feature = "synthesised-tools")]
83impl std::io::Write for SharedWriter {
84    fn write(&mut self, buf: &[u8]) -> std::io::Result<usize> {
85        self.0.lock().unwrap().extend_from_slice(buf);
86        Ok(buf.len())
87    }
88    fn flush(&mut self) -> std::io::Result<()> {
89        Ok(())
90    }
91}

93/// Result of a scheme evaluation with captured output streams.
94///
95/// Holds the raw `Result<Value>` so callers can stringify as needed.
96/// `format_eval()` uses `to_string()` (for scheme_eval).
97/// `format_tool()` uses `as_string()` unwrap (for synthesised tools that
98/// conventionally return scheme strings).
99///
100/// Both format methods produce:
101/// ```text
102/// result: <value or "error: ...">
103/// stdout: <captured stdout or "(empty)">
104/// stderr: <captured stderr or "(empty)">
105/// ```
106#[cfg(feature = "synthesised-tools")]
107pub(crate) struct CapturedOutput {
108    value: tein::Result<tein::Value>,
109    pub stdout: String,
110    pub stderr: String,
111}
112
113/// Shared formatting for structured output. Produces:
114/// `"result: <value_str>\nstdout: <stdout or "(empty)">\nstderr: <stderr or "(empty)">"`
115#[cfg(feature = "synthesised-tools")]
116fn format_output(value_str: &str, stdout: &str, stderr: &str) -> String {
117    format!(
118        "result: {}\nstdout: {}\nstderr: {}",
119        value_str,
120        if stdout.is_empty() { "(empty)" } else { stdout },
121        if stderr.is_empty() { "(empty)" } else { stderr },
122    )
123}
124
125#[cfg(feature = "synthesised-tools")]
126impl CapturedOutput {
127    /// Format for `scheme_eval` -- stringifies value with `to_string()`.
128    pub fn format_eval(&self) -> String {
129        let value_str = match &self.value {
130            Ok(val) => val.to_string(),
131            Err(e) => format!("error: {e}"),
132        };
133        format_output(&value_str, &self.stdout, &self.stderr)
134    }
135
136    /// Format for synthesised tools -- unwraps scheme strings via `as_string()`,
137    /// falling back to `to_string()` for non-string values.
138    pub fn format_tool(&self) -> String {
139        let value_str = match &self.value {
140            Ok(val) => match val.as_string() {
141                Some(s) => s.to_string(),
142                None => val.to_string(),
143            },
144            Err(e) => format!("error: {e}"),
145        };
146        format_output(&value_str, &self.stdout, &self.stderr)
147    }
148}

150/// Wrapper around `ThreadLocalContext` that captures stdout/stderr output.
151///
152/// Created by `build_tein_context`. The init closure wires `SharedWriter`-backed
153/// output ports as `current-output-port` and `current-error-port`. The ports
154/// survive across evaluations (tein's `set_current_output_port` persists).
155///
156/// Thread safety: `ThreadLocalContext` serialises all calls via its internal
157/// channel mutex, so no interleaving is possible between drain and read in
158/// `with_capture`.
159#[cfg(feature = "synthesised-tools")]
160pub struct TeinSession {
161    ctx: ThreadLocalContext,
162    stdout_buf: Arc<Mutex<Vec<u8>>>,
163    stderr_buf: Arc<Mutex<Vec<u8>>>,
164}
165
166#[cfg(feature = "synthesised-tools")]
167impl TeinSession {
168    /// Run a closure with stdout/stderr capture.
169    ///
170    /// Drain-eval-flush-read cycle:
171    /// 1. Clear buffers (drain stale output from prior calls)
172    /// 2. Run the closure (evaluate or call)
173    /// 3. Flush scheme ports to ensure buffers are complete
174    /// 4. Read captured output
175    ///
176    /// The closure receives `&ThreadLocalContext` for `evaluate` or `call`.
177    pub(crate) fn with_capture<F>(&self, f: F) -> CapturedOutput
178    where
179        F: FnOnce(&ThreadLocalContext) -> tein::Result<tein::Value>,
180    {
181        // 1. flush-then-drain: flush chibi's internal port buffers first (emitting
182        //    any stale buffered data into SharedWriter), then discard. This ensures
183        //    the capture window starts clean — no bleed from previous calls.
184        //    flush-output-port is R7RS (scheme base); flush-output is chibi-only and
185        //    unavailable in sandboxed contexts.
186        //    Flush errors are non-fatal but mean the drain may be incomplete; stale
187        //    output could bleed into this call's capture. No tracing dep is available,
188        //    so errors are silently discarded — investigate if bleed is observed.
189        let _ = self
190            .ctx
191            .evaluate("(flush-output-port (current-output-port))");
192        let _ = self
193            .ctx
194            .evaluate("(flush-output-port (current-error-port))");
195        self.stdout_buf.lock().unwrap().clear();
196        self.stderr_buf.lock().unwrap().clear();
197
198        // 2. run
199        let value = f(&self.ctx);
200
201        // 3. flush -- push any remaining buffered output into SharedWriter.
202        //    Same caveat: flush errors mean captured output may be truncated.
203        let _ = self
204            .ctx
205            .evaluate("(flush-output-port (current-output-port))");
206        let _ = self
207            .ctx
208            .evaluate("(flush-output-port (current-error-port))");
209
210        // 4. read
211        let stdout = String::from_utf8_lossy(&self.stdout_buf.lock().unwrap()).to_string();
212        let stderr = String::from_utf8_lossy(&self.stderr_buf.lock().unwrap()).to_string();
213
214        CapturedOutput {
215            value,
216            stdout,
217            stderr,
218        }
219    }
220
221    /// Delegate to inner context for internal calls (e.g. setting `%context-name%`,
222    /// resolving bindings) that don't need output capture.
223    pub(crate) fn evaluate(&self, code: &str) -> tein::Result<tein::Value> {
224        self.ctx.evaluate(code)
225    }
226
227    /// Delegate to inner context for calling procedures without capture.
228    /// Use when stdout/stderr output is not needed (e.g. internal resolution,
229    /// hook dispatch where only the return value matters).
230    pub(crate) fn call(
231        &self,
232        proc: &tein::Value,
233        args: &[tein::Value],
234    ) -> tein::Result<tein::Value> {
235        self.ctx.call(proc, args)
236    }
237}

323/// Top-level scheme preamble evaluated in every synthesised tool context.
324///
325/// Defines `%tool-registry%`, `%hook-registry%`, `%context-name%`, `define-tool`,
326/// `register-hook`, `harness-tools-docs`, and `hooks-docs` at the top level.
327///
328/// `hooks-docs` is generated from `HOOK_METADATA` — the canonical single source
329/// of truth for hook contracts. `harness-tools-docs` covers the harness API.
330/// Both are re-exported via `(harness docs)` as the canonical access path.
331///
332/// Call `(import (harness docs))` then `(describe hooks-docs)` to list all hooks,
333/// or `(module-doc hooks-docs 'pre_message)` for a specific hook's contract.
334/// Note: `(describe X)` takes an alist directly, not a symbol.
335///
336/// `define-tool` must be top-level (not inside a library) so its `set!` of
337/// `%tool-registry%` affects the top-level binding that rust reads post-evaluation.
338///
339/// Mutation site: if `define-tool` syntax changes, update `extract_multi_tools`
340/// which parses `%tool-registry%` entries. If `harness-tools-docs` entries change,
341/// update `chibi.md` and `AGENTS.md` accordingly.
342///
343/// Built once via `LazyLock` since `hooks-docs` is generated at runtime from
344/// `HOOK_METADATA`. The allocation is reused across all context builds.
345#[cfg(feature = "synthesised-tools")]
346pub(crate) static HARNESS_PREAMBLE: std::sync::LazyLock<String> = std::sync::LazyLock::new(|| {
347    let hooks_docs_alist = crate::tools::hooks::generate_hooks_docs_alist();
348    format!(
349        r#"
350(import (scheme base))
351
352;; accumulates define-tool entries. each entry is a list:
353;; (name-string description-string params-value execute-procedure)
354(define %tool-registry% '())
355
356;; accumulates hook registrations. each entry is a list:
357;; (hook-name-string handler-procedure)
358;; rust reads %hook-registry% after evaluation to populate Tool.hooks.
359(define %hook-registry% '())
360
361;; name of the calling context — mutated by execute_synthesised before each call.
362;; plugins read this to resolve /home/<ctx>/... VFS paths.
363(define %context-name% "")
364
365;; docs alist for public harness APIs — use (import (harness docs)) then
366;; (describe harness-tools-docs) or (module-doc harness-tools-docs 'define-tool).
367;; follows the same convention as introspect-docs, json-docs, etc.
368;; note: (describe X) takes an alist directly, NOT a symbol.
369(define harness-tools-docs
370  '((__module__ . "harness tools")
371    (define-tool . "macro: (define-tool name (description DESC) [(category CAT)] [(summary-params (PARAM ...))] (parameters PARAMS-ALIST) (execute (lambda (args) ...))) — registers a persistent tool; args is ((\"key\" . val) ...) alist. Optional: category is a string like \"network\" or \"shell\"; summary-params is a list of parameter names used to build the permission-prompt summary for network tools without a URL parameter.")
372    (call-tool . "procedure: (call-tool NAME ARGS-ALIST) -> string — invoke another registered tool; NAME is a string, ARGS-ALIST is ((\"key\" . \"val\") ...)")
373    (register-hook . "procedure: (register-hook HOOK-SYMBOL HANDLER) — register a hook callback; HOOK-SYMBOL e.g. 'pre_vfs_write, HANDLER is (lambda (payload) ...)")
374    (generate-id . "procedure: (generate-id) -> string — returns an 8-hex-char random identifier (uuid v4 prefix)")
375    (current-timestamp . "procedure: (current-timestamp) -> string — returns current UTC time as \"YYYYMMDD-HHMMz\"")))
376
377;; docs alist for all hook points — use (import (harness docs)) then
378;; (describe hooks-docs) to list all hooks, or (module-doc hooks-docs 'pre_message).
379;; generated from HOOK_METADATA (hooks.rs) — single source of truth.
380(define hooks-docs
381  {hooks_docs_alist})
382
383;; registers a tool: appends to %tool-registry% in definition order (LIFO via cons).
384;; rust reads %tool-registry% after evaluation; non-empty → multi-tool mode.
385(define-syntax define-tool
386  (syntax-rules (description category summary-params parameters execute)
387    ;; pattern 1: baseline (no category, no summary-params)
388    ((define-tool name
389       (description desc)
390       (parameters params)
391       (execute handler))
392     (set! %tool-registry%
393       (cons (list (symbol->string 'name) desc params handler #f #f)
394             %tool-registry%)))
395    ;; pattern 2: category only
396    ((define-tool name
397       (description desc)
398       (category cat)
399       (parameters params)
400       (execute handler))
401     (set! %tool-registry%
402       (cons (list (symbol->string 'name) desc params handler cat #f)
403             %tool-registry%)))
404    ;; pattern 3: summary-params only
405    ((define-tool name
406       (description desc)
407       (summary-params sp)
408       (parameters params)
409       (execute handler))
410     (set! %tool-registry%
411       (cons (list (symbol->string 'name) desc params handler #f sp)
412             %tool-registry%)))
413    ;; pattern 4: category + summary-params
414    ((define-tool name
415       (description desc)
416       (category cat)
417       (summary-params sp)
418       (parameters params)
419       (execute handler))
420     (set! %tool-registry%
421       (cons (list (symbol->string 'name) desc params handler cat sp)
422             %tool-registry%)))))
423
424;; registers a hook handler for a given hook point.
425;; hook-name is a symbol (e.g. 'pre_vfs_write).
426;; handler is a procedure taking one argument (the hook payload as an alist)
427;; and returning an alist (or '() for no-op).
428(define (register-hook hook-name handler)
429  (set! %hook-registry%
430    (cons (list (symbol->string hook-name) handler)
431          %hook-registry%)))
432"#,
433        hooks_docs_alist = hooks_docs_alist,
434    )

439/// Subset of ToolCallContext that is owned (no lifetimes) for thread-local storage.
440///
441/// Raw pointers are used because `ToolCallContext` has lifetimes. They are valid
442/// for the duration of `execute_synthesised`, which holds `CallContextGuard`.
443///
444/// Mutation sites: if `ToolCallContext` fields change, update this struct and
445/// `CallContextGuard::set`. Also update `call_tool_bridge` reconstruction.
446#[cfg(feature = "synthesised-tools")]
447struct ActiveCallContext {
448    app: *const crate::state::AppState,
449    context_name: String,
450    config: *const crate::config::ResolvedConfig,
451    project_root: std::path::PathBuf,
452    vfs: *const Vfs,
453    /// Non-empty string = `VfsCaller::Context(name)`, empty = `VfsCaller::System`.
454    vfs_caller_context: String,
455    /// Tokio runtime handle from the caller thread. Used by `call_tool_fn` to
456    /// schedule async work from the tein worker thread (which is not a tokio thread
457    /// and cannot use `block_in_place`).
458    runtime_handle: tokio::runtime::Handle,
459    /// Shared tool registry. Used by `call_tool_fn` for per-call dispatch.
460    /// Embedded per-call so concurrent tests never overwrite each other's registry.
461    registry: Arc<RwLock<ToolRegistry>>,
462}
463
464// SAFETY: the pointers in `ActiveCallContext` are only dereferenced on the
465// tein worker thread while `CallContextGuard` is alive (i.e. within a single
466// `execute_synthesised` call on that thread). No other thread accesses them.
467#[cfg(feature = "synthesised-tools")]
468unsafe impl Send for ActiveCallContext {}
469
470/// Per-tein-worker-thread call context. Keyed by the tein worker thread's `ThreadId`
471/// so concurrent synthesised tool calls from different contexts never collide.
472///
473/// `call_tool_fn` runs on the tein worker thread, so `std::thread::current().id()`
474/// gives the correct lookup key. `CallContextGuard` inserts on set and removes on drop.
475#[cfg(feature = "synthesised-tools")]
476static BRIDGE_CALL_CTX: std::sync::LazyLock<
477    std::sync::Mutex<std::collections::HashMap<std::thread::ThreadId, ActiveCallContext>>,
478> = std::sync::LazyLock::new(|| std::sync::Mutex::new(std::collections::HashMap::new()));
479
480/// RAII guard that inserts/removes an entry in the thread-ID-keyed `BRIDGE_CALL_CTX` map.
481/// Used in `execute_synthesised` to make the context available to the `call-tool`
482/// bridge without threading it through tein's FFI boundary.
483#[cfg(feature = "synthesised-tools")]
484pub(crate) struct CallContextGuard {
485    thread_id: std::thread::ThreadId,
486}
487
488#[cfg(feature = "synthesised-tools")]
489impl CallContextGuard {
490    pub(crate) fn set(
491        ctx: &crate::tools::registry::ToolCallContext<'_>,
492        registry: Arc<RwLock<ToolRegistry>>,
493        thread_id: std::thread::ThreadId,
494    ) -> Self {
495        BRIDGE_CALL_CTX.lock().unwrap().insert(
496            thread_id,
497            ActiveCallContext {
498                app: ctx.app as *const _,
499                context_name: ctx.context_name.to_string(),
500                config: ctx.config as *const _,
501                project_root: ctx.project_root.to_path_buf(),
502                vfs: ctx.vfs as *const _,
503                vfs_caller_context: match ctx.vfs_caller {
504                    VfsCaller::Context(name) => name.to_string(),
505                    VfsCaller::System => String::new(),
506                },
507                runtime_handle: tokio::runtime::Handle::current(),
508                registry,
509            },
510        );
511        CallContextGuard { thread_id }
512    }
513
514    /// Set from a `TeinHookContext` — used during hook dispatch to enable
515    /// `call-tool` and `(harness io)` from tein hook callbacks.
516    ///
517    /// Must be called from a tokio thread (uses `Handle::current()`).
518    pub(crate) fn set_from_hook_ctx(
519        ctx: &crate::tools::hooks::TeinHookContext<'_>,
520        worker_thread_id: std::thread::ThreadId,
521    ) -> Self {
522        BRIDGE_CALL_CTX.lock().unwrap().insert(
523            worker_thread_id,
524            ActiveCallContext {
525                app: ctx.app as *const _,
526                context_name: ctx.context_name.to_string(),
527                config: ctx.config as *const _,
528                project_root: ctx.project_root.to_path_buf(),
529                vfs: ctx.vfs as *const _,
530                vfs_caller_context: String::new(), // System caller for hook dispatch
531                runtime_handle: tokio::runtime::Handle::current(),
532                registry: Arc::clone(&ctx.registry),
533            },
534        );
535        CallContextGuard {
536            thread_id: worker_thread_id,
537        }
538    }
539}

550/// The `call-tool` foreign function: `(call-tool name args-alist) → string`
551///
552/// Reads `BRIDGE_CALL_CTX` (which carries the registry for the current call).
553/// Converts the scheme alist args to JSON, looks up the tool, and dispatches
554/// via `ToolRegistry::dispatch_impl` on the current tokio runtime.
555///
556/// Error handling: returns a scheme error string on failure (tein surfacees it
557/// as a scheme error condition). Does not panic.
558///
559/// Must be registered with `ctx.define_fn_variadic("call-tool", call_tool_bridge)`
560/// before `HARNESS_TOOLS_MODULE` is evaluated, so the re-export resolves.
561#[cfg(feature = "synthesised-tools")]
562#[tein::tein_fn(name = "call-tool")]
563fn call_tool_fn(name: String, args: Value) -> Result<String, String> {
564    let json_args = scheme_value_to_json(&args).map_err(|e| format!("args conversion: {e}"))?;
565
566    // Extract all needed data from the per-thread context while holding the lock,
567    // then drop the lock before dispatching (which may block_on async code).
568    let (
569        app_ptr,
570        context_name,
571        config_ptr,
572        project_root,
573        vfs_ptr,
574        vfs_caller_str,
575        runtime_handle,
576        registry,
577    ) = {
578        let tid = std::thread::current().id();
579        let guard = BRIDGE_CALL_CTX.lock().unwrap();
580        let active = guard.get(&tid).ok_or_else(|| {
581            "call-tool: no active call context for this thread (called outside tool execute?)"
582                .to_string()
583        })?;
584        (
585            active.app,
586            active.context_name.clone(),
587            active.config,
588            active.project_root.clone(),
589            active.vfs,
590            active.vfs_caller_context.clone(),
591            active.runtime_handle.clone(),
592            Arc::clone(&active.registry),
593        )
594        // guard drops here, releasing the lock
595    };
596
597    // SAFETY: pointers are valid for the duration of execute_synthesised,
598    // which holds CallContextGuard that set them. We reconstruct a
599    // ToolCallContext only for dispatch — no storage beyond this fn.
600    let call_ctx = unsafe {
601        crate::tools::registry::ToolCallContext {
602            app: &*app_ptr,
603            context_name: &context_name,
604            config: &*config_ptr,
605            project_root: &project_root,
606            vfs: &*vfs_ptr,
607            vfs_caller: if vfs_caller_str.is_empty() {
608                VfsCaller::System
609            } else {
610                VfsCaller::Context(&vfs_caller_str)
611            },
612        }
613    };
614
615    let tool_impl = {
616        let reg = registry.read().map_err(|e| format!("registry lock: {e}"))?;
617        let tool = reg
618            .get(&name)
619            .ok_or_else(|| format!("unknown tool: {name}"))?;
620        tool.r#impl.clone()
621    };
622
623    // Use the captured runtime handle directly rather than `vfs_block_on`:
624    // the tein worker thread is not a tokio thread, so `block_in_place`
625    // (which `vfs_block_on` uses) would panic. The captured handle from
626    // `CallContextGuard::set` (which runs on a tokio thread) is safe to
627    // `block_on` from this non-tokio worker thread. See commit 2016b193
628    // for the original `vfs_block_on` change and why this context differs.
629    runtime_handle
630        .block_on(ToolRegistry::dispatch_impl(
631            tool_impl, &name, &json_args, &call_ctx,
632        ))
633        .map_err(|e| format!("tool error: {e}"))
634}

661/// Helper: extract VFS handle and runtime handle from `BRIDGE_CALL_CTX` for the
662/// current thread. Returns `(runtime_handle, vfs_ptr)` after dropping the lock.
663///
664/// Called from IO FFI functions running on the tein worker thread.
665#[cfg(feature = "synthesised-tools")]
666fn io_bridge_ctx() -> Result<(tokio::runtime::Handle, *const Vfs), String> {
667    let tid = std::thread::current().id();
668    let guard = BRIDGE_CALL_CTX.lock().unwrap();
669    let active = guard.get(&tid).ok_or_else(|| {
670        "harness io: no active call context (called outside tool execute or hook dispatch?)"
671            .to_string()
672    })?;
673    let handle = active.runtime_handle.clone();
674    let vfs = active.vfs;
675    drop(guard);
676    Ok((handle, vfs))
677}

679/// `(io-read path)` — read a file. Returns string content or `#f` if not found.
680///
681/// Path dispatch: `"vfs://..."` → VFS with `VfsCaller::System`;
682/// bare absolute path → `tokio::fs::read_to_string`.
683#[cfg(feature = "synthesised-tools")]
684#[tein::tein_fn(name = "io-read")]
685fn io_read_fn(path: String) -> Result<Value, String> {
686    let (handle, vfs_ptr) = io_bridge_ctx()?;
687    if let Some(vfs_path_str) = path.strip_prefix("vfs://") {
688        let vp = VfsPath::new(vfs_path_str).map_err(|e| e.to_string())?;
689        // SAFETY: vfs_ptr comes from BRIDGE_CALL_CTX, set by CallContextGuard for the
690        // duration of execute_synthesised or hook dispatch. The Vfs lives in AppState
691        // (Arc-owned, session lifetime), so the reference is valid as long as the guard
692        // is alive.
693        let vfs = unsafe { &*vfs_ptr };
694        match handle.block_on(vfs.read(VfsCaller::System, &vp)) {
695            Ok(bytes) => Ok(Value::String(String::from_utf8_lossy(&bytes).into_owned())),
696            Err(e) if e.kind() == io::ErrorKind::NotFound => Ok(Value::Boolean(false)),
697            Err(e) => Err(e.to_string()),
698        }
699    } else {
700        match handle.block_on(tokio::fs::read_to_string(&path)) {
701            Ok(s) => Ok(Value::String(s)),
702            Err(e) if e.kind() == io::ErrorKind::NotFound => Ok(Value::Boolean(false)),
703            Err(e) => Err(e.to_string()),
704        }
705    }
706}

708/// `(io-write path data)` — write a file. Returns `#t` on success, raises on error.
709///
710/// VFS writes via `VfsCaller::System` (goes through `LocalBackend::write` →
711/// `safe_io::atomic_write`). Bare FS writes also use `safe_io::atomic_write`
712/// via `spawn_blocking`, matching the VFS backend's safety guarantees.
713#[cfg(feature = "synthesised-tools")]
714#[tein::tein_fn(name = "io-write")]
715fn io_write_fn(path: String, data: String) -> Result<Value, String> {
716    let (handle, vfs_ptr) = io_bridge_ctx()?;
717    if let Some(vfs_path_str) = path.strip_prefix("vfs://") {
718        let vp = VfsPath::new(vfs_path_str).map_err(|e| e.to_string())?;
719        // SAFETY: see io_read_fn — same pointer provenance and lifetime contract.
720        let vfs = unsafe { &*vfs_ptr };
721        handle
722            .block_on(vfs.write(VfsCaller::System, &vp, data.as_bytes()))
723            .map_err(|e| e.to_string())?;
724        Ok(Value::Boolean(true))
725    } else {
726        // The tein worker thread is a plain OS thread (not a tokio task), so a
727        // direct blocking call to safe_io::atomic_write is correct here.
728        // handle is only needed for VFS operations; bare FS writes are sync.
729        let _ = handle; // suppress unused warning in non-VFS branch
730        crate::safe_io::atomic_write(std::path::Path::new(&path), data.as_bytes())
731            .map_err(|e| e.to_string())?;
732        Ok(Value::Boolean(true))
733    }
734}

736/// `(io-append path data)` — append to a file. Returns `#t` on success, raises on error.
737///
738/// VFS append via `VfsCaller::System`. Bare FS appends match `LocalBackend::append`:
739/// `create(true) + append(true)` — creates the file if missing, no fsync required
740/// (appends are idempotent and not crash-atomic by design, same as VFS).
741#[cfg(feature = "synthesised-tools")]
742#[tein::tein_fn(name = "io-append")]
743fn io_append_fn(path: String, data: String) -> Result<Value, String> {
744    let (handle, vfs_ptr) = io_bridge_ctx()?;
745    if let Some(vfs_path_str) = path.strip_prefix("vfs://") {
746        let vp = VfsPath::new(vfs_path_str).map_err(|e| e.to_string())?;
747        // SAFETY: see io_read_fn — same pointer provenance and lifetime contract.
748        let vfs = unsafe { &*vfs_ptr };
749        handle
750            .block_on(vfs.append(VfsCaller::System, &vp, data.as_bytes()))
751            .map_err(|e| e.to_string())?;
752        Ok(Value::Boolean(true))
753    } else {
754        use tokio::io::AsyncWriteExt;
755        handle
756            .block_on(async {
757                let mut file = tokio::fs::OpenOptions::new()
758                    .create(true)
759                    .append(true)
760                    .open(&path)
761                    .await?;
762                file.write_all(data.as_bytes()).await
763            })
764            .map_err(|e| e.to_string())?;
765        Ok(Value::Boolean(true))
766    }
767}

769/// `(io-list path)` — list directory entries. Returns list of name strings sorted
770/// lexicographically, or empty list if path not found.
771///
772/// Both VFS and bare FS results are sorted to guarantee consistent ordering across
773/// backends (VFS via `LocalBackend` uses `read_dir` which is OS-order; bare FS
774/// likewise).
775#[cfg(feature = "synthesised-tools")]
776#[tein::tein_fn(name = "io-list")]
777fn io_list_fn(path: String) -> Result<Value, String> {
778    let (handle, vfs_ptr) = io_bridge_ctx()?;
779    if let Some(vfs_path_str) = path.strip_prefix("vfs://") {
780        let vp = VfsPath::new(vfs_path_str).map_err(|e| e.to_string())?;
781        // SAFETY: see io_read_fn — same pointer provenance and lifetime contract.
782        let vfs = unsafe { &*vfs_ptr };
783        match handle.block_on(vfs.list(VfsCaller::System, &vp)) {
784            Ok(mut entries) => {
785                entries.sort_by(|a, b| a.name.cmp(&b.name));
786                let names: Vec<Value> =
787                    entries.into_iter().map(|e| Value::String(e.name)).collect();
788                Ok(Value::List(names))
789            }
790            Err(e) if e.kind() == io::ErrorKind::NotFound => Ok(Value::List(vec![])),
791            Err(e) => Err(e.to_string()),
792        }
793    } else {
794        match handle.block_on(async {
795            let mut entries = Vec::new();
796            let mut dir = tokio::fs::read_dir(&path).await?;
797            while let Some(entry) = dir.next_entry().await? {
798                if let Some(name) = entry.file_name().to_str() {
799                    entries.push(name.to_string());
800                }
801            }
802            entries.sort();
803            Ok::<_, io::Error>(entries)
804        }) {
805            Ok(entries) => Ok(Value::List(
806                entries.into_iter().map(Value::String).collect(),
807            )),
808            Err(e) if e.kind() == io::ErrorKind::NotFound => Ok(Value::List(vec![])),
809            Err(e) => Err(e.to_string()),
810        }
811    }
812}

861/// Decompose Unix epoch seconds into (year, month, day, hour, minute) UTC.
862///
863/// Used by the `current-timestamp` harness helper to avoid a chrono dependency
864/// in the synthesised-tools module path.
865#[cfg(feature = "synthesised-tools")]
866fn secs_to_ymdhmz(mut s: u64) -> (u32, u32, u32, u32, u32) {
867    s /= 60; // discard seconds
868    let mi = (s % 60) as u32;
869    s /= 60;
870    let h = (s % 24) as u32;
871    s /= 24;
872    // Gregorian calendar decomposition
873    let z = s + 719_468;
874    let era = z / 146_097;
875    let doe = z - era * 146_097;
876    let yoe = (doe - doe / 1460 + doe / 36_524 - doe / 146_096) / 365;
877    let doy = doe - (365 * yoe + yoe / 4 - yoe / 100);
878    let mp = (5 * doy + 2) / 153;
879    let d = (doy - (153 * mp + 2) / 5 + 1) as u32;
880    let mo = if mp < 10 { mp + 3 } else { mp - 9 } as u32;
881    let y = (yoe + era * 400 + if mo <= 2 { 1 } else { 0 }) as u32;
882    (y, mo, d, h, mi)
883}

885/// Build a `TeinSession` for a synthesised tool, registering
886/// `call-tool`, the harness preamble, and `(harness tools)` module.
887///
888/// Returns the session and the tein worker thread's `ThreadId` (captured
889/// during init). The thread ID is the key for `BRIDGE_CALL_CTX` lookups.
890///
891/// Sandbox behaviour depends on `tier`:
892/// - `Sandboxed`: safe modules only, 10M step limit
893/// - `Unsandboxed`: full R7RS, no step limit (trusted tools only)
894#[cfg(feature = "synthesised-tools")]
895fn build_tein_context(
896    source: String,
897    tier: crate::config::SandboxTier,
898    http_prefixes: Option<Vec<String>>,
899    env_vars: Option<Vec<(String, String)>>,
900) -> io::Result<(TeinSession, std::thread::ThreadId)> {
901    let worker_thread_id = Arc::new(std::sync::Mutex::new(None::<std::thread::ThreadId>));
902    let tid_capture = Arc::clone(&worker_thread_id);
903
904    // Shared buffers for stdout/stderr capture. Arc clones go into the init
905    // closure (for wiring output ports) and into TeinSession (for reading).
906    let stdout_buf: Arc<Mutex<Vec<u8>>> = Arc::new(Mutex::new(Vec::new()));
907    let stderr_buf: Arc<Mutex<Vec<u8>>> = Arc::new(Mutex::new(Vec::new()));
908    let stdout_for_init = Arc::clone(&stdout_buf);
909    let stderr_for_init = Arc::clone(&stderr_buf);
910
911    let init = move |ctx: &Context| -> tein::Result<()> {
912        // Capture the tein worker thread's ID for BRIDGE_CALL_CTX keying.
913        *tid_capture.lock().unwrap() = Some(std::thread::current().id());
914        ctx.define_fn_variadic("call-tool", __tein_call_tool_fn)?;
915        ctx.define_fn_variadic("generate-id", __tein_generate_id_fn)?;
916        ctx.define_fn_variadic("current-timestamp", __tein_current_timestamp_fn)?;
917        ctx.evaluate(&HARNESS_PREAMBLE)?;
918        ctx.register_module(HARNESS_TOOLS_MODULE)
919            .map_err(|e| tein::Error::EvalError(format!("harness module: {e}")))?;
920        ctx.register_module(HARNESS_HOOKS_MODULE)
921            .map_err(|e| tein::Error::EvalError(format!("harness hooks module: {e}")))?;
922        ctx.register_module(HARNESS_DOCS_MODULE)
923            .map_err(|e| tein::Error::EvalError(format!("harness docs module: {e}")))?;
924        // (harness io) — privileged direct IO, available at Unsandboxed tier only.
925        // Sandboxed contexts will get a module-not-found error on (import (harness io)).
926        if tier == crate::config::SandboxTier::Unsandboxed {
927            ctx.define_fn_variadic("io-read", __tein_io_read_fn)?;
928            ctx.define_fn_variadic("io-write", __tein_io_write_fn)?;
929            ctx.define_fn_variadic("io-append", __tein_io_append_fn)?;
930            ctx.define_fn_variadic("io-list", __tein_io_list_fn)?;
931            ctx.define_fn_variadic("io-exists?", __tein_io_exists_fn)?;
932            ctx.define_fn_variadic("io-delete", __tein_io_delete_fn)?;
933            ctx.register_module(HARNESS_IO_MODULE)
934                .map_err(|e| tein::Error::EvalError(format!("harness io module: {e}")))?;
935        }
936        // Standard prelude — must run after all modules are registered (EVAL_PRELUDE
937        // imports (harness tools), which is only available post-register_module).
938        ctx.evaluate(EVAL_PRELUDE)?;
939        // Wire stdout/stderr capture ports. The SharedWriter clones share the
940        // same Arc buffers that TeinSession reads from. Ports persist across
941        // evaluations (confirmed by tein test suite).
942        let out_port = ctx.open_output_port(SharedWriter(stdout_for_init.clone()))?;
943        ctx.set_current_output_port(&out_port)?;
944        let err_port = ctx.open_output_port(SharedWriter(stderr_for_init.clone()))?;
945        ctx.set_current_error_port(&err_port)?;
946
947        ctx.evaluate(&source)?;
948        Ok(())
949    };
950
951    let ctx = match tier {
952        crate::config::SandboxTier::Sandboxed => {
953            let mut builder = Context::builder()
954                .standard_env()
955                .sandboxed(Modules::Safe)
956                .step_limit(10_000_000);
957            if let Some(ref prefixes) = http_prefixes {
958                let refs: Vec<&str> = prefixes.iter().map(|s| s.as_str()).collect();
959                builder = builder.http_allow(&refs);
960            }
961            if let Some(ref vars) = env_vars {
962                let refs: Vec<(&str, &str)> =
963                    vars.iter().map(|(k, v)| (k.as_str(), v.as_str())).collect();
964                builder = builder.environment_variables(&refs);
965            }
966            builder.build_managed(init)
967        }
968        crate::config::SandboxTier::Unsandboxed => {
969            // with_vfs_shadows() enables shadow modules (e.g. scheme/process-context,
970            // scheme/file) in non-sandboxed contexts. Required for (chibi diff) and
971            // other library modules that depend on scheme/process-context.
972            //
973            // `http_prefixes` is intentionally not applied here: unsandboxed contexts
974            // have unrestricted network access already, so an allowlist would be redundant.
975            // (A config entry under [tools.http.allow] for an unsandboxed path has no effect.)
976            let mut builder = Context::builder().standard_env().with_vfs_shadows();
977            if let Some(ref vars) = env_vars {
978                let refs: Vec<(&str, &str)> =
979                    vars.iter().map(|(k, v)| (k.as_str(), v.as_str())).collect();
980                builder = builder.environment_variables(&refs);
981            }
982            builder.build_managed(init)
983        }
984    }
985    .map_err(|e| io::Error::new(io::ErrorKind::InvalidData, format!("tein init: {e}")))?;
986
987    let tid = worker_thread_id
988        .lock()
989        .unwrap()
990        .expect("init closure must have run and captured thread ID");
991
992    let session = TeinSession {
993        ctx,
994        stdout_buf,
995        stderr_buf,
996    };
997    Ok((session, tid))
998}

1054/// Load one or more synthesised tools from scheme source.
1055///
1056/// Resolves sandbox tier from `tools_config` using longest-prefix match on
1057/// `vfs_path`. Evaluates `source` in a tein context configured accordingly.
1058/// Supports both convention format (`tool-name`, `tool-execute`, etc.) and
1059/// `define-tool` multi-tool format; backwards-compatible with both.
1060#[cfg(feature = "synthesised-tools")]
1061pub fn load_tools_from_source(
1062    source: &str,
1063    vfs_path: &VfsPath,
1064    registry: &Arc<RwLock<ToolRegistry>>,
1065    tools_config: &crate::config::ToolsConfig,
1066) -> io::Result<Vec<Tool>> {
1067    let source_owned = source.to_string();
1068    let tier = tools_config.resolve_tier(vfs_path.as_str());
1069    let env_vars = tools_config.resolve_env(vfs_path.as_str());
1070
1071    let http_result = tools_config.resolve_http_allow(vfs_path.as_str());
1072    let http_prefixes = match &http_result {
1073        crate::config::HttpAllowResult::Prefixes(p) => Some(p.clone()),
1074        _ => None, // NeedDeclared resolved after phase 1
1075    };
1076
1077    let (session, worker_thread_id) =
1078        build_tein_context(source_owned.clone(), tier, http_prefixes, env_vars.clone())?;
1079
1080    // Phase 2: trust-declared HTTP — read tool-http-allow, rebuild if needed
1081    let (session, worker_thread_id) =
1082        if matches!(http_result, crate::config::HttpAllowResult::NeedDeclared) {
1083            // Read tool-http-allow binding from phase 1 session
1084            let declared = session
1085                .evaluate("tool-http-allow")
1086                .ok()
1087                .and_then(|v| match v {
1088                    Value::List(items) => Some(
1089                        items
1090                            .iter()
1091                            .filter_map(|i| i.as_string().map(|s| s.to_string()))
1092                            .collect::<Vec<_>>(),
1093                    ),
1094                    _ => None,
1095                })
1096                .unwrap_or_default();
1097
1098            if let Some(trusted_prefixes) =
1099                tools_config.resolve_http_allow_with_declared(vfs_path.as_str(), &declared)
1100            {
1101                // Rebuild with the trusted prefixes
1102                build_tein_context(source_owned, tier, Some(trusted_prefixes), env_vars)?
1103            } else {
1104                (session, worker_thread_id)
1105            }
1106        } else {
1107            (session, worker_thread_id)
1108        };
1109
1110    // check if define-tool was used (%tool-registry% is non-empty list)
1111    let multi = session.evaluate("%tool-registry%").ok();
1112    let is_multi = matches!(
1113        &multi,
1114        Some(Value::List(items)) if !items.is_empty()
1115    );
1116
1117    if is_multi {
1118        extract_multi_tools(session, vfs_path, registry, worker_thread_id)
1119    } else {
1120        extract_single_tool(session, vfs_path, registry, worker_thread_id).map(|t| vec![t])
1121    }
1122}

1232/// Reads `%tool-registry%` (a LIFO list built via `cons`) and produces one
1233/// `Tool` per entry. All tools share the same tein context via `Arc`.
1234#[cfg(feature = "synthesised-tools")]
1235fn extract_multi_tools(
1236    session: TeinSession,
1237    vfs_path: &VfsPath,
1238    registry: &Arc<RwLock<ToolRegistry>>,
1239    worker_thread_id: std::thread::ThreadId,
1240) -> io::Result<Vec<Tool>> {
1241    let registry_val = session
1242        .evaluate("%tool-registry%")
1243        .map_err(|e| io::Error::other(format!("reading %tool-registry%: {e}")))?;
1244
1245    let entries = match registry_val {
1246        Value::List(items) => items,
1247        other => {
1248            return Err(io::Error::other(format!(
1249                "%%tool-registry%% is not a list: {other}"
1250            )));
1251        }
1252    };
1253
1254    let (hooks, hook_bindings) = extract_hook_registrations(&session)?;
1255    let context = Arc::new(session);
1256    let mut tools = Vec::with_capacity(entries.len());
1257
1258    // entries are in LIFO order (built via cons); reverse to get definition order
1259    for entry in entries.iter().rev() {
1260        let fields = match entry {
1261            Value::List(f) if f.len() >= 4 => f,
1262            other => {
1263                return Err(io::Error::other(format!(
1264                    "define-tool entry has unexpected shape: {other}"
1265                )));
1266            }
1267        };
1268
1269        let name = fields[0]
1270            .as_string()
1271            .ok_or_else(|| io::Error::other("define-tool: name not a string"))?
1272            .to_string();
1273        let description = fields[1]
1274            .as_string()
1275            .ok_or_else(|| io::Error::other("define-tool: description not a string"))?
1276            .to_string();
1277        let parameters = params_alist_to_json_schema(&fields[2])?;
1278
1279        if !fields[3].is_procedure() {
1280            return Err(io::Error::other(format!(
1281                "define-tool {name}: execute is not a procedure"
1282            )));
1283        }
1284
1285        // category (index 4, optional — absent in 4-element legacy entries)
1286        let category = if fields.len() > 4 {
1287            fields[4]
1288                .as_string()
1289                .map(ToolCategory::from_category_str)
1290                .unwrap_or(ToolCategory::Synthesised)
1291        } else {
1292            ToolCategory::Synthesised
1293        };
1294
1295        // summary_params (index 5, optional)
1296        let summary_params = if fields.len() > 5 {
1297            match &fields[5] {
1298                Value::List(items) => items
1299                    .iter()
1300                    .filter_map(|v| v.as_string().map(|s| s.to_string()))
1301                    .collect(),
1302                _ => vec![],
1303            }
1304        } else {
1305            vec![]
1306        };
1307
1308        // reject names that would produce invalid Scheme identifiers when
1309        // embedded in `%tool-execute-{name}%` (whitespace or parentheses break
1310        // the symbol syntax)
1311        if name
1312            .chars()
1313            .any(|c| c.is_whitespace() || c == '(' || c == ')')
1314        {
1315            return Err(io::Error::other(format!(
1316                "define-tool: name {name:?} contains invalid characters (whitespace or parens)"
1317            )));
1318        }
1319
1320        // bind the execute handler to a per-tool name so execute_synthesised
1321        // can find it by name. the context is shared across all tools in this file.
1322        let exec_binding = format!("%tool-execute-{name}%");
1323        // scheme-escape name before interpolating into a string literal
1324        let name_escaped = scheme_escape_string(&name);
1325        // %tool-registry% entries are (name desc params handler ...).
1326        // use list-ref to extract the handler (index 3) for this tool by name.
1327        // list-ref is in (scheme base) which the preamble already imports.
1328        context
1329            .evaluate(&format!(
1330                "(define {exec_binding} \
1331                 (list-ref \
1332                   (let loop ((reg %tool-registry%)) \
1333                     (if (string=? (car (car reg)) \"{name_escaped}\") \
1334                         (car reg) \
1335                         (loop (cdr reg)))) \
1336                   3))"
1337            ))
1338            .map_err(|e| io::Error::other(format!("binding {exec_binding}: {e}")))?;
1339
1340        tools.push(Tool {
1341            name,
1342            description,
1343            parameters,
1344            hooks: hooks.clone(),
1345            metadata: ToolMetadata::new(),
1346            summary_params,
1347            r#impl: ToolImpl::Synthesised {
1348                vfs_path: vfs_path.clone(),
1349                exec_binding: exec_binding.clone(),
1350                context: Arc::clone(&context),
1351                registry: Arc::clone(registry),
1352                worker_thread_id,
1353                hook_bindings: hook_bindings.clone(),
1354            },
1355            category,
1356        });
1357    }
1358
1359    Ok(tools)
1360}

1362/// Execute a synthesised tool by calling its bound execute procedure.
1363///
1364/// Converts JSON args to a scheme alist, resolves the `exec_binding` in the
1365/// session, and calls it with stdout/stderr capture. Returns structured output:
1366/// `"result: <value>\nstdout: <output>\nstderr: <output>"`.
1367///
1368/// Sets `BRIDGE_CALL_CTX` (with the per-call registry) via `CallContextGuard`
1369/// before calling into scheme so that `call-tool` can access the runtime context.
1370///
1371/// `%context-name%` is injected via `set!` on the top-level binding from
1372/// `HARNESS_PREAMBLE`. Concurrent calls to different `TeinSession`s are
1373/// safe because each session has its own `BRIDGE_CALL_CTX` slot (keyed by
1374/// `worker_thread_id`). Calls to the same session are serialised by
1375/// `ThreadLocalContext`'s internal mutex, so no interleaving can occur.
1376#[cfg(feature = "synthesised-tools")]
1377pub async fn execute_synthesised(
1378    session: &TeinSession,
1379    exec_binding: &str,
1380    call: &ToolCall<'_>,
1381    registry: Arc<RwLock<ToolRegistry>>,
1382    worker_thread_id: std::thread::ThreadId,
1383) -> io::Result<String> {
1384    let _guard = CallContextGuard::set(call.context, registry, worker_thread_id);
1385
1386    // Inject per-call context name before invoking the tool handler.
1387    // %context-name% mutates the top-level binding defined in HARNESS_PREAMBLE.
1388    // Safe because ThreadLocalContext serialises all calls to this context.
1389    let ctx_name_escaped = scheme_escape_string(call.context.context_name);
1390    session
1391        .evaluate(&format!("(set! %context-name% \"{ctx_name_escaped}\")"))
1392        .map_err(|e| io::Error::other(format!("inject %context-name%: {e}")))?;
1393
1394    let args_alist = json_args_to_scheme_alist(call.args)?;
1395    let exec_fn = session
1396        .evaluate(exec_binding)
1397        .map_err(|e| io::Error::other(format!("resolve {exec_binding}: {e}")))?;
1398
1399    // NOTE: similar capture pattern exists in run_scheme (eval.rs).
1400    // Changes here may need to be assessed for that callsite too.
1401    let captured = session.with_capture(|ctx| ctx.call(&exec_fn, &[args_alist]));
1402    Ok(captured.format_tool())
1403}

1429/// **Zones scanned:**
1430/// - `/tools/shared` — globally shared tools
1431/// - `/tools/home/<ctx>/` — per-context tools (one dir per context name)
1432/// - `/tools/flocks/<name>/` — flock-scoped tools
1433#[cfg(feature = "synthesised-tools")]
1434pub async fn scan_and_register(
1435    vfs: &Vfs,
1436    registry: &Arc<RwLock<ToolRegistry>>,
1437    tools_config: &crate::config::ToolsConfig,
1438) -> io::Result<()> {
1439    let mut zones = vec!["/tools/shared".to_string()];
1440
1441    // discover /tools/home/<ctx>/ directories
1442    if let Ok(entries) = vfs
1443        .list(VfsCaller::System, &VfsPath::new("/tools/home")?)
1444        .await
1445    {
1446        for entry in entries {
1447            zones.push(format!("/tools/home/{}", entry.name));
1448        }
1449    }
1450
1451    // discover /tools/flocks/<name>/ directories
1452    if let Ok(entries) = vfs
1453        .list(VfsCaller::System, &VfsPath::new("/tools/flocks")?)
1454        .await
1455    {
1456        for entry in entries {
1457            zones.push(format!("/tools/flocks/{}", entry.name));
1458        }
1459    }
1460
1461    for zone in &zones {
1462        scan_zone(vfs, registry, zone, tools_config).await?;
1463    }
1464    Ok(())
1465}

1467/// Scan a single VFS zone directory and register all `.scm` tool files found.
1468#[cfg(feature = "synthesised-tools")]
1469async fn scan_zone(
1470    vfs: &Vfs,
1471    registry: &Arc<RwLock<ToolRegistry>>,
1472    zone: &str,
1473    tools_config: &crate::config::ToolsConfig,
1474) -> io::Result<()> {
1475    let Ok(zone_path) = VfsPath::new(zone) else {
1476        return Ok(());
1477    };
1478    if !vfs
1479        .exists(VfsCaller::System, &zone_path)
1480        .await
1481        .unwrap_or(false)
1482    {
1483        return Ok(());
1484    }
1485    let entries = match vfs.list(VfsCaller::System, &zone_path).await {
1486        Ok(e) => e,
1487        Err(_) => return Ok(()), // zone unreadable — skip silently
1488    };
1489    for entry in entries {
1490        if entry.kind == VfsEntryKind::Directory {
1491            // recurse into subdirectories — ignore errors (zone boundary)
1492            let subzone = format!("{zone}/{}", entry.name);
1493            let _ = Box::pin(scan_zone(vfs, registry, &subzone, tools_config)).await;
1494            continue;
1495        }
1496        if !entry.name.ends_with(".scm") {
1497            continue;
1498        }
1499        let Ok(file_path) = VfsPath::new(&format!("{zone}/{}", entry.name)) else {
1500            continue;
1501        };
1502        let source = match vfs.read(VfsCaller::System, &file_path).await {
1503            Ok(b) => b,
1504            Err(_) => continue, // unreadable — skip silently
1505        };
1506        let Ok(source_str) = String::from_utf8(source) else {
1507            continue;
1508        };
1509        if let Ok(tools) = load_tools_from_source(&source_str, &file_path, registry, tools_config) {
1510            let mut reg = registry.write().unwrap();
1511            for tool in tools {
1512                reg.register(tool);
1513            }
1514        }
1515        // Err(_): invalid source — skip silently (caller can inspect via VFS)
1516    }
1517    Ok(())
1518}

1586/// Read `%hook-registry%` from a tein context and return (hooks, hook_bindings).
1587///
1588/// Each entry in `%hook-registry%` is `(hook-name-string handler-procedure)`.
1589/// For each valid entry, we:
1590/// 1. Parse the hook name string into a `HookPoint`.
1591/// 2. Bind the handler to `%hook-{hook_name}%` in the context.
1592/// 3. Record the mapping in `hook_bindings`.
1593///
1594/// Invalid hook names are warned and skipped (same as plugin hook parsing).
1595///
1596/// Mutation site: if `register-hook` alist shape changes, update this function
1597/// and `HARNESS_PREAMBLE` accordingly.
1598#[cfg(feature = "synthesised-tools")]
1599fn extract_hook_registrations(
1600    session: &TeinSession,
1601) -> io::Result<(
1602    Vec<super::hooks::HookPoint>,
1603    std::collections::HashMap<super::hooks::HookPoint, String>,
1604)> {
1605    let registry_val = session
1606        .evaluate("%hook-registry%")
1607        .map_err(|e| io::Error::other(format!("reading %hook-registry%: {e}")))?;
1608
1609    let entries = match registry_val {
1610        Value::List(items) if !items.is_empty() => items,
1611        _ => return Ok((vec![], std::collections::HashMap::new())),
1612    };
1613
1614    let mut hooks = Vec::new();
1615    let mut hook_bindings = std::collections::HashMap::new();
1616
1617    // entries are LIFO (via cons); reverse for definition order
1618    for entry in entries.iter().rev() {
1619        let fields = match entry {
1620            Value::List(f) if f.len() >= 2 => f,
1621            other => {
1622                eprintln!("[WARN] register-hook entry has unexpected shape: {other}");
1623                continue;
1624            }
1625        };
1626
1627        let hook_name = match fields[0].as_string() {
1628            Some(s) => s.to_string(),
1629            None => {
1630                eprintln!("[WARN] register-hook: hook name not a string");
1631                continue;
1632            }
1633        };
1634
1635        let hook_point = match hook_name.parse::<super::hooks::HookPoint>() {
1636            Ok(hp) => hp,
1637            Err(_) => {
1638                eprintln!("[WARN] Unknown hook '{hook_name}' in register-hook");
1639                continue;
1640            }
1641        };
1642
1643        if !fields[1].is_procedure() {
1644            eprintln!("[WARN] register-hook {hook_name}: handler is not a procedure");
1645            continue;
1646        }
1647
1648        // bind handler to a well-known name so execute_hook can find it
1649        let binding = format!("%hook-{hook_name}%");
1650        let hook_name_escaped = scheme_escape_string(&hook_name);
1651        // look up handler from %hook-registry% by name (first match in LIFO list = newest).
1652        // when the same hook point is registered multiple times, `define` overwrites on
1653        // each iteration (oldest-first), so the last-defined handler wins.
1654        session
1655            .evaluate(&format!(
1656                "(define {binding} \
1657             (cadr \
1658               (let loop ((reg %hook-registry%)) \
1659                 (if (string=? (car (car reg)) \"{hook_name_escaped}\") \
1660                     (car reg) \
1661                     (loop (cdr reg))))))"
1662            ))
1663            .map_err(|e| io::Error::other(format!("binding {binding}: {e}")))?;
1664
1665        // deduplicate hook points: only push to `hooks` once per hook point.
1666        // the binding name is the same regardless (`%hook-{hook_name}%`), so
1667        // or_insert_with is used purely to avoid duplicate entries in `hooks`.
1668        hook_bindings.entry(hook_point).or_insert_with(|| {
1669            hooks.push(hook_point);
1670            binding
1671        });
1672    }
1673
1674    Ok((hooks, hook_bindings))
1675}

1677/// Convert a scheme params alist to a JSON Schema object.
1678///
1679/// Input (scheme): `((name . ((type . "string") (description . "..."))) ...)`
1680///
1681/// In tein::Value terms, `(name . attrs-list)` where `attrs-list` is a proper
1682/// list becomes `List([Symbol("name"), Pair("type","string"), ...])` — tein
1683/// flattens `(head . proper-list)` into a proper list at the Value level.
1684/// Each entry is `List([name-sym, attr-pair, ...])` where head is the param
1685/// name and the rest are attribute pairs.
1686///
1687/// Output: `{"type": "object", "properties": {"name": {"type": "string", "description": "..."}}, "required": [...]}`
1688///
1689/// All parameters are required unless the inner alist contains `(required . #f)`.
1690#[cfg(feature = "synthesised-tools")]
1691fn params_alist_to_json_schema(val: &Value) -> io::Result<serde_json::Value> {
1692    let items = match val {
1693        Value::List(items) => items,
1694        Value::Nil => {
1695            return Ok(serde_json::json!({
1696                "type": "object",
1697                "properties": {},
1698                "required": []
1699            }));
1700        }
1701        other => {
1702            return Err(io::Error::new(
1703                io::ErrorKind::InvalidData,
1704                format!("tool-parameters must be a list, got: {other}"),
1705            ));
1706        }
1707    };
1708
1709    let mut properties = serde_json::Map::new();
1710    let mut required: Vec<serde_json::Value> = Vec::new();
1711
1712    for item in items {
1713        // Each entry is List([Symbol/String(name), attr-pair, ...]) — tein
1714        // flattens (name . attr-list) into a proper list when attr-list is proper.
1715        let elems = match item {
1716            Value::List(e) if !e.is_empty() => e,
1717            other => {
1718                return Err(io::Error::new(
1719                    io::ErrorKind::InvalidData,
1720                    format!("tool-parameters entry must be a non-empty list, got: {other}"),
1721                ));
1722            }
1723        };
1724        let name = match &elems[0] {
1725            Value::Symbol(s) | Value::String(s) => s.clone(),
1726            other => {
1727                return Err(io::Error::new(
1728                    io::ErrorKind::InvalidData,
1729                    format!("parameter name must be symbol or string, got: {other}"),
1730                ));
1731            }
1732        };
1733        // attrs: the tail of the list entry — pairs like (type . "string")
1734        let attrs = &elems[1..];
1735
1736        let mut prop = serde_json::Map::new();
1737        let mut is_required = true;
1738        for attr in attrs {
1739            let (akey, aval) = match attr {
1740                Value::Pair(k, v) => {
1741                    let k = match k.as_ref() {
1742                        Value::Symbol(s) | Value::String(s) => s.as_str(),
1743                        _ => continue,
1744                    };
1745                    (k.to_string(), v.as_ref())
1746                }
1747                _ => continue,
1748            };
1749            match akey.as_str() {
1750                "required" => {
1751                    if let Value::Boolean(b) = aval {
1752                        is_required = *b;
1753                    }
1754                }
1755                "type" => {
1756                    if let Some(s) = aval.as_string() {
1757                        prop.insert("type".into(), serde_json::Value::String(s.to_string()));
1758                    }
1759                }
1760                "description" => {
1761                    if let Some(s) = aval.as_string() {
1762                        prop.insert(
1763                            "description".into(),
1764                            serde_json::Value::String(s.to_string()),
1765                        );
1766                    }
1767                }
1768                _ => {} // forward-compat: ignore unknown attrs
1769            }
1770        }
1771
1772        if is_required {
1773            required.push(serde_json::Value::String(name.clone()));
1774        }
1775        properties.insert(name, serde_json::Value::Object(prop));
1776    }
1777
1778    Ok(serde_json::json!({
1779        "type": "object",
1780        "properties": properties,
1781        "required": required,
1782    }))
1783}

1785/// Convert a scheme value (from `call-tool` args) to a `serde_json::Value`.
1786///
1787/// Handles:
1788/// - alists (list of pairs) → JSON object
1789/// - plain lists → JSON array
1790/// - atoms → scalar JSON values
1791/// - `Nil` → `null`
1792///
1793/// Mutation site: if scheme value representation changes in tein, update here.
1794#[cfg(feature = "synthesised-tools")]
1795pub(crate) fn scheme_value_to_json(val: &Value) -> io::Result<serde_json::Value> {
1796    match val {
1797        Value::Nil => Ok(serde_json::Value::Null),
1798        Value::Boolean(b) => Ok(serde_json::Value::Bool(*b)),
1799        Value::Integer(n) => Ok(serde_json::json!(*n)),
1800        Value::Float(f) => Ok(serde_json::json!(*f)),
1801        Value::String(s) => Ok(serde_json::Value::String(s.clone())),
1802        Value::Symbol(s) => Ok(serde_json::Value::String(s.clone())),
1803        Value::Pair(k, v) => {
1804            // single pair → two-element array (or alist-like handling by caller)
1805            Ok(serde_json::Value::Array(vec![
1806                scheme_value_to_json(k)?,
1807                scheme_value_to_json(v)?,
1808            ]))
1809        }
1810        Value::List(items) => {
1811            // if every item is a pair → alist → JSON object
1812            let all_pairs = items
1813                .iter()
1814                .all(|item| matches!(item, Value::Pair(_, _) | Value::List(_)));
1815            if all_pairs && !items.is_empty() {
1816                // attempt alist → object conversion
1817                let mut map = serde_json::Map::new();
1818                let mut is_alist = true;
1819                for item in items {
1820                    match item {
1821                        Value::Pair(k, v) => {
1822                            let key = match k.as_ref() {
1823                                Value::String(s) | Value::Symbol(s) => s.clone(),
1824                                _ => {
1825                                    is_alist = false;
1826                                    break;
1827                                }
1828                            };
1829                            map.insert(key, scheme_value_to_json(v)?);
1830                        }
1831                        Value::List(pair_items) if pair_items.len() == 2 => {
1832                            let key = match &pair_items[0] {
1833                                Value::String(s) | Value::Symbol(s) => s.clone(),
1834                                _ => {
1835                                    is_alist = false;
1836                                    break;
1837                                }
1838                            };
1839                            map.insert(key, scheme_value_to_json(&pair_items[1])?);
1840                        }
1841                        _ => {
1842                            is_alist = false;
1843                            break;
1844                        }
1845                    }
1846                }
1847                if is_alist {
1848                    return Ok(serde_json::Value::Object(map));
1849                }
1850            }
1851            // plain list → array
1852            let arr: io::Result<Vec<_>> = items.iter().map(scheme_value_to_json).collect();
1853            Ok(serde_json::Value::Array(arr?))
1854        }
1855        other => Err(io::Error::other(format!(
1856            "unsupported scheme value type: {other}"
1857        ))),
1858    }
1859}
