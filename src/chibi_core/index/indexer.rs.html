1//! File walker and index updater for the codebase index.
2//!
3//! Walks the project tree (respecting `.gitignore`), detects changes via mtime + size,
4//! dispatches to language plugins for symbol extraction, and writes everything to the database.
5//! Core handles all DB writes — plugins never touch sqlite directly.
6
7use crate::tools::{HookPoint, Tool, execute_hook, execute_tool};
8use ignore::WalkBuilder;
9use rusqlite::Connection;
10use std::collections::HashMap;
11use std::io;
12use std::path::Path;
13
14/// Statistics returned after an index update.
15#[derive(Debug, Default)]
16pub struct IndexStats {
17    pub files_scanned: u32,
18    pub files_indexed: u32,
19    pub files_skipped: u32,
20    pub files_removed: u32,
21    pub symbols_added: u32,
22    pub refs_added: u32,
23}
24
25impl std::fmt::Display for IndexStats {
26    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
27        write!(
28            f,
29            "scanned: {}, indexed: {}, skipped (unchanged): {}, removed: {}, symbols: {}, refs: {}",
30            self.files_scanned,
31            self.files_indexed,
32            self.files_skipped,
33            self.files_removed,
34            self.symbols_added,
35            self.refs_added,
36        )
37    }
38}

49/// Extension-to-language mapping. Hardcoded for now; configurable later.
50const LANG_MAP: &[(&str, &str)] = &[
51    ("rs", "rust"),
52    ("py", "python"),
53    ("js", "javascript"),
54    ("ts", "typescript"),
55    ("tsx", "typescript"),
56    ("jsx", "javascript"),
57    ("rb", "ruby"),
58    ("go", "go"),
59    ("java", "java"),
60    ("c", "c"),
61    ("h", "c"),
62    ("cpp", "cpp"),
63    ("hpp", "cpp"),
64    ("cc", "cpp"),
65    ("zig", "zig"),
66    ("lua", "lua"),
67    ("sh", "shell"),
68    ("bash", "shell"),
69    ("toml", "toml"),
70    ("yaml", "yaml"),
71    ("yml", "yaml"),
72    ("json", "json"),
73    ("md", "markdown"),
74];

76/// Detect language from file extension.
77fn detect_language(path: &Path) -> Option<&'static str> {
78    let ext = path.extension()?.to_str()?;
79    LANG_MAP
80        .iter()
81        .find(|(e, _)| *e == ext)
82        .map(|(_, lang)| *lang)
83}

85/// Walk the project, detect changes, dispatch to plugins, update the database.
86///
87/// Language plugins follow the convention: tool named `lang_<language>` (e.g. `lang_rust`).
88/// They receive `{"files": [{"path": "...", "content": "..."}]}` on stdin and return
89/// `{"symbols": [...], "refs": [...]}` on stdout. Core handles all DB writes.
90pub fn update_index(
91    conn: &Connection,
92    project_root: &Path,
93    options: &IndexOptions,
94    tools: &[Tool],
95) -> io::Result<IndexStats> {
96    let mut stats = IndexStats::default();
97
98    // Collect all indexed paths so we can detect removals.
99    let mut existing_paths: HashMap<String, (i64, i64)> = HashMap::new();
100    {
101        let mut stmt = conn
102            .prepare("SELECT path, mtime, size FROM files")
103            .map_err(|e| io::Error::other(format!("failed to query files: {}", e)))?;
104        let rows = stmt
105            .query_map([], |row| {
106                Ok((
107                    row.get::<_, String>(0)?,
108                    row.get::<_, i64>(1)?,
109                    row.get::<_, i64>(2)?,
110                ))
111            })
112            .map_err(|e| io::Error::other(format!("failed to iterate files: {}", e)))?;
113        for row in rows {
114            let (path, mtime, size) =
115                row.map_err(|e| io::Error::other(format!("row error: {}", e)))?;
116            existing_paths.insert(path, (mtime, size));
117        }
118    }
119
120    // Walk the project tree, respecting .gitignore.
121    let walker = WalkBuilder::new(project_root)
122        .hidden(true) // skip hidden files by default
123        .git_ignore(true)
124        .build();
125
126    // Group files by language for batched plugin dispatch.
127    let mut files_by_lang: HashMap<String, Vec<(String, std::fs::Metadata)>> = HashMap::new();
128    let mut seen_paths: std::collections::HashSet<String> = std::collections::HashSet::new();
129
130    for entry in walker {
131        let entry = match entry {
132            Ok(e) => e,
133            Err(_) => continue,
134        };
135        // Skip directories.
136        if entry.file_type().map(|t| !t.is_file()).unwrap_or(true) {
137            continue;
138        }
139
140        let abs_path = entry.path();
141        let rel_path = abs_path
142            .strip_prefix(project_root)
143            .unwrap_or(abs_path)
144            .to_string_lossy()
145            .into_owned();
146
147        // Skip the .chibi directory itself.
148        if rel_path.starts_with(".chibi") {
149            continue;
150        }
151
152        stats.files_scanned += 1;
153        seen_paths.insert(rel_path.clone());
154
155        let meta = match std::fs::metadata(abs_path) {
156            Ok(m) => m,
157            Err(_) => continue,
158        };
159
160        let mtime = meta
161            .modified()
162            .ok()
163            .and_then(|t| t.duration_since(std::time::UNIX_EPOCH).ok())
164            .map(|d| d.as_secs() as i64)
165            .unwrap_or(0);
166        let size = meta.len() as i64;
167
168        // Change detection: skip if mtime and size unchanged (unless force).
169        if !options.force
170            && let Some(&(old_mtime, old_size)) = existing_paths.get(&rel_path)
171            && old_mtime == mtime
172            && old_size == size
173        {
174            stats.files_skipped += 1;
175            continue;
176        }
177
178        let lang = detect_language(abs_path).unwrap_or("unknown").to_string();
179
180        files_by_lang
181            .entry(lang)
182            .or_default()
183            .push((rel_path, meta));
184    }
185
186    // Process each language batch.
187    for (lang, files) in &files_by_lang {
188        // Find language plugin (tool named `lang_<language>`).
189        let plugin_name = format!("lang_{}", lang);
190        let plugin = tools.iter().find(|t| t.name == plugin_name);
191
192        for (rel_path, meta) in files {
193            let abs_path = project_root.join(rel_path);
194            let mtime = meta
195                .modified()
196                .ok()
197                .and_then(|t| t.duration_since(std::time::UNIX_EPOCH).ok())
198                .map(|d| d.as_secs() as i64)
199                .unwrap_or(0);
200            let size = meta.len() as i64;
201
202            // Upsert the file record.
203            conn.execute(
204                "INSERT INTO files (path, lang, mtime, size)
205                 VALUES (?1, ?2, ?3, ?4)
206                 ON CONFLICT(path) DO UPDATE SET
207                     lang = excluded.lang,
208                     mtime = excluded.mtime,
209                     size = excluded.size,
210                     indexed_at = datetime('now')",
211                rusqlite::params![rel_path, lang, mtime, size],
212            )
213            .map_err(|e| io::Error::other(format!("failed to upsert file: {}", e)))?;
214
215            let file_id: i64 = conn
216                .query_row("SELECT id FROM files WHERE path = ?1", [rel_path], |row| {
217                    row.get(0)
218                })
219                .map_err(|e| io::Error::other(format!("failed to get file id: {}", e)))?;
220
221            // Clear old symbols and refs for this file (cascade would handle it,
222            // but explicit delete is clearer for partial re-index).
223            conn.execute("DELETE FROM symbols WHERE file_id = ?1", [file_id])
224                .map_err(|e| io::Error::other(format!("failed to delete symbols: {}", e)))?;
225            conn.execute("DELETE FROM refs WHERE from_file_id = ?1", [file_id])
226                .map_err(|e| io::Error::other(format!("failed to delete refs: {}", e)))?;
227
228            // If a language plugin exists, dispatch to it for symbol extraction.
229            if let Some(plugin) = plugin {
230                let content = std::fs::read_to_string(&abs_path).unwrap_or_default();
231                let input = serde_json::json!({
232                    "files": [{"path": rel_path, "content": content}]
233                });
234
235                match execute_tool(plugin, &input) {
236                    Ok(output) => {
237                        if let Ok(parsed) = serde_json::from_str::<serde_json::Value>(&output) {
238                            stats.symbols_added += insert_symbols(conn, file_id, &parsed);
239                            stats.refs_added += insert_refs(conn, file_id, &parsed);
240                        }
241                        // Malformed output → graceful fallback (file indexed without symbols).
242                    }
243                    Err(_) => {
244                        // Plugin failure → file still indexed, just without symbols.
245                    }
246                }
247            }
248
249            stats.files_indexed += 1;
250
251            // Fire PostIndexFile hook (observe only — errors are non-fatal).
252            let hook_data = serde_json::json!({
253                "path": rel_path,
254                "lang": lang,
255                "symbol_count": stats.symbols_added,
256                "ref_count": stats.refs_added,
257            });
258            let _ = execute_hook(tools, HookPoint::PostIndexFile, &hook_data);
259        }
260    }
261
262    // Remove files that no longer exist on disk.
263    for path in existing_paths.keys() {
264        if !seen_paths.contains(path) {
265            conn.execute("DELETE FROM files WHERE path = ?1", [path])
266                .map_err(|e| io::Error::other(format!("failed to remove stale file: {}", e)))?;
267            stats.files_removed += 1;
268        }
269    }
270
271    Ok(stats)
272}

274/// Insert symbols from plugin output into the database. Returns count of symbols added.
275fn insert_symbols(conn: &Connection, file_id: i64, output: &serde_json::Value) -> u32 {
276    let symbols = match output.get("symbols").and_then(|v| v.as_array()) {
277        Some(arr) => arr,
278        None => return 0,
279    };
280
281    let mut count = 0u32;
282    for sym in symbols {
283        let name = sym.get("name").and_then(|v| v.as_str()).unwrap_or("");
284        let kind = sym.get("kind").and_then(|v| v.as_str()).unwrap_or("");
285        let line_start = sym.get("line_start").and_then(|v| v.as_i64()).unwrap_or(0);
286        let line_end = sym.get("line_end").and_then(|v| v.as_i64()).unwrap_or(0);
287        let signature = sym.get("signature").and_then(|v| v.as_str());
288        let visibility = sym.get("visibility").and_then(|v| v.as_str());
289
290        // Note: parent_id resolution (matching parent name → id) deferred to phase 6
291        // when we have a proper protocol. For now, parent_id is NULL.
292        let result = conn.execute(
293            "INSERT INTO symbols (file_id, name, kind, line_start, line_end, signature, visibility)
294             VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7)",
295            rusqlite::params![
296                file_id, name, kind, line_start, line_end, signature, visibility
297            ],
298        );
299
300        if result.is_ok() {
301            count += 1;
302        }
303    }
304    count
305}

307/// Insert refs from plugin output into the database. Returns count of refs added.
308fn insert_refs(conn: &Connection, file_id: i64, output: &serde_json::Value) -> u32 {
309    let refs = match output.get("refs").and_then(|v| v.as_array()) {
310        Some(arr) => arr,
311        None => return 0,
312    };
313
314    let mut count = 0u32;
315    for r in refs {
316        let from_line = r.get("from_line").and_then(|v| v.as_i64()).unwrap_or(0);
317        let to_name = r.get("to_name").and_then(|v| v.as_str()).unwrap_or("");
318        let kind = r.get("kind").and_then(|v| v.as_str());
319
320        let result = conn.execute(
321            "INSERT INTO refs (from_file_id, from_line, to_name, kind)
322             VALUES (?1, ?2, ?3, ?4)",
323            rusqlite::params![file_id, from_line, to_name, kind],
324        );
325
326        if result.is_ok() {
327            count += 1;
328        }
329    }
330    count
331}
