1//! Query interface for the codebase index.
2//!
3//! Provides symbol search, reference lookup, and index status reporting.
4//! All queries are read-only and return structured data.
5
6use rusqlite::Connection;
7use std::fmt;
8use std::path::Path;

10/// A symbol row returned from a query.
11#[derive(Debug)]
12pub struct SymbolRow {
13    pub id: i64,
14    pub file_path: String,
15    pub name: String,
16    pub kind: String,
17    pub line_start: i64,
18    pub line_end: i64,
19    pub signature: Option<String>,
20    pub visibility: Option<String>,
21}
22
23impl fmt::Display for SymbolRow {
24    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
25        write!(
26            f,
27            "{}:{}-{} {} {} {}",
28            self.file_path,
29            self.line_start,
30            self.line_end,
31            self.kind,
32            self.name,
33            self.signature.as_deref().unwrap_or("")
34        )
35    }
36}

38/// A reference row returned from a query.
39#[derive(Debug)]
40pub struct RefRow {
41    pub file_path: String,
42    pub from_line: i64,
43    pub to_name: String,
44    pub kind: Option<String>,
45}
46
47impl fmt::Display for RefRow {
48    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
49        write!(
50            f,
51            "{}:{} -> {} ({})",
52            self.file_path,
53            self.from_line,
54            self.to_name,
55            self.kind.as_deref().unwrap_or("unknown")
56        )
57    }
58}

60/// Options for querying symbols.
61#[derive(Debug, Default)]
62pub struct SymbolQuery {
63    /// Filter by symbol name (substring match, case-insensitive).
64    pub name: Option<String>,
65    /// Filter by symbol kind (exact match, e.g. "function", "struct").
66    pub kind: Option<String>,
67    /// Filter by file path (substring match).
68    pub file: Option<String>,
69    /// Maximum results to return.
70    pub limit: u32,
71}
72
73/// Query symbols from the index, filtering by name/kind/file.
74pub fn query_symbols(conn: &Connection, opts: &SymbolQuery) -> Vec<SymbolRow> {
75    let mut sql = String::from(
76        "SELECT s.id, f.path, s.name, s.kind, s.line_start, s.line_end, s.signature, s.visibility
77         FROM symbols s
78         JOIN files f ON s.file_id = f.id
79         WHERE 1=1",
80    );
81    let mut params: Vec<Box<dyn rusqlite::types::ToSql>> = Vec::new();
82
83    if let Some(ref name) = opts.name {
84        sql.push_str(" AND s.name LIKE ?");
85        params.push(Box::new(format!("%{}%", name)));
86    }
87    if let Some(ref kind) = opts.kind {
88        sql.push_str(" AND s.kind = ?");
89        params.push(Box::new(kind.clone()));
90    }
91    if let Some(ref file) = opts.file {
92        sql.push_str(" AND f.path LIKE ?");
93        params.push(Box::new(format!("%{}%", file)));
94    }
95
96    let limit = if opts.limit == 0 { 50 } else { opts.limit };
97    sql.push_str(&format!(" ORDER BY f.path, s.line_start LIMIT {}", limit));
98
99    let param_refs: Vec<&dyn rusqlite::types::ToSql> = params.iter().map(|p| p.as_ref()).collect();
100
101    let mut stmt = match conn.prepare(&sql) {
102        Ok(s) => s,
103        Err(_) => return Vec::new(),
104    };
105
106    let rows = match stmt.query_map(param_refs.as_slice(), |row| {
107        Ok(SymbolRow {
108            id: row.get(0)?,
109            file_path: row.get(1)?,
110            name: row.get(2)?,
111            kind: row.get(3)?,
112            line_start: row.get(4)?,
113            line_end: row.get(5)?,
114            signature: row.get(6)?,
115            visibility: row.get(7)?,
116        })
117    }) {
118        Ok(r) => r,
119        Err(_) => return Vec::new(),
120    };
121
122    rows.filter_map(|r| r.ok()).collect()
123}

125/// Find references to a given name in the index.
126pub fn query_refs(conn: &Connection, to_name: &str, limit: u32) -> Vec<RefRow> {
127    let limit = if limit == 0 { 50 } else { limit };
128
129    let mut stmt = match conn.prepare(
130        "SELECT f.path, r.from_line, r.to_name, r.kind
131         FROM refs r
132         JOIN files f ON r.from_file_id = f.id
133         WHERE r.to_name LIKE ?1
134         ORDER BY f.path, r.from_line
135         LIMIT ?2",
136    ) {
137        Ok(s) => s,
138        Err(_) => return Vec::new(),
139    };
140
141    let rows = match stmt.query_map(rusqlite::params![format!("%{}%", to_name), limit], |row| {
142        Ok(RefRow {
143            file_path: row.get(0)?,
144            from_line: row.get(1)?,
145            to_name: row.get(2)?,
146            kind: row.get(3)?,
147        })
148    }) {
149        Ok(r) => r,
150        Err(_) => return Vec::new(),
151    };
152
153    rows.filter_map(|r| r.ok()).collect()
154}

156/// Return a human-readable summary of the index status.
157pub fn index_status(conn: &Connection, project_root: &Path) -> String {
158    let file_count: u32 = conn
159        .query_row("SELECT COUNT(*) FROM files", [], |row| row.get(0))
160        .unwrap_or(0);
161    let symbol_count: u32 = conn
162        .query_row("SELECT COUNT(*) FROM symbols", [], |row| row.get(0))
163        .unwrap_or(0);
164    let ref_count: u32 = conn
165        .query_row("SELECT COUNT(*) FROM refs", [], |row| row.get(0))
166        .unwrap_or(0);
167
168    // Language breakdown.
169    let lang_stats: Vec<(String, u32)> = conn
170        .prepare("SELECT COALESCE(lang, 'unknown'), COUNT(*) FROM files GROUP BY lang ORDER BY COUNT(*) DESC")
171        .ok()
172        .map(|mut stmt| {
173            stmt.query_map([], |row| Ok((row.get(0)?, row.get(1)?)))
174                .ok()
175                .map(|rows| rows.filter_map(|r| r.ok()).collect())
176                .unwrap_or_default()
177        })
178        .unwrap_or_default();
179
180    let mut out = format!(
181        "Index: {} ({})\n  files: {}, symbols: {}, refs: {}",
182        project_root.display(),
183        if file_count == 0 { "empty" } else { "active" },
184        file_count,
185        symbol_count,
186        ref_count,
187    );
188
189    if !lang_stats.is_empty() {
190        out.push_str("\n  languages: ");
191        let parts: Vec<String> = lang_stats
192            .iter()
193            .map(|(l, c)| format!("{} ({})", l, c))
194            .collect();
195        out.push_str(&parts.join(", "));
196    }
197
198    out
199}
