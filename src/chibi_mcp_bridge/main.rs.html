23/// Heartbeat interval for lockfile freshness (seconds).
24const HEARTBEAT_SECS: u64 = 30;

26/// Lockfile content written atomically on startup.
27#[derive(serde::Serialize)]
28struct LockContent {
29    pid: u32,
30    address: String,
31    started: u64,
32    heartbeat_secs: u64,
33    timestamp: u64,
34}

36/// Resolve chibi home directory: `CHIBI_HOME` env > `~/.chibi`
37fn chibi_home() -> PathBuf {
38    if let Ok(h) = std::env::var("CHIBI_HOME") {
39        return PathBuf::from(h);
40    }
41    dirs::home_dir()
42        .expect("could not determine home directory")
43        .join(".chibi")
44}

46/// Read `api_key` from `<home>/config.toml` (chibi's main config).
47///
48/// Returns `None` if the file is missing, unparseable, or has no key set.
49fn read_api_key(home: &Path) -> Option<String> {
50    let content = fs::read_to_string(home.join("config.toml")).ok()?;
51    let table: toml::Table = content.parse().ok()?;
52    table.get("api_key")?.as_str().map(String::from)
53}

55/// Write bridge lockfile atomically (O_CREAT | O_EXCL).
56///
57/// If a lockfile already exists, checks whether it is stale (heartbeat
58/// timestamp older than 1.5x the heartbeat interval). Stale lockfiles are
59/// removed and retried. Returns `AlreadyExists` only when another bridge
60/// instance is genuinely running.
61fn write_lockfile(home: &Path, addr: &SocketAddr) -> std::io::Result<PathBuf> {
62    let lock_path = home.join("mcp-bridge.lock");
63
64    if let Some(parent) = lock_path.parent() {
65        fs::create_dir_all(parent)?;
66    }
67
68    let now = SystemTime::now()
69        .duration_since(UNIX_EPOCH)
70        .unwrap_or(Duration::ZERO)
71        .as_secs();
72
73    let content = LockContent {
74        pid: std::process::id(),
75        address: addr.to_string(),
76        started: now,
77        heartbeat_secs: HEARTBEAT_SECS,
78        timestamp: now,
79    };
80
81    let json = serde_json::to_string(&content).map_err(std::io::Error::other)?;
82
83    // Try to create exclusively first.
84    match OpenOptions::new()
85        .write(true)
86        .create_new(true)
87        .open(&lock_path)
88    {
89        Ok(mut file) => {
90            file.write_all(json.as_bytes())?;
91            file.sync_all()?;
92            Ok(lock_path)
93        }
94        Err(e) if e.kind() == std::io::ErrorKind::AlreadyExists => {
95            // Check if the existing lock is stale (heartbeat expired).
96            if is_lockfile_stale(&lock_path) {
97                eprintln!("[mcp-bridge] removing stale lockfile");
98                fs::remove_file(&lock_path)?;
99                // Retry once.
100                let mut file = OpenOptions::new()
101                    .write(true)
102                    .create_new(true)
103                    .open(&lock_path)?;
104                file.write_all(json.as_bytes())?;
105                file.sync_all()?;
106                return Ok(lock_path);
107            }
108            Err(e)
109        }
110        Err(e) => Err(e),
111    }
112}

114/// Check if the lockfile is stale (heartbeat timestamp older than 1.5x interval,
115/// or PID no longer running).
116fn is_lockfile_stale(lock_path: &Path) -> bool {
117    let content = match fs::read_to_string(lock_path) {
118        Ok(c) => c,
119        Err(_) => return true,
120    };
121    let lock: serde_json::Value = match serde_json::from_str(&content) {
122        Ok(v) => v,
123        Err(_) => return true,
124    };
125
126    // PID liveness: instant detection of a crashed bridge without waiting for
127    // the heartbeat to expire.
128    #[cfg(target_os = "linux")]
129    {
130        if let Some(pid) = lock.get("pid").and_then(|v| v.as_u64())
131            && !Path::new(&format!("/proc/{pid}")).exists()
132        {
133            return true;
134        }
135    }
136
137    let heartbeat_secs = lock
138        .get("heartbeat_secs")
139        .and_then(|v| v.as_u64())
140        .unwrap_or(HEARTBEAT_SECS);
141    let timestamp = match lock.get("timestamp").and_then(|v| v.as_u64()) {
142        Some(t) => t,
143        None => return true,
144    };
145    let now = SystemTime::now()
146        .duration_since(UNIX_EPOCH)
147        .unwrap_or(Duration::ZERO)
148        .as_secs();
149    let stale_threshold = (heartbeat_secs as f64 * 1.5) as u64;
150    now.saturating_sub(timestamp) > stale_threshold
151}

153/// Touch the lockfile by updating its timestamp field.
154fn touch_lockfile(lock_path: &Path) -> std::io::Result<()> {
155    let content = fs::read_to_string(lock_path)?;
156    let mut lock: serde_json::Value =
157        serde_json::from_str(&content).map_err(std::io::Error::other)?;
158    let now = SystemTime::now()
159        .duration_since(UNIX_EPOCH)
160        .unwrap_or(Duration::ZERO)
161        .as_secs();
162    lock["timestamp"] = serde_json::json!(now);
163    let json = serde_json::to_string(&lock).map_err(std::io::Error::other)?;
164    fs::write(lock_path, json)?;
165    Ok(())
166}
167
168/// Remove the lockfile on shutdown.
169fn remove_lockfile(lock_path: &Path) {
170    let _ = fs::remove_file(lock_path);
171}

173/// Handle a single TCP connection: read one JSON request, dispatch via Bridge,
174/// write one JSON response.
175async fn handle_connection(
176    stream: &mut tokio::net::TcpStream,
177    bridge: &Bridge,
178) -> Result<(), Box<dyn std::error::Error>> {
179    let mut buf = Vec::with_capacity(4096);
180    stream.read_to_end(&mut buf).await?;
181
182    let request: protocol::Request = serde_json::from_slice(&buf)?;
183    let response = bridge.handle_request(request).await;
184
185    let response_json = serde_json::to_string(&response)?;
186    stream.write_all(response_json.as_bytes()).await?;
187    stream.shutdown().await?;
188
189    Ok(())
190}

192#[tokio::main]
193async fn main() -> Result<(), Box<dyn std::error::Error>> {
194    let home = chibi_home();
195    let config = BridgeConfig::load(&home);
196    let api_key = read_api_key(&home);
197
198    // Start MCP servers
199    let mut server_manager = ServerManager::new();
200    for (name, server_config) in &config.servers {
201        if let Err(e) = server_manager.start_server(name, server_config).await {
202            eprintln!("[mcp-bridge] failed to start server '{name}': {e}");
203        }
204    }
205
206    // Summary cache: shared between background generation and request handling.
207    // `None` when summaries are disabled â€” Bridge skips substitution entirely.
208    let all_tools = server_manager.list_all_tools();
209    let summary_cache = if config.summary.enabled {
210        let cache = Arc::new(Mutex::new(cache::SummaryCache::load(&home)));
211        if !all_tools.is_empty() {
212            let bg_cache = Arc::clone(&cache);
213            let summary_model = config.summary.model.clone();
214            let bg_api_key = api_key.clone();
215            tokio::spawn(async move {
216                let count = summary::fill_cache_gaps(
217                    &bg_cache,
218                    &all_tools,
219                    &summary_model,
220                    bg_api_key.as_deref(),
221                )
222                .await;
223                if count > 0 {
224                    eprintln!("[mcp-bridge] generated {count} new tool summaries");
225                }
226            });
227        }
228        Some(cache)
229    } else {
230        None
231    };
232
233    let bridge = Arc::new(Bridge {
234        server_manager,
235        summary_cache,
236    });
237
238    let listener = TcpListener::bind("127.0.0.1:0").await?;
239    let addr = listener.local_addr()?;
240    eprintln!("[mcp-bridge] listening on {addr}");
241
242    let lock_path = write_lockfile(&home, &addr)?;
243    eprintln!("[mcp-bridge] lockfile: {}", lock_path.display());
244
245    let idle_timeout = Duration::from_secs(config.idle_timeout_minutes * 60);
246    let last_activity = Arc::new(Mutex::new(Instant::now()));
247
248    // Watchdog: heartbeat + idle timeout. Returns when idle timeout is reached,
249    // causing the main select! to exit gracefully instead of process::exit().
250    let watchdog_activity = Arc::clone(&last_activity);
251    let watchdog_lock_path = lock_path.clone();
252    let watchdog = async move {
253        let mut heartbeat_elapsed = Duration::ZERO;
254        let tick = Duration::from_secs(10);
255        let heartbeat_interval = Duration::from_secs(HEARTBEAT_SECS);
256        loop {
257            tokio::time::sleep(tick).await;
258            heartbeat_elapsed += tick;
259
260            // Touch lockfile at heartbeat interval
261            if heartbeat_elapsed >= heartbeat_interval {
262                heartbeat_elapsed = Duration::ZERO;
263                if let Err(e) = touch_lockfile(&watchdog_lock_path) {
264                    eprintln!("[mcp-bridge] failed to touch lockfile: {e}");
265                }
266            }
267
268            let elapsed = watchdog_activity.lock().await.elapsed();
269            if elapsed >= idle_timeout {
270                eprintln!("[mcp-bridge] idle timeout reached, shutting down");
271                break;
272            }
273        }
274    };
275
276    // Accept loop: handle incoming connections until the watchdog signals shutdown.
277    let accept_loop = async {
278        loop {
279            let (mut stream, peer) = match listener.accept().await {
280                Ok(conn) => conn,
281                Err(e) => {
282                    eprintln!("[mcp-bridge] accept error: {e}");
283                    continue;
284                }
285            };
286            eprintln!("[mcp-bridge] connection from {peer}");
287            *last_activity.lock().await = Instant::now();
288
289            let bridge = Arc::clone(&bridge);
290            tokio::spawn(async move {
291                if let Err(e) = handle_connection(&mut stream, &bridge).await {
292                    eprintln!("[mcp-bridge] error handling connection: {e}");
293                }
294            });
295        }
296    };
297
298    tokio::select! {
299        () = watchdog => {}
300        () = accept_loop => {}
301    }
302
303    remove_lockfile(&lock_path);
304    Ok(())
305}

397    #[test]
398    fn lockfile_stale_detection() {
399        let tmp = TempDir::new().unwrap();
400        let lock_path = tmp.path().join("mcp-bridge.lock");
401
402        // Fresh lockfile should not be stale
403        let now = SystemTime::now()
404            .duration_since(UNIX_EPOCH)
405            .unwrap()
406            .as_secs();
407        let content = serde_json::json!({
408            "pid": 1, "address": "127.0.0.1:9999",
409            "started": now, "heartbeat_secs": 30, "timestamp": now,
410        });
411        fs::write(&lock_path, content.to_string()).unwrap();
412        assert!(!is_lockfile_stale(&lock_path));
413
414        // Old timestamp should be stale (60s ago, threshold is 45s)
415        let content = serde_json::json!({
416            "pid": 1, "address": "127.0.0.1:9999",
417            "started": now, "heartbeat_secs": 30, "timestamp": now - 60,
418        });
419        fs::write(&lock_path, content.to_string()).unwrap();
420        assert!(is_lockfile_stale(&lock_path));
421
422        // 40s ago should not be stale (under 45s threshold)
423        let content = serde_json::json!({
424            "pid": 1, "address": "127.0.0.1:9999",
425            "started": now, "heartbeat_secs": 30, "timestamp": now - 40,
426        });
427        fs::write(&lock_path, content.to_string()).unwrap();
428        assert!(!is_lockfile_stale(&lock_path));
429
430        // Invalid content should be treated as stale
431        fs::write(&lock_path, "not json").unwrap();
432        assert!(is_lockfile_stale(&lock_path));
433
434        // Missing timestamp field should be treated as stale
435        let content = serde_json::json!({"pid": 1, "address": "127.0.0.1:9999"});
436        fs::write(&lock_path, content.to_string()).unwrap();
437        assert!(is_lockfile_stale(&lock_path));
438    }
