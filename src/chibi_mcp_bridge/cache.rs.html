1//! Summary cache with schema-hash invalidation.
2//!
3//! Caches LLM-generated tool summaries keyed by `server:tool:schema_hash`.
4//! Persisted as JSONL in `CHIBI_HOME/mcp-bridge/cache.jsonl`.
5
6use sha2::{Digest, Sha256};
7use std::collections::HashMap;
8use std::io;
9use std::path::{Path, PathBuf};
10
11/// Persistent summary cache backed by a JSONL file.
12pub struct SummaryCache {
13    entries: HashMap<String, String>,
14    path: PathBuf,
15}
16
17/// Single cache entry for JSONL serialisation.
18#[derive(serde::Serialize, serde::Deserialize)]
19struct CacheEntry {
20    key: String,
21    summary: String,
22}
23
24/// Build a cache key from server name, tool name, and schema hash.
25fn cache_key(server: &str, tool: &str, schema: &serde_json::Value) -> String {
26    let schema_str = serde_json::to_string(schema).unwrap_or_default();
27    let hash = Sha256::digest(schema_str.as_bytes());
28    // Use first 8 bytes (16 hex chars) for a compact but collision-resistant key
29    let hex: String = hash[..8].iter().map(|b| format!("{b:02x}")).collect();
30    format!("{server}:{tool}:{hex}")
31}
32
33impl SummaryCache {
34    /// Load cache from disk, or create an empty cache if the file doesn't exist.
35    pub fn load(home: &Path) -> Self {
36        let path = home.join("mcp-bridge").join("cache.jsonl");
37        let entries = match std::fs::read_to_string(&path) {
38            Ok(content) => content
39                .lines()
40                .filter_map(|line| serde_json::from_str::<CacheEntry>(line).ok())
41                .map(|e| (e.key, e.summary))
42                .collect(),
43            Err(_) => HashMap::new(),
44        };
45        Self { entries, path }
46    }
47
48    /// Save the cache to disk as JSONL.
49    pub fn save(&self) -> io::Result<()> {
50        if let Some(parent) = self.path.parent() {
51            std::fs::create_dir_all(parent)?;
52        }
53        let content: String = self
54            .entries
55            .iter()
56            .map(|(key, summary)| {
57                serde_json::to_string(&CacheEntry {
58                    key: key.clone(),
59                    summary: summary.clone(),
60                })
61                .unwrap_or_default()
62            })
63            .collect::<Vec<_>>()
64            .join("\n");
65        std::fs::write(&self.path, content)
66    }
67
68    /// Look up a cached summary for a tool.
69    pub fn get(&self, server: &str, tool: &str, schema: &serde_json::Value) -> Option<&str> {
70        let key = cache_key(server, tool, schema);
71        self.entries.get(&key).map(|s| s.as_str())
72    }
73
74    /// Store a summary in the cache.
75    pub fn set(&mut self, server: &str, tool: &str, schema: &serde_json::Value, summary: String) {
76        let key = cache_key(server, tool, schema);
77        self.entries.insert(key, summary);
78    }
79
80    /// Number of cached entries.
81    #[cfg(test)]
82    pub fn len(&self) -> usize {
83        self.entries.len()
84    }
85}

87#[cfg(test)]
88mod tests {
89    use super::*;
90    use serde_json::json;
91
92    #[test]
93    fn empty_cache_returns_none() {
94        let tmp = tempfile::tempdir().unwrap();
95        let cache = SummaryCache::load(tmp.path());
96        assert!(cache.get("srv", "tool", &json!({})).is_none());
97    }
98
99    #[test]
100    fn set_get_roundtrip() {
101        let tmp = tempfile::tempdir().unwrap();
102        let mut cache = SummaryCache::load(tmp.path());
103        let schema = json!({"type": "object"});
104        cache.set("srv", "greet", &schema, "says hello".into());
105        assert_eq!(cache.get("srv", "greet", &schema), Some("says hello"));
106    }
107
108    #[test]
109    fn different_schema_hash_is_cache_miss() {
110        let tmp = tempfile::tempdir().unwrap();
111        let mut cache = SummaryCache::load(tmp.path());
112        let schema_v1 = json!({"type": "object", "properties": {"a": {}}});
113        let schema_v2 = json!({"type": "object", "properties": {"a": {}, "b": {}}});
114        cache.set("srv", "tool", &schema_v1, "v1 summary".into());
115        assert_eq!(cache.get("srv", "tool", &schema_v1), Some("v1 summary"));
116        assert!(cache.get("srv", "tool", &schema_v2).is_none());
117    }
118
119    #[test]
120    fn save_and_reload() {
121        let tmp = tempfile::tempdir().unwrap();
122        let schema = json!({"type": "object"});
123
124        {
125            let mut cache = SummaryCache::load(tmp.path());
126            cache.set("srv", "tool1", &schema, "summary one".into());
127            cache.set("srv", "tool2", &schema, "summary two".into());
128            cache.save().unwrap();
129        }
130
131        let cache = SummaryCache::load(tmp.path());
132        assert_eq!(cache.get("srv", "tool1", &schema), Some("summary one"));
133        assert_eq!(cache.get("srv", "tool2", &schema), Some("summary two"));
134        assert_eq!(cache.len(), 2);
135    }
136
137    #[test]
138    fn missing_file_returns_empty_cache() {
139        let tmp = tempfile::tempdir().unwrap();
140        let cache = SummaryCache::load(tmp.path());
141        assert_eq!(cache.len(), 0);
142    }
143}
