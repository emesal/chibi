117/// Detect terminal rendering capabilities from environment variables
118fn detect_terminal_capability() -> TerminalCapability {
119    // Check COLORTERM for truecolor support
120    if let Ok(colorterm) = std::env::var("COLORTERM") {
121        let ct = colorterm.to_lowercase();
122        if ct.contains("truecolor") || ct.contains("24bit") {
123            return TerminalCapability::Truecolor;
124        }
125    }
126
127    // Check TERM for color support level
128    if let Ok(term) = std::env::var("TERM") {
129        let t = term.to_lowercase();
130        if t.contains("truecolor") || t.contains("24bit") {
131            return TerminalCapability::Truecolor;
132        }
133        if t.contains("256color") {
134            return TerminalCapability::Ansi256;
135        }
136        if t.contains("color") {
137            return TerminalCapability::Ansi16;
138        }
139    }
140
141    // Default to basic ANSI if we can't determine capability
142    TerminalCapability::Ansi16
143}

145/// Resolve the rendering mode based on config and terminal capabilities
146fn resolve_render_mode(
147    mode: ConfigImageRenderMode,
148    enable_truecolor: bool,
149    enable_ansi: bool,
150    enable_ascii: bool,
151) -> ImageRenderMode {
152    match mode {
153        ConfigImageRenderMode::Truecolor if enable_truecolor => ImageRenderMode::Truecolor,
154        ConfigImageRenderMode::Ansi if enable_ansi => ImageRenderMode::Ansi,
155        ConfigImageRenderMode::Ascii if enable_ascii => ImageRenderMode::Ascii,
156        ConfigImageRenderMode::Placeholder => ImageRenderMode::Placeholder,
157        ConfigImageRenderMode::Auto => {
158            let cap = detect_terminal_capability();
159            match cap {
160                TerminalCapability::Truecolor if enable_truecolor => ImageRenderMode::Truecolor,
161                TerminalCapability::Truecolor
162                | TerminalCapability::Ansi256
163                | TerminalCapability::Ansi16
164                    if enable_ansi =>
165                {
166                    ImageRenderMode::Ansi
167                }
168                _ if enable_ascii => ImageRenderMode::Ascii,
169                _ => ImageRenderMode::Placeholder,
170            }
171        }
172        _ => {
173            // Disabled mode, fallback to auto logic
174            resolve_render_mode(
175                ConfigImageRenderMode::Auto,
176                enable_truecolor,
177                enable_ansi,
178                enable_ascii,
179            )
180        }
181    }
182}

235    /// Render a list of parse events, intercepting image events when enabled.
236    fn render_events(
237        pipeline: &mut RenderPipeline,
238        events: &[ParseEvent],
239        render_images: bool,
240        terminal_width: usize,
241        fetch_config: &ImageFetchConfig,
242        display_config: &ImageDisplayConfig,
243        render_mode: ImageRenderMode,
244    ) -> io::Result<()> {
245        for event in events {
246            if render_images
247                && let ParseEvent::Image { url, .. } = event
248                && try_render_image(
249                    url,
250                    terminal_width,
251                    fetch_config,
252                    display_config,
253                    render_mode,
254                )
255                .is_ok()
256            {
257                continue;
258            }
259            pipeline
260                .renderer
261                .render_event(event)
262                .map_err(|e| io::Error::other(format!("Render error: {}", e)))?;
263        }
264        Ok(())
265    }

267    /// Buffer chunk, render complete lines. Passthrough if not TTY.
268    pub fn write_chunk(&mut self, chunk: &str) -> io::Result<()> {
269        if chunk.is_empty() {
270            return Ok(());
271        }
272
273        let pipeline = match self.pipeline.as_mut() {
274            Some(p) => p,
275            None => {
276                // Passthrough: write raw bytes directly
277                let mut lock = io::stdout().lock();
278                lock.write_all(chunk.as_bytes())?;
279                lock.flush()?;
280                return Ok(());
281            }
282        };
283
284        self.line_buffer.push_str(chunk);
285
286        // Process all complete lines
287        while let Some(newline_pos) = self.line_buffer.find('\n') {
288            let line = self.line_buffer[..newline_pos]
289                .trim_end_matches('\r')
290                .to_string();
291            self.line_buffer = self.line_buffer[newline_pos + 1..].to_string();
292
293            let events = pipeline.parser.parse_line(&line);
294            Self::render_events(
295                pipeline,
296                &events,
297                self.render_images,
298                self.terminal_width,
299                &self.fetch_config,
300                &self.display_config,
301                self.render_mode,
302            )?;
303        }
304
305        // Prevent unbounded buffer growth: force-flush if buffer exceeds limit
306        if self.line_buffer.len() > MAX_LINE_BUFFER_BYTES {
307            let line = std::mem::take(&mut self.line_buffer);
308            let line = line.trim_end_matches('\r');
309            let events = pipeline.parser.parse_line(line);
310            Self::render_events(
311                pipeline,
312                &events,
313                self.render_images,
314                self.terminal_width,
315                &self.fetch_config,
316                &self.display_config,
317                self.render_mode,
318            )?;
319        }
320
321        Ok(())
322    }
323
324    /// Flush remaining partial line buffer. Call at end of response.
325    pub fn finish(&mut self) -> io::Result<()> {
326        let pipeline = match self.pipeline.as_mut() {
327            Some(p) => p,
328            None => return Ok(()),
329        };
330
331        // Flush any remaining partial line
332        if !self.line_buffer.is_empty() {
333            let line = std::mem::take(&mut self.line_buffer);
334            let line = line.trim_end_matches('\r');
335            let events = pipeline.parser.parse_line(line);
336            Self::render_events(
337                pipeline,
338                &events,
339                self.render_images,
340                self.terminal_width,
341                &self.fetch_config,
342                &self.display_config,
343                self.render_mode,
344            )?;
345        }
346
347        // Finalize parser (close any open blocks)
348        let events = pipeline.parser.finalize();
349        Self::render_events(
350            pipeline,
351            &events,
352            self.render_images,
353            self.terminal_width,
354            &self.fetch_config,
355            &self.display_config,
356            self.render_mode,
357        )?;
358
359        Ok(())
360    }

363/// Decode a `data:image/...;base64,...` URI into a `DynamicImage`.
364///
365/// Expects the input with the `data:` prefix already stripped (i.e., starts
366/// with `image/...`).
367fn decode_data_uri_image(rest: &str) -> io::Result<image::DynamicImage> {
368    let (mime, payload) = rest
369        .split_once(";base64,")
370        .ok_or_else(|| io::Error::other("data URI missing ;base64, delimiter"))?;
371
372    if !mime.starts_with("image/") {
373        return Err(io::Error::other(format!(
374            "data URI MIME type is not an image: {}",
375            mime
376        )));
377    }
378
379    let bytes = base64::engine::general_purpose::STANDARD
380        .decode(payload)
381        .map_err(|e| io::Error::other(format!("invalid base64 in data URI: {}", e)))?;
382
383    image::load_from_memory(&bytes)
384        .map_err(|e| io::Error::other(format!("failed to decode image from data URI: {}", e)))
385}

387/// Fetch a remote image over HTTP(S) and decode it.
388fn fetch_remote_image(url: &str, config: &ImageFetchConfig) -> io::Result<image::DynamicImage> {
389    if url.starts_with("http://") && !config.allow_http {
390        return Err(io::Error::other(
391            "plain HTTP image URLs are not allowed (set image_allow_http = true to enable)",
392        ));
393    }
394
395    // Try cache first
396    if let Some(ref cache_dir) = config.cache_dir
397        && let Some(cached) = crate::image_cache::cache_get(cache_dir, url)
398    {
399        return image::load_from_memory(&cached)
400            .map_err(|e| io::Error::other(format!("failed to decode cached image: {}", e)));
401    }
402
403    // Fetch from network — spawn on a separate task to isolate timeout/cancellation
404    // handling. block_in_place temporarily parks the calling thread while awaiting
405    // the result, which is necessary for synchronous rendering.
406    let url_owned = url.to_string();
407    let max_bytes = config.max_download_bytes;
408    let timeout = config.fetch_timeout_seconds;
409    let allow_http = config.allow_http;
410    let bytes = tokio::task::block_in_place(|| {
411        tokio::runtime::Handle::current().block_on(async {
412            tokio::task::spawn(fetch_image_bytes(url_owned, max_bytes, timeout, allow_http))
413                .await
414                .map_err(|e| io::Error::other(format!("image fetch task failed: {}", e)))?
415        })
416    })?;
417
418    // Store in cache (best-effort)
419    if let Some(ref cache_dir) = config.cache_dir {
420        let _ = crate::image_cache::cache_put(
421            cache_dir,
422            url,
423            &bytes,
424            config.cache_max_bytes,
425            config.cache_max_age_days,
426        );
427    }
428
429    image::load_from_memory(&bytes)
430        .map_err(|e| io::Error::other(format!("failed to decode fetched image: {}", e)))
431}

433/// Asynchronously fetch image bytes from a URL with size and timeout limits.
434async fn fetch_image_bytes(
435    url: String,
436    max_bytes: usize,
437    timeout_seconds: u64,
438    allow_http: bool,
439) -> io::Result<Vec<u8>> {
440    use futures_util::StreamExt;
441
442    // Build a redirect policy that prevents HTTPS→HTTP downgrades
443    let redirect_policy = reqwest::redirect::Policy::custom(move |attempt| {
444        if attempt.previous().len() >= 5 {
445            attempt.error("too many redirects (max 5)")
446        } else if !allow_http {
447            // Block HTTPS→HTTP downgrade
448            if let Some(prev) = attempt.previous().last()
449                && prev.scheme() == "https"
450                && attempt.url().scheme() == "http"
451            {
452                return attempt.error("redirect from HTTPS to HTTP is not allowed");
453            }
454            attempt.follow()
455        } else {
456            attempt.follow()
457        }
458    });
459
460    let client = reqwest::Client::builder()
461        .timeout(std::time::Duration::from_secs(timeout_seconds))
462        .redirect(redirect_policy)
463        .build()
464        .map_err(|e| io::Error::other(format!("failed to build HTTP client: {}", e)))?;
465
466    let response = client
467        .get(url)
468        .send()
469        .await
470        .map_err(|e| io::Error::other(format!("image fetch failed: {}", e)))?;
471
472    if !response.status().is_success() {
473        return Err(io::Error::other(format!(
474            "image fetch returned HTTP {}",
475            response.status()
476        )));
477    }
478
479    // Validate Content-Type if present
480    if let Some(ct) = response.headers().get(reqwest::header::CONTENT_TYPE)
481        && let Ok(ct_str) = ct.to_str()
482        && !ct_str.starts_with("image/")
483    {
484        return Err(io::Error::other(format!(
485            "remote URL Content-Type is not an image: {}",
486            ct_str
487        )));
488    }
489
490    // Early reject if Content-Length exceeds limit
491    if let Some(len) = response.content_length()
492        && len as usize > max_bytes
493    {
494        return Err(io::Error::other(format!(
495            "image too large: Content-Length {} exceeds limit {}",
496            len, max_bytes
497        )));
498    }
499
500    // Stream body with size enforcement
501    let mut stream = response.bytes_stream();
502    let mut buf = Vec::new();
503
504    while let Some(chunk_result) = stream.next().await {
505        let chunk =
506            chunk_result.map_err(|e| io::Error::other(format!("image download error: {}", e)))?;
507        if buf.len() + chunk.len() > max_bytes {
508            return Err(io::Error::other(format!(
509                "image download exceeded size limit of {} bytes",
510                max_bytes
511            )));
512        }
513        buf.extend_from_slice(&chunk);
514    }
515
516    Ok(buf)
517}

519/// Attempt to render an image inline with the appropriate mode
520fn try_render_image(
521    url: &str,
522    term_width: usize,
523    fetch_config: &ImageFetchConfig,
524    display_config: &ImageDisplayConfig,
525    render_mode: ImageRenderMode,
526) -> io::Result<()> {
527    // Early return for placeholder mode
528    if matches!(render_mode, ImageRenderMode::Placeholder) {
529        return Err(io::Error::other("placeholder mode"));
530    }
531
532    // Load the image
533    let img = if let Some(rest) = url.strip_prefix("data:") {
534        decode_data_uri_image(rest)?
535    } else if url.starts_with("http://") || url.starts_with("https://") {
536        fetch_remote_image(url, fetch_config)?
537    } else {
538        let path = url.strip_prefix("file://").unwrap_or(url);
539        image::open(path)
540            .map_err(|e| io::Error::other(format!("Failed to open image {}: {}", path, e)))?
541    };
542
543    let (orig_w, orig_h) = (img.width(), img.height());
544    let term_w = term_width as u32;
545
546    // Apply width percent constraint
547    let effective_width = term_w * display_config.max_width_percent.min(100) / 100;
548    // Height limit in pixels (truecolor renderer prints 2 pixels per terminal line)
549    let max_height_pixels = display_config.max_height_lines * 2;
550
551    let (new_w, mut new_h) = imgcatr::ops::image_resized_size(
552        (orig_w, orig_h),
553        (effective_width, max_height_pixels),
554        true,
555    );
556
557    // Clamp height if it still exceeds the limit
558    if new_h > max_height_pixels {
559        new_h = max_height_pixels;
560    }
561
562    // Ensure height is even (truecolor renderer prints pixel pairs)
563    if new_h % 2 != 0 {
564        new_h += 1;
565    }
566
567    let resized = imgcatr::ops::resize_image(&img, (new_w, new_h));
568
569    // Render with the appropriate mode
570    match render_mode {
571        ImageRenderMode::Truecolor => render_truecolor(&resized, term_width, display_config),
572        ImageRenderMode::Ansi => render_ansi(&resized, term_width, display_config),
573        ImageRenderMode::Ascii => render_ascii(&resized, term_width, display_config),
574        ImageRenderMode::Placeholder => unreachable!(),
575    }
576}

578/// Render image with 24-bit truecolor ANSI codes
579fn render_truecolor(
580    img: &image::DynamicImage,
581    term_width: usize,
582    display_config: &ImageDisplayConfig,
583) -> io::Result<()> {
584    let new_w = img.width() as usize;
585    let new_h = img.height() as usize;
586    let pad = calculate_padding(new_w, term_width, display_config.alignment);
587
588    let mut stdout = io::stdout().lock();
589    writeln!(stdout)?;
590
591    if pad == 0 {
592        imgcatr::ops::write_ansi_truecolor(&mut stdout, img);
593    } else {
594        let padding: String = " ".repeat(pad);
595        let pixels = img.to_rgba8();
596        let mut y = 0;
597        while y + 1 < new_h {
598            write!(stdout, "{}", padding)?;
599            for x in 0..new_w {
600                let top = pixels.get_pixel(x as u32, y as u32);
601                let bot = pixels.get_pixel(x as u32, (y + 1) as u32);
602                write!(
603                    stdout,
604                    "\x1b[38;2;{};{};{};48;2;{};{};{}m\u{2580}",
605                    top[0], top[1], top[2], bot[0], bot[1], bot[2]
606                )?;
607            }
608            writeln!(stdout, "\x1b[0m")?;
609            y += 2;
610        }
611        if y < new_h {
612            write!(stdout, "{}", padding)?;
613            for x in 0..new_w {
614                let top = pixels.get_pixel(x as u32, y as u32);
615                write!(
616                    stdout,
617                    "\x1b[38;2;{};{};{}m\u{2580}",
618                    top[0], top[1], top[2]
619                )?;
620            }
621            writeln!(stdout, "\x1b[0m")?;
622        }
623    }
624
625    writeln!(stdout)?;
626    stdout.flush()?;
627    Ok(())
628}

630/// Render image with 16-color ANSI codes
631fn render_ansi(
632    img: &image::DynamicImage,
633    term_width: usize,
634    display_config: &ImageDisplayConfig,
635) -> io::Result<()> {
636    let new_w = img.width() as usize;
637    let pad = calculate_padding(new_w, term_width, display_config.alignment);
638
639    let mut stdout = io::stdout().lock();
640    writeln!(stdout)?;
641
642    // Use imgcatr's ANSI color approximation
643    use imgcatr::util::ANSI_COLOURS_WHITE_BG;
644
645    if pad == 0 {
646        imgcatr::ops::write_ansi(&mut stdout, img, &ANSI_COLOURS_WHITE_BG);
647    } else {
648        // Manual rendering with padding
649        let padding: String = " ".repeat(pad);
650        let colourtable = imgcatr::ops::create_colourtable(
651            img,
652            &ANSI_COLOURS_WHITE_BG,
653            imgcatr::util::bg_colours_for(&ANSI_COLOURS_WHITE_BG),
654        );
655
656        for line in colourtable {
657            write!(stdout, "{}", padding)?;
658            for (upper_clr, lower_clr) in line {
659                write!(
660                    stdout,
661                    "{}{}\u{2580}",
662                    imgcatr::util::ANSI_COLOUR_ESCAPES[upper_clr],
663                    imgcatr::util::ANSI_BG_COLOUR_ESCAPES[lower_clr]
664                )?;
665            }
666            writeln!(stdout, "{}", imgcatr::util::ANSI_RESET_ATTRIBUTES)?;
667        }
668    }
669
670    writeln!(stdout)?;
671    stdout.flush()?;
672    Ok(())
673}

675/// Render image as ASCII art
676fn render_ascii(
677    img: &image::DynamicImage,
678    term_width: usize,
679    display_config: &ImageDisplayConfig,
680) -> io::Result<()> {
681    let (width, height) = img.dimensions();
682    let pad = calculate_padding(width as usize, term_width, display_config.alignment);
683    let padding: String = " ".repeat(pad);
684
685    let mut stdout = io::stdout().lock();
686    writeln!(stdout)?;
687
688    // Convert to RGBA8 to ensure we can read pixels safely
689    let img = img.to_rgba8();
690
691    for y in (0..height).step_by(2) {
692        write!(stdout, "{}", padding)?;
693        for x in (0..width).step_by(1) {
694            let pix = img.get_pixel(x, y);
695            let mut intensity = pix[0] / 3 + pix[1] / 3 + pix[2] / 3;
696            // Handle transparency
697            if pix[3] == 0 {
698                intensity = 0;
699            }
700            write!(stdout, "{}", ascii_char(intensity))?;
701        }
702        writeln!(stdout)?;
703    }
704
705    writeln!(stdout)?;
706    stdout.flush()?;
707    Ok(())
708}

710/// Map intensity to ASCII character
711fn ascii_char(intensity: u8) -> &'static str {
712    let index = (intensity / 32).min(7);
713    [" ", ".", ",", "-", "~", "+", "=", "@"][index as usize]
714}

716/// Calculate left padding for image alignment
717fn calculate_padding(image_cols: usize, term_width: usize, alignment: ImageAlignment) -> usize {
718    match alignment {
719        ImageAlignment::Center => term_width.saturating_sub(image_cols) / 2,
720        ImageAlignment::Right => term_width.saturating_sub(image_cols),
721        ImageAlignment::Left => 0,
722    }
723}

1246    #[test]
1247    fn render_ascii_output_lines_match_image_height_div_two() {
1248        // render_ascii steps y by 2, so a 4-row image produces 2 content lines.
1249        // We verify this by counting the lines it would produce (height / 2)
1250        // and that the character width per line equals the image width.
1251        let width: u32 = 10;
1252        let height: u32 = 4;
1253        let img = image::RgbImage::from_pixel(width, height, image::Rgb([128, 128, 128]));
1254        let dynamic: image::DynamicImage = img.into();
1255
1256        // The expected content lines (excluding surrounding blank lines):
1257        // height stepped by 2 = height / 2 iterations
1258        let expected_content_lines = height / 2;
1259        assert_eq!(
1260            expected_content_lines, 2,
1261            "sanity: 4-row image produces 2 ASCII content lines"
1262        );
1263
1264        // Each content line has exactly `width` characters of ASCII art.
1265        // Verify ascii_char produces one char per pixel column.
1266        let rgba = dynamic.to_rgba8();
1267        for y in (0..height).step_by(2) {
1268            let mut char_count = 0u32;
1269            for x in 0..width {
1270                let pix = rgba.get_pixel(x, y);
1271                let intensity = pix[0] / 3 + pix[1] / 3 + pix[2] / 3;
1272                let ch = ascii_char(intensity);
1273                assert!(!ch.is_empty());
1274                char_count += 1;
1275            }
1276            assert_eq!(
1277                char_count, width,
1278                "each ASCII line must have exactly image_width characters"
1279            );
1280        }
1281    }
