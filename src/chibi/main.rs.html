26/// Build the interactive permission handler for gated operations.
27///
28/// Prompts the user via `/dev/tty` (not stdin, which may be piped) for Y/n
29/// confirmation on file writes and shell execution. Default-allow on Enter
30/// (empty input). Returns fail-safe deny if no TTY is available.
31fn build_interactive_permission_handler() -> PermissionHandler {
32    Box::new(|hook_data: &serde_json::Value| {
33        use chibi_core::json_ext::JsonExt;
34
35        let tool_name = hook_data.get_str_or("tool_name", "unknown");
36        let display = hook_data
37            .get_str("path")
38            .or_else(|| hook_data.get_str("command"))
39            .unwrap_or("(no details)");
40
41        eprint!("[{}] {} [Y/n] ", tool_name, display);
42        io::stderr().flush().ok();
43
44        // Read from /dev/tty so piped stdin doesn't interfere
45        let approved = match std::fs::File::open("/dev/tty") {
46            Ok(tty) => {
47                let mut reader = io::BufReader::new(tty);
48                let mut response = String::new();
49                if io::BufRead::read_line(&mut reader, &mut response).is_ok() {
50                    // Default-allow: only deny on explicit "n" or "no"
51                    !matches!(response.trim().to_lowercase().as_str(), "n" | "no")
52                } else {
53                    false
54                }
55            }
56            Err(_) => false, // no TTY = fail-safe deny
57        };
58
59        Ok(approved)
60    })
61}

63/// Build a trust-mode permission handler that auto-approves all operations.
64///
65/// Used with `-t`/`--trust` for headless/automation scenarios where all
66/// permission-gated tools should execute without prompting.
67fn build_trust_permission_handler() -> PermissionHandler {
68    Box::new(|_hook_data: &serde_json::Value| Ok(true))
69}
70
71/// Select the appropriate permission handler based on trust mode.
72fn select_permission_handler(trust: bool) -> PermissionHandler {
73    if trust {
74        build_trust_permission_handler()
75    } else {
76        build_interactive_permission_handler()
77    }
78}

108/// Generate a unique context name for `-c new` or `-c new:prefix`
109/// Format: [prefix_]YYYYMMDD_HHMMSS[_N]
110fn generate_new_context_name(chibi: &Chibi, prefix: Option<&str>) -> String {
111    use chrono::Local;
112
113    let timestamp = Local::now().format("%Y%m%d_%H%M%S").to_string();
114    let base_name = match prefix {
115        Some(p) => format!("{}_{}", p, timestamp),
116        None => timestamp,
117    };
118
119    // Check for collisions
120    let existing = chibi.list_contexts();
121    if !existing.contains(&base_name) {
122        return base_name;
123    }
124
125    // Append _N until we find an unused name
126    let mut n = 2;
127    loop {
128        let candidate = format!("{}_{}", base_name, n);
129        if !existing.contains(&candidate) {
130            return candidate;
131        }
132        n += 1;
133    }
134}

136/// Resolve "new" or "new:prefix" or "-" context names
137fn resolve_context_name(chibi: &Chibi, session: &Session, name: &str) -> io::Result<String> {
138    if name == "-" {
139        session.get_previous()
140    } else if name == "new" {
141        Ok(generate_new_context_name(chibi, None))
142    } else if let Some(prefix) = name.strip_prefix("new:") {
143        if prefix.is_empty() {
144            return Err(io::Error::new(
145                ErrorKind::InvalidInput,
146                "Prefix cannot be empty in 'new:prefix'",
147            ));
148        }
149        Ok(generate_new_context_name(chibi, Some(prefix)))
150    } else {
151        Ok(name.to_string())
152    }
153}

155/// Build CLI ResolvedConfig from Chibi facade.
156/// Combines core config with presentation settings from cli.toml.
157///
158/// When `context_window_limit` is unset (0), resolves it from ratatoskr's
159/// model registry via a synchronous (no-network) lookup.
160fn resolve_cli_config(
161    chibi: &Chibi,
162    context_name: &str,
163    username_override: Option<&str>,
164) -> io::Result<ResolvedConfig> {
165    let mut core = chibi.resolve_config(context_name, username_override)?;
166    let cli = load_cli_config(chibi.home_dir(), Some(context_name))?;
167
168    // Resolve context_window_limit from ratatoskr registry if still unknown
169    chibi_core::gateway::ensure_context_window(&mut core);
170
171    Ok(ResolvedConfig {
172        core,
173        render_markdown: cli.render_markdown,
174        verbose: cli.verbose,
175        hide_tool_calls: cli.hide_tool_calls,
176        show_thinking: cli.show_thinking,
177        image: cli.image,
178        markdown_style: cli.markdown_style,
179    })
180}

182/// Resolve CLI-specific context names in Command variants.
183///
184/// Handles "-" (previous context), "new", and "new:prefix" in optional
185/// context name fields. Core doesn't know about these â€” they're CLI
186/// conveniences tied to session state.
187fn resolve_command_names(
188    command: &Command,
189    chibi: &Chibi,
190    session: &Session,
191) -> io::Result<Command> {
192    // Helper: resolve an optional name, defaulting to working_context
193    let resolve_opt = |name: &Option<String>| -> io::Result<Option<String>> {
194        match name {
195            Some(n) => Ok(Some(resolve_context_name(chibi, session, n)?)),
196            None => Ok(None),
197        }
198    };
199
200    match command {
201        Command::DestroyContext { name } => Ok(Command::DestroyContext {
202            name: resolve_opt(name)?,
203        }),
204        Command::ArchiveHistory { name } => Ok(Command::ArchiveHistory {
205            name: resolve_opt(name)?,
206        }),
207        Command::CompactContext { name } => Ok(Command::CompactContext {
208            name: resolve_opt(name)?,
209        }),
210        Command::RenameContext { old, new } => Ok(Command::RenameContext {
211            old: resolve_opt(old)?,
212            new: new.clone(),
213        }),
214        Command::ShowLog { context, count } => Ok(Command::ShowLog {
215            context: resolve_opt(context)?,
216            count: *count,
217        }),
218        Command::Inspect { context, thing } => Ok(Command::Inspect {
219            context: resolve_opt(context)?,
220            thing: thing.clone(),
221        }),
222        Command::SetSystemPrompt { context, prompt } => Ok(Command::SetSystemPrompt {
223            context: resolve_opt(context)?,
224            prompt: prompt.clone(),
225        }),
226        Command::ClearCache { name } => Ok(Command::ClearCache {
227            name: resolve_opt(name)?,
228        }),
229        Command::CheckInbox { context } => Ok(Command::CheckInbox {
230            context: resolve_context_name(chibi, session, context)?,
231        }),
232        // All other commands pass through unchanged
233        _ => Ok(command.clone()),
234    }
235}

237/// Execute from ChibiInput.
238///
239/// Handles CLI-specific concerns (context selection, session, username overrides,
240/// help/version, image cache) then delegates command dispatch to core's
241/// `execute_command()`. Session updates are driven by `CommandEffect`.
242async fn execute_from_input(
243    input: ChibiInput,
244    chibi: &mut Chibi,
245    session: &mut Session,
246    force_markdown: bool,
247) -> io::Result<()> {
248    let mut did_action = false;
249
250    // Pre-resolution verbose: used for diagnostics before full config is resolved.
251    let early_verbose = input.verbose_flag;
252
253    // --- CLI-specific: context selection ---
254    // working_context: the context we're actually operating on this invocation
255    // implied_context: persisted in session.json, what you get when no context is specified
256    let working_context = match &input.context {
257        ContextSelection::Current => session.implied_context.clone(),
258        ContextSelection::Ephemeral { name } => {
259            let actual_name = resolve_context_name(chibi, session, name)?;
260            chibi.app.ensure_context_dir(&actual_name)?;
261            if early_verbose {
262                eprintln!("[Using ephemeral context: {}]", actual_name);
263            }
264            actual_name
265        }
266        ContextSelection::Switch { name, persistent } => {
267            if name == "-" {
268                session.swap_with_previous()?;
269            } else {
270                let actual_name = resolve_context_name(chibi, session, name)?;
271                session.switch_context(actual_name);
272            }
273            chibi.app.ensure_context_dir(&session.implied_context)?;
274
275            if *persistent {
276                session.save(chibi.home_dir())?;
277                chibi.save()?;
278            }
279            if early_verbose {
280                eprintln!("[Switched to context: {}]", &session.implied_context);
281            }
282            did_action = true;
283            session.implied_context.clone()
284        }
285    };
286
287    // --- CLI-specific: username override ---
288    let ephemeral_username: Option<&str> = match &input.username_override {
289        Some(UsernameOverride::Persistent(username)) => {
290            let mut local_config = chibi.app.load_local_config(&working_context)?;
291            local_config.username = Some(username.clone());
292            chibi
293                .app
294                .save_local_config(&working_context, &local_config)?;
295            if early_verbose {
296                eprintln!(
297                    "[Username '{}' saved to context '{}']",
298                    username, working_context
299                );
300            }
301            did_action = true;
302            None // persistent was saved, no runtime override needed
303        }
304        Some(UsernameOverride::Ephemeral(username)) => Some(username.as_str()),
305        None => None,
306    };
307
308    // --- CLI-specific: intercept ShowHelp / ShowVersion ---
309    match &input.command {
310        Command::ShowHelp => {
311            Cli::print_help();
312            return Ok(());
313        }
314        Command::ShowVersion => {
315            println!("chibi {}", env!("CARGO_PKG_VERSION"));
316            return Ok(());
317        }
318        _ => {}
319    }
320
321    // --- resolve CLI-specific context names in command fields ---
322    let command = resolve_command_names(&input.command, chibi, session)?;
323
324    // --- resolve config ---
325    let mut cli_config = resolve_cli_config(chibi, &working_context, ephemeral_username)?;
326
327    // Apply per-invocation config overrides (-s/--set)
328    if !input.config_overrides.is_empty() {
329        cli_config
330            .core
331            .apply_overrides_from_pairs(&input.config_overrides)
332            .map_err(|e| io::Error::new(ErrorKind::InvalidInput, e))?;
333    }
334
335    // CLI flags override cli.toml values
336    if input.verbose_flag {
337        cli_config.verbose = true;
338    }
339    if input.hide_tool_calls_flag {
340        cli_config.hide_tool_calls = true;
341    }
342    if input.show_thinking_flag {
343        cli_config.show_thinking = true;
344    }
345
346    // Derive presentation flags from CLI config
347    let verbose = cli_config.verbose;
348    let show_tool_calls = !cli_config.hide_tool_calls || verbose;
349    let show_thinking = cli_config.show_thinking || verbose;
350
351    let md_config = if cli_config.render_markdown && !input.raw {
352        Some(md_config_from_resolved(
353            &cli_config,
354            chibi.home_dir(),
355            force_markdown,
356        ))
357    } else {
358        None
359    };
360
361    // Build output handler now that markdown config is resolved.
362    // OutputHandler::emit_markdown renders via MarkdownStream when md_config is present,
363    // enabling markdown display in ShowLog (emit_entry calls emit_markdown for messages).
364    let handler = match md_config.clone() {
365        Some(cfg) => OutputHandler::with_markdown(verbose, cfg),
366        None => OutputHandler::new(verbose),
367    };
368
369    let mut sink =
370        CliResponseSink::new(&handler, md_config, verbose, show_tool_calls, show_thinking);
371
372    // --- delegate to core ---
373    let effect = chibi_core::execute_command(
374        chibi,
375        &working_context,
376        &command,
377        &input.flags,
378        &cli_config.core,
379        &handler,
380        &mut sink,
381    )
382    .await?;
383    if !matches!(command, Command::NoOp) {
384        did_action = true;
385    }
386
387    // --- CLI-specific: handle CommandEffect for session updates ---
388    match &effect {
389        CommandEffect::ContextDestroyed(name) => {
390            if session
391                .handle_context_destroyed(name, |n| chibi.app.context_dir(n).exists())
392                .is_some()
393            {
394                session.save(chibi.home_dir())?;
395            }
396        }
397        CommandEffect::ContextRenamed { old, new } => {
398            if session.implied_context == *old {
399                session.implied_context = new.clone();
400                session.save(chibi.home_dir())?;
401            }
402            if session.previous_context.as_deref() == Some(old.as_str()) {
403                session.previous_context = Some(new.clone());
404                session.save(chibi.home_dir())?;
405            }
406        }
407        CommandEffect::InspectConfigField {
408            context: ctx,
409            field,
410        } => {
411            let mut cfg = resolve_cli_config(chibi, ctx, ephemeral_username)?;
412            // Re-apply per-invocation overrides so -s is visible via inspect
413            if !input.config_overrides.is_empty() {
414                cfg.core
415                    .apply_overrides_from_pairs(&input.config_overrides)
416                    .map_err(|e| io::Error::new(ErrorKind::InvalidInput, e))?;
417            }
418            match cfg.get_field(field) {
419                Some(value) => handler.emit_result(&value),
420                None => handler.emit_result("(not set)"),
421            }
422        }
423        CommandEffect::InspectConfigList { context: _ } => {
424            handler.emit_result("Inspectable items:");
425            for name in ["system_prompt", "reflection", "todos", "goals", "home"] {
426                handler.emit_result(&format!("  {}", name));
427            }
428            for field in ResolvedConfig::list_fields() {
429                handler.emit_result(&format!("  {}", field));
430            }
431        }
432        CommandEffect::None => {}
433    }
434
435    // --- CLI-specific: check if auto-destroy removed our session context ---
436    if !chibi.app.context_dir(&session.implied_context).exists() {
437        session.implied_context = "default".to_string();
438        session.previous_context = None;
439        session.save(chibi.home_dir())?;
440    }
441
442    // --- CLI-specific: image cache cleanup ---
443    if cli_config.image.cache_enabled {
444        let image_cache_dir = chibi.home_dir().join("image_cache");
445        match image_cache::cleanup_image_cache(
446            &image_cache_dir,
447            cli_config.image.cache_max_bytes,
448            cli_config.image.cache_max_age_days,
449        ) {
450            Ok(removed) if removed > 0 => {
451                if verbose {
452                    eprintln!(
453                        "[Image cache cleanup: removed {} entries (max {} days, max {} MB)]",
454                        removed,
455                        cli_config.image.cache_max_age_days,
456                        cli_config.image.cache_max_bytes / (1024 * 1024),
457                    );
458                }
459            }
460            _ => {}
461        }
462    }
463
464    // --- no-action check ---
465    if !did_action && matches!(command, Command::NoOp) {
466        return Err(io::Error::new(
467            ErrorKind::InvalidInput,
468            "No operation specified",
469        ));
470    }
471
472    Ok(())
473}

475/// Extract --home flag value from args (before full CLI parsing)
476fn extract_home_override(args: &[String]) -> Option<PathBuf> {
477    let mut iter = args.iter();
478    while let Some(arg) = iter.next() {
479        if arg == "--home" {
480            return iter.next().map(PathBuf::from);
481        }
482        if let Some(path) = arg.strip_prefix("--home=") {
483            return Some(PathBuf::from(path));
484        }
485    }
486    None
487}

503#[tokio::main]
504async fn main() -> io::Result<()> {
505    // Check for early flags (before full CLI parsing)
506    let args: Vec<String> = std::env::args().collect();
507
508    let trust_mode = args.iter().any(|a| a == "--trust" || a == "-t");
509    let home_override = extract_home_override(&args);
510    let project_root_override = extract_project_root_override(&args);
511
512    // Parse CLI arguments to ChibiInput
513    let input = cli::parse()?;
514
515    // Handle --debug md=<FILENAME> early (renders markdown and quits, implies -x)
516    if let Some(path) = &input.md_file {
517        let content = std::fs::read_to_string(path).map_err(|e| {
518            io::Error::new(
519                ErrorKind::NotFound,
520                format!("Failed to read file '{}': {}", path, e),
521            )
522        })?;
523        let mut md_cfg = md_config_defaults(true);
524        md_cfg.force_render = input.force_markdown;
525        render_markdown_output(&content, md_cfg)?;
526        return Ok(());
527    }
528
529    // Handle --debug force-markdown
530    let force_markdown = input.force_markdown;
531
532    // Plain handler for load-time events (auto-destroy, cache cleanup, MCP).
533    // A markdown-aware handler is built inside execute_from_input once config is resolved.
534    let load_output = OutputHandler::new(input.verbose_flag);
535
536    let mut chibi = Chibi::load_with_options(
537        LoadOptions {
538            home: home_override,
539            project_root: project_root_override,
540        },
541        &load_output,
542    )?;
543    chibi.set_permission_handler(select_permission_handler(trust_mode));
544    let mut session = Session::load(chibi.home_dir())?;
545
546    execute_from_input(input, &mut chibi, &mut session, force_markdown).await
547}
