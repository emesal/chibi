9#[derive(Debug, Clone, Serialize, Deserialize)]
10pub struct ImageCacheMetadata {
11    pub url: String,
12    pub key: String,
13    pub size_bytes: u64,
14    pub created_at: u64,
15    pub last_accessed_at: u64,
16}

18fn cache_key(url: &str) -> String {
19    let mut hasher = Sha256::new();
20    hasher.update(url.as_bytes());
21    format!("{:x}", hasher.finalize())
22}
23
24fn img_path(cache_dir: &Path, key: &str) -> PathBuf {
25    cache_dir.join(format!("{}.img", key))
26}
27
28fn meta_path(cache_dir: &Path, key: &str) -> PathBuf {
29    cache_dir.join(format!("{}.meta.json", key))
30}
31
32fn now_unix() -> u64 {
33    SystemTime::now()
34        .duration_since(UNIX_EPOCH)
35        .unwrap_or_default()
36        .as_secs()
37}

39/// Return cached image bytes on hit, updating `last_accessed_at` (best-effort).
40/// Returns `None` on miss or any read error.
41pub fn cache_get(cache_dir: &Path, url: &str) -> Option<Vec<u8>> {
42    let key = cache_key(url);
43    let img = img_path(cache_dir, &key);
44    let meta = meta_path(cache_dir, &key);
45
46    let bytes = fs::read(&img).ok()?;
47    let meta_bytes = fs::read(&meta).ok()?;
48    let mut metadata: ImageCacheMetadata = serde_json::from_slice(&meta_bytes).ok()?;
49
50    // Best-effort update of last_accessed_at using atomic write pattern
51    metadata.last_accessed_at = now_unix();
52    let meta_tmp = cache_dir.join(format!("{}.meta.json.tmp", key));
53    if let Ok(json_str) = serde_json::to_string(&metadata) {
54        let _ = fs::write(&meta_tmp, json_str).and_then(|_| fs::rename(&meta_tmp, &meta));
55    }
56
57    Some(bytes)
58}

60/// Store image bytes in the cache. Atomic write via `.tmp` rename.
61/// Triggers cleanup after writing.
62pub fn cache_put(
63    cache_dir: &Path,
64    url: &str,
65    bytes: &[u8],
66    max_bytes: u64,
67    max_age_days: u64,
68) -> io::Result<()> {
69    fs::create_dir_all(cache_dir)?;
70
71    let key = cache_key(url);
72    let img = img_path(cache_dir, &key);
73    let meta = meta_path(cache_dir, &key);
74
75    let now = now_unix();
76    let metadata = ImageCacheMetadata {
77        url: url.to_string(),
78        key: key.clone(),
79        size_bytes: bytes.len() as u64,
80        created_at: now,
81        last_accessed_at: now,
82    };
83
84    let meta_json =
85        serde_json::to_string(&metadata).map_err(|e| io::Error::other(format!("{}", e)))?;
86
87    // Atomic write with fsync (prevents corruption on crash)
88    safe_io::atomic_write(&img, bytes)?;
89    safe_io::atomic_write_text(&meta, &meta_json)?;
90
91    // Trigger cleanup (best-effort)
92    let _ = cleanup_image_cache(cache_dir, max_bytes, max_age_days);
93
94    Ok(())
95}

97/// Two-phase eviction: age-based, then LRU by size.
98/// Also removes orphan `.tmp` files and entries with corrupt sidecars.
99/// Returns the number of entries removed.
100pub fn cleanup_image_cache(
101    cache_dir: &Path,
102    max_bytes: u64,
103    max_age_days: u64,
104) -> io::Result<usize> {
105    if !cache_dir.exists() {
106        return Ok(0);
107    }
108
109    let now = now_unix();
110    let max_age_secs = max_age_days * 86400;
111    let mut removed = 0;
112
113    // Remove orphan .tmp files
114    for entry in fs::read_dir(cache_dir)? {
115        let entry = entry?;
116        let name = entry.file_name();
117        let name_str = name.to_string_lossy();
118        if name_str.ends_with(".tmp") {
119            let _ = fs::remove_file(entry.path());
120        }
121    }
122
123    // Collect all cache entries by reading .meta.json files
124    let mut entries: Vec<(ImageCacheMetadata, PathBuf, PathBuf)> = Vec::new();
125
126    for entry in fs::read_dir(cache_dir)? {
127        let entry = entry?;
128        let name = entry.file_name();
129        let name_str = name.to_string_lossy();
130
131        if !name_str.ends_with(".meta.json") {
132            continue;
133        }
134
135        let meta_file = entry.path();
136        let key = name_str.trim_end_matches(".meta.json").to_string();
137        let img_file = img_path(cache_dir, &key);
138
139        // Try to parse metadata
140        let meta_bytes = match fs::read(&meta_file) {
141            Ok(b) => b,
142            Err(_) => {
143                // Corrupt — remove both files
144                let _ = fs::remove_file(&meta_file);
145                let _ = fs::remove_file(&img_file);
146                removed += 1;
147                continue;
148            }
149        };
150
151        let metadata: ImageCacheMetadata = match serde_json::from_slice(&meta_bytes) {
152            Ok(m) => m,
153            Err(_) => {
154                // Corrupt sidecar — remove both files
155                let _ = fs::remove_file(&meta_file);
156                let _ = fs::remove_file(&img_file);
157                removed += 1;
158                continue;
159            }
160        };
161
162        // Check if .img file exists; if not, remove orphan metadata
163        if !img_file.exists() {
164            let _ = fs::remove_file(&meta_file);
165            removed += 1;
166            continue;
167        }
168
169        entries.push((metadata, img_file, meta_file));
170    }
171
172    // Phase 1: age eviction
173    let mut remaining: Vec<(ImageCacheMetadata, PathBuf, PathBuf)> = Vec::new();
174    for (meta, img_file, meta_file) in entries {
175        if now.saturating_sub(meta.created_at) > max_age_secs {
176            let _ = fs::remove_file(&img_file);
177            let _ = fs::remove_file(&meta_file);
178            removed += 1;
179        } else {
180            remaining.push((meta, img_file, meta_file));
181        }
182    }
183
184    // Phase 2: LRU eviction if over size limit
185    let total_size: u64 = remaining.iter().map(|(m, _, _)| m.size_bytes).sum();
186    if total_size > max_bytes {
187        // Sort by last_accessed_at ascending (oldest accessed first)
188        remaining.sort_by_key(|(m, _, _)| m.last_accessed_at);
189
190        let mut current_size = total_size;
191        for (meta, img_file, meta_file) in &remaining {
192            if current_size <= max_bytes {
193                break;
194            }
195            let _ = fs::remove_file(img_file);
196            let _ = fs::remove_file(meta_file);
197            current_size = current_size.saturating_sub(meta.size_bytes);
198            removed += 1;
199        }
200    }
201
202    Ok(removed)
203}

205/// Remove the entire cache directory.
206#[allow(dead_code)]
207pub fn clear_image_cache(cache_dir: &Path) -> io::Result<()> {
208    if cache_dir.exists() {
209        fs::remove_dir_all(cache_dir)?;
210    }
211    Ok(())
212}
