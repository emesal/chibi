1//! Output handling for the CLI.
2//!
3//! This module provides `OutputHandler` which handles outputting
4//! text results and diagnostics for the terminal.
5//! Implements `OutputSink` directly — all output goes through trait methods.
6
7use chibi_core::OutputSink;
8use chibi_core::context::TranscriptEntry;
9use chibi_core::output::CommandEvent;
10use std::io::{self, IsTerminal, Write};
11
12use crate::markdown::{MarkdownConfig, MarkdownStream};
13
14/// CLI output handler — text to stdout, diagnostics to stderr.
15///
16/// Implements `OutputSink` directly; all output goes through trait methods.
17/// Always operates in text mode — JSON output belongs to chibi-json.
18///
19/// When `md_config` is `Some`, `emit_markdown` renders content through
20/// `MarkdownStream` (used by `emit_entry` for message content in ShowLog).
21pub struct OutputHandler {
22    verbose: bool,
23    md_config: Option<MarkdownConfig>,
24}
25
26impl OutputHandler {
27    /// Create a new output handler without markdown rendering.
28    pub fn new(verbose: bool) -> Self {
29        Self {
30            verbose,
31            md_config: None,
32        }
33    }
34
35    /// Create a new output handler with markdown rendering enabled.
36    pub fn with_markdown(verbose: bool, md_config: MarkdownConfig) -> Self {
37        Self {
38            verbose,
39            md_config: Some(md_config),
40        }
41    }
42}
43
44impl OutputSink for OutputHandler {
45    fn emit_result(&self, content: &str) {
46        println!("{}", content);
47    }
48
49    fn emit_event(&self, event: CommandEvent) {
50        if !self.verbose {
51            return;
52        }
53        let text = match &event {
54            CommandEvent::AutoDestroyed { count } => {
55                format!("[Auto-destroyed {} expired context(s)]", count)
56            }
57            CommandEvent::CacheCleanup {
58                removed,
59                max_age_days,
60            } => format!(
61                "[Auto-cleanup: removed {} old cache entries (older than {} days)]",
62                removed,
63                max_age_days + 1
64            ),
65            CommandEvent::SystemPromptSet { context } => {
66                format!("[System prompt set for context '{}']", context)
67            }
68            CommandEvent::UsernameSaved { username, context } => {
69                format!("[Username '{}' saved to context '{}']", username, context)
70            }
71            CommandEvent::ModelSet { model, context } => {
72                format!("[Model '{}' saved to context '{}']", model, context)
73            }
74            CommandEvent::InboxEmpty { context } => {
75                format!("[No messages in inbox for '{}']", context)
76            }
77            CommandEvent::InboxProcessing { count, context } => format!(
78                "[Processing {} message(s) from inbox for '{}']",
79                count, context
80            ),
81            CommandEvent::AllInboxesEmpty => "[No messages in any inbox.]".to_string(),
82            CommandEvent::InboxesProcessed { count } => {
83                format!("[Processed inboxes for {} context(s).]", count)
84            }
85            CommandEvent::McpToolsLoaded { count } => format!("[MCP: {} tools loaded]", count),
86            CommandEvent::McpBridgeUnavailable { reason } => {
87                format!("[MCP: bridge unavailable: {}]", reason)
88            }
89            CommandEvent::CompactionStarted {
90                context,
91                message_count,
92            } => format!("[Compacting '{}': {} messages]", context, message_count),
93            CommandEvent::CompactionComplete {
94                context,
95                archived,
96                remaining,
97            } => format!(
98                "[Compaction complete '{}': {} archived, {} remaining]",
99                context, archived, remaining
100            ),
101            CommandEvent::RollingCompactionDecision { archived } => {
102                format!(
103                    "[Rolling compaction: LLM selected {} messages to archive]",
104                    archived
105                )
106            }
107            CommandEvent::RollingCompactionFallback { drop_percentage } => format!(
108                "[Rolling compaction: LLM decision failed, falling back to dropping oldest {}%]",
109                drop_percentage
110            ),
111            CommandEvent::RollingCompactionComplete {
112                archived,
113                remaining,
114            } => format!(
115                "[Rolling compaction complete: {} archived, {} remaining]",
116                archived, remaining
117            ),
118            CommandEvent::LoadSummary {
119                builtin_count,
120                builtin_names,
121                plugin_count,
122                plugin_names,
123            } => {
124                let mut lines = format!(
125                    "[Built-in ({}): {}]",
126                    builtin_count,
127                    builtin_names.join(", ")
128                );
129                if *plugin_count == 0 {
130                    lines.push_str("\n[No plugins loaded]");
131                } else {
132                    lines.push_str(&format!(
133                        "\n[Plugins ({}): {}]",
134                        plugin_count,
135                        plugin_names.join(", ")
136                    ));
137                }
138                lines
139            }
140        };
141        eprintln!("{}", text);
142    }
143
144    fn newline(&self) {
145        println!();
146    }
147
148    fn emit_entry(&self, entry: &TranscriptEntry) -> io::Result<()> {
149        use chibi_core::context;
150
151        match entry.entry_type.as_str() {
152            context::ENTRY_TYPE_MESSAGE => {
153                self.emit_result(&format!("[{}]", entry.from.to_uppercase()));
154                self.emit_markdown(&entry.content)?;
155                self.newline();
156            }
157            context::ENTRY_TYPE_TOOL_CALL => {
158                if self.verbose {
159                    self.emit_result(&format!("[TOOL CALL: {}]\n{}\n", entry.to, entry.content));
160                } else {
161                    let args_preview = if entry.content.len() > 60 {
162                        format!("{}...", &entry.content[..60])
163                    } else {
164                        entry.content.clone()
165                    };
166                    self.emit_result(&format!("[TOOL: {}] {}", entry.to, args_preview));
167                }
168            }
169            context::ENTRY_TYPE_TOOL_RESULT => {
170                if self.verbose {
171                    self.emit_result(&format!(
172                        "[TOOL RESULT: {}]\n{}\n",
173                        entry.from, entry.content
174                    ));
175                } else {
176                    let size = entry.content.len();
177                    let size_str = if size > 1024 {
178                        format!("{:.1}kb", size as f64 / 1024.0)
179                    } else {
180                        format!("{}b", size)
181                    };
182                    self.emit_result(&format!("  -> {}", size_str));
183                }
184            }
185            "compaction" => {
186                if self.verbose {
187                    self.emit_result(&format!("[COMPACTION]: {}\n", entry.content));
188                }
189            }
190            _ => {
191                if self.verbose {
192                    self.emit_result(&format!(
193                        "[{}]: {}\n",
194                        entry.entry_type.to_uppercase(),
195                        entry.content
196                    ));
197                }
198            }
199        }
200        Ok(())
201    }
202
203    fn confirm(&self, prompt: &str) -> bool {
204        let stdin = io::stdin();
205        if !stdin.is_terminal() {
206            return false;
207        }
208
209        eprint!("{} [y/N] ", prompt);
210        io::stderr().flush().ok();
211
212        let mut input = String::new();
213        if stdin.read_line(&mut input).is_err() {
214            return false;
215        }
216
217        matches!(input.trim().to_lowercase().as_str(), "y" | "yes")
218    }
219
220    fn emit_markdown(&self, content: &str) -> io::Result<()> {
221        if let Some(cfg) = &self.md_config {
222            let mut md = MarkdownStream::new(cfg.clone());
223            md.write_chunk(content)?;
224            md.finish()?;
225        } else {
226            self.emit_result(content);
227        }
228        Ok(())
229    }
230}
